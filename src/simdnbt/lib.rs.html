30    fn gzip_decode(src: &[u8]) -> Vec<u8> {
31        let mut src_slice = src;
32        let mut decoded_src_decoder = GzDecoder::new(&mut src_slice);
33        let mut decoded_src = Vec::new();
34        decoded_src_decoder.read_to_end(&mut decoded_src).unwrap();
35        decoded_src
36    }
37
38    fn test_decodes_equally(src: &[u8]) {
39        let nbt_borrow = crate::borrow::read(&mut Cursor::new(src))
40            .unwrap()
41            .unwrap()
42            .as_compound()
43            .to_owned();
44        let nbt_owned = crate::owned::read(&mut Cursor::new(src))
45            .unwrap()
46            .unwrap()
47            .as_compound();
48        // (there's another test in owned to make sure that PartialEq actually works)
49        assert_eq!(nbt_borrow, nbt_owned);
50    }
