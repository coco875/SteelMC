19pub enum CommandNode {
20    Root {
21        children: Vec<i32>,
22    },
23    Literal {
24        children: Vec<i32>,
25        redirects_to: Option<i32>,
26        name: &'static str,
27        is_executable: bool,
28    },
29    Argument {
30        children: Vec<i32>,
31        redirects_to: Option<i32>,
32        name: &'static str,
33        is_executable: bool,
34        parser: ArgumentType,
35        suggestions_type: Option<SuggestionType>,
36    },
37}
38
39impl CommandNode {
40    const FLAG_IS_EXECUTABLE: u8 = 4;
41    const FLAG_HAS_REDIRECT: u8 = 8;
42    const FLAG_HAS_SUGGESTION_TYPE: u8 = 16;
43
44    pub fn new_root() -> Self {
45        Self::Root {
46            children: Vec::new(),
47        }
48    }
49
50    pub fn new_literal(info: CommandNodeInfo, name: &'static str) -> Self {
51        Self::Literal {
52            children: info.children,
53            name,
54            is_executable: info.is_executable,
55            redirects_to: info.redirects_to,
56        }
57    }
58
59    pub fn new_argument(
60        info: CommandNodeInfo,
61        name: &'static str,
62        argument: (ArgumentType, Option<SuggestionType>),
63    ) -> Self {
64        Self::Argument {
65            children: info.children,
66            name,
67            is_executable: info.is_executable,
68            redirects_to: info.redirects_to,
69            parser: argument.0,
70            suggestions_type: argument.1,
71        }
72    }
73
74    fn flags(&self) -> u8 {
75        let (mut flags, is_executable, has_redirect, has_suggestions_type) = match self {
76            CommandNode::Root { .. } => (0, false, false, false),
77            CommandNode::Literal {
78                is_executable,
79                redirects_to,
80                ..
81            } => (1, *is_executable, redirects_to.is_some(), false),
82            CommandNode::Argument {
83                is_executable,
84                redirects_to: r,
85                suggestions_type,
86                ..
87            } => (2, *is_executable, r.is_some(), suggestions_type.is_some()),
88        };
89
90        if is_executable {
91            flags |= Self::FLAG_IS_EXECUTABLE
92        }
93        if has_redirect {
94            flags |= Self::FLAG_HAS_REDIRECT
95        }
96        if has_suggestions_type {
97            flags |= Self::FLAG_HAS_SUGGESTION_TYPE
98        }
99        flags
100    }
101
102    pub fn set_children(&mut self, children: Vec<i32>) {
103        match self {
104            CommandNode::Root { children: c } => *c = children,
105            CommandNode::Literal { children: c, .. } => *c = children,
106            CommandNode::Argument { children: c, .. } => *c = children,
107        }
108    }
109
110    fn children(&self) -> &[i32] {
111        match self {
112            CommandNode::Root { children } => children,
113            CommandNode::Literal { children, .. } => children,
114            CommandNode::Argument { children, .. } => children,
115        }
116    }
117
118    fn redirects_to(&self) -> &Option<i32> {
119        match self {
120            CommandNode::Root { .. } => &None,
121            CommandNode::Literal { redirects_to, .. } => redirects_to,
122            CommandNode::Argument { redirects_to, .. } => redirects_to,
123        }
124    }
125
126    fn name(&self) -> Option<&'static str> {
127        match self {
128            CommandNode::Root { .. } => None,
129            CommandNode::Literal { name, .. } => Some(*name),
130            CommandNode::Argument { name, .. } => Some(*name),
131        }
132    }
133}
134
135impl WriteTo for CommandNode {
136    fn write(&self, writer: &mut impl Write) -> Result<()> {
137        writer.write_all(&self.flags().to_be_bytes())?;
138        let children = self.children();
139        VarInt(children.len() as i32).write(writer)?;
140        for child in children {
141            VarInt(*child).write(writer)?;
142        }
143
144        if let Some(redirects_to) = self.redirects_to() {
145            VarInt(*redirects_to).write(writer)?;
146        }
147
148        if let Some(name) = self.name() {
149            name.write_prefixed::<VarInt>(writer)?;
150        }
151
152        if let CommandNode::Argument {
153            parser,
154            suggestions_type,
155            ..
156        } = self
157        {
158            parser.write(writer)?;
159
160            if let Some(suggestions_type) = suggestions_type {
161                suggestions_type.as_str().write_prefixed::<VarInt>(writer)?;
162            }
163        }
164
165        Ok(())
166    }
167}

169pub struct CommandNodeInfo {
170    children: Vec<i32>,
171    is_executable: bool,
172    redirects_to: Option<i32>,
173}
174
175impl CommandNodeInfo {
176    pub fn new(children: Vec<i32>) -> Self {
177        Self {
178            children,
179            is_executable: false,
180            redirects_to: None,
181        }
182    }
183
184    pub fn new_executable() -> Self {
185        Self {
186            children: Vec::new(),
187            is_executable: true,
188            redirects_to: None,
189        }
190    }
191
192    pub fn new_redirect(redirects_to: i32) -> Self {
193        Self {
194            children: Vec::new(),
195            is_executable: false,
196            redirects_to: Some(redirects_to),
197        }
198    }
199
200    pub fn chain(mut self, mut other: Self) -> Self {
201        self.children.append(&mut other.children);
202        self.is_executable |= other.is_executable;
203        self
204    }

295#[derive(Debug, Clone, Copy)]
296pub enum ArgumentStringTypeBehavior {
297    SingleWord,
298    QuotablePhrase,
299    GreedyPhrase,
300}
301
302impl WriteTo for ArgumentType {
303    fn write(&self, writer: &mut impl Write) -> Result<()> {
304        // Safety: Since Self is repr(u32), it is guaranteed to hold the discriminant in the first 4 bytes
305        // See https://doc.rust-lang.org/reference/items/enumerations.html#pointer-casting
306        let id = unsafe { *(self as *const Self as *const i32) };
307        VarInt(id).write(writer)?;
308
309        match self {
310            Self::Float { min, max } => Self::write_min_max(*min, *max, writer),
311            Self::Double { min, max } => Self::write_min_max(*min, *max, writer),
312            Self::Integer { min, max } => Self::write_min_max(*min, *max, writer),
313            Self::Long { min, max } => Self::write_min_max(*min, *max, writer),
314            Self::String { behavior } => {
315                let i = match behavior {
316                    ArgumentStringTypeBehavior::SingleWord => 0,
317                    ArgumentStringTypeBehavior::QuotablePhrase => 1,
318                    ArgumentStringTypeBehavior::GreedyPhrase => 2,
319                };
320                VarInt(i).write(writer)
321            }
322            Self::Entity { flags } => flags.write(writer),
323            Self::ScoreHolder { flags } => flags.write(writer),
324            Self::Time { min } => min.write(writer),
325            Self::ResourceOrTag { identifier } => identifier.write_prefixed::<VarInt>(writer),
326            Self::ResourceOrTagKey { identifier } => identifier.write_prefixed::<VarInt>(writer),
327            Self::Resource { identifier } => identifier.write_prefixed::<VarInt>(writer),
328            Self::ResourceKey { identifier } => identifier.write_prefixed::<VarInt>(writer),
329            _ => Ok(()),
330        }
331    }
332}
333
334impl ArgumentType {
335    fn write_min_max<T: WriteTo>(
336        min: Option<T>,
337        max: Option<T>,
338        writer: &mut impl Write,
339    ) -> Result<()> {
340        // none = 0
341        // min = 1
342        // max = 2
343        // min & max = 3
344        (min.is_some() as u8 + max.is_some() as u8 + max.is_some() as u8).write(writer)?;
345
346        if let Some(min) = min {
347            min.write(writer)?;
348        }
349        if let Some(max) = max {
350            max.write(writer)?;
351        }
352
353        Ok(())
354    }
355}
