1use crate::{ConcatMixed, ConstChoice, ConstCtOption, Int, Uint};
2
3impl<const LIMBS: usize> Uint<LIMBS> {
4    /// Compute "wide" multiplication between an [`Uint`] and [`Int`] as 3-tuple `(lo, hi, negate)`.
5    /// The `(lo, hi)` components contain the _magnitude of the product_, with sizes
6    /// corresponding to the sizes of the operands; `negate` indicates whether the result should be
7    /// negated when converted from [`Uint`] to [`Int`].
8    ///
9    /// Note: even if `negate` is truthy, the magnitude might be zero!
10    pub const fn widening_mul_int<const RHS_LIMBS: usize>(
11        &self,
12        rhs: &Int<RHS_LIMBS>,
13    ) -> (Uint<LIMBS>, Uint<RHS_LIMBS>, ConstChoice) {
14        let (rhs_abs, rhs_sgn) = rhs.abs_sign();
15        let (lo, hi) = self.widening_mul(&rhs_abs);
16        (lo, hi, rhs_sgn)
17    }
18
19    /// Multiply `self` by [`Int`] `rhs`, returning a concatenated "wide" result.
20    pub const fn concatenating_mul_int<const RHS_LIMBS: usize, const WIDE_LIMBS: usize>(
21        &self,
22        rhs: &Int<RHS_LIMBS>,
23    ) -> Int<WIDE_LIMBS>
24    where
25        Uint<LIMBS>: ConcatMixed<Uint<RHS_LIMBS>, MixedOutput = Uint<WIDE_LIMBS>>,
26    {
27        let (rhs_abs, rhs_sign) = rhs.abs_sign();
28        let product_abs = self.concatenating_mul(&rhs_abs);
29
30        // always fits
31        *product_abs.wrapping_neg_if(rhs_sign).as_int()
32    }
33
34    /// Checked multiplication of `self` with [`Int`] `rhs`.
35    pub fn checked_mul_int<const RHS_LIMBS: usize>(
36        &self,
37        rhs: &Int<RHS_LIMBS>,
38    ) -> ConstCtOption<Int<LIMBS>> {
39        let (lo, hi, is_negative) = self.widening_mul_int(rhs);
40        Int::new_from_abs_sign(lo, is_negative).and_choice(hi.is_nonzero().not())
41    }
42}

44#[cfg(test)]
45mod tests {
46    use crate::{ConstChoice, I64, I128, I256, U64, U128};
47
48    #[test]
49    fn widening_mul_int() {
50        assert_eq!(
51            U128::MAX.widening_mul_int(&I64::from_i64(-55)),
52            (
53                U128::from_be_hex("FFFFFFFFFFFFFFFFFFFFFFFFFFFFFFC9"),
54                U64::from_u64(54),
55                ConstChoice::TRUE
56            )
57        )
58    }
59
60    #[test]
61    fn concatenating_mul_int() {
62        assert_eq!(
63            U128::MAX.concatenating_mul_int(&I128::from_i64(-55)),
64            I256::from_be_hex("FFFFFFFFFFFFFFFFFFFFFFFFFFFFFFC900000000000000000000000000000037")
65        );
66        assert_eq!(
67            U128::MAX.concatenating_mul_int(&I128::MAX),
68            I256::from_be_hex("7FFFFFFFFFFFFFFFFFFFFFFFFFFFFFFE80000000000000000000000000000001")
69        );
70        assert_eq!(
71            U128::MAX.concatenating_mul_int(&I128::MIN),
72            I256::from_be_hex("8000000000000000000000000000000080000000000000000000000000000000")
73        );
74    }
75
76    #[test]
77    fn checked_mul_int() {
78        assert_eq!(
79            U64::from_be_hex("00000000FFFFFFFF")
80                .checked_mul_int(&I64::from_be_hex("FFFFFFFF80000000"))
81                .unwrap(),
82            I64::from_be_hex("8000000080000000")
83        );
84        assert!(bool::from(U64::MAX.checked_mul_int(&I128::ONE).is_none()));
85    }
