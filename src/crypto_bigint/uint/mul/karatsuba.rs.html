1//! Karatsuba multiplication:
2//!
3//! This is a method which reduces the complexity of multiplication from O(n^2) to O(n^1.585).
4//! For smaller numbers, it is best to stick to schoolbook multiplication, taking advantage
5//! of better cache locality and avoiding recursion.
6//!
7//! In general, we consider the multiplication of two numbers of an equal size, `n` bits.
8//! Setting b = 2^(n/2), then we can decompose the values:
9//!   x•y = (x0 + x1•b)(y0 + y1•b)
10//!
11//! This equation is equivalent to a linear combination of three products of size `n/2`, which
12//! may each be reduced by applying the same optimization.
13//! Setting z0 = x0y0, z1 = (x0 + x1)(y1 + y0), z2 = x1y1:
14//!   x•y = z0 + (z1 - z0 - z2)•b + z2•b^2
15//!
16//! Considering each sub-product as a tuple of integers `(lo, hi)`, the product is calculated as
17//! follows (with appropriate carries):
18//!   [z0.0, z0.1 + z1.0 - z0.0 - z2.0, z1.1 - z0.1 + z2.0 - z2.1, z2.1]
19//!
20//! Squaring uses a similar optimization, breaking the operation down into two half-size
21//! squarings and a half-size multiplication:
22//!
23//!   x^2 = (x0 + x1•b)^2 = x0^2 + 2x0x1•b + (x1•b)^2
24//!
25//! The wrapping implementations of these methods employ variations on these algorithms. Because
26//! the fixed-size implementations for power-of-two `Uint` sizes are optimized well by the
27//! compiler, the dynamic implementations break down large multiplications into calls to these
28//! optimized methods.

35/// Compute a wide multiplication result for fixed-size arguments. Common power-of-two
36/// limb counts are implemented explicitly and used as a basis for general multiplication.
37///
38/// The limb counts of `lhs` and `rhs` must be `LHS` and `RHS` respectively.
39pub const fn widening_mul_fixed<const LHS: usize, const RHS: usize>(
40    lhs: &UintRef,
41    rhs: &UintRef,
42) -> (Uint<LHS>, Uint<RHS>) {
43    debug_assert!(lhs.nlimbs() == LHS && rhs.nlimbs() == RHS);
44
45    // Perform the core Karatsuba multiplication for fixed-size arguments, returning
46    // a wide result. The size of the arguments is halved on each recursion.
47    #[inline]
48    const fn reduce<const LHS: usize, const RHS: usize, const HALF: usize>(
49        x: &UintRef,
50        y: &UintRef,
51    ) -> (Uint<LHS>, Uint<RHS>) {
52        assert!(LHS <= RHS && LHS == HALF * 2);
53        let (x0, x1) = x.split_at(HALF);
54        let (y0, y1) = y.split_at(HALF);
55
56        // Calculate z0 = x0•y0
57        let z0 = widening_mul_fixed(x0, y0);
58        // Calculate z2 = x1•y1
59        let z2 = widening_mul_fixed(x1, y1);
60
61        // Calculate z1 = (x0 + x1)(y0 + y1)
62        let (mut l0, mut l1) = (Uint::<HALF>::ZERO, Uint::<HALF>::ZERO);
63        let (mut l0c, mut l1c) = (Limb::ZERO, Limb::ZERO);
64        let mut i = 0;
65        while i < HALF {
66            (l0.limbs[i], l0c) = x0.0[i].carrying_add(x1.0[i], l0c);
67            (l1.limbs[i], l1c) = y0.0[i].carrying_add(y1.0[i], l1c);
68            i += 1;
69        }
70        let z1 = widening_mul_fixed(l0.as_uint_ref(), l1.as_uint_ref());
71
72        // Middle terms of the result
73        let (mut s0, mut s1) = (z0.1, z2.0);
74        let (mut c, mut carry);
75
76        // Add z1•b
77        (s0, c) = s0.carrying_add(&z1.0, Limb::ZERO);
78        (s1, c) = s1.carrying_add(&z1.1, c);
79        carry = c;
80        // Correct for overflowing terms in z1 by adding (l0c•l1 + l1c•l0)•b^2
81        (s1, c) = s1.carrying_add(
82            &Uint::select(&Uint::ZERO, &l0, l1c.is_nonzero()),
83            Limb::ZERO,
84        );
85        carry = carry.wrapping_add(c);
86        (s1, c) = s1.carrying_add(
87            &Uint::select(&Uint::ZERO, &l1, l0c.is_nonzero()),
88            Limb::ZERO,
89        );
90        carry = carry.wrapping_add(c);
91        carry = carry.wrapping_add(l0c.bitand(l1c));
92
93        // Subtract (z0 + z2)•b
94        (s0, c) = s0.borrowing_sub(&z0.0, Limb::ZERO);
95        (s1, c) = s1.borrowing_sub(&z0.1, c);
96        carry = carry.wrapping_add(c);
97        (s0, c) = s0.borrowing_sub(&z2.0, Limb::ZERO);
98        (s1, c) = s1.borrowing_sub(&z2.1, c);
99        carry = carry.wrapping_add(c);
100
101        (
102            concat_wide(&z0.0, &s0),
103            concat_wide(&s1, &z2.1.overflowing_add_limb(carry).0),
104        )
105    }
106
107    // Handle relatively small integers
108    if LHS < MIN_STARTING_LIMBS || RHS < MIN_STARTING_LIMBS {
109        let (mut lo, mut hi) = (Uint::ZERO, Uint::ZERO);
110        schoolbook::mul_wide(
111            lhs.as_slice(),
112            rhs.as_slice(),
113            lo.as_mut_limbs(),
114            hi.as_mut_limbs(),
115        );
116        (lo, hi)
117    }
118    // Handle optimized integer sizes. These calls are determined statically, so
119    // only the relevant implementation should be inlined.
120    else if LHS == RHS {
121        match LHS {
122            16 => reduce::<LHS, RHS, 8>(lhs, rhs),
123            32 => reduce::<LHS, RHS, 16>(lhs, rhs),
124            64 => reduce::<LHS, RHS, 32>(lhs, rhs),
125            128 => reduce::<LHS, RHS, 64>(lhs, rhs),
126            256 => reduce::<LHS, RHS, 128>(lhs, rhs),
127            _ => {
128                let mut lo_hi = [[Limb::ZERO; LHS]; 2];
129                wrapping_mul(lhs, rhs, UintRef::new_flattened_mut(&mut lo_hi), false);
130                (Uint::new(lo_hi[0]), Uint::new(lo_hi[1]).resize::<RHS>())
131            }
132        }
133    }
134    // If LHS < RHS, then multiply the evenly matched limbs (which may be optimized) and
135    // add the product of the remaining RHS limbs with the LHS.
136    else if LHS < RHS {
137        let (y0, y1) = rhs.split_at(LHS);
138        let (lo, mut hi) = resize_wide(widening_mul_fixed::<LHS, LHS>(lhs, y0));
139        wrapping_mul(lhs, y1, hi.as_mut_uint_ref(), true);
140        (lo, hi)
141    }
142    // When LHS > RHS, swap the arguments and adjust the result.
143    else {
144        // LHS > RHS, swap arguments
145        let (lo, hi) = widening_mul_fixed::<RHS, LHS>(rhs, lhs);
146        // Need to repartition from (RHS, LHS) to (LHS, RHS)
147        let mut lo = lo.resize::<LHS>();
148        lo.as_mut_uint_ref()
149            .trailing_mut(RHS)
150            .copy_from(hi.as_uint_ref().leading(LHS - RHS));
151        (
152            lo,
153            hi.wrapping_shr_by_limbs_vartime((LHS - RHS) as u32)
154                .resize::<RHS>(),
155        )
156    }
157}

159/// Compute a wrapping multiplication result for a fixed LHS and dynamic RHS argument. Common
160/// power-of-two limb counts are implemented explicitly and used as a basis for general multiplication.
161///
162/// The limb count of `lhs` must be `LHS`.
163#[inline]
164pub const fn wrapping_mul_fixed<const LHS: usize>(
165    lhs: &UintRef,
166    rhs: &UintRef,
167) -> (Uint<LHS>, Limb) {
168    debug_assert!(lhs.nlimbs() == LHS);
169
170    // Perform an optimized Karatsuba multiplication returning the lower half of the limbs
171    // of the product.
172    //   x•y = (x0 + x1•b)(y0 + y1•b)
173    //       = x0y0 + x0y1•b + y0x1•b + x1y1•b^2
174    // For the wrapped product, computing the upper half of (x0y1•b + 0x1•b) and x1y1•b^2
175    // can be skipped, resulting in two more half-size wrapping multiplications.
176    #[inline]
177    const fn reduce<const LHS: usize, const HALF: usize>(
178        x: &UintRef,
179        y: &UintRef,
180    ) -> (Uint<LHS>, Limb) {
181        debug_assert!(LHS == HALF * 2);
182        let (x0, x1) = x.split_at(HALF);
183        let (y0, y1) = y.leading(LHS).split_at(HALF);
184
185        // Calculate z0 = x0•y0
186        let z0 = widening_mul_fixed::<HALF, HALF>(x0, y0);
187        // Calculate z1 = x0•y1
188        let (z1, z1c) = wrapping_mul_fixed::<HALF>(x0, y1);
189        // Calculate z2 = x1•y0
190        let (z2, z2c) = wrapping_mul_fixed::<HALF>(x1, y0);
191
192        let (hi, c1) = z0.1.carrying_add(&z1, Limb::ZERO);
193        let (hi, c2) = hi.carrying_add(&z2, Limb::ZERO);
194        let carry = z1c.wrapping_add(z2c).wrapping_add(c1).wrapping_add(c2);
195
196        (concat_wide(&z0.0, &hi), carry)
197    }
198
199    // Handle smaller integer sizes
200    if LHS < MIN_STARTING_LIMBS || rhs.nlimbs() < MIN_STARTING_LIMBS {
201        let mut lo = Uint::ZERO;
202        let carry = schoolbook::wrapping_mul_add(lhs.as_slice(), rhs.as_slice(), lo.as_mut_limbs());
203        return (lo, carry);
204    }
205    // Because only matching limbs are considered, any LHS <= RHS is treated as a fixed-size
206    // wrapping multiplication.
207    else if LHS <= rhs.nlimbs() {
208        match LHS {
209            16 => return reduce::<LHS, 8>(lhs, rhs),
210            32 => return reduce::<LHS, 16>(lhs, rhs),
211            64 => return reduce::<LHS, 32>(lhs, rhs),
212            128 => return reduce::<LHS, 64>(lhs, rhs),
213            256 => return reduce::<LHS, 128>(lhs, rhs),
214            _ => {}
215        }
216    }
217
218    // For LHS > RHS and less optimized sizes, defer to the dynamic implementation.
219    let mut lo = Uint::ZERO;
220    let carry = wrapping_mul(lhs, rhs, lo.as_mut_uint_ref(), false);
221    (lo, carry)
222}

224/// Compute a wide squaring result for a fixed-size argument. Common power-of-two limb counts
225/// are implemented explicitly and used as a basis for general squaring.
226///
227/// The limb count of `uint` must be `LIMBS`.
228pub const fn widening_square_fixed<const LIMBS: usize>(
229    uint: &UintRef,
230) -> (Uint<LIMBS>, Uint<LIMBS>) {
231    debug_assert!(
232        uint.nlimbs() == LIMBS,
233        "invalid arguments to widening_square_fixed"
234    );
235
236    // Perform an optimized Karatsuba squaring returning the a wide result.
237    // The size of the arguments is halved on each recursion.
238    #[inline]
239    const fn reduce<const LIMBS: usize, const HALF: usize>(
240        x: &UintRef,
241    ) -> (Uint<LIMBS>, Uint<LIMBS>) {
242        debug_assert!(LIMBS == HALF * 2);
243        let (x0, x1) = x.split_at(HALF);
244
245        // Calculate z0 = x0^2
246        let z0 = widening_square_fixed::<HALF>(x0);
247        // Calculate z1 = x0•x1
248        let mut z1 = widening_mul_fixed::<HALF, HALF>(x0, x1);
249        // Calculate z2 = x1^2
250        let z2 = widening_square_fixed::<HALF>(x1);
251
252        let (mut c, mut carry);
253        // Double z1
254        (z1.0, c) = z1.0.overflowing_shl1();
255        (z1.1, carry) = z1.1.carrying_shl1(c);
256        // Add z0.1, z2.0 to z1
257        (z1.0, c) = z1.0.carrying_add(&z0.1, Limb::ZERO);
258        (z1.1, c) = z1.1.carrying_add(&z2.0, c);
259        carry = carry.wrapping_add(c);
260
261        (
262            concat_wide(&z0.0, &z1.0),
263            concat_wide(&z1.1, &z2.1.overflowing_add_limb(carry).0),
264        )
265    }
266
267    // Handle smaller integer sizes
268    if LIMBS < MIN_STARTING_LIMBS {
269        let (mut lo, mut hi) = (Uint::ZERO, Uint::ZERO);
270        schoolbook::square_wide(uint.as_slice(), lo.as_mut_limbs(), hi.as_mut_limbs());
271        (lo, hi)
272    }
273    // Forward to optimized implementations or the dynamic implementation. This choice should
274    // be determined statically, resulting in an optimized method for each integer size.
275    else {
276        match LIMBS {
277            16 => reduce::<LIMBS, 8>(uint),
278            32 => reduce::<LIMBS, 16>(uint),
279            64 => reduce::<LIMBS, 32>(uint),
280            128 => reduce::<LIMBS, 64>(uint),
281            256 => reduce::<LIMBS, 128>(uint),
282            _ => {
283                let mut lo_hi = [[Limb::ZERO; LIMBS]; 2];
284                wrapping_square(uint, UintRef::new_flattened_mut(&mut lo_hi));
285                (Uint::new(lo_hi[0]), Uint::new(lo_hi[1]))
286            }
287        }
288    }
289}

291/// Compute a wrapping square for a fixed-size argument. This method defers to the dynamic
292/// implementation, but inlining results in the selection of an efficient algorithm.
293///
294/// The limb count of `uint` must be `LIMBS`.
295#[inline]
296pub const fn wrapping_square_fixed<const LIMBS: usize>(uint: &UintRef) -> (Uint<LIMBS>, Limb) {
297    let mut lo = Uint::ZERO;
298    let carry = wrapping_square(uint, lo.as_mut_uint_ref());
299    (lo, carry)
300}

302/// Multiply two limb slices, placing the potentially-truncated result in `out`.
303///
304/// `lhs` and `rhs` may have different lengths.
305/// `out` must have a limb count less than or equal to the width of the wide product of `lhs` and `rhs`.
306/// `add` determines whether the result is simply written to `out`, or added to its current value.
307///
308/// The return value is the carry result of the wrapped multiplication (and addition, if performed).
309#[inline]
310pub const fn wrapping_mul(lhs: &UintRef, rhs: &UintRef, out: &mut UintRef, add: bool) -> Limb {
311    assert!(
312        lhs.nlimbs() + rhs.nlimbs() >= out.nlimbs(),
313        "invalid arguments to wrapping_mul"
314    );
315
316    // Use the optimized multiplication for `LIMBS` limbs to break down the calculation of the product.
317    const fn reduce<const LIMBS: usize>(
318        x: &UintRef,
319        y: &UintRef,
320        out: &mut UintRef,
321        add: bool,
322    ) -> Limb {
323        let out_len = out.nlimbs();
324
325        // Perform a wrapping multiplication, returning up to the limb count of `x`.
326        // The inputs `x` and `y` are split such that the products of their terms line up nicely
327        // with the boundary of the wrapped result.
328        // The product is computed as:
329        //   x•y = (x0 + x1•b)(y0 + y1•c)
330        //       = x0y + x1y0•b + x1y1•bc
331        // where x1 and y0 have the same (power-of-two) size, the product `x1y0•b` wraps evenly at
332        // the boundary of x's limb count, and `x1y1•bc` is entirely outside the wrapped result.
333        if out_len <= x.nlimbs() {
334            let (x0, x1) = x.leading(out_len).split_at(out_len - LIMBS);
335            let y0 = y.leading(LIMBS);
336
337            // Compute `x1y0` and place it at the end of the output buffer.
338            let (z1, mut carry) = wrapping_mul_fixed::<LIMBS>(x1, y0);
339            let assign = out.trailing_mut(out_len - LIMBS);
340            if add {
341                let c = assign.carrying_add_assign(z1.as_uint_ref(), Limb::ZERO);
342                carry = carry.wrapping_add(c);
343            } else {
344                assign.copy_from(z1.as_uint_ref());
345            }
346
347            // Add `x0y` if necessary.
348            if !x0.is_empty() {
349                let c = wrapping_mul(x0, y, out, true);
350                carry = carry.wrapping_add(c)
351            }
352            carry
353        }
354        // Perform a wide (but possibly truncated) multiplication, returning up the limb count of `x•y`.
355        // The product is computed as:
356        //   x•y = (x0 + x1•b)(y0 + y1•b)
357        //       = x0y0 + x1y•b + x0y1•b
358        else {
359            let (x0, x1) = x.split_at(LIMBS);
360            let y_len = if y.nlimbs() < out_len {
361                y.nlimbs()
362            } else {
363                out_len
364            };
365            let (y0, y1) = y.leading(y_len).split_at(LIMBS);
366
367            let z0 = widening_mul_fixed::<LIMBS, LIMBS>(x0, y0);
368            let (assign, tail) = out.split_at_mut(if out.nlimbs() < LIMBS * 2 {
369                out.nlimbs()
370            } else {
371                LIMBS * 2
372            });
373            let (lo, hi) = assign.split_at_mut(LIMBS);
374            let mut carry = if add {
375                let mut carry = lo.carrying_add_assign(z0.0.as_uint_ref(), Limb::ZERO);
376                carry = hi.carrying_add_assign(z0.1.as_uint_ref().leading(hi.nlimbs()), carry);
377                tail.add_assign_limb(carry)
378            } else {
379                lo.copy_from(z0.0.as_uint_ref());
380                hi.copy_from(z0.1.as_uint_ref().leading(hi.nlimbs()));
381                Limb::ZERO
382            };
383
384            // Add x1y•b if necessary.
385            if !x1.is_empty() {
386                let c = wrapping_mul(x1, y, out.trailing_mut(LIMBS), true);
387                carry = carry.wrapping_add(c);
388            }
389            // Add x0y1•b if necessary.
390            if !y1.is_empty() {
391                let tail_len = out_len - LIMBS;
392                let assign_len = if y_len < tail_len { y_len } else { tail_len };
393                let (assign, tail) = out.trailing_mut(LIMBS).split_at_mut(assign_len);
394                let c = wrapping_mul(y1, x0, assign, true);
395                let c = tail.add_assign_limb(c);
396                carry = carry.wrapping_add(c);
397            }
398            carry
399        }
400    }
401
402    let overlap = if lhs.nlimbs() < rhs.nlimbs() {
403        lhs.nlimbs()
404    } else {
405        rhs.nlimbs()
406    };
407    let overlap = if overlap < out.nlimbs() {
408        overlap
409    } else {
410        out.nlimbs()
411    };
412    let split = previous_power_of_2(overlap);
413
414    // Handle smaller sized integers
415    if split < MIN_STARTING_LIMBS {
416        return schoolbook::wrapping_mul_add(lhs.as_slice(), rhs.as_slice(), out.as_mut_slice());
417    }
418
419    // Select an optimized implementation for a fixed number of limbs
420    match split {
421        16 => reduce::<16>(lhs, rhs, out, add),
422        32 => reduce::<32>(lhs, rhs, out, add),
423        64 => reduce::<64>(lhs, rhs, out, add),
424        128 => reduce::<128>(lhs, rhs, out, add),
425        _ => reduce::<256>(lhs, rhs, out, add),
426    }
427}

429/// Square a limb slice, placing the potentially-truncated result in `out`.
430///
431/// `out` must have a limb count less than or equal to the width of the wide squaring of `uint`.
432#[inline]
433pub(crate) const fn wrapping_square(uint: &UintRef, out: &mut UintRef) -> Limb {
434    assert!(
435        out.nlimbs() <= uint.nlimbs() * 2,
436        "invalid arguments to wrapping_square"
437    );
438
439    // Use the optimized squaring for `LIMBS` limbs to break down the calculation of the product.
440    const fn reduce<const LIMBS: usize>(x: &UintRef, out: &mut UintRef) -> Limb {
441        let (x0, x1) = x.split_at(LIMBS);
442        let (lo, hi) = out.split_at_mut(LIMBS);
443
444        // Add z0 = x0^2
445        let z0 = widening_square_fixed::<LIMBS>(x0);
446        lo.copy_from(z0.0.as_uint_ref());
447
448        // Add z1 = 2x0•x1•b
449        if hi.nlimbs() <= LIMBS {
450            let (z1, _carry) = wrapping_mul_fixed::<LIMBS>(x0, x1);
451            let z1 = z1.overflowing_shl1().0;
452            let z2 = z0.1.wrapping_add(&z1);
453            let (z2, tail) = z2.as_uint_ref().split_at(hi.nlimbs());
454            hi.copy_from(z2);
455            if tail.is_empty() {
456                Limb::ZERO
457            } else {
458                tail.0[0]
459            }
460        } else {
461            let (z01, z2) = hi.split_at_mut(LIMBS);
462            z01.copy_from(z0.1.as_uint_ref());
463            wrapping_square(x1, z2);
464            let mut dx0 = Uint::<LIMBS>::ZERO;
465            dx0.as_mut_uint_ref().copy_from(x0);
466            let (dx0, dx0_hi) = dx0.overflowing_shl1();
467            let z2_len = if z2.nlimbs() < x1.nlimbs() {
468                z2.nlimbs()
469            } else {
470                x1.nlimbs()
471            };
472            let mut carry = z2.leading_mut(z2_len).conditional_add_assign(
473                x1.leading(z2_len),
474                Limb::ZERO,
475                dx0_hi.is_nonzero(),
476            );
477            let (z1, z1tail) = hi.split_at_mut(LIMBS + z2_len);
478            let c = wrapping_mul(dx0.as_uint_ref(), x1, z1, true);
479            carry = carry.wrapping_add(c);
480            z1tail.add_assign_limb(carry)
481        }
482    }
483
484    // Restrict the size of the input to the size of the output
485    let x = if uint.nlimbs() >= out.nlimbs() {
486        uint.leading(out.nlimbs())
487    } else {
488        uint
489    };
490
491    // Handle smaller integer sizes
492    if x.nlimbs() <= MIN_STARTING_LIMBS {
493        return schoolbook::wrapping_square(x.as_slice(), out.as_mut_slice());
494    }
495
496    // Select an optimized 'split' such that a wide multiplication will not
497    // calculate too many unnecessary limbs. We select the largest power of 2
498    // <= the output size, and step back to the previous power if it exceeds the
499    // input size or more than MIN_STARTING_LIMBS limbs would be truncated.
500    let mut split = previous_power_of_2(out.nlimbs());
501    if split > x.nlimbs() || 2 * split >= out.nlimbs() + MIN_STARTING_LIMBS {
502        split /= 2;
503    }
504
505    // Select an optimized implementation for a fixed number of limbs
506    match split {
507        16 => reduce::<16>(x, out),
508        32 => reduce::<32>(x, out),
509        64 => reduce::<64>(x, out),
510        128 => reduce::<128>(x, out),
511        _ => reduce::<256>(x, out),
512    }
513}

515/// Join two `Uints` into an output `Uint`, potentially larger than both in size.
516#[inline]
517const fn concat_wide<const LIMBS: usize, const HALF: usize>(
518    lo: &Uint<HALF>,
519    hi: &Uint<HALF>,
520) -> Uint<LIMBS> {
521    assert!(LIMBS >= HALF * 2);
522    let mut res = Uint::<LIMBS>::ZERO;
523    let (lo_mut, hi_mut) = res
524        .as_mut_uint_ref()
525        .leading_mut(HALF * 2)
526        .split_at_mut(HALF);
527    lo_mut.copy_from_slice(lo.as_limbs());
528    hi_mut.copy_from_slice(hi.as_limbs());
529    res
530}
531
532/// Resize a pair of `Uint`s.
533#[inline(always)]
534const fn resize_wide<const LIMBS: usize, const LHS: usize, const RHS: usize>(
535    (lo, hi): (Uint<LIMBS>, Uint<LIMBS>),
536) -> (Uint<LHS>, Uint<RHS>) {
537    assert!(LHS == LIMBS && RHS >= LIMBS);
538    (lo.resize(), hi.resize())
539}

541/// Compute the power of 2 `p` such that `p <= value < p*2`.
542#[inline]
543const fn previous_power_of_2(value: usize) -> usize {
544    if value == 0 {
545        0
546    } else {
547        1usize << value.ilog2()
548    }
549}
