1//! [`Uint`] modular addition operations.
2
3use crate::{AddMod, Limb, NonZero, Uint};
4
5impl<const LIMBS: usize> Uint<LIMBS> {
6    /// Computes `self + rhs mod p`.
7    ///
8    /// Assumes `self + rhs` as unbounded integer is `< 2p`.
9    pub const fn add_mod(&self, rhs: &Self, p: &NonZero<Self>) -> Self {
10        let (w, carry) = self.carrying_add(rhs, Limb::ZERO);
11        w.sub_mod_with_carry(carry, p.as_ref(), p.as_ref())
12    }
13
14    /// Computes `self + rhs mod p` for the special modulus
15    /// `p = MAX+1-c` where `c` is small enough to fit in a single [`Limb`].
16    ///
17    /// Assumes `self + rhs` as unbounded integer is `< 2p`.
18    pub const fn add_mod_special(&self, rhs: &Self, c: Limb) -> Self {
19        // `Uint::carrying_add` also works with a carry greater than 1.
20        let (out, carry) = self.carrying_add(rhs, c);
21
22        // If overflow occurred, then above addition of `c` already accounts
23        // for the overflow. Otherwise, we need to subtract `c` again, which
24        // in that case cannot underflow.
25        let l = carry.0.wrapping_sub(1) & c.0;
26        out.wrapping_sub(&Self::from_word(l))
27    }
28
29    /// Computes `self + self mod p`.
30    ///
31    /// Assumes `self` as unbounded integer is `< p`.
32    pub const fn double_mod(&self, p: &NonZero<Self>) -> Self {
33        let (w, carry) = self.overflowing_shl1();
34        w.sub_mod_with_carry(carry, p.as_ref(), p.as_ref())
35    }
36}
37
38impl<const LIMBS: usize> AddMod for Uint<LIMBS> {
39    type Output = Self;
40
41    fn add_mod(&self, rhs: &Self, p: &NonZero<Self>) -> Self {
42        debug_assert!(self < p.as_ref());
43        debug_assert!(rhs < p.as_ref());
44        self.add_mod(rhs, p)
45    }
46}

53    #[test]
54    fn add_mod_nist_p256() {
55        let a =
56            U256::from_be_hex("44acf6b7e36c1342c2c5897204fe09504e1e2efb1a900377dbc4e7a6a133ec56");
57        let b =
58            U256::from_be_hex("d5777c45019673125ad240f83094d4252d829516fac8601ed01979ec1ec1a251");
59        let n =
60            U256::from_be_hex("ffffffff00000000ffffffffffffffffbce6faada7179e84f3b9cac2fc632551")
61                .to_nz()
62                .unwrap();
63
64        let actual = a.add_mod(&b, &n);
65        let expected =
66            U256::from_be_hex("1a2472fde50286541d97ca6a3592dd75beb9c9646e40c511b82496cfc3926956");
67
68        assert_eq!(expected, actual);
69    }

83    macro_rules! test_add_mod_special {
84        ($size:expr, $test_name:ident) => {
85            #[test]
86            fn $test_name() {
87                let mut rng = rand_chacha::ChaCha8Rng::seed_from_u64(1);
88                let moduli = [
89                    NonZero::<Limb>::random(&mut rng),
90                    NonZero::<Limb>::random(&mut rng),
91                ];
92
93                for special in &moduli {
94                    let p =
95                        &NonZero::new(Uint::ZERO.wrapping_sub(&Uint::from(special.get()))).unwrap();
96
97                    let minus_one = p.wrapping_sub(&Uint::ONE);
98
99                    let base_cases = [
100                        (Uint::ZERO, Uint::ZERO, Uint::ZERO),
101                        (Uint::ONE, Uint::ZERO, Uint::ONE),
102                        (Uint::ZERO, Uint::ONE, Uint::ONE),
103                        (minus_one, Uint::ONE, Uint::ZERO),
104                        (Uint::ONE, minus_one, Uint::ZERO),
105                    ];
106                    for (a, b, c) in &base_cases {
107                        let x = a.add_mod_special(b, *special.as_ref());
108                        assert_eq!(*c, x, "{} + {} mod {} = {} != {}", a, b, p, x, c);
109                    }
110
111                    for _i in 0..100 {
112                        let a = Uint::<$size>::random_mod(&mut rng, p);
113                        let b = Uint::<$size>::random_mod(&mut rng, p);
114
115                        let c = a.add_mod_special(&b, *special.as_ref());
116                        assert!(c < **p, "not reduced: {} >= {} ", c, p);
117
118                        let expected = a.add_mod(&b, p);
119                        assert_eq!(c, expected, "incorrect result");
120                    }
121                }
122            }
123        };
124    }
