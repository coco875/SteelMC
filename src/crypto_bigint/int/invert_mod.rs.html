1//! Operations related to computing the inverse of an [`Int`] modulo a [`Uint`].
2
3use subtle::CtOption;
4
5use crate::{ConstCtOption, Int, InvertMod, NonZero, Odd, Uint};
6
7impl<const LIMBS: usize> Int<LIMBS> {
8    /// Computes the multiplicative inverse of `self` mod `modulus`, where `modulus` is odd.
9    pub fn invert_odd_mod(&self, modulus: &Odd<Uint<LIMBS>>) -> ConstCtOption<Uint<LIMBS>> {
10        let (abs, sgn) = self.abs_sign();
11        let maybe_inv = abs.invert_odd_mod(modulus);
12        let (abs_inv, inv_is_some) = maybe_inv.components_ref();
13        // Note: when `self` is negative and modulus is non-zero, then
14        // self^{-1} % modulus = modulus - |self|^{-1} % modulus
15        ConstCtOption::new(
16            Uint::select(abs_inv, &modulus.wrapping_sub(abs_inv), sgn),
17            inv_is_some,
18        )
19    }
20
21    /// Computes the multiplicative inverse of `self` mod `modulus`.
22    ///
23    /// Returns some if an inverse exists, otherwise none.
24    pub const fn invert_mod(&self, modulus: &NonZero<Uint<LIMBS>>) -> ConstCtOption<Uint<LIMBS>> {
25        let (abs, sgn) = self.abs_sign();
26        let maybe_inv = abs.invert_mod(modulus);
27        let (abs_inv, inv_is_some) = maybe_inv.components_ref();
28        // Note: when `self` is negative and modulus is non-zero, then
29        // self^{-1} % modulus = modulus - |self|^{-1} % modulus
30        ConstCtOption::new(
31            Uint::select(abs_inv, &modulus.as_ref().wrapping_sub(abs_inv), sgn),
32            inv_is_some,
33        )
34    }
35}
36
37impl<const LIMBS: usize> InvertMod<NonZero<Uint<LIMBS>>> for Int<LIMBS>
38where
39    Uint<LIMBS>: InvertMod<Output = Uint<LIMBS>>,
40{
41    type Output = Uint<LIMBS>;
42
43    fn invert_mod(&self, modulus: &NonZero<Uint<LIMBS>>) -> CtOption<Self::Output> {
44        Self::invert_mod(self, modulus).into()
45    }
46}
