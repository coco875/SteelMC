1//! Modular symbol calculation for integers in Montgomery form with a constant modulus.
2
3use crate::JacobiSymbol;
4
5use super::MontyForm;
6
7impl<const LIMBS: usize> MontyForm<LIMBS> {
8    /// Compute the Jacobi symbol `(self|modulus)`.
9    ///
10    /// For a prime modulus, this corresponds to the Legendre symbol and indicates
11    /// whether `self` is quadratic residue.
12    pub const fn jacobi_symbol(&self) -> JacobiSymbol {
13        self.retrieve().jacobi_symbol(self.params().modulus())
14    }
15
16    /// Compute the Jacobi symbol `(self|modulus)`.
17    ///
18    /// For a prime modulus, this corresponds to the Legendre symbol and indicates
19    /// whether `self` is quadratic residue.
20    ///
21    /// This method is variable-time with respect to the value of `self`.
22    pub const fn jacobi_symbol_vartime(&self) -> JacobiSymbol {
23        self.retrieve()
24            .jacobi_symbol_vartime(self.params().modulus())
25    }
26}
27
28#[cfg(test)]
29mod tests {
30    use crate::{
31        JacobiSymbol, Odd, U256,
32        modular::{MontyForm, MontyParams},
33    };
34
35    const PARAMS: MontyParams<{ U256::LIMBS }> =
36        MontyParams::new_vartime(Odd::<U256>::from_be_hex(
37            "2523648240000001BA344D80000000086121000000000013A700000000000013",
38        ));
39
40    #[test]
41    fn jacobi_quad_residue() {
42        let x =
43            U256::from_be_hex("14BFAE46F4026E97C7A3FCD889B379A5F025719911C994A594FC6C5092AC58B1");
44        let x_mod = MontyForm::new(&x, PARAMS);
45
46        let jac = x_mod.jacobi_symbol();
47        let jac_vartime = x_mod.jacobi_symbol_vartime();
48        assert_eq!(jac, JacobiSymbol::One);
49        assert_eq!(jac, jac_vartime);
50    }
51
52    #[test]
53    fn jacobi_quad_nonresidue() {
54        let x =
55            U256::from_be_hex("1D2EFB21D283A2DDE77004B9DE9A9624F7B15CEEF055CD02E9EF1A9F1B76F253");
56        let x_mod = MontyForm::new(&x, PARAMS);
57
58        let jac = x_mod.jacobi_symbol();
59        let jac_vartime = x_mod.jacobi_symbol_vartime();
60        assert_eq!(jac, JacobiSymbol::MinusOne);
61        assert_eq!(jac, jac_vartime);
62    }
63}
