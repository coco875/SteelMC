4/// Binary GCD update step.
5///
6/// This is a condensed, constant time execution of the following algorithm:
7/// ```text
8/// if a mod 2 == 1
9///    if a < b
10///        (a, b) ← (b, a)
11///    a ← a - b
12/// a ← a/2
13/// ```
14///
15/// Note: assumes `b` to be odd. Might yield an incorrect result if this is not the case.
16///
17/// Ref: Pornin, Algorithm 1, L3-9, <https://eprint.iacr.org/2020/972.pdf>.
18#[inline(always)]
19pub(super) const fn bingcd_step<const LIMBS: usize>(
20    a: &mut Uint<LIMBS>,
21    b: &mut Uint<LIMBS>,
22) -> (ConstChoice, ConstChoice, Word) {
23    let a_b_mod_4 = (a.limbs[0].0 & b.limbs[0].0) & 3;
24
25    let a_odd = a.is_odd();
26    let (a_sub_b, borrow) = a.borrowing_sub(&Uint::select(&Uint::ZERO, b, a_odd), Limb::ZERO);
27    let swap = borrow.is_nonzero();
28    *b = Uint::select(b, a, swap);
29    *a = a_sub_b.wrapping_neg_if(swap).shr1();
30
31    // (b|a) = -(a|b) iff a = b = 3 mod 4 (quadratic reciprocity)
32    let mut jacobi_neg = swap.if_true_word(a_b_mod_4 & (a_b_mod_4 >> 1) & 1);
33
34    // (2a|b) = -(a|b) iff b = ±3 mod 8
35    // b is always odd, so we ignore the lower bit and check that bits 2 and 3 are '01' or '10'
36    let b_lo = b.limbs[0].0;
37    jacobi_neg ^= ((b_lo >> 1) ^ (b_lo >> 2)) & 1;
38
39    (a_odd, swap, jacobi_neg)
40}

43    /// The minimal number of binary GCD iterations required to guarantee successful completion.
44    pub(crate) const MINIMAL_BINGCD_ITERATIONS: u32 = 2 * Self::BITS - 1;

46    /// Computes `gcd(self, rhs)`, leveraging (a constant time implementation of) the classic
47    /// Binary GCD algorithm.
48    ///
49    /// Note: this algorithm is efficient for [Uint]s with relatively few `LIMBS`.

53    #[inline]
54    pub(crate) const fn classic_bingcd(&self, rhs: &Uint<LIMBS>) -> Self {
55        self.classic_bingcd_(rhs).0
56    }

58    /// Computes `gcd(self, rhs)`, leveraging (a constant time implementation of) the classic
59    /// Binary GCD algorithm.
60    ///
61    /// Note: this algorithm is efficient for [Uint]s with relatively few `LIMBS`.
62    ///
63    /// Ref: Pornin, Optimized Binary GCD for Modular Inversion, Algorithm 1.
64    /// <https://eprint.iacr.org/2020/972.pdf>
65    ///
66    /// This method returns a pair consisting of the GCD and the sign of the Jacobi symbol,
67    /// 0 for positive and 1 for negative.
68    #[inline(always)]
69    pub(crate) const fn classic_bingcd_(&self, rhs: &Uint<LIMBS>) -> (Self, Word) {
70        // (self, rhs) corresponds to (m, y) in the Algorithm 1 notation.
71        let (mut a, mut b) = (*rhs, *self.as_ref());
72        let mut i = 0;
73        let mut jacobi_neg = 0;
74        while i < Self::MINIMAL_BINGCD_ITERATIONS {
75            jacobi_neg ^= bingcd_step(&mut a, &mut b).2;
76            i += 1;
77        }
78
79        let gcd = b
80            .to_odd()
81            .expect("gcd of an odd value with something else is always odd");
82        (gcd, jacobi_neg)
83    }

85    /// Variable time equivalent of [`Self::classic_bingcd`].
86    #[inline]
87    pub(crate) const fn classic_bingcd_vartime(&self, rhs: &Uint<LIMBS>) -> Self {
88        self.classic_bingcd_vartime_(rhs).0
89    }

91    /// Variable time equivalent of [`Self::classic_bingcd_`].
92    #[inline(always)]
93    pub(crate) const fn classic_bingcd_vartime_(&self, rhs: &Uint<LIMBS>) -> (Self, Word) {
94        // (self, rhs) corresponds to (m, y) in the Algorithm 1 notation.
95        let (mut a, mut b) = (*rhs, *self.as_ref());
96        let mut jacobi_neg = 0;
97        while !a.is_zero_vartime() {
98            jacobi_neg ^= bingcd_step(&mut a, &mut b).2;
99        }
100
101        let gcd = b
102            .to_odd()
103            .expect("gcd of an odd value with something else is always odd");
104        (gcd, jacobi_neg)
105    }

107    /// Computes `gcd(self, rhs)`, leveraging the optimized Binary GCD algorithm.
108    ///
109    /// Note: this algorithm becomes more efficient than the classical algorithm for [Uint]s with
110    /// relatively many `LIMBS`. A best-effort threshold is presented in [Self::bingcd].
111    ///
112    /// Note: the full algorithm has an additional parameter; this function selects the best-effort
113    /// value for this parameter. You might be able to further tune your performance by calling the
114    /// [Self::optimized_bingcd_] function directly.
115    ///
116    /// Ref: Pornin, Optimized Binary GCD for Modular Inversion, Algorithm 2.
117    /// <https://eprint.iacr.org/2020/972.pdf>
118    #[inline(always)]
119    pub(crate) const fn optimized_bingcd(&self, rhs: &Uint<LIMBS>) -> Self {
120        self.optimized_bingcd_::<{ U64::BITS }, { U64::LIMBS }, { U128::LIMBS }>(rhs, U64::BITS - 1)
121            .0
122    }

124    /// Computes `gcd(self, rhs)`, leveraging the optimized Binary GCD algorithm.
125    ///
126    /// Ref: Pornin, Optimized Binary GCD for Modular Inversion, Algorithm 2.
127    /// <https://eprint.iacr.org/2020/972.pdf>
128    ///
129    /// In summary, the optimized algorithm does not operate on `self` and `rhs` directly, but
130    /// instead of condensed summaries that fit in few registers. Based on these summaries, an
131    /// update matrix is constructed by which `self` and `rhs` are updated in larger steps.
132    ///
133    /// This function is generic over the following three values:
134    /// - `K`: the number of bits used when summarizing `self` and `rhs` for the inner loop. The
135    ///   `K` top bits and `K` least significant bits are selected. It is recommended to keep
136    ///   `K` close to a (multiple of) the number of bits that fit in a single register.
137    /// - `LIMBS_K`: should be chosen as the minimum number s.t. [`Uint::<LIMBS>::BITS`] `≥ K`,
138    /// - `LIMBS_2K`: should be chosen as the minimum number s.t. [`Uint::<LIMBS>::BITS`] `≥ 2K`.
139    ///
140    /// This method returns a pair consisting of the GCD and the sign of the Jacobi symbol,
141    /// 0 for positive and 1 for negative.
142    #[inline(always)]
143    pub(crate) const fn optimized_bingcd_<
144        const K: u32,
145        const LIMBS_K: usize,
146        const LIMBS_2K: usize,
147    >(
148        &self,
149        rhs: &Uint<LIMBS>,
150        batch_max: u32,
151    ) -> (Self, Word) {
152        let (mut a, mut b) = (*self.as_ref(), *rhs);
153        let mut jacobi_neg = 0;
154
155        let mut iterations = Self::MINIMAL_BINGCD_ITERATIONS;
156        while iterations > 0 {
157            let batch = u32_min(iterations, batch_max);
158            iterations -= batch;
159
160            // Construct a_ and b_ as the summary of a and b, respectively.
161            let n = u32_max(2 * K, a.bitor(&b).bits());
162            let a_ = a.compact::<K, LIMBS_2K>(n);
163            let b_ = b.compact::<K, LIMBS_2K>(n);
164
165            // Compute the batch update matrix from a_ and b_.
166            let (.., matrix, j_neg) = a_
167                .to_odd()
168                .expect("a_ is always odd")
169                .partial_binxgcd::<LIMBS_K>(&b_, batch, ConstChoice::FALSE);
170            jacobi_neg ^= j_neg;
171
172            // Update `a` and `b` using the update matrix.
173            // Safe to use vartime: the number of doublings is the same as the batch size.
174            let (updated_a, updated_b) = matrix.extended_apply_to_vartime::<LIMBS>((a, b));
175            (a, _) = updated_a.dropped_abs_sign();
176            (b, _) = updated_b.dropped_abs_sign();
177        }
178
179        (
180            a.to_odd()
181                .expect("gcd of an odd value with something else is always odd"),
182            jacobi_neg,
183        )
184    }

186    /// Variable time equivalent of [`Self::optimized_bingcd`].
187    #[inline(always)]
188    pub(crate) const fn optimized_bingcd_vartime(&self, rhs: &Uint<LIMBS>) -> Self {
189        self.optimized_bingcd_vartime_::<{ U64::BITS }, { U64::LIMBS }, { U128::LIMBS }>(
190            rhs,
191            U64::BITS - 1,
192        )
193        .0
194    }

196    /// Variable time equivalent of [`Self::optimized_bingcd_`].
197    #[inline(always)]
198    pub(crate) const fn optimized_bingcd_vartime_<
199        const K: u32,
200        const LIMBS_K: usize,
201        const LIMBS_2K: usize,
202    >(
203        &self,
204        rhs: &Uint<LIMBS>,
205        batch_max: u32,
206    ) -> (Self, Word) {
207        let (mut a, mut b) = (*self.as_ref(), *rhs);
208        let mut jacobi_neg = 0;
209
210        while !b.is_zero_vartime() {
211            // Construct a_ and b_ as the summary of a and b, respectively.
212            let n = u32_max(2 * K, u32_max(a.bits_vartime(), b.bits_vartime()));
213            let mut a_ = a.compact_vartime::<K, LIMBS_2K>(n);
214            let mut b_ = b.compact_vartime::<K, LIMBS_2K>(n);
215
216            if n <= Uint::<LIMBS_2K>::BITS {
217                loop {
218                    jacobi_neg ^= bingcd_step(&mut b_, &mut a_).2;
219                    if b_.is_zero_vartime() {
220                        break;
221                    }
222                }
223                a = a_.resize();
224                break;
225            }
226
227            // Compute the batch update matrix from a_ and b_.
228            let (.., matrix, j_neg) = a_
229                .to_odd()
230                .expect("a_ is always odd")
231                .partial_binxgcd::<LIMBS_K>(&b_, batch_max, ConstChoice::FALSE);
232            jacobi_neg ^= j_neg;
233
234            // Update `a` and `b` using the update matrix.
235            let (updated_a, updated_b) = matrix.extended_apply_to_vartime((a, b));
236            (a, _) = updated_a.dropped_abs_sign();
237            (b, _) = updated_b.dropped_abs_sign();
238        }
239
240        (
241            a.to_odd()
242                .expect("gcd of an odd value with something else is always odd"),
243            jacobi_neg,
244        )
245    }
