14/// 2x2-Matrix with integers for elements.
15///
16/// ### Representation
17/// The internal state represents the matrix
18/// ```text
19/// [ m00  m01 ]
20/// [ m10  m11 ]
21/// ```
22#[derive(Debug, Clone, Copy, PartialEq)]
23pub(crate) struct IntMatrix<const LIMBS: usize> {
24    m00: ExtraLimbInt<LIMBS>,
25    m01: ExtraLimbInt<LIMBS>,
26    m10: ExtraLimbInt<LIMBS>,
27    m11: ExtraLimbInt<LIMBS>,
28}
29
30impl<const LIMBS: usize> Unit for IntMatrix<LIMBS> {
31    const UNIT: Self = Self {
32        m00: ExtraLimbInt::ONE,
33        m01: ExtraLimbInt::ZERO,
34        m10: ExtraLimbInt::ZERO,
35        m11: ExtraLimbInt::ONE,
36    };
37}

39impl<const LIMBS: usize> IntMatrix<LIMBS> {
40    /// Negate the top row of this matrix if `negate`; otherwise do nothing.
41    pub(super) const fn conditional_negate_top_row(&mut self, negate: ConstChoice) {
42        self.m00 = self.m00.wrapping_neg_if(negate);
43        self.m01 = self.m01.wrapping_neg_if(negate);
44    }
45
46    /// Negate the bottom row of this matrix if `negate`; otherwise do nothing.
47    pub(super) const fn conditional_negate_bottom_row(&mut self, negate: ConstChoice) {
48        self.m10 = self.m10.wrapping_neg_if(negate);
49        self.m11 = self.m11.wrapping_neg_if(negate);
50    }
51
52    pub(super) const fn to_pattern_matrix(self) -> PatternMatrix<LIMBS> {
53        let (abs_m00, m00_is_negative) = self.m00.abs_sign();
54        let (abs_m01, m01_is_negative) = self.m01.abs_sign();
55        let (abs_m10, m10_is_negative) = self.m10.abs_sign();
56        let (abs_m11, m11_is_negative) = self.m11.abs_sign();
57
58        // Construct the pattern.
59        let m00_is_zero = abs_m00.is_zero();
60        let m01_is_zero = abs_m01.is_zero();
61        let pattern_vote_1 = m00_is_zero.not().and(m00_is_negative.not());
62        let pattern_vote_2 = m01_is_zero.not().and(m01_is_negative);
63
64        let m00_and_m01_are_zero = m00_is_zero.and(m01_is_zero);
65        let m10_is_zero = abs_m10.is_zero();
66        let m11_is_zero = abs_m11.is_zero();
67        let pattern_vote_3 = m00_and_m01_are_zero.and(m10_is_zero.not().and(m10_is_negative));
68        let pattern_vote_4 = m00_and_m01_are_zero.and(m11_is_zero.not().and(m11_is_negative.not()));
69        let pattern = pattern_vote_1
70            .or(pattern_vote_2)
71            .or(pattern_vote_3)
72            .or(pattern_vote_4);
73
74        PatternMatrix {
75            m00: abs_m00.checked_drop_extension().expect("m00 fits"),
76            m01: abs_m01.checked_drop_extension().expect("m01 fits"),
77            m10: abs_m10.checked_drop_extension().expect("m10 fits"),
78            m11: abs_m11.checked_drop_extension().expect("m11 fits"),
79            pattern,
80        }
81    }
82}

84/// 2x2-Matrix where either the diagonal or off-diagonal elements are negative.
85///
86/// The internal state represents the matrix
87/// ```text
88///      true                 false
89/// [  m00 -m01 ]         [ -m00  m01 ]
90/// [ -m10  m11 ]   or    [  m10 -m11 ]
91/// ```
92/// depending on whether `pattern` is respectively truthy or not.
93#[derive(Debug, Clone, Copy, PartialEq)]
94pub(crate) struct PatternMatrix<const LIMBS: usize> {
95    pub m00: Uint<LIMBS>,
96    pub m01: Uint<LIMBS>,
97    pub m10: Uint<LIMBS>,
98    pub m11: Uint<LIMBS>,
99    pub pattern: ConstChoice,
100}
101
102impl<const LIMBS: usize> PatternMatrix<LIMBS> {
103    pub const UNIT: Self = Self {
104        m00: Uint::ONE,
105        m01: Uint::ZERO,
106        m10: Uint::ZERO,
107        m11: Uint::ONE,
108        pattern: ConstChoice::TRUE,
109    };
110
111    /// Apply this matrix to a vector of [Uint]s, returning the result as a vector of
112    /// [ExtendedInt]s.
113    #[inline]
114    pub(crate) const fn extended_apply_to<const VEC_LIMBS: usize>(
115        &self,
116        vec: Vector<Uint<VEC_LIMBS>>,
117    ) -> Vector<ExtendedInt<VEC_LIMBS, LIMBS>> {
118        let (a, b) = vec;
119        let m00a = ExtendedInt::from_product(a, self.m00);
120        let m10a = ExtendedInt::from_product(a, self.m10);
121        let m01b = ExtendedInt::from_product(b, self.m01);
122        let m11b = ExtendedInt::from_product(b, self.m11);
123        (
124            m00a.wrapping_sub(&m01b).wrapping_neg_if(self.pattern.not()),
125            m11b.wrapping_sub(&m10a).wrapping_neg_if(self.pattern.not()),
126        )
127    }
128
129    /// Wrapping apply this matrix to `rhs`. Return the result in a [`IntMatrix<RHS_LIMBS>`].
130    #[inline]
131    pub(super) const fn mul_int_matrix<const RHS_LIMBS: usize>(
132        &self,
133        rhs: &IntMatrix<RHS_LIMBS>,
134    ) -> IntMatrix<RHS_LIMBS> {
135        let a0 = rhs.m00.wrapping_mul((&self.m00, &self.pattern.not()));
136        let a1 = rhs.m10.wrapping_mul((&self.m01, &self.pattern));
137        let m00 = a0.wrapping_add(&a1);
138
139        let b0 = rhs.m01.wrapping_mul((&self.m00, &self.pattern.not()));
140        let b1 = rhs.m11.wrapping_mul((&self.m01, &self.pattern));
141        let m01 = b0.wrapping_add(&b1);
142
143        let c0 = rhs.m00.wrapping_mul((&self.m10, &self.pattern));
144        let c1 = rhs.m10.wrapping_mul((&self.m11, &self.pattern.not()));
145        let m10 = c0.wrapping_add(&c1);
146
147        let d0 = rhs.m01.wrapping_mul((&self.m10, &self.pattern));
148        let d1 = rhs.m11.wrapping_mul((&self.m11, &self.pattern.not()));
149        let m11 = d0.wrapping_add(&d1);
150
151        IntMatrix { m00, m01, m10, m11 }
152    }
153
154    /// Swap the rows of this matrix if `swap` is truthy. Otherwise, do nothing.
155    #[inline]
156    pub(crate) const fn conditional_swap_rows(&mut self, swap: ConstChoice) {
157        Uint::conditional_swap(&mut self.m00, &mut self.m10, swap);
158        Uint::conditional_swap(&mut self.m01, &mut self.m11, swap);
159        self.pattern = self.pattern.xor(swap);
160    }
161
162    /// Subtract the bottom row from the top if `subtract` is truthy. Otherwise, do nothing.
163    #[inline]
164    pub(crate) const fn conditional_subtract_bottom_row_from_top(&mut self, subtract: ConstChoice) {
165        // Note: because the signs of the internal representation are stored in `pattern`,
166        // subtracting one row from another involves _adding_ these rows instead.
167        self.m00 = Uint::select(&self.m00, &self.m00.wrapping_add(&self.m10), subtract);
168        self.m01 = Uint::select(&self.m01, &self.m01.wrapping_add(&self.m11), subtract);
169    }
170
171    /// Subtract the right column from the left if `subtract` is truthy. Otherwise, do nothing.
172    #[inline]
173    pub(crate) const fn conditional_subtract_right_column_from_left(
174        &mut self,
175        subtract: ConstChoice,
176    ) {
177        // Note: because the signs of the internal representation are stored in `pattern`,
178        // subtracting one column from another involves _adding_ these columns instead.
179        self.m00 = Uint::select(&self.m00, &self.m00.wrapping_add(&self.m01), subtract);
180        self.m10 = Uint::select(&self.m10, &self.m10.wrapping_add(&self.m11), subtract);
181    }
182
183    /// If `add` is truthy, add the right column to the left. Otherwise, do nothing.
184    #[inline]
185    pub(crate) const fn conditional_add_right_column_to_left(&mut self, add: ConstChoice) {
186        // Note: because the signs of the internal representation are stored in `pattern`,
187        // subtracting one column from another involves _adding_ these columns instead.
188        self.m00 = Uint::select(&self.m00, &self.m01.wrapping_sub(&self.m00), add);
189        self.m10 = Uint::select(&self.m10, &self.m11.wrapping_sub(&self.m10), add);
190    }
191
192    /// Double the bottom row of this matrix if `double` is truthy. Otherwise, do nothing.
193    #[inline]
194    pub(crate) const fn conditional_double_bottom_row(&mut self, double: ConstChoice) {
195        self.m10 = Uint::select(&self.m10, &self.m10.overflowing_shl1().0, double);
196        self.m11 = Uint::select(&self.m11, &self.m11.overflowing_shl1().0, double);
197    }
198
199    /// Negate the elements in this matrix if `negate` is truthy. Otherwise, do nothing.
200    #[inline]
201    pub(crate) const fn conditional_negate(&mut self, negate: ConstChoice) {
202        self.pattern = self.pattern.xor(negate);
203    }
204}

225/// Variation on [`PatternMatrix`], where the contents of the matrix need to be divided by
226/// `2^k`.
227/// The internal state represents the matrix
228/// ```text
229///      true                       false
230/// [  m00 -m01 ]               [ -m00  m01 ]
231/// [ -m10  m11 ] / 2^k   or    [  m10 -m11 ] / 2^k
232/// ```
233/// depending on whether `pattern` is respectively truthy or not.
234///
235/// Since some of the operations conditionally increase `k`, this struct furthermore keeps track of
236/// `k_upper_bound`; an upper bound on the value of `k`.
237#[derive(Debug, Clone, Copy, PartialEq)]
238pub(crate) struct DividedPatternMatrix<const LIMBS: usize> {
239    pub(super) inner: PatternMatrix<LIMBS>,
240    pub k: u32,
241    pub k_upper_bound: u32,
242}
243
244impl<const LIMBS: usize> DividedPatternMatrix<LIMBS> {
245    /// The unit matrix.
246    pub const UNIT: Self = Self {
247        inner: PatternMatrix::UNIT,
248        k: 0,
249        k_upper_bound: 0,
250    };
251
252    /// Apply this matrix to a vector of [Uint]s, returning the result as a vector of
253    /// [ExtendedInt]s.
254    #[inline]
255    #[allow(unused)] // save for optimized xgcd
256    pub const fn extended_apply_to<const VEC_LIMBS: usize, const UPPER_BOUND: u32>(
257        &self,
258        vec: Vector<Uint<VEC_LIMBS>>,
259    ) -> Vector<ExtendedInt<VEC_LIMBS, LIMBS>> {
260        let (a, b) = self.inner.extended_apply_to(vec);
261        (
262            a.bounded_div_2k::<UPPER_BOUND>(self.k),
263            b.bounded_div_2k::<UPPER_BOUND>(self.k),
264        )
265    }
266
267    /// Apply this matrix to a vector of [Uint]s, returning the result as a vector of
268    /// [ExtendedInt]s.
269    #[inline]
270    pub const fn extended_apply_to_vartime<const VEC_LIMBS: usize>(
271        &self,
272        vec: Vector<Uint<VEC_LIMBS>>,
273    ) -> Vector<ExtendedInt<VEC_LIMBS, LIMBS>> {
274        let (a, b) = self.inner.extended_apply_to(vec);
275        (a.div_2k_vartime(self.k), b.div_2k_vartime(self.k))
276    }
277
278    /// Multiply `self` with `rhs`. Return the result as a [`DividedIntMatrix<LIMBS>`].
279    #[inline]
280    pub const fn mul_int_matrix<const RHS_LIMBS: usize>(
281        &self,
282        rhs: &DividedIntMatrix<RHS_LIMBS>,
283    ) -> DividedIntMatrix<RHS_LIMBS> {
284        DividedIntMatrix {
285            inner: self.inner.mul_int_matrix(&rhs.inner),
286            k: self.k + rhs.k,
287            k_upper_bound: self.k_upper_bound + rhs.k_upper_bound,
288        }
289    }
290
291    /// Swap the rows of this matrix if `swap` is truthy. Otherwise, do nothing.
292    #[inline]
293    pub const fn conditional_swap_rows(&mut self, swap: ConstChoice) {
294        self.inner.conditional_swap_rows(swap);
295    }
296
297    /// Swap the rows of this matrix.
298    #[inline]
299    pub const fn swap_rows(&mut self) {
300        self.conditional_swap_rows(ConstChoice::TRUE);
301    }
302
303    /// Subtract the bottom row from the top if `subtract` is truthy. Otherwise, do nothing.
304    #[inline]
305    pub const fn conditional_subtract_bottom_row_from_top(&mut self, subtract: ConstChoice) {
306        self.inner
307            .conditional_subtract_bottom_row_from_top(subtract);
308    }
309
310    /// Double the bottom row of this matrix if `double` is truthy. Otherwise, do nothing.
311    #[inline]
312    pub const fn conditional_double_bottom_row(&mut self, double: ConstChoice) {
313        self.inner.conditional_double_bottom_row(double);
314        self.k = double.select_u32(self.k, self.k + 1);
315        self.k_upper_bound += 1;
316    }
317}
