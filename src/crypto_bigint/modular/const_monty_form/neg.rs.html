1//! Negations of integers in Montgomery form with a constant modulus.
2
3use super::{ConstMontyForm, ConstMontyParams};
4use core::ops::Neg;
5
6impl<MOD: ConstMontyParams<LIMBS>, const LIMBS: usize> ConstMontyForm<MOD, LIMBS> {
7    /// Negates the number.
8    pub const fn neg(&self) -> Self {
9        Self {
10            montgomery_form: self
11                .montgomery_form
12                .neg_mod(MOD::PARAMS.modulus.as_nz_ref()),
13            phantom: self.phantom,
14        }
15    }
16}
17
18impl<MOD: ConstMontyParams<LIMBS>, const LIMBS: usize> Neg for ConstMontyForm<MOD, LIMBS> {
19    type Output = Self;
20    fn neg(self) -> Self {
21        ConstMontyForm::neg(&self)
22    }
23}
24
25impl<MOD: ConstMontyParams<LIMBS>, const LIMBS: usize> Neg for &ConstMontyForm<MOD, LIMBS> {
26    type Output = ConstMontyForm<MOD, LIMBS>;
27    fn neg(self) -> ConstMontyForm<MOD, LIMBS> {
28        ConstMontyForm::neg(self)
29    }
30}

50        let x_mod = Fe::new(&x);
