4/// Based on Algorithm 14.36 in Handbook of Applied Cryptography
5/// <https://cacr.uwaterloo.ca/hac/about/chap14.pdf>
6///
7/// Multiply `x` and `y` in Montgomery form, producing `x•y•R^-1 mod modulus + a•modulus`.
8///
9/// This algorithm roughly corresponds to the Finely Integrated Operand Scanning (FIOS)
10/// method of "Analyzing and Comparing Montgomery Multiplication Algorithms" by Koc et al
11/// <https://www.microsoft.com/en-us/research/wp-content/uploads/1996/01/j37acmon.pdf>
12/// but using wide words to track the intermediate products and carry.
13///
14/// The final conditional subtraction of the modulus to produce a result in the range
15/// `[0, modulus)` is not performed here, and must be performed by the caller. In some
16/// cases this may be deferred, as demonstrated by the `almost_montgomery_mul` method used
17/// in `BoxedMontyMultiplier`.
18#[inline(always)]
19pub const fn montgomery_multiply_inner(
20    x: &[Limb],
21    y: &[Limb],
22    out: &mut [Limb],
23    modulus: &[Limb],
24    mod_neg_inv: Limb,
25) -> Limb {
26    let nlimbs = modulus.len();
27    assert!(nlimbs == x.len() && nlimbs == y.len() && nlimbs == out.len());
28
29    let mut meta_carry = 0;
30
31    let mut i = 0;
32    while i < nlimbs {
33        let xi = x[i];
34        // A[0] + x[i]y[0] <= (2^64 - 1)^2 + (2^64 - 1) = 2^128 - 2^64
35        let axy = (xi.0 as WideWord) * (y[0].0 as WideWord) + out[0].0 as WideWord;
36        let u = (axy as Word).wrapping_mul(mod_neg_inv.0);
37
38        let mut carry;
39        // A[0] + x[i]y[0] + um[0] <= (2^128 - 1) + (2^128 - 2^64) = 2^129 - 2^64 - 1
40        let (a, c) = ((u as WideWord) * (modulus[0].0 as WideWord)).overflowing_add(axy);
41        // carry <= (2^129 - 2^64 - 1) / 2^64 <= 2^65 - 2
42        carry = ((c as WideWord) << Word::BITS) | (a >> Word::BITS);
43
44        let mut j = 1;
45        while j < nlimbs {
46            // A[j] + x[i]y[j] <= (2^64 - 1)^2 + (2^64 - 1) = 2^128 - 2^64
47            let axy = (xi.0 as WideWord) * (y[j].0 as WideWord) + out[j].0 as WideWord;
48            // um[j] + carry <= (2^64 - 1)^2 + (2^65 - 2) = 2^128 - 1
49            let umc = (u as WideWord) * (modulus[j].0 as WideWord) + carry;
50            let (ab, c) = axy.overflowing_add(umc);
51            out[j - 1] = Limb(ab as Word);
52            // carry <= (2^129 - 2^64 - 1) / 2^64 <= 2^65 - 2
53            carry = ((c as WideWord) << Word::BITS) | (ab >> Word::BITS);
54            j += 1;
55        }
56
57        carry += meta_carry;
58        (out[nlimbs - 1], meta_carry) = (Limb(carry as Word), carry >> Word::BITS);
59
60        i += 1;
61    }
62
63    Limb(meta_carry as Word)
64}

66/// Computes the Montgomery product of `a` and `b` modulo `modulus`, where
67/// `a` and `b` are in Montgomery form.
68pub(crate) const fn mul_montgomery_form<const LIMBS: usize>(
69    a: &Uint<LIMBS>,
70    b: &Uint<LIMBS>,
71    modulus: &Odd<Uint<LIMBS>>,
72    mod_neg_inv: Limb,
73) -> Uint<LIMBS> {
74    let mut out = Uint::<LIMBS>::ZERO;
75    let carry = montgomery_multiply_inner(
76        &a.limbs,
77        &b.limbs,
78        &mut out.limbs,
79        &modulus.0.limbs,
80        mod_neg_inv,
81    );
82    out.sub_mod_with_carry(carry, modulus.as_ref(), modulus.as_ref())
83}

85/// Computes the Montgomery squaring of `a` modulo `modulus` where
86/// `a` is in Montgomery form.
87pub(crate) const fn square_montgomery_form<const LIMBS: usize>(
88    a: &Uint<LIMBS>,
89    modulus: &Odd<Uint<LIMBS>>,
90    mod_neg_inv: Limb,
91) -> Uint<LIMBS> {
92    // One case in which it appears to be more efficient to perform a wide squaring and reduction.
93    if LIMBS == 4 {
94        let lower_upper = a.square_wide();
95        return montgomery_reduction(&lower_upper, modulus, mod_neg_inv);
96    }
97
98    let mut out = Uint::<LIMBS>::ZERO;
99    let carry = montgomery_multiply_inner(
100        &a.limbs,
101        &a.limbs,
102        &mut out.limbs,
103        &modulus.0.limbs,
104        mod_neg_inv,
105    );
106    out.sub_mod_with_carry(carry, modulus.as_ref(), modulus.as_ref())
107}
