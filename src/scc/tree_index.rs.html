24/// Scalable concurrent B-plus tree.
25///
26/// [`TreeIndex`] is a concurrent B-plus tree variant optimized for read operations. Read
27/// operations, such as read iteration over entries, are neither blocked nor interrupted by other
28/// threads or tasks. Write operations, such as insert and remove, do not block if structural
29/// changes are not required.
30///
31/// ## Note
32///
33/// [`TreeIndex`] methods are linearizable. However, its iterator methods are not; [`Iter`] and
34/// [`Range`] are only guaranteed to observe events that happened before the first call to
35/// [`Iterator::next`].
36///
37/// ## The key features of [`TreeIndex`]
38///
39/// * Lock-free read: read and scan operations are never blocked and do not modify shared data.
40/// * Near lock-free write: write operations do not block unless a structural change is needed.
41/// * No busy waiting: each node has a wait queue to avoid spinning.
42/// * Immutability: the data in the container is immutable until it becomes unreachable.
43///
44/// ## The key statistics for [`TreeIndex`]
45///
46/// * The maximum number of entries that a leaf can contain: 14.
47/// * The maximum number of leaves or child nodes a node can contain: 15.
48///
49/// ## Locking behavior
50///
51/// Read access is always lock-free and non-blocking. Write access to an entry is also lock-free and
52/// non-blocking as long as no structural changes are required. However, when nodes are split or
53/// merged by a write operation, other write operations on keys in the affected range are blocked.
54///
55/// ### Synchronous methods in an asynchronous code block
56///
57/// It is generally not recommended to use blocking methods, such as [`TreeIndex::insert_sync`], in
58/// an asynchronous code block or [`poll`](std::future::Future::poll), since it may lead to
59/// deadlocks or performance degradation.
60///
61/// ### Unwind safety
62///
63/// [`TreeIndex`] is impervious to out-of-memory errors and panics in user-specified code under one
64/// condition; `K::drop` and `V::drop` must not panic.
65pub struct TreeIndex<K, V> {
66    root: AtomicShared<Node<K, V>>,
67}

164    /// Inserts a key-value pair.
165    ///
166    /// # Errors
167    ///
168    /// Returns an error along with the supplied key-value pair if the key exists.
169    ///
170    /// # Examples
171    ///
172    /// ```
173    /// use scc::TreeIndex;
174    ///
175    /// let treeindex: TreeIndex<u64, u32> = TreeIndex::new();
176    /// let future_insert = treeindex.insert_async(1, 10);
177    /// ```
178    #[inline]
179    pub async fn insert_async(&self, mut key: K, mut val: V) -> Result<(), (K, V)> {
180        let mut pinned_async_wait = pin!(AsyncWait::default());
181        let mut new_root = None;
182        loop {
183            let need_await = {
184                let guard = Guard::new();
185                let root_ptr = self.root.load(Acquire, &guard);
186                if let Some(root_ref) = root_ptr.as_ref() {
187                    match root_ref.insert(key, val, &mut pinned_async_wait, &guard) {
188                        Ok(r) => match r {
189                            InsertResult::Success => return Ok(()),
190                            InsertResult::Frozen(k, v) | InsertResult::Retry(k, v) => {
191                                key = k;
192                                val = v;
193                                root_ref.cleanup_link(&key, false, &guard);
194                                true
195                            }
196                            InsertResult::Duplicate(k, v) => return Err((k, v)),
197                            InsertResult::Full(k, v) => {
198                                let (k, v) = Node::split_root(root_ptr, &self.root, k, v, &guard);
199                                key = k;
200                                val = v;
201                                continue;
202                            }
203                            InsertResult::Retired(k, v) => {
204                                key = k;
205                                val = v;
206                                !Node::cleanup_root(&self.root, &mut pinned_async_wait, &guard)
207                            }
208                        },
209                        Err((k, v)) => {
210                            key = k;
211                            val = v;
212                            true
213                        }
214                    }
215                } else {
216                    false
217                }
218            };
219
220            if need_await {
221                pinned_async_wait.wait().await;
222            }
223
224            let node = if let Some(new_root) = new_root.take() {
225                new_root
226            } else {
227                Shared::new(Node::new_leaf_node())
228            };
229            if let Err((node, _)) = self.root.compare_exchange(
230                Ptr::null(),
231                (Some(node), Tag::None),
232                AcqRel,
233                Acquire,
234                &Guard::new(),
235            ) {
236                new_root = node;
237            }
238        }
239    }

241    /// Inserts a key-value pair.
242    ///
243    /// # Errors
244    ///
245    /// Returns an error along with the supplied key-value pair if the key exists.
246    ///
247    /// # Examples
248    ///
249    /// ```
250    /// use scc::TreeIndex;
251    ///
252    /// let treeindex: TreeIndex<u64, u32> = TreeIndex::new();
253    ///
254    /// assert!(treeindex.insert_sync(1, 10).is_ok());
255    /// assert_eq!(treeindex.insert_sync(1, 11).err().unwrap(), (1, 11));
256    /// assert_eq!(treeindex.peek_with(&1, |k, v| *v).unwrap(), 10);
257    /// ```
258    #[inline]
259    pub fn insert_sync(&self, mut key: K, mut val: V) -> Result<(), (K, V)> {
260        let mut new_root = None;
261        loop {
262            let guard = Guard::new();
263            let root_ptr = self.root.load(Acquire, &guard);
264            if let Some(root_ref) = root_ptr.as_ref() {
265                match root_ref.insert(key, val, &mut (), &guard) {
266                    Ok(r) => match r {
267                        InsertResult::Success => return Ok(()),
268                        InsertResult::Frozen(k, v) | InsertResult::Retry(k, v) => {
269                            key = k;
270                            val = v;
271                            root_ref.cleanup_link(&key, false, &guard);
272                        }
273                        InsertResult::Duplicate(k, v) => return Err((k, v)),
274                        InsertResult::Full(k, v) => {
275                            let (k, v) = Node::split_root(root_ptr, &self.root, k, v, &guard);
276                            key = k;
277                            val = v;
278                            continue;
279                        }
280                        InsertResult::Retired(k, v) => {
281                            key = k;
282                            val = v;
283                            let _result = Node::cleanup_root(&self.root, &mut (), &guard);
284                        }
285                    },
286                    Err((k, v)) => {
287                        key = k;
288                        val = v;
289                    }
290                }
291            }
292
293            let node = if let Some(new_root) = new_root.take() {
294                new_root
295            } else {
296                Shared::new(Node::new_leaf_node())
297            };
298            if let Err((node, _)) = self.root.compare_exchange(
299                Ptr::null(),
300                (Some(node), Tag::None),
301                AcqRel,
302                Acquire,
303                &guard,
304            ) {
305                new_root = node;
306            }
307        }
308    }

357    /// Removes a key-value pair if the given condition is met.
358    ///
359    /// Returns `false` if the key does not exist or the condition was not met. Returns `true` if
360    /// the key existed and the condition was met after marking the entry unreachable; the memory
361    /// will be reclaimed later.
362    ///
363    /// # Examples
364    ///
365    /// ```
366    /// use scc::TreeIndex;
367    ///
368    /// let treeindex: TreeIndex<u64, u32> = TreeIndex::new();
369    /// let future_remove = treeindex.remove_if_async(&1, |v| *v == 0);
370    /// ```
371    #[inline]
372    pub async fn remove_if_async<Q, F: FnMut(&V) -> bool>(&self, key: &Q, mut condition: F) -> bool
373    where
374        Q: Comparable<K> + ?Sized,
375    {
376        let mut pinned_async_wait = pin!(AsyncWait::default());
377        let mut removed = false;
378        loop {
379            {
380                let guard = Guard::new();
381                if let Some(root_ref) = self.root.load(Acquire, &guard).as_ref() {
382                    if let Ok(result) = root_ref.remove_if::<_, _, _>(
383                        key,
384                        &mut condition,
385                        &mut pinned_async_wait,
386                        &guard,
387                    ) {
388                        if matches!(result, RemoveResult::Cleanup) {
389                            root_ref.cleanup_link(key, false, &guard);
390                        }
391                        match result {
392                            RemoveResult::Success => return true,
393                            RemoveResult::Cleanup | RemoveResult::Retired => {
394                                if Node::cleanup_root(&self.root, &mut pinned_async_wait, &guard) {
395                                    return true;
396                                }
397                                removed = true;
398                            }
399                            RemoveResult::Fail => {
400                                if removed {
401                                    if Node::cleanup_root(
402                                        &self.root,
403                                        &mut pinned_async_wait,
404                                        &guard,
405                                    ) {
406                                        return true;
407                                    }
408                                } else {
409                                    return false;
410                                }
411                            }
412                            RemoveResult::Frozen => (),
413                        }
414                    }
415                } else {
416                    return removed;
417                }
418            }
419            pinned_async_wait.wait().await;
420        }
421    }

443    where
444        Q: Comparable<K> + ?Sized,
445    {
446        let mut removed = false;
447        loop {
448            let guard = Guard::new();
449            if let Some(root_ref) = self.root.load(Acquire, &guard).as_ref() {
450                if let Ok(result) =
451                    root_ref.remove_if::<_, _, _>(key, &mut condition, &mut (), &guard)
452                {
453                    if matches!(result, RemoveResult::Cleanup) {
454                        root_ref.cleanup_link(key, false, &guard);
455                    }
456                    match result {
457                        RemoveResult::Success => return true,
458                        RemoveResult::Cleanup | RemoveResult::Retired => {
459                            if Node::cleanup_root(&self.root, &mut (), &guard) {
460                                return true;
461                            }
462                            removed = true;
463                        }
464                        RemoveResult::Fail => {
465                            if removed {
466                                if Node::cleanup_root(&self.root, &mut (), &guard) {
467                                    return true;
468                                }
469                            } else {
470                                return false;
471                            }
472                        }
473                        RemoveResult::Frozen => (),
474                    }
475                }
476            } else {
477                return removed;
478            }
479        }

482    /// Removes keys in the specified range.
483    ///
484    /// This method removes internal nodes that are definitely contained in the specified range
485    /// first, and then removes remaining entries individually.
486    ///
487    /// # Note
488    ///
489    /// Internally, multiple internal node locks need to be acquired, thus making this method
490    /// susceptible to lock starvation.
491    ///
492    /// # Examples
493    ///
494    /// ```
495    /// use scc::TreeIndex;
496    ///
497    /// let treeindex: TreeIndex<u64, u32> = TreeIndex::new();
498    ///
499    /// for k in 2..8 {
500    ///     assert!(treeindex.insert_sync(k, 1).is_ok());
501    /// }
502    ///
503    /// let future_remove_range = treeindex.remove_range_async(3..8);
504    /// ```
505    #[inline]
506    pub async fn remove_range_async<Q, R: RangeBounds<Q>>(&self, range: R)
507    where
508        Q: Comparable<K> + ?Sized,
509    {
510        let mut pinned_async_wait = pin!(AsyncWait::default());
511        let start_unbounded = matches!(range.start_bound(), Unbounded);
512
513        loop {
514            {
515                let guard = Guard::new();
516
517                // Remove internal nodes, and individual entries in affected leaves.
518                //
519                // It takes O(N) to traverse sub-trees on the range border.
520                if let Some(root_ref) = self.root.load(Acquire, &guard).as_ref() {
521                    if let Ok(num_children) = root_ref.remove_range(
522                        &range,
523                        start_unbounded,
524                        None,
525                        None,
526                        &mut pinned_async_wait,
527                        &guard,
528                    ) {
529                        if num_children >= 2
530                            || Node::cleanup_root(&self.root, &mut pinned_async_wait, &guard)
531                        {
532                            // Completed removal and cleaning up the root.
533                            return;
534                        }
535                    }
536                } else {
537                    // Nothing to remove.
538                    return;
539                }
540            }
541            pinned_async_wait.wait().await;
542        }
543    }

545    /// Removes keys in the specified range.
546    ///
547    /// This method removes internal nodes that are definitely contained in the specified range
548    /// first, and then removes remaining entries individually.
549    ///
550    /// # Note
551    ///
552    /// Internally, multiple internal node locks need to be acquired, thus making this method
553    /// susceptible to lock starvation.
554    ///
555    /// # Examples
556    ///
557    /// ```
558    /// use scc::TreeIndex;
559    ///
560    /// let treeindex: TreeIndex<u64, u32> = TreeIndex::new();
561    ///
562    /// for k in 2..8 {
563    ///     assert!(treeindex.insert_sync(k, 1).is_ok());
564    /// }
565    ///
566    /// treeindex.remove_range_sync(3..8);
567    ///
568    /// assert!(treeindex.contains(&2));
569    /// assert!(!treeindex.contains(&3));
570    /// ```
571    #[inline]
572    pub fn remove_range_sync<Q, R: RangeBounds<Q>>(&self, range: R)
573    where
574        Q: Comparable<K> + ?Sized,
575    {
576        let start_unbounded = matches!(range.start_bound(), Unbounded);
577        let guard = Guard::new();
578
579        // Remove internal nodes, and individual entries in affected leaves.
580        //
581        // It takes O(N) to traverse sub-trees on the range border.
582        while let Some(root_ref) = self.root.load(Acquire, &guard).as_ref() {
583            if let Ok(num_children) =
584                root_ref.remove_range(&range, start_unbounded, None, None, &mut (), &guard)
585            {
586                if num_children < 2 && !Node::cleanup_root(&self.root, &mut (), &guard) {
587                    continue;
588                }
589                break;
590            }
591        }
592    }

713    /// # Examples

720    /// ```

724        self.iter(&guard).count()
725    }

744    /// Returns an [`Iter`].
745    ///

747    /// are scanned in ascending order, and key-value pairs that have existed since the invocation
748    /// of the method are guaranteed to be visited if they are not removed. However, it is possible
749    /// to visit removed key-value pairs momentarily.

881impl<'g, K, V> Iterator for Iter<'_, 'g, K, V>
882where
883    K: 'static + Clone + Ord,
884    V: 'static + Clone,
885{
886    type Item = (&'g K, &'g V);
887
888    #[inline]
889    fn next(&mut self) -> Option<Self::Item> {
890        // Starts scanning.
891        if self.leaf_scanner.is_none() {
892            let root_ptr = self.root.load(Acquire, self.guard);
893            if let Some(root_ref) = root_ptr.as_ref() {
894                if let Some(scanner) = root_ref.min(self.guard) {
895                    self.leaf_scanner.replace(scanner);
896                }
897            } else {
898                return None;
899            }
900        }
901
902        // Go to the next entry.
903        if let Some(mut scanner) = self.leaf_scanner.take() {
904            let min_allowed_key = scanner.get().map(|(key, _)| key);
905            if let Some(result) = scanner.next() {
906                self.leaf_scanner.replace(scanner);
907                return Some(result);
908            }
909            // Go to the next leaf node.
910            if let Some(new_scanner) = scanner.jump(min_allowed_key, self.guard) {
911                if let Some(entry) = new_scanner.get() {
912                    self.leaf_scanner.replace(new_scanner);
913                    return Some(entry);
914                }
915            }
916        }
917        None
918    }
919}

949impl<'g, K, V, Q, R> Range<'_, 'g, K, V, Q, R>
950where
951    K: 'static + Clone + Ord,
952    V: 'static + Clone,
953    Q: Comparable<K> + ?Sized,
954    R: RangeBounds<Q>,
955{
956    #[inline]
957    fn next_unbounded(&mut self) -> Option<(&'g K, &'g V)> {
958        if self.leaf_scanner.is_none() {
959            // Start scanning.
960            let root_ptr = self.root.load(Acquire, self.guard);
961            if let Some(root_ref) = root_ptr.as_ref() {
962                let min_allowed_key = match self.bounds.start_bound() {
963                    Excluded(key) | Included(key) => Some(key),
964                    Unbounded => {
965                        self.check_lower_bound = false;
966                        None
967                    }
968                };
969                let mut leaf_scanner = min_allowed_key
970                    .and_then(|min_allowed_key| root_ref.max_le_appr(min_allowed_key, self.guard));
971                if leaf_scanner.is_none() {
972                    // No `min_allowed_key` is supplied, or no keys smaller than or equal to
973                    // `min_allowed_key` found.
974                    if let Some(mut scanner) = root_ref.min(self.guard) {
975                        // It's possible that the leaf has just been emptied, so go to the next.
976                        scanner.next();
977                        while scanner.get().is_none() {
978                            scanner = scanner.jump(None, self.guard)?;
979                        }
980                        leaf_scanner.replace(scanner);
981                    }
982                }
983                if let Some(leaf_scanner) = leaf_scanner {
984                    if let Some(result) = leaf_scanner.get() {
985                        self.set_check_upper_bound(&leaf_scanner);
986                        self.leaf_scanner.replace(leaf_scanner);
987                        return Some(result);
988                    }
989                }
990            }
991        }
992
993        // Go to the next entry.
994        if let Some(mut leaf_scanner) = self.leaf_scanner.take() {
995            let min_allowed_key = leaf_scanner.get().map(|(key, _)| key);
996            if let Some(result) = leaf_scanner.next() {
997                self.leaf_scanner.replace(leaf_scanner);
998                return Some(result);
999            }
1000            // Go to the next leaf node.
1001            if let Some(new_scanner) = leaf_scanner.jump(min_allowed_key, self.guard) {
1002                if let Some(entry) = new_scanner.get() {
1003                    self.set_check_upper_bound(&new_scanner);
1004                    self.leaf_scanner.replace(new_scanner);
1005                    return Some(entry);
1006                }
1007            }
1008        }
1009
1010        None
1011    }
1012
1013    #[inline]
1014    fn set_check_upper_bound(&mut self, scanner: &Scanner<K, V>) {
1015        self.check_upper_bound = match self.bounds.end_bound() {
1016            Excluded(key) => scanner
1017                .max_key()
1018                .is_some_and(|max_key| key.compare(max_key).is_le()),
1019            Included(key) => scanner
1020                .max_key()
1021                .is_some_and(|max_key| key.compare(max_key).is_lt()),
1022            Unbounded => false,
1023        };
1024    }
1025}

1030        f.debug_struct("Range")
1031            .field("root", &self.root)

1039impl<'g, K, V, Q, R> Iterator for Range<'_, 'g, K, V, Q, R>
1040where
1041    K: 'static + Clone + Ord,
1042    V: 'static + Clone,
1043    Q: Comparable<K> + ?Sized,
1044    R: RangeBounds<Q>,
1045{
1046    type Item = (&'g K, &'g V);
1047
1048    #[inline]
1049    fn next(&mut self) -> Option<Self::Item> {
1050        while let Some((k, v)) = self.next_unbounded() {
1051            if self.check_lower_bound {
1052                match self.bounds.start_bound() {
1053                    Excluded(key) => {
1054                        if key.compare(k).is_ge() {
1055                            continue;
1056                        }
1057                    }
1058                    Included(key) => {
1059                        if key.compare(k).is_gt() {
1060                            continue;
1061                        }
1062                    }
1063                    Unbounded => (),
1064                }
1065            }
1066            self.check_lower_bound = false;
1067            if self.check_upper_bound {
1068                match self.bounds.end_bound() {
1069                    Excluded(key) => {
1070                        if key.compare(k).is_gt() {
1071                            return Some((k, v));
1072                        }
1073                    }
1074                    Included(key) => {
1075                        if key.compare(k).is_ge() {
1076                            return Some((k, v));
1077                        }
1078                    }
1079                    Unbounded => {
1080                        return Some((k, v));
1081                    }
1082                }
1083                break;
1084            }
1085            return Some((k, v));
1086        }
1087        None
1088    }
1089}
