13/// [`BucketArray`] is a special purpose array to manage [`Bucket`] and [`DataBlock`].
14pub struct BucketArray<K, V, L: LruList, const TYPE: char> {
15    buckets: NonNull<Bucket<K, V, L, TYPE>>,
16    data_blocks: NonNull<DataBlock<K, V, BUCKET_LEN>>,
17    array_len: usize,
18    hash_offset: u16,
19    sample_size_mask: u16,
20    bucket_ptr_offset: u16,
21    old_array: AtomicShared<BucketArray<K, V, L, TYPE>>,
22    garbage_link: AtomicShared<BucketArray<K, V, L, TYPE>>,
23    num_cleared_buckets: AtomicUsize,
24}

26impl<K, V, L: LruList, const TYPE: char> BucketArray<K, V, L, TYPE> {
27    /// Creates a new [`BucketArray`] of the given capacity.
28    ///
29    /// `capacity` is the desired number of entries, not the length of the bucket array.
30    pub(crate) fn new(
31        capacity: usize,
32        old_array: AtomicShared<BucketArray<K, V, L, TYPE>>,
33    ) -> Self {
34        let log2_array_len = Self::calculate_log2_array_size(capacity);
35        assert_ne!(log2_array_len, 0);
36
37        let array_len = 1_usize << log2_array_len;
38        unsafe {
39            let (bucket_size, bucket_array_allocation_size, bucket_array_layout) =
40                Self::bucket_array_layout(array_len);
41            let Some(unalignedbucket_array_ptr) = NonNull::new(alloc_zeroed(bucket_array_layout))
42            else {
43                panic!("memory allocation failure: {bucket_array_allocation_size} bytes");
44            };
45            let bucket_array_ptr_offset = bucket_size.next_power_of_two()
46                - (unalignedbucket_array_ptr.addr().get() % bucket_size.next_power_of_two());
47            assert!(
48                bucket_array_ptr_offset + bucket_size * array_len <= bucket_array_allocation_size,
49            );
50            assert_eq!(
51                (unalignedbucket_array_ptr.addr().get() + bucket_array_ptr_offset)
52                    % bucket_size.next_power_of_two(),
53                0
54            );
55
56            #[allow(clippy::cast_ptr_alignment)] // The alignment was just asserted.
57            let buckets = unalignedbucket_array_ptr
58                .add(bucket_array_ptr_offset)
59                .cast::<Bucket<K, V, L, TYPE>>();
60            let bucket_array_ptr_offset = u16::try_from(bucket_array_ptr_offset).unwrap_or(0);
61
62            let data_block_array_layout = Layout::from_size_align(
63                size_of::<DataBlock<K, V, BUCKET_LEN>>() * array_len,
64                align_of::<[DataBlock<K, V, BUCKET_LEN>; 0]>(),
65            )
66            .unwrap();
67
68            // In case the below data block allocation fails, deallocate the bucket array.
69            let mut alloc_guard = ExitGuard::new(false, |allocated| {
70                if !allocated {
71                    dealloc(
72                        unalignedbucket_array_ptr.cast::<u8>().as_ptr(),
73                        bucket_array_layout,
74                    );
75                }
76            });
77
78            let Some(data_blocks) =
79                NonNull::new(alloc(data_block_array_layout).cast::<DataBlock<K, V, BUCKET_LEN>>())
80            else {
81                panic!(
82                    "memory allocation failure: {} bytes",
83                    data_block_array_layout.size()
84                );
85            };
86            *alloc_guard = true;
87
88            Self {
89                buckets,
90                data_blocks,
91                array_len,
92                hash_offset: u16::try_from(u64::BITS).unwrap_or(64) - u16::from(log2_array_len),
93                sample_size_mask: u16::from(log2_array_len).next_power_of_two() - 1,
94                bucket_ptr_offset: bucket_array_ptr_offset,
95                old_array,
96                garbage_link: AtomicShared::null(),
97                num_cleared_buckets: AtomicUsize::new(0),
98            }
99        }
100    }

102    /// Returns the number of [`Bucket`] instances in the [`BucketArray`].
103    #[inline]
104    pub(crate) const fn len(&self) -> usize {
105        self.array_len
106    }
107
108    /// Returns the number of entry slots in the bucket array.
109    #[inline]
110    pub(crate) const fn num_slots(&self) -> usize {
111        self.array_len * BUCKET_LEN
112    }
113
114    /// Calculates the [`Bucket`] index for the hash value.
115    #[allow(clippy::cast_possible_truncation)] // Intended truncation.
116    #[inline]
117    pub(crate) const fn calculate_bucket_index(&self, hash: u64) -> usize {
118        // Take the upper n-bits to make sure that a single bucket is spread across a few adjacent
119        // buckets when the hash table is resized.
120        (hash >> self.hash_offset) as usize
121    }
122
123    /// Returns the minimum capacity.
124    #[inline]
125    pub const fn minimum_capacity() -> usize {
126        BUCKET_LEN << 1
127    }

129    /// Returns a reference to its rehashing metadata.
130    #[inline]
131    pub(crate) const fn rehashing_metadata(&self) -> &AtomicUsize {

135    /// Checks if the bucket is allowed to initiate sampling.
136    #[inline]
137    pub(crate) const fn initiate_sampling(&self, index: usize) -> bool {
138        (index & self.sample_size_mask as usize) == 0
139    }

141    /// Returns the recommended sampling size.
142    #[inline]
143    pub(crate) const fn sample_size(&self) -> usize {
144        (self.sample_size_mask + 1) as usize
145    }

147    /// Returns a reference to a [`Bucket`] at the given position.
148    #[inline]

150        debug_assert!(index < self.len());

157        debug_assert!(index < self.len());
158        unsafe { self.data_blocks.add(index) }
159    }

161    /// Returns a reference to the old array pointer.
162    #[inline]
163    pub(crate) const fn bucket_link(&self) -> &AtomicShared<BucketArray<K, V, L, TYPE>> {
164        &self.old_array
165    }
166
167    /// Returns a reference to the garbage link pointer.
168    #[inline]
169    pub(crate) const fn garbage_link(&self) -> &AtomicShared<BucketArray<K, V, L, TYPE>> {
170        &self.garbage_link
171    }
172
173    /// Returns the recommended sampling size.
174    #[inline]
175    pub(crate) fn full_sample_size(&self) -> usize {
176        (self.sample_size() * self.sample_size()).min(self.len())
177    }
178
179    /// Returns `true` if the old array exists.
180    #[inline]
181    pub(crate) fn has_old_array(&self) -> bool {
182        !self.old_array.is_null(Acquire)
183    }
184
185    /// Returns a [`Ptr`] to the old array.
186    #[inline]
187    pub(crate) fn old_array<'g>(&self, guard: &'g Guard) -> Ptr<'g, BucketArray<K, V, L, TYPE>> {
188        self.old_array.load(Acquire, guard)
189    }
190
191    /// Calculates the layout of the memory block for an array of `T`.
192    #[inline]
193    const fn bucket_array_layout(array_len: usize) -> (usize, usize, Layout) {
194        let size_of_t = size_of::<Bucket<K, V, L, TYPE>>();
195        let aligned_size = size_of_t.next_power_of_two();
196        let allocation_size = aligned_size + array_len * size_of_t;
197        (size_of_t, allocation_size, unsafe {
198            // Intentionally misaligned in order to take full advantage of demand paging.
199            Layout::from_size_align_unchecked(allocation_size, 1)
200        })
201    }
202
203    /// Calculates logâ‚‚ of the array size from the given capacity.
204    ///
205    /// Returns a non-zero `u8`, even when `capacity < 2 * BUCKET_LEN`.
206    #[inline]
207    fn calculate_log2_array_size(capacity: usize) -> u8 {
208        let adjusted_capacity = capacity.clamp(64, (usize::MAX / 2) - (BUCKET_LEN - 1));
209        let required_buckets = adjusted_capacity.div_ceil(BUCKET_LEN).next_power_of_two();
210        let log2_capacity = usize::BITS as usize - (required_buckets.leading_zeros() as usize) - 1;
211
212        // `2^log2_capacity * BUCKET_LEN >= capacity`.
213        debug_assert!(log2_capacity < (usize::BITS as usize));
214        debug_assert!((1_usize << log2_capacity) * BUCKET_LEN >= adjusted_capacity);
215        u8::try_from(log2_capacity).unwrap_or(0)
216    }

219impl<K, V, L: LruList, const TYPE: char> Drop for BucketArray<K, V, L, TYPE> {
220    fn drop(&mut self) {
221        if !self.old_array.is_null(Relaxed) {
222            unsafe {
223                self.old_array
224                    .swap((None, Tag::None), Relaxed)
225                    .0
226                    .map(|a| a.drop_in_place());
227            }
228        }
229
230        let num_cleared_buckets = if TYPE == INDEX {
231            0
232        } else {
233            self.num_cleared_buckets.load(Relaxed)
234        };
235        if num_cleared_buckets < self.array_len {
236            for index in num_cleared_buckets..self.array_len {
237                self.bucket(index).drop_entries(self.data_block(index));
238            }
239        }
240
241        unsafe {
242            dealloc(
243                self.buckets
244                    .cast::<u8>()
245                    .sub(self.bucket_ptr_offset as usize)
246                    .as_ptr(),
247                Self::bucket_array_layout(self.array_len).2,
248            );
249            dealloc(
250                self.data_blocks.cast::<u8>().as_ptr(),
251                Layout::from_size_align(
252                    size_of::<DataBlock<K, V, BUCKET_LEN>>() * self.array_len,
253                    align_of::<[DataBlock<K, V, BUCKET_LEN>; 0]>(),
254                )
255                .unwrap(),
256            );
257        }
258    }
259}
