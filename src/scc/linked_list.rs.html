7/// [`LinkedList`] is a trait implementing a lock-free singly linked list.
8pub trait LinkedList: Sized {
9    /// Returns a reference to the forward link.
10    ///
11    /// The pointer value may be tagged if [`Self::mark`] or [`Self::delete_self`] has been
12    /// invoked. The [`AtomicShared`] must only be updated through [`LinkedList`] in order to keep
13    /// the linked list consistent.
14    fn link_ref(&self) -> &AtomicShared<Self>;
15
16    /// Returns `true` if `self` is reachable and not marked.
17    ///
18    /// # Examples
19    ///
20    /// ```
21    /// use std::sync::atomic::Ordering::Relaxed;
22    ///
23    /// use scc::{AtomicShared, LinkedList, Tag};
24    ///
25    /// #[derive(Default)]
26    /// struct L(AtomicShared<L>, usize);
27    /// impl LinkedList for L {
28    ///     fn link_ref(&self) -> &AtomicShared<L> {
29    ///         &self.0
30    ///     }
31    /// }
32    ///
33    /// let head: L = L::default();
34    /// assert!(head.is_clear(Relaxed));
35    /// assert!(head.mark(Relaxed));
36    /// assert!(!head.is_clear(Relaxed));
37    /// assert!(head.delete_self(Relaxed));
38    /// assert!(!head.is_clear(Relaxed));
39    /// ```
40    #[inline]
41    fn is_clear(&self, order: Ordering) -> bool {
42        self.link_ref().tag(order) == Tag::None
43    }
44
45    /// Marks `self` with an internal flag to denote that `self` is in a special state.
46    ///
47    /// Returns `false` if a flag has already been set on `self`.
48    ///
49    /// # Examples
50    ///
51    /// ```
52    /// use std::sync::atomic::Ordering::Relaxed;
53    ///
54    /// use scc::{AtomicShared, LinkedList};
55    ///
56    /// #[derive(Default)]
57    /// struct L(AtomicShared<L>, usize);
58    /// impl LinkedList for L {
59    ///     fn link_ref(&self) -> &AtomicShared<L> {
60    ///         &self.0
61    ///     }
62    /// }
63    ///
64    /// let head: L = L::default();
65    /// assert!(head.mark(Relaxed));
66    /// ```
67    #[inline]
68    fn mark(&self, order: Ordering) -> bool {
69        self.link_ref()
70            .update_tag_if(Tag::First, |ptr| ptr.tag() == Tag::None, order, Relaxed)
71    }
72
73    /// Removes any mark from `self`.
74    ///
75    /// Returns `false` if no flag has been set on `self`.
76    ///
77    /// # Examples
78    ///
79    /// ```
80    /// use std::sync::atomic::Ordering::Relaxed;
81    ///
82    /// use scc::{AtomicShared, LinkedList};
83    ///
84    /// #[derive(Default)]
85    /// struct L(AtomicShared<L>, usize);
86    /// impl LinkedList for L {
87    ///     fn link_ref(&self) -> &AtomicShared<L> {
88    ///         &self.0
89    ///     }
90    /// }
91    ///
92    /// let head: L = L::default();
93    /// assert!(!head.unmark(Relaxed));
94    /// assert!(head.mark(Relaxed));
95    /// assert!(head.unmark(Relaxed));
96    /// assert!(!head.is_marked(Relaxed));
97    /// ```
98    #[inline]
99    fn unmark(&self, order: Ordering) -> bool {
100        self.link_ref()
101            .update_tag_if(Tag::None, |ptr| ptr.tag() == Tag::First, order, Relaxed)
102    }
103
104    /// Returns `true` if `self` has a mark on it.
105    ///
106    /// # Examples
107    ///
108    /// ```
109    /// use std::sync::atomic::Ordering::Relaxed;
110    ///
111    /// use scc::{AtomicShared, LinkedList};
112    ///
113    /// #[derive(Default)]
114    /// struct L(AtomicShared<L>, usize);
115    /// impl LinkedList for L {
116    ///     fn link_ref(&self) -> &AtomicShared<L> {
117    ///         &self.0
118    ///     }
119    /// }
120    ///
121    /// let head: L = L::default();
122    /// assert!(!head.is_marked(Relaxed));
123    /// assert!(head.mark(Relaxed));
124    /// assert!(head.is_marked(Relaxed));
125    /// ```
126    #[inline]
127    fn is_marked(&self, order: Ordering) -> bool {
128        self.link_ref().tag(order) == Tag::First
129    }
130
131    /// Deletes `self`.
132    ///
133    /// Returns `false` if `self` is already marked as deleted.
134    ///
135    /// # Examples
136    ///
137    /// ```
138    /// use std::sync::atomic::Ordering::Relaxed;
139    ///
140    /// use scc::{AtomicShared, Guard, LinkedList, Shared};
141    ///
142    /// #[derive(Default)]
143    /// struct L(AtomicShared<L>, usize);
144    /// impl LinkedList for L {
145    ///     fn link_ref(&self) -> &AtomicShared<L> {
146    ///         &self.0
147    ///     }
148    /// }
149    ///
150    /// let guard = Guard::new();
151    ///
152    /// let head: L = L::default();
153    /// let tail: Shared<L> = Shared::new(L::default());
154    /// assert!(head.push_back(tail.clone(), false, Relaxed, &guard).is_ok());
155    ///
156    /// tail.delete_self(Relaxed);
157    /// assert!(head.next_ptr(Relaxed, &guard).as_ref().is_none());
158    /// ```
159    #[inline]
160    fn delete_self(&self, order: Ordering) -> bool {
161        self.link_ref().update_tag_if(
162            Tag::Second,
163            |ptr| {
164                let tag = ptr.tag();
165                tag == Tag::None || tag == Tag::First
166            },
167            order,
168            Relaxed,
169        )
170    }
171
172    /// Returns `true` if `self` has been deleted.
173    ///
174    /// # Examples
175    ///
176    /// ```
177    /// use std::sync::atomic::Ordering::Relaxed;
178    ///
179    /// use scc::{AtomicShared, LinkedList};
180    ///
181    /// #[derive(Default)]
182    /// struct L(AtomicShared<L>, usize);
183    /// impl LinkedList for L {
184    ///     fn link_ref(&self) -> &AtomicShared<L> {
185    ///         &self.0
186    ///     }
187    /// }
188    ///
189    /// let entry: L = L::default();
190    /// assert!(!entry.is_deleted(Relaxed));
191    /// entry.delete_self(Relaxed);
192    /// assert!(entry.is_deleted(Relaxed));
193    /// ```
194    #[inline]
195    fn is_deleted(&self, order: Ordering) -> bool {
196        let tag = self.link_ref().tag(order);
197        tag == Tag::Second || tag == Tag::Both
198    }
199
200    /// Appends the given entry to `self` and returns a pointer to the entry.
201    ///
202    /// If `mark` is `true`, it atomically sets an internal flag on `self` when updating
203    /// the linked list, otherwise it removes the mark.
204    ///
205    /// # Errors
206    ///
207    /// Returns the supplied [`Shared`] when it finds `self` deleted.
208    ///
209    /// # Examples
210    ///
211    /// ```
212    /// use std::sync::atomic::Ordering::{Relaxed, Release};
213    ///
214    /// use scc::{AtomicShared, Guard, LinkedList, Shared};
215    ///
216    /// #[derive(Default)]
217    /// struct L(AtomicShared<L>, usize);
218    /// impl LinkedList for L {
219    ///     fn link_ref(&self) -> &AtomicShared<L> {
220    ///         &self.0
221    ///     }
222    /// }
223    ///
224    /// let guard = Guard::new();
225    ///
226    /// let head: L = L::default();
227    /// assert!(head.push_back(Shared::new(L::default()), true, Release, &guard).is_ok());
228    /// assert!(head.is_marked(Relaxed));
229    /// assert!(head.push_back(Shared::new(L::default()), false, Release, &guard).is_ok());
230    /// assert!(!head.is_marked(Relaxed));
231    ///
232    /// head.delete_self(Relaxed);
233    /// assert!(!head.is_marked(Relaxed));
234    /// assert!(head.push_back(Shared::new(L::default()), false, Release, &guard).is_err());
235    /// ```
236    #[inline]
237    fn push_back<'g>(
238        &self,
239        mut entry: Shared<Self>,
240        mark: bool,
241        order: Ordering,
242        guard: &'g Guard,
243    ) -> Result<Ptr<'g, Self>, Shared<Self>> {
244        let new_tag = if mark { Tag::First } else { Tag::None };
245        let mut next_ptr = self.link_ref().load(Relaxed, guard);
246
247        loop {
248            let tag = next_ptr.tag();
249            if tag == Tag::Second || tag == Tag::Both {
250                break;
251            }
252
253            entry
254                .link_ref()
255                .swap((next_ptr.get_shared(), Tag::None), Relaxed);
256            match self.link_ref().compare_exchange_weak(
257                next_ptr,
258                (Some(entry), new_tag),
259                order,
260                Relaxed,
261                guard,
262            ) {
263                Ok((_, updated)) => {
264                    return Ok(updated);
265                }
266                Err((passed, actual)) => {
267                    entry = unsafe { passed.unwrap_unchecked() };
268                    next_ptr = actual;
269                }
270            }
271        }
272
273        // `current` has been deleted.
274        Err(entry)
275    }
276
277    /// Returns the closest next valid entry.
278    ///
279    /// It unlinks deleted entries until it reaches a valid one.
280    ///
281    /// # Examples
282    ///
283    /// ```
284    /// use std::sync::atomic::Ordering::{Acquire, Relaxed, Release};
285    ///
286    /// use scc::{AtomicShared, Guard, LinkedList, Shared};
287    ///
288    /// #[derive(Default)]
289    /// struct L(AtomicShared<L>, usize);
290    /// impl LinkedList for L {
291    ///     fn link_ref(&self) -> &AtomicShared<L> {
292    ///         &self.0
293    ///     }
294    /// }
295    ///
296    /// let guard = Guard::new();
297    ///
298    /// let head: L = L::default();
299    /// assert!(
300    ///     head.push_back(Shared::new(L(AtomicShared::null(), 1)), false, Release, &guard).is_ok());
301    /// head.mark(Relaxed);
302    ///
303    /// let next_ptr = head.next_ptr(Acquire, &guard);
304    /// assert_eq!(next_ptr.as_ref().unwrap().1, 1);
305    /// assert!(head.is_marked(Relaxed));
306    /// ```
307    #[inline]
308    fn next_ptr<'g>(&self, order: Ordering, guard: &'g Guard) -> Ptr<'g, Self> {
309        next_ptr_recursive(self, order, 64, guard)
310    }
311
312    /// Returns a [`Shared`] handle to the closest next valid entry.
313    ///
314    /// It unlinks deleted entries until it reaches a valid one.
315    ///
316    /// # Examples
317    ///
318    /// ```
319    /// use std::sync::atomic::Ordering::{Acquire, Relaxed, Release};
320    ///
321    /// use scc::{AtomicShared, Guard, LinkedList, Shared};
322    ///
323    /// #[derive(Default)]
324    /// struct L(AtomicShared<L>, usize);
325    /// impl LinkedList for L {
326    ///     fn link_ref(&self) -> &AtomicShared<L> {
327    ///         &self.0
328    ///     }
329    /// }
330    ///
331    /// let guard = Guard::new();
332    ///
333    /// let head: L = L::default();
334    /// assert!(
335    ///     head.push_back(Shared::new(L(AtomicShared::null(), 1)), false, Release, &guard).is_ok());
336    /// head.mark(Relaxed);
337    ///
338    /// let next_shared = head.next_shared(Acquire, &guard);
339    /// assert_eq!(next_shared.unwrap().1, 1);
340    /// assert!(head.is_marked(Relaxed));
341    /// ```
342    #[inline]
343    fn next_shared(&self, order: Ordering, guard: &Guard) -> Option<Shared<Self>> {
344        let mut next_ptr = self.next_ptr(order, guard);
345        let mut next_entry = next_ptr.get_shared();
346        while !next_ptr.is_null() && next_entry.is_none() {
347            // The entry was released in the mean time.
348            next_ptr = next_ptr
349                .as_ref()
350                .map_or_else(Ptr::null, |n| n.next_ptr(Acquire, guard));
351            next_entry = next_ptr.get_shared();
352        }
353        next_entry
354    }
355}

357/// [`Entry`] stores an instance of `T` and a link to the next entry.
358pub struct Entry<T> {
359    /// `instance` is always `Some` unless [`Self::take_inner`] is called.
360    instance: Option<T>,
361    /// `next` points to the next entry in a linked list.
362    next: AtomicShared<Self>,
363}
364
365impl<T> Entry<T> {
366    /// Extracts the inner instance of `T`.
367    ///
368    /// # Safety
369    ///
370    /// This method has to be called at most once per [`Entry`], and the caller needs to make sure
371    /// that the [`Entry`] is not accessed via [`LinkedList`] methods.
372    ///
373    /// # Examples
374    ///
375    /// ```
376    /// use scc::Stack;
377    ///
378    /// let stack: Stack<usize> = Stack::default();
379    ///
380    /// stack.push(37);
381    ///
382    /// let mut entry = stack.pop().unwrap();
383    /// let pushed = unsafe { entry.get_mut().unwrap().take_inner() };
384    /// assert_eq!(pushed, 37);
385    /// ```
386    #[inline]
387    pub unsafe fn take_inner(&mut self) -> T {
388        unsafe { self.instance.take().unwrap_unchecked() }
389    }
390
391    #[inline]
392    pub(super) fn new(val: T) -> Self {
393        Self {
394            instance: Some(val),
395            next: AtomicShared::default(),
396        }
397    }
398
399    /// Returns a reference to `next`.
400    #[inline]
401    pub(super) fn next(&self) -> &AtomicShared<Self> {
402        &self.next
403    }
404}

462            if let Some(next_entry) = self.next.load(Relaxed, &guard).as_ref() {
463                next_ptr_recursive(next_entry, Relaxed, 64, &guard);
464            }
465        }

496/// Recursively cleans up the linked list starting from the supplied head.
497fn next_ptr_recursive<'g, T: LinkedList>(
498    head: &T,
499    order: Ordering,
500    mut depth: usize,
501    guard: &'g Guard,
502) -> Ptr<'g, T> {
503    let mut head_next_ptr = head.link_ref().load(order, guard);
504    let mut head_tag = head_next_ptr.tag();
505    loop {
506        let mut next_ptr = head_next_ptr;
507        let next_valid_ptr = loop {
508            if let Some(next_ref) = next_ptr.as_ref() {
509                let next_next_ptr = next_ref.link_ref().load(order, guard);
510                if next_next_ptr.tag() != Tag::Second {
511                    break next_ptr;
512                }
513                if depth != 0 {
514                    break next_ptr_recursive(next_ref, order, depth - 1, guard);
515                }
516                next_ptr = next_next_ptr;
517            } else {
518                break Ptr::null();
519            }
520        };
521
522        // Do not allow a recursive call more than once.
523        depth = 0;
524
525        // Updates its link if there is an invalid entry between itself and the next valid one.
526        if next_valid_ptr.with_tag(head_tag) != head_next_ptr {
527            let next_valid_entry = next_valid_ptr.get_shared();
528            if !next_valid_ptr.is_null() && next_valid_entry.is_none() {
529                // The entry was unlinked in the meantime.
530                head_next_ptr = head.link_ref().load(order, guard);
531                head_tag = head_next_ptr.tag();
532                continue;
533            }
534            match head.link_ref().compare_exchange(
535                head_next_ptr,
536                (next_valid_entry, head_tag),
537                AcqRel,
538                Acquire,
539                guard,
540            ) {
541                Ok((prev, _)) => {
542                    if let Some(removed) = prev {
543                        let _: bool = removed.release();
544                    }
545                }
546                Err((_, actual)) => {
547                    head_next_ptr = actual;
548                    head_tag = actual.tag();
549                    continue;
550                }
551            }
552        }
553
554        return next_valid_ptr;
555    }
