10/// [`AsyncGuard`] is used when an asynchronous task needs to be suspended without invalidating any
11/// references.
12///
13/// The validity of those references must be checked and verified by the user.
14#[derive(Debug, Default)]
15pub(crate) struct AsyncGuard {
16    /// [`Guard`] that can be dropped without invalidating any references.
17    guard: UnsafeCell<Option<Guard>>,
18}
19
20#[derive(Debug)]
21pub(crate) struct AsyncWait {
22    /// Allows the user to await the lock anywhere in the code.
23    pager: Pager<'static, Lock>,
24}
25
26/// [`TryWait`] allows [`AsyncWait`] to be used in synchronous methods.
27pub(crate) trait TryWait {
28    /// Registers the [`Pager`] in the [`Lock`], or synchronously waits for the [`Lock`] to be
29    /// available.
30    fn try_wait(&mut self, lock: &Lock);
31}
32
33impl AsyncGuard {
34    /// Returns `true` if the [`AsyncGuard`] contains a valid [`Guard`].
35    #[inline]
36    pub(crate) fn has_guard(&self) -> bool {
37        unsafe { (*self.guard.get()).is_some() }
38    }
39
40    /// Returns or creates a new [`Guard`].
41    ///
42    /// # Safety
43    ///
44    /// The caller must ensure that any references derived from the returned [`Guard`] do not
45    /// outlive the underlying instance.
46    #[inline]
47    pub(crate) fn guard(&self) -> &Guard {
48        unsafe { (*self.guard.get()).get_or_insert_with(Guard::new) }
49    }
50
51    /// Resets the [`AsyncGuard`] to its initial state.
52    #[inline]
53    pub(crate) fn reset(&self) {
54        unsafe {
55            *self.guard.get() = None;
56        }
57    }
58
59    /// Loads the content of the [`AtomicShared`] without exposing the [`Guard`].
60    #[inline]
61    pub(crate) fn load<T>(&self, atomic_ptr: &AtomicShared<T>, mo: Ordering) -> Option<&T> {
62        atomic_ptr.load(mo, self.guard()).as_ref()
63    }
64
65    /// Checks if the reference is valid.
66    #[inline]
67    pub(crate) fn check_ref<T>(&self, atomic_ptr: &AtomicShared<T>, r: &T, mo: Ordering) -> bool {
68        atomic_ptr
69            .load(mo, self.guard())
70            .as_ref()
71            .is_some_and(|s| ptr::eq(s, r))
72    }
73}

75// SAFETY: this is the sole purpose of `AsyncGuard`; Send-safety should be ensured by the user,
76// e.g., the `AsyncGuard` should always be reset before the task is suspended.
77unsafe impl Send for AsyncGuard {}
78unsafe impl Sync for AsyncGuard {}

80impl AsyncWait {
81    /// Awaits the [`Lock`] to be available.
82    #[inline]
83    pub async fn wait(self: &mut Pin<&mut Self>) {
84        let this = unsafe { ptr::read(self) };
85        let mut pinned_pager = unsafe { Pin::new_unchecked(&mut this.get_unchecked_mut().pager) };
86        let _result = pinned_pager.poll_async().await;
87    }
88}
89
90impl Default for AsyncWait {
91    #[inline]
92    fn default() -> Self {
93        Self {
94            pager: unsafe {
95                std::mem::transmute::<Pager<'_, Lock>, Pager<'static, Lock>>(Pager::default())
96            },
97        }
98    }
99}
100
101impl TryWait for Pin<&mut AsyncWait> {
102    #[inline]
103    fn try_wait(&mut self, lock: &Lock) {
104        let this = unsafe { ptr::read(self) };
105        let mut pinned_pager = unsafe {
106            let pager_ref = std::mem::transmute::<&mut Pager<'static, Lock>, &mut Pager<Lock>>(
107                &mut this.get_unchecked_mut().pager,
108            );
109            Pin::new_unchecked(pager_ref)
110        };
111        lock.register_pager(&mut pinned_pager, Mode::WaitExclusive, false);
112    }
113}
114
115impl TryWait for () {
116    #[inline]
117    fn try_wait(&mut self, lock: &Lock) {
118        let mut pinned_pager = pin!(Pager::default());
119        lock.register_pager(&mut pinned_pager, Mode::WaitExclusive, true);
120        let _: Result<_, _> = pinned_pager.poll_sync();
121    }
122}
