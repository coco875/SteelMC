14/// Decide how prime candidates are manipulated by setting certain bits before primality testing,
15/// influencing the range of the prime.
16#[derive(Debug, Clone, Copy)]
17pub enum SetBits {
18    /// Set the most significant bit, thus limiting the range to `[MAX/2 + 1, MAX]`.
19    ///
20    /// In other words, all candidates will have the same bit size.
21    Msb,
22    /// Set two most significant bits, limiting the range to `[MAX - MAX/4 + 1, MAX]`.
23    ///
24    /// This is useful in the RSA case because a product of two such numbers will have a guaranteed bit size.
25    TwoMsb,
26    /// No additional bits set; uses the full range `[1, MAX]`.
27    None,
28}
29
30/// Returns a random odd integer up to the given bit length.
31///
32/// The `set_bits` parameter decides which extra bits are set, which decides the range of the number.
33///
34/// Returns an error variant if `bit_length` is greater than the maximum allowed for `T`
35/// (applies to fixed-length types).
36pub fn random_odd_integer<T, R>(rng: &mut R, bit_length: NonZeroU32, set_bits: SetBits) -> Result<Odd<T>, Error>
37where
38    T: Unsigned + RandomBits,
39    R: CryptoRng + ?Sized,
40{
41    let bit_length = bit_length.get();
42
43    let mut random = T::try_random_bits(rng, bit_length).map_err(|err| match err {
44        RandomBitsError::RandCore(_) => unreachable!("`rng` impls `CryptoRng` and therefore is infallible"),
45        RandomBitsError::BitsPrecisionMismatch { .. } => {
46            unreachable!("we are not requesting a specific `bits_precision`")
47        }
48        RandomBitsError::BitLengthTooLarge {
49            bit_length,
50            bits_precision,
51        } => Error::BitLengthTooLarge {
52            bit_length,
53            bits_precision,
54        },
55    })?;
56
57    // Make it odd
58    // `bit_length` is non-zero, so the 0-th bit exists.
59    random.set_bit_vartime(0, true);
60
61    // Will not overflow since `bit_length` is ensured to be within the size of the integer
62    // (checked within the `T::try_random_bits()` call).
63    // `bit_length - 1`-th bit exists since `bit_length` is non-zero.
64    match set_bits {
65        SetBits::None => {}
66        SetBits::Msb => random.set_bit_vartime(bit_length - 1, true),
67        SetBits::TwoMsb => {
68            random.set_bit_vartime(bit_length - 1, true);
69            // We could panic here, but since the primary purpose of `TwoMsb` is to ensure the bit length
70            // of the product of two numbers, ignoring this for `bit_length = 1` leads to the desired result.
71            if bit_length > 1 {
72                random.set_bit_vartime(bit_length - 2, true);
73            }
74        }
75    }
76
77    Ok(Odd::new(random).expect("the number is odd by construction"))
78}

80// The type we use to calculate incremental residues.
81// Should be >= `SmallPrime` in size.
82type Residue = u32;
83
84// The maximum increment that won't overflow the type we use to calculate residues of increments:
85// we need `(max_prime - 1) + max_incr <= Type::MAX`.
86const INCR_LIMIT: Residue = Residue::MAX - LAST_SMALL_PRIME as Residue + 1;
87
88/// An iterator returning numbers with up to and including given bit length,
89/// starting from a given number, that are not multiples of the first 2048 small primes.
90#[derive(Clone, Debug, PartialEq, Eq)]
91pub struct SmallFactorsSieve<T: Unsigned> {
92    // Instead of dividing a big integer by small primes every time (which is slow),
93    // we keep a "base" and a small increment separately,
94    // so that we can only calculate the residues of the increment.
95    base: T,
96    incr: Residue,
97    incr_limit: Residue,
98    safe_primes: bool,
99    residues: Vec<SmallPrime>,
100    max_bit_length: u32,
101    produces_nothing: bool,
102    starts_from_exception: bool,
103    last_round: bool,
104}
105
106impl<T> SmallFactorsSieve<T>
107where
108    T: Unsigned,
109{
110    /// Creates a new sieve, iterating from `start` and until the last number with `max_bit_length`
111    /// bits, producing numbers that are not non-trivial multiples of a list of small primes in the
112    /// range `[2, start)` (`safe_primes = false`) or `[2, start/2)` (`safe_primes = true`).
113    ///
114    /// Note that `start` is adjusted to `2`, or the next `1 mod 2` number (`safe_primes = false`);
115    /// and `5`, or `3 mod 4` number (`safe_primes = true`).
116    ///
117    /// Panics if `max_bit_length` greater than the precision of `start`.
118    ///
119    /// If `safe_primes` is `true`, both the returned `n` and `n/2` are sieved.
120    pub fn new(start: T, max_bit_length: NonZeroU32, safe_primes: bool) -> Result<Self, Error> {
121        let max_bit_length = max_bit_length.get();
122
123        if max_bit_length > start.bits_precision() {
124            return Err(Error::BitLengthTooLarge {
125                bit_length: max_bit_length,
126                bits_precision: start.bits_precision(),
127            });
128        }
129
130        // If we are targeting safe primes, iterate over the corresponding
131        // possible Germain primes (`n/2`), reducing the task to that with `safe_primes = false`.
132        let (max_bit_length, mut start) = if safe_primes {
133            (max_bit_length - 1, start.wrapping_shr_vartime(1))
134        } else {
135            (max_bit_length, start)
136        };
137
138        // This is easier than making all the methods generic enough to handle these corner cases.
139        let produces_nothing = max_bit_length < start.bits_vartime() || max_bit_length < 2;
140
141        // Add the exception to the produced candidates - the only one that doesn't fit
142        // the general pattern of incrementing the base by 2.
143        let mut starts_from_exception = false;
144        if start <= T::from(2u32) {
145            starts_from_exception = true;
146            start = T::from(3u32);
147        } else {
148            // Adjust the start so that we hit odd numbers when incrementing it by 2.
149            start |= T::one();
150        }
151
152        // Only calculate residues by primes up to and not including `start`, because when we only
153        // have the resiude, we cannot distinguish between a prime itself and a multiple of that
154        // prime.
155        let residues_len = if T::from(LAST_SMALL_PRIME) <= start {
156            SMALL_PRIMES.len()
157        } else {
158            // `start` is smaller than the last prime in the list so casting `start` to a `u16` is
159            // safe. We need to find out how many residues we can use.
160            let start_small = start.as_ref()[0].0 as SmallPrime;
161            SMALL_PRIMES.partition_point(|x| *x < start_small)
162        };
163
164        Ok(Self {
165            base: start,
166            incr: 0, // This will ensure that `update_residues()` is called right away.
167            incr_limit: 0,
168            safe_primes,
169            residues: vec![0; residues_len],
170            max_bit_length,
171            produces_nothing,
172            starts_from_exception,
173            last_round: false,
174        })
175    }
176
177    fn update_residues(&mut self) -> bool {
178        if self.incr_limit != 0 && self.incr <= self.incr_limit {
179            return true;
180        }
181
182        if self.last_round {
183            return false;
184        }
185
186        // Set the new base.
187        // Should not overflow since `incr` is never greater than `incr_limit`,
188        // and the latter is chosen such that it doesn't overflow when added to `base`
189        // (see the rest of this method).
190        self.base = self
191            .base
192            .checked_add(&self.incr.into())
193            .expect("Does not overflow by construction");
194
195        self.incr = 0;
196
197        // Re-calculate residues. This is taking up most of the sieving time.
198        for (i, rec) in RECIPROCALS.iter().enumerate().take(self.residues.len()) {
199            let rem = self.base.rem_limb_with_reciprocal(rec);
200            self.residues[i] = rem.0 as SmallPrime;
201        }
202
203        // Find the increment limit.
204        let max_value = match T::one_like(&self.base)
205            .overflowing_shl_vartime(self.max_bit_length)
206            .into()
207        {
208            Some(val) => val,
209            None => T::one_like(&self.base),
210        };
211        let incr_limit = max_value.wrapping_sub(&self.base);
212        self.incr_limit = if incr_limit > T::from(INCR_LIMIT) {
213            INCR_LIMIT
214        } else {
215            // We are close to `2^max_bit_length - 1`.
216            // Mark this round as the last.
217            self.last_round = true;
218            // Can unwrap here since we just checked above that `incr_limit <= INCR_LIMIT`,
219            // and `INCR_LIMIT` fits into `Residue`.
220            let incr_limit_small: Residue = incr_limit.as_ref()[0]
221                .0
222                .try_into()
223                .expect("the increment limit should fit within `Residue`");
224            incr_limit_small
225        };
226
227        true
228    }
229
230    // Returns `true` if the current `base + incr` is divisible by any of the small primes.
231    fn current_is_composite(&self) -> bool {
232        self.residues.iter().enumerate().any(|(i, m)| {
233            let d = SMALL_PRIMES[i] as Residue;
234            let r = (*m as Residue + self.incr) % d;
235
236            // A trick from "Safe Prime Generation with a Combined Sieve" by Michael J. Wiener
237            // (https://eprint.iacr.org/2003/186).
238            // Remember that the check above was for the `(n - 1)/2`;
239            // If `(n - 1)/2 mod d == (d - 1)/2`, it means that `n mod d == 0`.
240            // In other words, we are checking the remainder of `n mod d`
241            // for virtually no additional cost.
242            r == 0 || (self.safe_primes && r == (d - 1) >> 1)
243        })
244    }
245
246    // Returns the restored `base + incr` if it is not composite (wrt the small primes),
247    // and bumps the increment unconditionally.
248    fn maybe_next(&mut self) -> Option<T> {
249        let result = if self.current_is_composite() {
250            None
251        } else {
252            match self.base.checked_add(&self.incr.into()).into_option() {
253                Some(mut num) => {
254                    if self.safe_primes {
255                        // Divide by 2 and ensure it's odd with an OR.
256                        num = num.wrapping_shl_vartime(1) | T::one_like(&self.base);
257                    }
258                    Some(num)
259                }
260                None => None,
261            }
262        };
263
264        self.incr += 2;
265        result
266    }
267
268    fn next(&mut self) -> Option<T> {
269        // Corner cases handled here
270        if self.produces_nothing {
271            return None;
272        }
273
274        if self.starts_from_exception {
275            self.starts_from_exception = false;
276            return Some(T::from(if self.safe_primes { 5u32 } else { 2u32 }));
277        }
278
279        // Main loop
280
281        while self.update_residues() {
282            match self.maybe_next() {
283                Some(x) => return Some(x),
284                None => continue,
285            };
286        }
287        None
288    }
289}

302/// A type producing sieves for random prime generation.
303pub trait SieveFactory {
304    /// The type of items returning by the sieves.
305    type Item;
306
307    /// The resulting sieve.
308    type Sieve: Iterator<Item = Self::Item>;
309
310    /// Makes a sieve given an RNG and the previous exhausted sieve (if any).
311    ///
312    /// Returning `None` signals that the prime generation should stop.
313    fn make_sieve<R>(
314        &mut self,
315        rng: &mut R,
316        previous_sieve: Option<&Self::Sieve>,
317    ) -> Result<Option<Self::Sieve>, Error>
318    where
319        R: CryptoRng + ?Sized;
320}
321
322/// A sieve returning numbers that are not multiples of a set of small factors.
323#[derive(Debug, Clone, Copy)]
324pub struct SmallFactorsSieveFactory<T> {
325    max_bit_length: NonZeroU32,
326    safe_primes: bool,
327    set_bits: SetBits,
328    phantom: PhantomData<T>,
329}
330
331impl<T> SmallFactorsSieveFactory<T>
332where
333    T: Unsigned + RandomBits,
334{
335    /// Creates a factory that produces sieves returning numbers of at most `max_bit_length` bits
336    /// that are not divisible by a number of small factors.
337    ///
338    /// Some bits may be guaranteed to set depending on the requested `set_bits`.
339    ///
340    /// Depending on the requested `flavor`, additional filters may be applied.
341    pub fn new(flavor: Flavor, max_bit_length: u32, set_bits: SetBits) -> Result<Self, Error> {
342        match flavor {
343            Flavor::Any => {
344                if max_bit_length < 2 {
345                    return Err(Error::BitLengthTooSmall {
346                        bit_length: max_bit_length,
347                        flavor,
348                    });
349                }
350            }
351            Flavor::Safe => {
352                if max_bit_length < 3 {
353                    return Err(Error::BitLengthTooSmall {
354                        bit_length: max_bit_length,
355                        flavor,
356                    });
357                }
358            }
359        }
360        let max_bit_length = NonZero::new(max_bit_length).expect("`bit_length` should be non-zero");
361        Ok(Self {
362            max_bit_length,
363            safe_primes: match flavor {
364                Flavor::Any => false,
365                Flavor::Safe => true,
366            },
367            set_bits,
368            phantom: PhantomData,
369        })
370    }
371}

373impl<T> SieveFactory for SmallFactorsSieveFactory<T>
374where
375    T: Unsigned + RandomBits,
376{
377    type Item = T;
378    type Sieve = SmallFactorsSieve<T>;
379    fn make_sieve<R>(
380        &mut self,
381        rng: &mut R,
382        _previous_sieve: Option<&Self::Sieve>,
383    ) -> Result<Option<Self::Sieve>, Error>
384    where
385        R: CryptoRng + ?Sized,
386    {
387        let start = random_odd_integer::<T, _>(rng, self.max_bit_length, self.set_bits)?;
388        Ok(Some(SmallFactorsSieve::new(
389            start.get(),
390            self.max_bit_length,
391            self.safe_primes,
392        )?))
393    }
394}
