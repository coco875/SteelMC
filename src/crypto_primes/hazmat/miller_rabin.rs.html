11/// Precomputed data used to perform Miller-Rabin primality test[^Pomerance1980].
12///
13/// The numbers that pass it are commonly called "strong probable primes"
14/// (or "strong pseudoprimes" if they are, in fact, composite).
15///
16/// The implementation satisfies the FIPS.186-5 standard[^FIPS].
17///
18/// [^Pomerance1980]: C. Pomerance, J. L. Selfridge, S. S. Wagstaff "The Pseudoprimes to 25*10^9",
19///   Math. Comp. 35 1003-1026 (1980),
20///   DOI: [10.2307/2006210](https://dx.doi.org/10.2307/2006210)
21///
22/// [^FIPS]: FIPS-186.5 standard, <https://nvlpubs.nist.gov/nistpubs/FIPS/NIST.FIPS.186-5.pdf>
23#[derive(Clone, Debug, PartialEq, Eq)]
24pub struct MillerRabin<T: Unsigned> {
25    // The odd number that may or may not be a prime.
26    candidate: T,
27    /// The number of bits necessesary to represent the candidate. Note: this is not the number of
28    /// bits used by a `T` in memory.
29    bits: u32,
30    /// Pre-computed parameters for the Montgomery form of `T`.
31    montgomery_params: <<T as Unsigned>::Monty as Monty>::Params,
32    /// The number 1 in Montgomery form.
33    one: <T as Unsigned>::Monty,
34    /// The number -1 in Montgomery form.
35    minus_one: <T as Unsigned>::Monty,
36    /// The `s` exponent in the Miller-Rabin test, that finds `s` and `d` odd s.t. `candidate - 1 ==
37    /// 2^s * d` (the pair `s` and `d` is unique).
38    s: u32,
39    /// The `d` factor in the Miller-Rabin test, that finds `s` and `d` odd s.t. `candidate -
40    /// 1 == 2^s * d` (the pair `s` and `d` is unique).
41    d: T,
42}
43
44impl<T: Unsigned + RandomMod> MillerRabin<T> {
45    /// Initializes a Miller-Rabin test for `candidate`.
46    pub fn new(candidate: Odd<T>) -> Self {
47        let params = <T as Unsigned>::Monty::new_params_vartime(candidate.clone());
48        let m_one = <T as Unsigned>::Monty::one(params.clone());
49        let m_minus_one = -m_one.clone();
50
51        let one = T::one_like(candidate.as_ref());
52
53        // Find `s` and odd `d` such that `candidate - 1 == 2^s * d`.
54        let (s, d) = if candidate.as_ref() == &one {
55            (0, one)
56        } else {
57            let candidate_minus_one = candidate.wrapping_sub(&one);
58            let s = candidate_minus_one.trailing_zeros_vartime();
59            // Will not overflow because `candidate` is odd and greater than 1.
60            let d = candidate_minus_one
61                .overflowing_shr_vartime(s)
62                .expect("shifting by `s` is within range by construction: `candidate` is odd and greater than 1");
63            (s, d)
64        };
65
66        Self {
67            bits: candidate.bits_vartime(),
68            candidate: candidate.get(),
69            montgomery_params: params,
70            one: m_one,
71            minus_one: m_minus_one,
72            s,
73            d,
74        }
75    }
76
77    /// Perform a Miller-Rabin check with a given base.
78    pub fn test(&self, base: &T) -> Primality {
79        // One could check here if `gcd(base, candidate) == 1` and return `Composite` otherwise.
80        // In practice it doesn't make any performance difference in normal operation.
81
82        let base = <T as Unsigned>::Monty::new(base.clone(), self.montgomery_params.clone());
83
84        // Implementation detail: bounded exp gets faster every time we decrease the bound
85        // by the window length it uses, which is currently 4 bits.
86        // So even when the bound isn't low enough that the number can fit
87        // in a smaller number of limbs, there is still a performance gain
88        // from specifying the bound.
89        let mut test = base.pow_bounded_exp(&self.d, self.bits);
90
91        if test == self.one || test == self.minus_one {
92            return Primality::ProbablyPrime;
93        }
94        for _ in 1..self.s {
95            test = test.square();
96            if test == self.one {
97                return Primality::Composite;
98            } else if test == self.minus_one {
99                return Primality::ProbablyPrime;
100            }
101        }
102        Primality::Composite
103    }
104
105    /// Perform a Miller-Rabin check with base 2.
106    pub fn test_base_two(&self) -> Primality {
107        self.test(&T::from_limb_like(Limb::from(2u32), &self.candidate))
108    }
109
110    /// Perform a Miller-Rabin check with a random base (in the range `[2, candidate-2]`,
111    /// because the test holds trivially for bases 1 or `candidate-1`) drawn using the provided RNG.
112    ///
113    /// *Note:* if `candidate == 1` or `candidate == 3` (which would make the above range contain no numbers)
114    /// no check is actually performed, since we already know the result
115    /// ([`Primality::Composite`] for 1, [`Primality::Prime`] for 3).
116    pub fn test_random_base<R: CryptoRng + ?Sized>(&self, rng: &mut R) -> Primality {
117        if equals_primitive(&self.candidate, 1) {
118            // As per standard convention
119            return Primality::Composite;
120        }
121
122        if equals_primitive(&self.candidate, 3) {
123            // As per standard convention
124            return Primality::Prime;
125        }
126
127        // The candidate is odd, so by now it is guaranteed to be >= 5.
128        let range = self.candidate.wrapping_sub(&T::from(3u32));
129        let range_nonzero = CTNonZero::new(range).expect("the range should be non-zero by construction");
130        // This should not overflow as long as `random_mod()` behaves according to the contract
131        // (that is, returns a number within the given range).
132        let random = T::random_mod(rng, &range_nonzero)
133            .checked_add(&T::from(2u32))
134            .expect("addition should not overflow by construction");
135        self.test(&random)
136    }
137}

139/**
140Returns the probability `p_{k,t}` of an odd `k`-bit integer passing `t` rounds of MR testing with random bases
141is actually composite.
142
143Taken from FIPS-186.5[^FIPS], Section C.1, Eq. (2); FIPS in turn quotes Damgård et al[^Damgard].
144There it can be found as Eq. (4.1), with the factors bounded by Proposition 1 and Proposition 2 in Section 4.
145
146**Warning**:[^FIPS]
147Care must be taken when using the phrase "error probability" in connection with
148the recommended number of rounds of MR testing. The probability that a composite number of bit length `k`
149survives `t` rounds of Miller-Rabin testing is not the same as `p_{k,t}`, which is the probability that a
150number surviving `t` rounds of Miller-Rabin testing is composite. Ordinarily, the latter probability
151is the one that should be of most interest to a party responsible for generating primes, while the
152former may be more important to a party responsible for validating the primality of a number
153generated by someone else. However, for sufficiently large `k` (e.g., `k ≥ 51`), it can be shown that
154`p_{k,t} ≤ 4^{–t}` under the same assumptions concerning the selection of candidates as those made to
155obtain formula for `p_{k,t}` (see Damgård et al[^Damgard]).
156In such cases, `t = ceil(–log2(p_target)/2)` rounds of Miller-Rabin testing
157can be used to both generate and validate primes with `p_target` serving as an upper bound on both
158the probability that the generation process yields a composite number and the probability that a
159composite number would survive an attempt to validate its primality.
160
161[^FIPS]: FIPS-186.5 standard, <https://nvlpubs.nist.gov/nistpubs/FIPS/NIST.FIPS.186-5.pdf>
162[^Damgard]:
163    Damgård I, Landrock P, Pomerance C (1993) Average Case Error Estimates for the Strong
164    Probable Prime Test. Mathematics of Computation 61(203):177-194 (1993),
165    <https://www.ams.org/journals/mcom/1993-61-203/S0025-5718-1993-1189518-9/S0025-5718-1993-1189518-9.pdf>
166*/
167const fn pseudoprime_probability(k: u32, t: u32, cap_m: u32) -> f64 {
168    // Simplify Eq. (2) from [^FIPS] by factoring out `2^{k-2}`, which makes it
169    // p_{k,t} = 2.000743 ln(2) k 2^{-2} (
170    //     2^{-Mt} +
171    //     8 (pi^2 - 6) / 3 * sum(m=3..M) sum(j=2..m) 2^{m - (m-1)t - j - (k-1)/j})
172
173    // `2.00743 ln(2) k` comes from Proposition 2 in [^Damgard], from the estimate for `(pi(2^k) - pi(2^{k-1}))^{-1}`.
174
175    // Calculate the powers of 2 under the summations.
176    //
177    // Technically, only a few terms really contribute to the result, and the rest are extremely small in comparison.
178    // But finding out which ones specifically is a little messy.
179    // Can be done if more performance is desired.
180    let mut s = 0.;
181    let mut m = 3i32;
182    while m <= cap_m as i32 {
183        let mut j = 2i32;
184        while j <= m {
185            // Note that we are using `two_powf_upper_bound()` which means we are getting a slightly larger result,
186            // and therefore very slightly overestimating the resulting probability.
187            // This means safety is not compromised.
188            s += two_powf_upper_bound((m - (m - 1) * (t as i32) - j) as f64 - (k - 1) as f64 / j as f64);
189            j += 1;
190        }
191        m += 1;
192    }
193
194    const PI: f64 = core::f64::consts::PI;
195
196    // `2.00743 * ln(2) * 2^(-2)`
197    const COEFF: f64 = 0.3478611111678627;
198
199    COEFF * k as f64 * (1. / two_powi(cap_m * t) + 8. * (PI * PI - 6.) / 3. * s)
200}

202/// For a candidate of size `bit_length`, returns the minimum number of Miller-Rabin tests with random bases
203/// required for the probability of hitting a pseudoprime (that is, a composite for which all M-R tests pass)
204/// to be smaller than `2^{-log2_target}`.
205///
206/// Returns `None` if the number of iterations could not be found for the given bounds.
207///
208/// This function implements the formula prescribed by the FIPS.186-5 standard[^FIPS].
209///
210/// [^FIPS]: FIPS-186.5 standard, <https://nvlpubs.nist.gov/nistpubs/FIPS/NIST.FIPS.186-5.pdf>
211pub const fn minimum_mr_iterations(bit_length: u32, log2_target: u32) -> Option<usize> {
212    let mut t = 1;
213    while t <= log2_target.div_ceil(2) {
214        let cap_m_limit = floor_sqrt(4 * (bit_length - 1)) - 1;
215
216        let mut cap_m = 3;
217        while cap_m <= cap_m_limit {
218            let p = pseudoprime_probability(bit_length, t, cap_m);
219            if p < 1. / two_powi(log2_target) {
220                return Some(t as usize);
221            }
222            cap_m += 1;
223        }
224
225        t += 1;
226    }
227
228    None
229}

266    fn random_checks<T: Unsigned + RandomMod, R: CryptoRng + ?Sized>(
267        rng: &mut R,
268        mr: &MillerRabin<T>,
269        count: usize,
270    ) -> usize {
271        (0..count)
272            .map(|_| -> usize { mr.test_random_base(rng).is_probably_prime().into() })
273            .sum()
274    }

276    fn test_composites(numbers: &[u32], expected_result: bool) {
277        let mut rng = ChaCha8Rng::from_seed(*b"01234567890123456789012345678901");
278        for num in numbers.iter() {
279            let base_2_false_positive = is_spsp(*num);
280            let actual_expected_result = if base_2_false_positive { true } else { expected_result };
281
282            // A random base MR test is expected to report a composite as a prime
283            // with about 1/4 probability. So we're expecting less than
284            // 35 out of 100 false positives, seems to work.
285
286            let mr = MillerRabin::new(Odd::new(U64::from(*num)).unwrap());
287            assert_eq!(mr.test_base_two().is_probably_prime(), actual_expected_result);
288            let reported_prime = random_checks(&mut rng, &mr, 100);
289            assert!(
290                reported_prime < 35,
291                "reported as prime in {reported_prime} out of 100 tests",
292            );
293        }
294    }
