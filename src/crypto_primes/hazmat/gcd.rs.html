2use crypto_bigint::{Limb, NonZero as CTNonZero, Unsigned, Word};

4/// Calculates the greatest common divisor of `n` and `m`.
5/// By definition, `gcd(0, m) == m`.
6/// `n` must be non-zero.
7pub(crate) fn gcd_vartime<T>(n: &T, m: NonZero<Word>) -> Word
8where
9    T: Unsigned,
10{
11    let m = m.get();
12    // This we can check since it doesn't affect the return type,
13    // even though `n` will not be 0 either in the application.
14    if n.is_zero().into() {
15        return m;
16    }
17
18    // Normalize input: the resulting (a, b) are both small, a >= b, and b != 0.
19    let (a, b): (Word, Word) = if n.bits() > Word::BITS {
20        // `m` is non-zero, so we can unwrap.
21        let r = n.rem_limb(CTNonZero::new(Limb::from(m)).expect("divisor should be non-zero here"));
22        (m, r.0)
23    } else {
24        // In this branch `n` is `Word::BITS` bits or shorter,
25        // so we can safely take the first limb.
26        let n = n.as_ref()[0].0;
27        if n > m { (n, m) } else { (m, n) }
28    };
29
30    // Now do standard binary GCD on the two u64s
31    binary_gcd(a, b)
32}
33
34// Binary GCD lifted verbatim from [1], minus the base checks.
35// The identities mentioned in the comments are the following:
36// 1. `gcd(n, 0) = n`: everything divides 0 and n is the largest number that divides n.
37// 2. `gcd(2n, 2m) = 2*gcd(n, m)`: 2 is a common divisor
38// 3. `gcd(n, 2m) = gcd(n, m)` if m is odd: 2 is then not a common divisor.
39// 4. `gcd(n, m) = gcd(n, m - n)` if n, m odd and n <= m
40//
41// As GCD is commutative `gcd(n, m) = gcd(m, n)` those identities still apply if the operands are swapped.
42//
43// [1]: https://en.wikipedia.org/wiki/Binary_GCD_algorithm
44fn binary_gcd(mut n: Word, mut m: Word) -> Word {
45    // Using identities 2 and 3:
46    // gcd(2ⁱn, 2ʲm) = 2ᵏ gcd(n, m) with n, m odd and k = min(i, j)
47    // 2ᵏ is the greatest power of two that divides both 2ⁱn and 2ʲm
48    let i = n.trailing_zeros();
49    n >>= i;
50    let j = m.trailing_zeros();
51    m >>= j;
52    let k = core::cmp::min(i, j);
53
54    loop {
55        // Swap if necessary so n ≤ m
56        if n > m {
57            core::mem::swap(&mut n, &mut m);
58        }
59
60        // Identity 4: gcd(n, m) = gcd(n, m-n) as n ≤ m and n, m are both odd
61        m -= n;
62        // m is now even
63
64        if m == 0 {
65            // Identity 1: gcd(n, 0) = n
66            // The shift by k is necessary to add back the 2ᵏ factor that was removed before the loop
67            return n << k;
68        }
69
70        // Identity 3: gcd(n, 2ʲ m) = gcd(n, m) as n is odd
71        m >>= m.trailing_zeros();
72    }
73}

86    fn corner_cases() {
87        assert_eq!(gcd_vartime(&U128::from(0u64), NonZero::new(5).unwrap()), 5);
88        assert_eq!(gcd_vartime(&U128::from(1u64), NonZero::new(11 * 13 * 19).unwrap()), 1);
89        assert_eq!(gcd_vartime(&U128::from(7u64 * 11 * 13), NonZero::new(1).unwrap()), 1);
90        assert_eq!(
91            gcd_vartime(&U128::from(7u64 * 11 * 13), NonZero::new(11 * 13 * 19).unwrap()),
92            11 * 13
93        );
94    }
