5/// Estimate the number of primes smaller than x using the asymptotic expansion of `Li(x)` with 4 terms, i.e.:
6///
7///   $$\pi(x) \approx \frac{x}{\ln x} \left(1 + \frac{1!}{\ln x} + \frac{2!}{\ln^2 x} + \frac{3!}{\ln^3 x}\right)$$
8///
9/// For values of x up to ~$10^{29}$, consider using precalculated values for $\pi(x)$ from
10/// e.g. <https://sweet.ua.pt/tos/primes.html>.
11///
12/// # Error considerations
13///
14/// In addition to the usual floating point math limitations, there are two components to the error:
15///
16/// 1. The truncation error from using a limited number of terms (4) of the asymptotic expansion of Li(x): $|Li(x) -
17///    Li_{approx}(x)|$
18/// 2. The theoretical error, given by the absolute difference between $\pi(x)$ and Li(x): $|\pi(x) - Li(x)|$
19///
20/// ## Truncation Error
21///
22/// The size of the truncation error for the 4-term approximation $Li_{approx}(x)$ is roughly the size of the first
23/// omitted term, which is $\frac{24x}{\ln^5 x}$.
24///
25/// | Bits | Truncation error (abs)     | Truncation error (rel) |
26/// | :--- | :------------------------- | :----------------------|
27/// | 1024 | $\approx 2^{981}$          | $\approx 2^{-33}$      |
28/// | 2048 | $\approx 2^{2000}$         | $\approx 2^{-37}$      |
29/// | 4096 | $\approx 2^{4043}$         | $\approx 2^{-41}$      |
30///
31/// *(Relative error is calculated as the first omitted term divided by $Li_{approx}(x)$)*
32///
33/// ## Theoretical Error
34///
35/// Assuming RH, we can use Schoenfeld’s bound $|\pi(x) - \text{Li}(x)| < \frac{\sqrt{x} \ln x}{8\pi}$.
36/// This gives a very small theoretical error bound compared to the best known unconditional bounds[^Trudgian2014].
37///
38/// | Bits | Error Bound (abs) | Error Bound (rel) |
39/// | :--- | :---------------- | :-----------------|
40/// | 1024 | < $2^{517}$       | < $2^{-497}$      |
41/// | 2048 | < $2^{1030}$      | < $2^{-1007}$     |
42/// | 4096 | < $2^{2055}$      | < $2^{-2029}$     |
43///
44/// *(Relative error bound is calculated as Schoenfeld’s Bound (abs) divided by $Li_{approx}(x)$)*
45///
46/// ## Discussion
47///
48/// Assuming RH, the dominant error term in estimating $\pi(x)$ with $Li_{approx}(x)$ comes from truncating the asymptotic
49/// expansion of Li(x). While this truncation error is extremely small in relative terms, it is large in absolute terms.
50/// Improving the Li(x) approximation to be the same order of magnitude as the Schoenfeld bound would require using
51/// hundreds of terms from the asymptotic series (or employing more complex methods like continued fractions for the
52/// remainder, which is beyond the scope of this simple approximation). In relative terms the error bound is
53/// approximately $2^{-33}$ (for 1024 bits) down to $\sim 2^{-41}$ (for 4096 bits). This corresponds to an extremely small
54/// percentage error (significantly less than $10^{-8}$ %).
55///
56/// It should be noted that while Li(x) is generally smaller than $\pi(x)$ for 'small' x, it is known that the sign of
57/// $\pi(x) - Li(x)$ changes infinitely often. It has been proven that there must be a crossing below $\sim 10^{316}$
58/// ($\sim 2^{1051}$), which is well within the ranges used in this library. Thus, users should be aware that the
59/// estimate provided here can be both greater than and smaller than the actual value of $\pi(x)$.
60///
61/// [^Dusart2010]: P. Dusart, "Estimates of Some Functions Over Primes without R.H.",
62///   [arXiv:1002.0442](https://arxiv.org/pdf/1002.0442) (2010)
63///
64/// [^Dusart2018]: P. Dusart, "Explicit estimates of some functions over primes",
65///   The Ramanujan Journal 45(1) 227-251 (2018),
66///   DOI: [10.1007/s11139-016-9839-4](https://link.springer.com/article/10.1007/s11139-016-9839-4)
67///
68/// [^Wikipedia-pcf]: <https://en.wikipedia.org/wiki/Prime-counting_function>
69///
70/// [^Schoenfeld1976]: L. Schoenfeld, "Sharper Bounds for the Chebyshev Functions θ(x) and ψ(x). II",
71///   Math. Comp. 30(134) 337-360 (1976),
72///   DOI: [10.2307/2005976](https://www.jstor.org/stable/2005976)
73///
74/// [^Trudgian2014]: T. Trudgian, "Updating the error term in the prime number theorem",
75///   [arXiv:1401.2689](https://arxiv.org/abs/1401.2689) (2014)
76pub fn estimate_primecount<const LIMBS: usize, const RHS_LIMBS: usize>(x: &Uint<LIMBS>) -> Uint<LIMBS>
77where
78    Uint<LIMBS>: Concat<Output = Uint<RHS_LIMBS>>,
79    Uint<RHS_LIMBS>: Split<Output = Uint<LIMBS>>,
80{
81    // Number of bits to scale by (fractional bits during division)
82    // On 64 bit CPUs and x is a U64, we use 32 bits (1 limb * 32). Else 64.
83    // On 32 bit CPUs and x is a U64, we use 32 bits (2 limb * 16). Else 64.
84    #[cfg(target_pointer_width = "32")]
85    let scale_bits = (LIMBS as u32 * 16).min(64);
86    #[cfg(target_pointer_width = "64")]
87    let scale_bits = (LIMBS as u32 * 32).min(64);
88    let total_scale_bits = 2 * scale_bits;
89    // Scaling factor for the denominators of the expansion terms, 2^64.
90    let denom_scale_factor = (1u128 << scale_bits) as f64;
91
92    let ln_x = ln(x);
93
94    if !ln_x.is_finite() || ln_x <= 1.0 {
95        return Uint::ZERO;
96    }
97
98    // Calculate f64 denominators L, L^2, L^3, L^4
99    let ln_x_2 = ln_x * ln_x;
100    let ln_x_3 = ln_x_2 * ln_x;
101    let ln_x_4 = ln_x_3 * ln_x;
102
103    // Scale up a float by 2^64, round, cast to u128 and then to a wide `Uint`.
104    let f64_to_scaled_uint = |value: f64| -> NonZero<Uint<RHS_LIMBS>> {
105        let scaled = value * denom_scale_factor;
106        let scaled = libm::round(scaled).max(1.0) as u128;
107        let denom = Uint::<RHS_LIMBS>::from_u128(scaled);
108        NonZero::new(denom).expect("max(1.0) ensures value is at least 1")
109    };
110
111    let d1 = f64_to_scaled_uint(ln_x);
112    let d2 = f64_to_scaled_uint(ln_x_2);
113    let d3 = f64_to_scaled_uint(ln_x_3);
114    let d4 = f64_to_scaled_uint(ln_x_4);
115
116    let x_wide: Uint<RHS_LIMBS> = x.concat(&Uint::ZERO);
117    let x_wide_scaled = x_wide.wrapping_shl_vartime(total_scale_bits);
118
119    let term1_scaled = x_wide_scaled.wrapping_div(&d1);
120    let term2_scaled = x_wide_scaled.wrapping_div(&d2);
121
122    // Term 3: (2x << total_scale_bits) / d3  (Factorial 2!)
123    let term3_scaled = x_wide_scaled.wrapping_shl_vartime(1).wrapping_div(&d3);
124
125    // Term 4: (6x << total_scale_bits) / d4  (Factorial 3!)
126    let six = Uint::<LIMBS>::from(6u64);
127    let x_times_6 = x_wide_scaled.saturating_mul(&six);
128    let term4_scaled = x_times_6.wrapping_div(&d4);
129
130    let sum_scaled = term1_scaled
131        .wrapping_add(&term2_scaled)
132        .wrapping_add(&term3_scaled)
133        .wrapping_add(&term4_scaled);
134
135    // Descale by right-shifting
136    let li_x = sum_scaled >> scale_bits;
137    let (lo, hi) = li_x.split();
138    assert_eq!(hi, Uint::ZERO, "De-scaling should leave the high half zero");
139    lo
140}

209    fn assert_bit_difference<const LIMBS: usize>(candidate: Uint<LIMBS>, reference: Uint<LIMBS>, min_bit_diff: u32) {
210        let delta = if reference > candidate {
211            reference - candidate
212        } else {
213            candidate - reference
214        };
215        assert!(
216            reference.bits_vartime() - delta.bits_vartime() >= min_bit_diff,
217            "Estimate not close enough: delta has {} bits, reference has {} bits. Difference should be >= {}\nEstimate: {candidate},\nReference: {reference}",
218            delta.bits_vartime(),
219            reference.bits_vartime(),
220            min_bit_diff
221        );
222    }

278    fn uint_to_u128<const LIMBS: usize>(x: &Uint<LIMBS>) -> u128 {
279        let limbs = x.as_limbs();
280        #[cfg(target_pointer_width = "32")]
281        return (limbs[3].0 as u128) << 96
282            | (limbs[2].0 as u128) << 64
283            | (limbs[1].0 as u128) << 32
284            | limbs[0].0 as u128;
285        #[cfg(target_pointer_width = "64")]
286        return ((limbs[1].0 as u128) << 64) | limbs[0].0 as u128;
287    }
