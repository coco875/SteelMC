13/// Probabilistically checks if the given number is prime using the provided RNG
14/// according to FIPS-186.5[^FIPS] standard.
15///
16/// Performed checks:
17/// - `mr_iterations` of Miller-Rabin check with random bases;
18/// - Regular Lucas check with Selfridge base (see [`SelfridgeBase`] for details), if `add_lucas_test` is `true`.
19///
20/// See [`MillerRabin`] and [`lucas_test`] for more details about the checks;
21/// use [`minimum_mr_iterations`](`crate::hazmat::minimum_mr_iterations`)
22/// to calculate the number of required iterations.
23///
24/// [^FIPS]: FIPS-186.5 standard, <https://nvlpubs.nist.gov/nistpubs/FIPS/NIST.FIPS.186-5.pdf>
25pub fn is_prime<T>(
26    rng: &mut (impl CryptoRng + ?Sized),
27    flavor: Flavor,
28    candidate: &T,
29    mr_iterations: usize,
30    add_lucas_test: bool,
31) -> bool
32where
33    T: Unsigned + RandomMod,
34{
35    match flavor {
36        Flavor::Any => {}
37        Flavor::Safe => return is_safe_prime(rng, candidate, mr_iterations, add_lucas_test),
38    }
39
40    if equals_primitive(candidate, 1) {
41        return false;
42    }
43
44    if equals_primitive(candidate, 2) {
45        return true;
46    }
47
48    let odd_candidate: Odd<T> = match Odd::new(candidate.clone()).into() {
49        Some(x) => x,
50        None => return false,
51    };
52
53    // The random base test only makes sense when `candidate > 3`.
54    if !equals_primitive(candidate, 3) {
55        let mr = MillerRabin::new(odd_candidate.clone());
56        for _ in 0..mr_iterations {
57            if !mr.test_random_base(rng).is_probably_prime() {
58                return false;
59            }
60        }
61    }
62
63    if add_lucas_test {
64        match lucas_test(odd_candidate, SelfridgeBase, LucasCheck::Strong) {
65            Primality::Composite => return false,
66            Primality::Prime => return true,
67            _ => {}
68        }
69    }
70
71    true
72}

74/// Probabilistically checks if the given number is a safe prime using the provided RNG
75/// according to FIPS-186.5[^FIPS] standard.
76///
77/// See [`fips_is_prime`] for details about the performed checks.
78///
79/// [^FIPS]: FIPS-186.5 standard, <https://nvlpubs.nist.gov/nistpubs/FIPS/NIST.FIPS.186-5.pdf>
80fn is_safe_prime<T>(
81    rng: &mut (impl CryptoRng + ?Sized),
82    candidate: &T,
83    mr_iterations: usize,
84    add_lucas_test: bool,
85) -> bool
86where
87    T: Unsigned + RandomMod,
88{
89    // Since, by the definition of safe prime, `(candidate - 1) / 2` must also be prime,
90    // and therefore odd, `candidate` has to be equal to 3 modulo 4.
91    // 5 is the only exception, so we check for it.
92    if equals_primitive(candidate, 5) {
93        return true;
94    }
95
96    // Safe primes are always of the form 4k + 3 (i.e. n ≡ 3 mod 4)
97    // The last two digits of a binary number give you its value modulo 4.
98    // Primes p=4n+3 will always end in 11 in binary because p ≡ 3 mod 4.
99    if candidate.as_ref()[0].0 & 3 != 3 {
100        return false;
101    }
102
103    is_prime(rng, Flavor::Any, candidate, mr_iterations, add_lucas_test)
104        && is_prime(
105            rng,
106            Flavor::Any,
107            &candidate.wrapping_shr_vartime(1),
108            mr_iterations,
109            add_lucas_test,
110        )
111}
