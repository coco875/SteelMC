15/// Converts a slice of bytes to a string, including invalid characters.
16///
17/// The algorithm is as follows:
18///
19/// - If the input is valid MUTF-8, but also valid UTF-8, the function will
20///   return <code>[Cow::Borrowed]\(&[str]\)</code>.
21/// - If the input is valid MUTF-8, but invalid UTF-8, the function will return
22///   <code>[Cow::Owned]\([String]\)</code>. This case has the potential to
23///   panic.
24/// - If the input is not valid MUTF-8, the function will return
25///   <code>[Cow::Owned]\([String]\)</code>, where the best attempt at decoding
26///   the input strictly as MUTF-8 will be made, with any invalid bytes being
27///   replaced with the [U+FFFD REPLACEMENT CHARACTER] (ÔøΩ). This case has the
28///   potential to panic.
29///
30/// **NOTE:** This function is significantly slower than [`decode_lossy`], but
31/// is more "correct" in the sense that valid UTF-8 that is not valid MUTF-8
32/// will be treated as invalid input and properly replaced with the [U+FFFD
33/// REPLACEMENT CHARACTER] (ÔøΩ). If you don't need this guarantee, and can
34/// tolerate valid UTF-8 that is not valid MUTF-8, use [`decode_lossy`] instead.
35///
36/// [U+FFFD REPLACEMENT CHARACTER]: char::REPLACEMENT_CHARACTER
37///
38/// # Panics
39///
40/// This function will panic if the buffer required to decode the input exceeds
41/// [`isize::MAX`] bytes.
42///
43/// # Examples
44///
45/// If the input is valid MUTF-8, but also valid UTF-8, the function will return
46/// a borrowed string:
47///
48/// ```
49/// # extern crate alloc;
50/// use alloc::borrow::Cow;
51///
52/// use simd_cesu8::mutf8;
53///
54/// let bytes = b"Hello, world!";
55/// let decoded = mutf8::decode_lossy_strict(bytes);
56/// assert_eq!(decoded, Cow::Borrowed("Hello, world!"));
57/// ```
58///
59/// If the input is valid MUTF-8, but invalid UTF-8, the function will return an
60/// owned string:
61///
62/// ```
63/// # extern crate alloc;
64/// use alloc::borrow::Cow;
65/// use alloc::string::String;
66///
67/// use simd_cesu8::mutf8;
68///
69/// let bytes = [0xc0, 0x80];
70/// let decoded = mutf8::decode_lossy_strict(&bytes);
71/// assert_eq!(decoded, Cow::<str>::Owned(String::from("\0")));
72/// ```
73///
74/// If the input is not valid MUTF-8, the function will return an owned string:
75///
76/// ```
77/// # extern crate alloc;
78/// use alloc::borrow::Cow;
79///
80/// use simd_cesu8::mutf8;
81///
82/// // NOTE: This is the first half of MUTF-8's byte pair followed by a null
83/// // byte in MUTF-8.
84/// let bytes = [0xc0, 0xc0, 0x80];
85/// let decoded = mutf8::decode_lossy_strict(&bytes);
86/// assert_eq!(decoded, Cow::<str>::Owned(String::from("ÔøΩ\0")));
87/// ```
88///
89/// Unlike [`decode_lossy`], this function will treat valid UTF-8 that is not
90/// valid MUTF-8 as an invalid input:
91///
92/// ```
93/// # extern crate alloc;
94/// use alloc::borrow::Cow;
95/// use alloc::string::String;
96///
97/// use simd_cesu8::mutf8;
98///
99/// let bytes = [0x00, 0x00, 0x00];
100/// assert_eq!(core::str::from_utf8(&bytes), Ok("\0\0\0"));
101///
102/// let decoded = mutf8::decode_lossy_strict(&bytes);
103/// assert_eq!(decoded, Cow::<str>::Owned(String::from("ÔøΩÔøΩÔøΩ")));
104/// ```
105#[must_use]
106#[inline]
107pub fn decode_lossy_strict(bytes: &[u8]) -> Cow<'_, str> {
108    if contains_null_or_utf8_4_byte_char_header(bytes) || from_utf8(bytes).is_err() {
109        let result = internal::decode(bytes, DecodeOptions {
110            flavor: Flavor::Mutf8,
111            lossy: true,
112        });
113
114        // SAFETY: When `lossy` is `true`, the function will always return a
115        // valid string.
116        let string = unsafe { result.unwrap_unchecked() };
117
118        Cow::Owned(string)
119    } else {
120        // SAFETY: We know that `bytes` is a valid UTF-8 string.
121        Cow::Borrowed(unsafe { core::str::from_utf8_unchecked(bytes) })
122    }
123}

233/// Converts a slice of bytes to a string.
234///
235/// The algorithm is as follows:
236///
237/// - If the input is valid MUTF-8, but also valid UTF-8, the function will
238///   return <code>[Cow::Borrowed]\(&[str]\)</code>.
239/// - If the input is valid MUTF-8, but invalid UTF-8, the function will return
240///   <code>[Cow::Owned]\([String]\)</code>. This case has the potential to
241///   panic.
242///
243/// # Errors
244///
245/// If the input is not valid MUTF-8, the function will return a
246/// [`DecodingError`].
247///
248/// # Panics
249///
250/// This function will panic if the buffer required to decode the input exceeds
251/// [`isize::MAX`] bytes.
252///
253/// # Examples
254///
255/// If the input is valid MUTF-8, but also valid UTF-8, the function will return
256/// a borrowed string:
257///
258/// ```
259/// # extern crate alloc;
260/// use alloc::borrow::Cow;
261///
262/// use simd_cesu8::mutf8;
263///
264/// # fn main() -> Result<(), simd_cesu8::DecodingError> {
265/// let bytes = b"Hello, world!";
266/// let decoded = mutf8::decode_strict(bytes)?;
267/// assert_eq!(decoded, Cow::Borrowed("Hello, world!"));
268/// # Ok(())
269/// # }
270/// ```
271///
272/// If the input is valid MUTF-8, but invalid UTF-8, the function will return an
273/// owned string:
274///
275/// ```
276/// # extern crate alloc;
277/// use alloc::borrow::Cow;
278/// use alloc::string::String;
279///
280/// use simd_cesu8::mutf8;
281///
282/// # fn main() -> Result<(), simd_cesu8::DecodingError> {
283/// let bytes = [0xc0, 0x80];
284/// let decoded = mutf8::decode_strict(&bytes)?;
285/// assert_eq!(decoded, Cow::<str>::Owned(String::from("\0")));
286/// # Ok(())
287/// # }
288/// ```
289///
290/// Unlike [`decode`], this function will treat valid UTF-8 that is not valid
291/// MUTF-8 as an invalid input:
292///
293/// ```
294/// use simd_cesu8::mutf8;
295///
296/// let bytes = [0x00];
297/// assert_eq!(core::str::from_utf8(&bytes), Ok("\0"));
298///
299/// let result = mutf8::decode_strict(&bytes);
300/// assert!(result.is_err());
301/// ```
302#[inline]
303pub fn decode_strict(bytes: &[u8]) -> Result<Cow<'_, str>, DecodingError> {
304    if contains_null_or_utf8_4_byte_char_header(bytes) || from_utf8(bytes).is_err() {
305        let string = internal::decode(bytes, DecodeOptions {
306            flavor: Flavor::Mutf8,
307            lossy: false,
308        })?;
309
310        Ok(Cow::Owned(string))
311    } else {
312        // SAFETY: We know that `bytes` is a valid UTF-8 string.
313        let string = unsafe { core::str::from_utf8_unchecked(bytes) };
314        Ok(Cow::Borrowed(string))
315    }
316}

412/// Encodes a string to MUTF-8.
413///
414/// The algorithm is as follows:
415///
416/// - If the input, as UTF-8, is also valid MUTF-8, the function will return
417///   <code>[Cow::Borrowed]\([&\[u8\]][slice]\)</code>.
418/// - If the input, as UTF-8, is not valid MUTF-8, the function will return
419///   <code>[Cow::Owned]\([Vec]<[u8]>\)</code>. This case has the potential to
420///   panic.
421///
422/// # Panics
423///
424/// This function will panic if the buffer required to encode the input exceeds
425/// [`isize::MAX`] bytes.
426///
427/// # Examples
428///
429/// ```
430/// # extern crate alloc;
431/// use alloc::borrow::Cow;
432///
433/// use simd_cesu8::mutf8;
434///
435/// let null = "\0";
436/// assert_eq!(null.len(), 1);
437/// assert_eq!(null.as_bytes(), &[0x00]);
438/// assert_eq!(mutf8::encode(null), Cow::<[u8]>::Owned(vec![0xc0, 0x80]));
439///
440/// let single_byte  = "\u{0045}";
441/// assert_eq!(single_byte, "E");
442/// assert_eq!(single_byte.len(), 1);
443/// assert_eq!(single_byte.as_bytes(), &[0x45]);
444/// assert_eq!(mutf8::encode(single_byte), Cow::Borrowed(&[0x45]));
445///
446/// let two_bytes = "\u{0205}";
447/// assert_eq!(two_bytes, "»Ö");
448/// assert_eq!(two_bytes.len(), 2);
449/// assert_eq!(two_bytes.as_bytes(), &[0xc8, 0x85]);
450/// assert_eq!(mutf8::encode(two_bytes), Cow::Borrowed(&[0xc8, 0x85]));
451///
452/// let three_bytes = "\u{20ac}";
453/// assert_eq!(three_bytes, "‚Ç¨");
454/// assert_eq!(three_bytes.len(), 3);
455/// assert_eq!(three_bytes.as_bytes(), &[0xe2, 0x82, 0xac]);
456/// assert_eq!(
457///     mutf8::encode(three_bytes),
458///     Cow::Borrowed(&[0xe2, 0x82, 0xac])
459/// );
460///
461/// let four_bytes = "\u{10400}";
462/// assert_eq!(four_bytes, "êêÄ");
463/// assert_eq!(four_bytes.len(), 4);
464/// assert_eq!(four_bytes.as_bytes(), &[0xf0, 0x90, 0x90, 0x80]);
465/// assert_eq!(
466///     mutf8::encode(four_bytes),
467///     Cow::<[u8]>::Owned(vec![0xed, 0xa0, 0x81, 0xed, 0xb0, 0x80])
468/// );
469#[must_use]
470#[inline]
471pub fn encode(value: &str) -> Cow<'_, [u8]> {
472    if contains_null_or_utf8_4_byte_char_header(value.as_bytes()) {
473        Cow::Owned(internal::encode(value, Flavor::Mutf8))
474    } else {
475        Cow::Borrowed(value.as_bytes())
476    }
477}
