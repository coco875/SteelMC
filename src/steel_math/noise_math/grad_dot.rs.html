5/// Gather gradient components for 4 hashes into separate x/y/z SIMD vectors,
6/// then compute the dot product with the given position vectors.
7#[inline]
8#[must_use]
9pub fn grad_dot_4x(hashes: [usize; 4], x: f64x4, y: f64x4, z: f64x4) -> f64x4 {
10    let mut gx = [0.0f64; 4];
11    let mut gy = [0.0f64; 4];
12    let mut gz = [0.0f64; 4];
13    for i in 0..4 {
14        let g = &GRADIENT[hashes[i] & 15];
15        gx[i] = g[0];
16        gy[i] = g[1];
17        gz[i] = g[2];
18    }
19    f64x4::from_array(gx) * x + f64x4::from_array(gy) * y + f64x4::from_array(gz) * z
20}

22/// Calculate the dot product of a gradient vector and the position vector.
23#[expect(clippy::inline_always, reason = "hot-path noise primitive")]
24#[inline(always)]
25#[must_use]
26pub fn grad_dot(hash: usize, x: f64, y: f64, z: f64) -> f64 {
27    let g = &GRADIENT[hash & 15];
28    g[0] * x + g[1] * y + g[2] * z
29}
