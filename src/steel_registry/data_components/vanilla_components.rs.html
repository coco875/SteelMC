7/// Equipment slot for the equippable component.
8#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
9pub enum EquippableSlot {
10    Head,
11    Chest,
12    Legs,
13    Feet,
14    Body,
15    Mainhand,
16    Offhand,
17    Saddle,
18}
19
20impl EquippableSlot {
21    /// Parses an equipment slot from a string (as used in items.json).
22    #[must_use]
23    pub fn parse(s: &str) -> Option<Self> {
24        match s {
25            "head" => Some(Self::Head),
26            "chest" => Some(Self::Chest),
27            "legs" => Some(Self::Legs),
28            "feet" => Some(Self::Feet),
29            "body" => Some(Self::Body),
30            "mainhand" => Some(Self::Mainhand),
31            "offhand" => Some(Self::Offhand),
32            "saddle" => Some(Self::Saddle),
33            _ => None,
34        }
35    }
36
37    /// Returns the string representation of this slot.
38    #[must_use]
39    pub const fn as_str(&self) -> &'static str {
40        match self {
41            Self::Head => "head",
42            Self::Chest => "chest",
43            Self::Legs => "legs",
44            Self::Feet => "feet",
45            Self::Body => "body",
46            Self::Mainhand => "mainhand",
47            Self::Offhand => "offhand",
48            Self::Saddle => "saddle",
49        }
50    }
51
52    /// Returns true if this is a humanoid armor slot.
53    #[must_use]
54    pub const fn is_humanoid_armor(&self) -> bool {
55        matches!(self, Self::Head | Self::Chest | Self::Legs | Self::Feet)
56    }
57}

65/// A single rule within a Tool component.
66/// Rules are evaluated in order; the first matching rule determines the speed/drop behavior.
67#[derive(Debug, Clone, PartialEq)]
68pub struct ToolRule {
69    /// The blocks this rule applies to (can be a tag like "#minecraft:mineable/pickaxe",
70    /// a single block like "minecraft:cobweb", or a list of blocks).
71    pub blocks: Vec<Identifier>,
72    /// The mining speed for these blocks. If None, uses the tool's default_mining_speed.
73    pub speed: Option<f32>,
74    /// Whether the tool is "correct" for dropping items from these blocks.
75    /// If None, falls back to the block's requiresCorrectToolForDrops property.
76    pub correct_for_drops: Option<bool>,
77}
78
79impl ToolRule {
80    /// Creates a rule that sets both mining speed and marks the tool as correct for drops.
81    #[must_use]
82    pub fn mines_and_drops(blocks: Vec<Identifier>, speed: f32) -> Self {
83        Self {
84            blocks,
85            speed: Some(speed),
86            correct_for_drops: Some(true),
87        }
88    }
89
90    /// Creates a rule that explicitly denies drops (e.g., incorrect tool tier).
91    #[must_use]
92    pub fn denies_drops(blocks: Vec<Identifier>) -> Self {
93        Self {
94            blocks,
95            speed: None,
96            correct_for_drops: Some(false),
97        }
98    }
99
100    /// Creates a rule that only overrides the mining speed.
101    #[must_use]
102    pub fn override_speed(blocks: Vec<Identifier>, speed: f32) -> Self {
103        Self {
104            blocks,
105            speed: Some(speed),
106            correct_for_drops: None,
107        }
108    }
109}

111/// The tool component data - defines mining speed and drop behavior for blocks.
112#[derive(Debug, Clone, PartialEq)]
113pub struct Tool {
114    /// Rules evaluated in order to determine mining speed and drop behavior.
115    pub rules: Vec<ToolRule>,
116    /// Default mining speed when no rule matches.
117    pub default_mining_speed: f32,
118    /// Damage to apply to the item per block mined.
119    pub damage_per_block: i32,
120    /// Whether the tool can destroy blocks in creative mode.
121    pub can_destroy_blocks_in_creative: bool,
122}
123
124impl Default for Tool {
125    fn default() -> Self {
126        Self {
127            rules: Vec::new(),
128            default_mining_speed: 1.0,
129            damage_per_block: 1,
130            can_destroy_blocks_in_creative: true,
131        }
132    }
133}
134
135impl Tool {
136    /// Returns the mining speed for a block state.
137    /// Evaluates rules in order; returns the first matching rule's speed,
138    /// or `default_mining_speed` if no rule matches.
139    #[must_use]
140    pub fn get_mining_speed(&self, block_state_id: BlockStateId) -> f32 {
141        for rule in &self.rules {
142            if let Some(speed) = rule.speed
143                && rule.matches_block(block_state_id)
144            {
145                return speed;
146            }
147        }
148        self.default_mining_speed
149    }
150
151    /// Returns true if this tool is "correct" for getting drops from the block.
152    /// Evaluates rules in order; returns the first matching rule's `correct_for_drops`,
153    /// or `false` if no rule explicitly matches.
154    #[must_use]
155    pub fn is_correct_for_drops(&self, block_state_id: BlockStateId) -> bool {
156        for rule in &self.rules {
157            if let Some(correct) = rule.correct_for_drops
158                && rule.matches_block(block_state_id)
159            {
160                return correct;
161            }
162        }
163        false
164    }
165}
166
167impl ToolRule {
168    /// Checks if this rule matches a block state.
169    /// Handles both direct block identifiers and block tags (prefixed with #).
170    #[must_use]
171    pub fn matches_block(&self, block_state_id: BlockStateId) -> bool {
172        use crate::REGISTRY;
173
174        let Some(block) = REGISTRY.blocks.by_state_id(block_state_id) else {
175            return false;
176        };
177
178        for block_id in &self.blocks {
179            let id_str = format!("{}:{}", block_id.namespace, block_id.path);
180
181            // Check if it's a tag reference (starts with #)
182            if let Some(tag_str) = id_str.strip_prefix('#') {
183                if let Ok(tag_id) = Identifier::from_str(tag_str)
184                    && REGISTRY.blocks.is_in_tag(block, &tag_id)
185                {
186                    return true;
187                }
188            } else {
189                // Direct block match
190                if block.key == *block_id {
191                    return true;
192                }
193            }
194        }
195
196        false
197    }
198}
