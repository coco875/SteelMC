12pub trait BlockStateExt {
13    fn get_block(&self) -> BlockRef;
14    fn is_air(&self) -> bool;
15    fn has_block_entity(&self) -> bool;
16    fn get_value<T, P: Property<T>>(&self, property: &P) -> T;
17    fn set_value<T, P: Property<T>>(&self, property: &P, value: T) -> BlockStateId;
18    fn get_property_str(&self, name: &str) -> Option<String>;
19    fn get_collision_shape(&self) -> &'static [blocks::shapes::AABB];
20    fn get_outline_shape(&self) -> &'static [blocks::shapes::AABB];
21    /// Checks if this block face is sturdy enough to support other blocks.
22    /// Uses `SupportType::Full` by default.
23    fn is_face_sturdy(&self, direction: Direction) -> bool;
24    /// Checks if this block face is sturdy for the given support type.
25    fn is_face_sturdy_for(&self, direction: Direction, support_type: SupportType) -> bool;
26}
27
28impl BlockStateExt for BlockStateId {
29    fn get_block(&self) -> BlockRef {
30        REGISTRY
31            .blocks
32            .by_state_id(*self)
33            .expect("Expected a valid state id")
34    }
35
36    fn is_air(&self) -> bool {
37        self.get_block().config.is_air
38    }
39
40    fn has_block_entity(&self) -> bool {
41        // TODO: Implement when block entities are added
42        false
43    }
44
45    fn get_value<T, P: Property<T>>(&self, property: &P) -> T {
46        REGISTRY.blocks.get_property(*self, property)
47    }
48
49    fn set_value<T, P: Property<T>>(&self, property: &P, value: T) -> BlockStateId {
50        REGISTRY.blocks.set_property(*self, property, value)
51    }
52
53    fn get_property_str(&self, name: &str) -> Option<String> {
54        REGISTRY
55            .blocks
56            .get_properties(*self)
57            .into_iter()
58            .find(|(n, _)| *n == name)
59            .map(|(_, v)| v.to_string())
60    }
61
62    fn get_collision_shape(&self) -> &'static [blocks::shapes::AABB] {
63        REGISTRY.blocks.get_collision_shape(*self)
64    }
65
66    fn get_outline_shape(&self) -> &'static [blocks::shapes::AABB] {
67        REGISTRY.blocks.get_outline_shape(*self)
68    }
69
70    fn is_face_sturdy(&self, direction: Direction) -> bool {
71        self.is_face_sturdy_for(direction, SupportType::Full)
72    }
73
74    fn is_face_sturdy_for(&self, direction: Direction, support_type: SupportType) -> bool {
75        let shape = self.get_collision_shape();
76        blocks::shapes::is_face_sturdy(shape, direction, support_type)
77    }
78}
