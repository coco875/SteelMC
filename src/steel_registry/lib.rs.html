361    pub fn init(&self, value: Registry) -> Result<(), Registry> {

367        self.0.get_or_init(f)
368    }

381/// Trait for types stored in a registry, allowing self-lookup of their numeric ID.
382pub trait RegistryEntry: 'static {
383    fn key(&self) -> &Identifier;
384    fn try_id(&self) -> Option<usize>;
385
386    /// # Panics
387    /// Panics if the entry is not registered.
388    fn id(&self) -> usize {
389        self.try_id().expect("entry not found in registry")
390    }
391}
392
393/// Generic trait for registries with a typed entry.
394///
395/// `Entry` is the concrete type (e.g. `Block`); all lookups return `&'static Entry`
396/// to enforce cheap pointer copies and prevent expensive clones.
397pub trait RegistryExt {
398    type Entry: RegistryEntry;
399
400    fn freeze(&mut self);
401    fn by_id(&self, id: usize) -> Option<&'static Self::Entry>;
402    fn by_key(&self, key: &Identifier) -> Option<&'static Self::Entry>;
403    fn id_from_key(&self, key: &Identifier) -> Option<usize>;
404    fn len(&self) -> usize;
405    fn is_empty(&self) -> bool;
406}
407
408/// Trait for registries that support tagging entries.
409pub trait TaggedRegistryExt: RegistryExt {
410    fn register_tag(&mut self, tag: Identifier, keys: &[&'static str]);
411    fn modify_tag(&mut self, tag: &Identifier, f: impl FnOnce(Vec<Identifier>) -> Vec<Identifier>);
412    fn is_in_tag(&self, entry: &'static Self::Entry, tag: &Identifier) -> bool;
413    fn get_tag(&self, tag: &Identifier) -> Option<Vec<&'static Self::Entry>>;
414    fn iter_tag(&self, tag: &Identifier) -> impl Iterator<Item = &'static Self::Entry> + '_;
415    fn tag_keys(&self) -> impl Iterator<Item = &Identifier> + '_;
416}
417
418/// Implements `RegistryExt` for a registry type.
419///
420/// Expects `$id_field` to be `Vec<&'static $Entry>`.
421#[macro_export]
422macro_rules! impl_registry_ext {
423    ($Registry:ty, $Entry:ty, $id_field:ident, $key_field:ident) => {
424        impl $crate::RegistryExt for $Registry {
425            type Entry = $Entry;
426
427            fn freeze(&mut self) {
428                self.allows_registering = false;
429            }
430
431            fn by_id(&self, id: usize) -> Option<&'static $Entry> {
432                self.$id_field.get(id).copied()
433            }
434
435            fn by_key(&self, key: &steel_utils::Identifier) -> Option<&'static $Entry> {
436                self.$key_field
437                    .get(key)
438                    .and_then(|&id| self.$id_field.get(id).copied())
439            }
440
441            fn id_from_key(&self, key: &steel_utils::Identifier) -> Option<usize> {
442                self.$key_field.get(key).copied()
443            }
444
445            fn len(&self) -> usize {
446                self.$id_field.len()
447            }
448
449            fn is_empty(&self) -> bool {
450                self.$id_field.is_empty()
451            }
452        }
453    };
454}

456/// Implements `RegistryEntry` for an entry type via hash map lookup.
457#[macro_export]
458macro_rules! impl_registry_entry {
459    ($Entry:ty, $global_field:ident) => {
460        impl $crate::RegistryEntry for $Entry {
461            fn key(&self) -> &steel_utils::Identifier {
462                &self.key
463            }
464
465            fn try_id(&self) -> Option<usize> {
466                use $crate::RegistryExt;
467                $crate::REGISTRY.$global_field.id_from_key(&self.key)
468            }
469        }
470    };
471}
472
473/// Implements both `RegistryExt` and `RegistryEntry` for a standard registry.
474#[macro_export]
475macro_rules! impl_registry {
476    ($Registry:ty, $Entry:ty, $id_field:ident, $key_field:ident, $global_field:ident) => {
477        $crate::impl_registry_ext!($Registry, $Entry, $id_field, $key_field);
478        $crate::impl_registry_entry!($Entry, $global_field);
479    };
480}

482/// Implements `TaggedRegistryExt` for a registry with tag support.
483#[macro_export]
484macro_rules! impl_tagged_registry {
485    ($Registry:ty, $key_field:ident, $entity_name:literal) => {
486        impl $crate::TaggedRegistryExt for $Registry {
487            fn register_tag(&mut self, tag: steel_utils::Identifier, keys: &[&'static str]) {
488                assert!(
489                    self.allows_registering,
490                    "Cannot register tags after registry has been frozen"
491                );
492
493                let identifiers: Vec<steel_utils::Identifier> = keys
494                    .iter()
495                    .filter_map(|key| {
496                        let ident = steel_utils::registry::registry_vanilla_or_custom_tag(key);
497                        if self.$key_field.contains_key(&ident) {
498                            Some(ident)
499                        } else {
500                            None
501                        }
502                    })
503                    .collect();
504
505                self.tags.insert(tag, identifiers);
506            }
507
508            fn modify_tag(
509                &mut self,
510                tag: &steel_utils::Identifier,
511                f: impl FnOnce(Vec<steel_utils::Identifier>) -> Vec<steel_utils::Identifier>,
512            ) {
513                let existing = self.tags.remove(tag).unwrap_or_default();
514                let entries = f(existing)
515                    .into_iter()
516                    .filter(|key| {
517                        let exists = self.$key_field.contains_key(key);
518                        if !exists {
519                            tracing::error!(
520                                "{} {} not found in registry, skipping from tag {}",
521                                $entity_name,
522                                key,
523                                tag,
524                            );
525                        }
526                        exists
527                    })
528                    .collect();
529                self.tags.insert(tag.clone(), entries);
530            }
531
532            fn is_in_tag(
533                &self,
534                entry: &'static Self::Entry,
535                tag: &steel_utils::Identifier,
536            ) -> bool {
537                self.tags
538                    .get(tag)
539                    .is_some_and(|entries| entries.contains(&entry.key))
540            }
541
542            fn get_tag(&self, tag: &steel_utils::Identifier) -> Option<Vec<&'static Self::Entry>> {
543                use $crate::RegistryExt;
544                self.tags.get(tag).map(|idents| {
545                    idents
546                        .iter()
547                        .filter_map(|ident| self.by_key(ident))
548                        .collect()
549                })
550            }
551
552            fn iter_tag(
553                &self,
554                tag: &steel_utils::Identifier,
555            ) -> impl Iterator<Item = &'static Self::Entry> + '_ {
556                use $crate::RegistryExt;
557                self.tags
558                    .get(tag)
559                    .into_iter()
560                    .flat_map(|v| v.iter().filter_map(|ident| self.by_key(ident)))
561            }
562
563            fn tag_keys(&self) -> impl Iterator<Item = &steel_utils::Identifier> + '_ {
564                self.tags.keys()
565            }
566        }
567    };
568}
