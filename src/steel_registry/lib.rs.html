227pub struct RegistryLock(OnceLock<Registry>);
228
229impl RegistryLock {
230    #[allow(clippy::result_large_err)]
231    pub fn init(&self, value: Registry) -> Result<(), Registry> {
232        self.0.set(value)
233    }
234
235    #[cfg(test)]
236    pub(crate) fn get_or_init(&self, f: impl FnOnce() -> Registry) -> &Registry {
237        self.0.get_or_init(f)
238    }
239}
240
241impl Deref for RegistryLock {
242    type Target = Registry;
243
244    fn deref(&self) -> &Self::Target {
245        self.0.get().expect("Registry not init")
246    }
247}

314impl Debug for Registry {
315    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
316        f.write_str("Registry {")
317            .and_then(|_| f.write_fmt(format_args!("Blocks Loaded: {}", self.blocks.len())))
318            .and_then(|_| f.write_str("}"))
319    }
