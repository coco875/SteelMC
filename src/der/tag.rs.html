13/// Indicator bit for constructed form encoding (i.e. vs primitive form)
14const CONSTRUCTED_FLAG: u8 = 0b100000;

16/// Types which have a constant ASN.1 [`Tag`].
17pub trait FixedTag {
18    /// ASN.1 tag
19    const TAG: Tag;
20}
21
22/// Types which have an ASN.1 [`Tag`].
23#[diagnostic::on_unimplemented(note = "Consider adding impl of `FixedTag` to `{Self}`")]
24pub trait Tagged {
25    /// Get the ASN.1 tag that this type is encoded with.
26    fn tag(&self) -> Tag;
27}
28
29/// Types which are [`FixedTag`] always have a known [`Tag`] type.
30impl<T: FixedTag + ?Sized> Tagged for T {
31    fn tag(&self) -> Tag {
32        T::TAG
33    }
34}

36/// Types which have a constant ASN.1 constructed bit.
37///
38/// Auto-implemented on all types that implement [`FixedTag`].
39#[diagnostic::on_unimplemented(note = "Consider adding impl of `FixedTag` to `{Self}`")]
40pub trait IsConstructed {
41    /// ASN.1 constructed bit
42    const CONSTRUCTED: bool;
43}

50/// ASN.1 tags.
51///
52/// Tags are the leading identifier octet of the Tag-Length-Value encoding
53/// used by ASN.1 DER and identify the type of the subsequent value.
54///
55/// They are described in X.690 Section 8.1.2: Identifier octets, and
56/// structured as follows:
57///
58/// ```text
59/// | Class | P/C | Tag Number |
60/// ```
61///
62/// - Bits 8/7: [`Class`]
63/// - Bit 6: primitive (0) or constructed (1)
64/// - Bits 5-1: tag number
65#[cfg_attr(feature = "arbitrary", derive(arbitrary::Arbitrary))]
66#[derive(Copy, Clone, Eq, Hash, PartialEq, PartialOrd, Ord)]
67#[non_exhaustive]
68pub enum Tag {
69    /// `BOOLEAN` tag: `1`.
70    Boolean,
71
72    /// `INTEGER` tag: `2`.
73    Integer,
74
75    /// `BIT STRING` tag: `3`.
76    BitString,
77
78    /// `OCTET STRING` tag: `4`.
79    OctetString,
80
81    /// `NULL` tag: `5`.
82    Null,
83
84    /// `OBJECT IDENTIFIER` tag: `6`.
85    ObjectIdentifier,
86
87    /// `REAL` tag: `9`.
88    Real,
89
90    /// `ENUMERATED` tag: `10`.
91    Enumerated,
92
93    /// `UTF8String` tag: `12`.
94    Utf8String,
95
96    /// `RELATIVE OID` tag: `13`.
97    RelativeOid,
98
99    /// `SEQUENCE` tag: `16`.
100    Sequence,
101
102    /// `SET` and `SET OF` tag: `17`.
103    Set,
104
105    /// `NumericString` tag: `18`.
106    NumericString,
107
108    /// `PrintableString` tag: `19`.
109    PrintableString,
110
111    /// `TeletexString` tag: `20`.
112    TeletexString,
113
114    /// `VideotexString` tag: `21`.
115    VideotexString,
116
117    /// `IA5String` tag: `22`.
118    Ia5String,
119
120    /// `UTCTime` tag: `23`.
121    UtcTime,
122
123    /// `GeneralizedTime` tag: `24`.
124    GeneralizedTime,
125
126    /// `VisibleString` tag: `26`.
127    VisibleString,

129    /// `GeneralString` tag: `27`.
130    GeneralString,

133    BmpString,

168    /// Decode a [`Tag`] in addition to returning the value of the constructed bit.
169    pub(crate) fn decode_with_constructed_bit<'a>(
170        reader: &mut impl Reader<'a>,
171    ) -> Result<(Self, bool)> {
172        let first_byte = reader.read_byte()?;
173        let is_constructed = first_byte & CONSTRUCTED_FLAG != 0;
174
175        let tag = match first_byte {
176            0x01 => Tag::Boolean,
177            0x02 => Tag::Integer,
178            0x03 => Tag::BitString,
179            0x04 => Tag::OctetString,
180            0x05 => Tag::Null,
181            0x06 => Tag::ObjectIdentifier,
182            0x09 => Tag::Real,
183            0x0A => Tag::Enumerated,
184            0x0C => Tag::Utf8String,
185            0x0D => Tag::RelativeOid,
186            0x12 => Tag::NumericString,
187            0x13 => Tag::PrintableString,
188            0x14 => Tag::TeletexString,
189            0x15 => Tag::VideotexString,
190            0x16 => Tag::Ia5String,
191            0x17 => Tag::UtcTime,
192            0x18 => Tag::GeneralizedTime,
193            0x1A => Tag::VisibleString,
194            0x1B => Tag::GeneralString,
195            0x1E => Tag::BmpString,
196            #[cfg(feature = "ber")]
197            0x24 if reader.encoding_rules().is_ber() => Tag::OctetString,
198            0x30 => Tag::Sequence, // constructed
199            0x31 => Tag::Set,      // constructed
200            0x40..=0x7F => {
201                let (constructed, number) = parse_parts(first_byte, reader)?;
202
203                Tag::Application {
204                    constructed,
205                    number,
206                }
207            }
208            0x80..=0xBF => {
209                let (constructed, number) = parse_parts(first_byte, reader)?;
210
211                Tag::ContextSpecific {
212                    constructed,
213                    number,
214                }
215            }
216            0xC0..=0xFF => {
217                let (constructed, number) = parse_parts(first_byte, reader)?;
218
219                Tag::Private {
220                    constructed,
221                    number,
222                }
223            }
224            // universal tag in long form
225            0x1F => return Err(reader.error(ErrorKind::TagNumberInvalid)),
226            byte => return Err(reader.error(ErrorKind::TagUnknown { byte })),
227        };
228
229        Ok((tag, is_constructed))
230    }

232    /// Peek at the next byte in the reader and attempt to decode it as a [`Tag`] value.
233    ///
234    /// Does not modify the reader's state.
235    pub fn peek<'a>(reader: &impl Reader<'a>) -> Result<Self> {
236        Self::decode(&mut reader.clone())
237    }

239    /// Returns true if given context-specific (or any given class) tag number matches the peeked tag.
240    pub(crate) fn peek_matches<'a, R: Reader<'a>>(
241        reader: &mut R,
242        expected_class: Class,
243        expected_tag_number: TagNumber,
244    ) -> Result<bool> {
245        if reader.is_finished() {
246            return Ok(false);
247        }
248
249        let tag = Self::peek(reader)?;
250        Ok(tag.class() == expected_class && tag.number() == expected_tag_number)
251    }

258            Ok(self)

261        }

371fn parse_parts<'a, R: Reader<'a>>(first_byte: u8, reader: &mut R) -> Result<(bool, TagNumber)> {
372    let constructed = first_byte & CONSTRUCTED_FLAG != 0;
373    let first_number_part = first_byte & TagNumber::MASK;
374
375    if first_number_part != TagNumber::MASK {
376        return Ok((constructed, TagNumber(first_number_part.into())));
377    }
378
379    let mut multi_byte_tag_number = 0;
380
381    for i in 0..Tag::MAX_SIZE - 1 {
382        let byte = reader.read_byte()?;
383        multi_byte_tag_number |= u32::from(byte & 0x7F);
384
385        if byte & 0x80 == 0 {
386            if multi_byte_tag_number < u32::from(TagNumber::MASK) {
387                return Err(reader.error(ErrorKind::TagNumberInvalid));
388            }
389
390            return Ok((constructed, TagNumber(multi_byte_tag_number)));
391        } else if i == 0 && multi_byte_tag_number == 0 {
392            // 8.1.2.4.2c says "bits 7 to 1 of the first subsequent octet shall not all be zero"
393            return Err(reader.error(ErrorKind::TagNumberInvalid));
394        }
395
396        if multi_byte_tag_number.leading_zeros() < 7 {
397            return Err(reader.error(ErrorKind::TagNumberInvalid));
398        }
399
400        multi_byte_tag_number <<= 7;
401    }
402
403    // missing terminator byte
404    Err(reader.error(ErrorKind::TagNumberInvalid))
405}

407impl Encode for Tag {
408    #[allow(clippy::cast_possible_truncation)]
409    fn encoded_len(&self) -> Result<Length> {
410        let number = self.number().value();
411
412        let length = if number <= 30 {
413            Length::ONE
414        } else {
415            Length::new(number.ilog2() / 7 + 2)
416        };
417
418        Ok(length)
419    }
420
421    fn encode(&self, writer: &mut impl Writer) -> Result<()> {
422        let mut first_byte = (self.class() as u8) | (u8::from(self.is_constructed()) << 5);
423
424        let number = self.number().value();
425
426        if number < u32::from(TagNumber::MASK) {
427            first_byte |= (number & 0x1F) as u8;
428            writer.write_byte(first_byte)?;
429        } else {
430            first_byte |= TagNumber::MASK;
431            writer.write_byte(first_byte)?;
432
433            let extra_bytes = number.ilog2() / 7 + 1;
434
435            for shift in (0..extra_bytes).rev() {
436                let mut byte = ((number >> (shift * 7)) & 0x7f) as u8;
437
438                if shift != 0 {
439                    byte |= 0x80;
440                }
441
442                writer.write_byte(byte)?;
443            }
444        }
445
446        Ok(())
447    }
448}

647        // T-REC-X.680-202102
648        // 8.6 The canonical order for tags is based on the outermost tag of each type and is defined as follows:
649        // a) those elements or alternatives with universal class tags shall appear first, followed by those with
650        // application class tags, followed by those with context-specific tags, followed by those with private class
651        // tags;
652        // b) within each class of tags, the elements or alternatives shall appear in ascending order of their tag
653        // numbers.
654        assert_eq!(Tag::Boolean.der_cmp(&Tag::Integer), Ok(Ordering::Less));
655        assert_eq!(Tag::Integer.der_cmp(&Tag::Null), Ok(Ordering::Less));

657        assert_eq!(Tag::Sequence.der_cmp(&Tag::Ia5String), Ok(Ordering::Less));
658        assert_eq!(Tag::Ia5String.der_cmp(&Tag::BmpString), Ok(Ordering::Less));
