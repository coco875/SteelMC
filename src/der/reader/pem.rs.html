1//! Streaming PEM reader.
2
3use super::{Reader, position::Position};
4use crate::{EncodingRules, Error, ErrorKind, Length, Result};
5use pem_rfc7468::Decoder;
6
7/// `Reader` type which decodes PEM on-the-fly.
8#[cfg(feature = "pem")]
9#[derive(Clone)]
10pub struct PemReader<'i> {
11    /// Inner PEM decoder.
12    decoder: Decoder<'i>,
13
14    /// Encoding rules to apply when decoding the input.
15    encoding_rules: EncodingRules,
16
17    /// Position tracker.
18    position: Position,
19}
20
21#[cfg(feature = "pem")]
22impl<'i> PemReader<'i> {
23    /// Create a new PEM reader which decodes data on-the-fly.
24    ///
25    /// Uses the default 64-character line wrapping.
26    pub fn new(pem: &'i [u8]) -> Result<Self> {
27        let decoder = Decoder::new(pem)?;
28        let input_len = Length::try_from(decoder.remaining_len())?;
29
30        Ok(Self {
31            decoder,
32            encoding_rules: EncodingRules::default(),
33            position: Position::new(input_len),
34        })
35    }
36
37    /// Get the PEM label which will be used in the encapsulation boundaries
38    /// for this document.
39    pub fn type_label(&self) -> &'i str {
40        self.decoder.type_label()
41    }
42}
43
44#[cfg(feature = "pem")]
45impl<'i> Reader<'i> for PemReader<'i> {
46    fn encoding_rules(&self) -> EncodingRules {
47        self.encoding_rules
48    }
49
50    fn input_len(&self) -> Length {
51        self.position.input_len()
52    }
53
54    fn position(&self) -> Length {
55        self.position.current()
56    }
57
58    fn read_nested<T, F, E>(&mut self, len: Length, f: F) -> core::result::Result<T, E>
59    where
60        F: FnOnce(&mut Self) -> core::result::Result<T, E>,
61        E: From<Error>,
62    {
63        let resumption = self.position.split_nested(len)?;
64        let ret = f(self);
65        self.position.resume_nested(resumption);
66        ret
67    }
68
69    fn read_slice(&mut self, _len: Length) -> Result<&'i [u8]> {
70        // Can't borrow from PEM because it requires decoding
71        Err(self.error(ErrorKind::Reader))
72    }
73
74    fn read_into<'o>(&mut self, buf: &'o mut [u8]) -> Result<&'o [u8]> {
75        self.position.advance(Length::try_from(buf.len())?)?;
76        self.decoder.decode(buf)?;
77        Ok(buf)
78    }
79}
