26/// Builder for creating a persistent chunk with its own palettes.
27struct ChunkBuilder<'a> {
28    block_states: Vec<PersistentBlockState>,
29    biomes: Vec<Identifier>,
30    registry: &'a Registry,
31}

33impl<'a> ChunkBuilder<'a> {
34    const fn new(registry: &'a Registry) -> Self {
35        Self {
36            block_states: Vec::new(),
37            biomes: Vec::new(),
38            registry,
39        }
40    }

42    /// Ensures a block state exists in the chunk's palette, returning its index.
43    fn ensure_block_state(&mut self, block_id: BlockStateId) -> u16 {
44        // Get block and properties from registry
45        let block = self
46            .registry
47            .blocks
48            .by_state_id(block_id)
49            .expect("Invalid block state ID");
50        let properties = self.registry.blocks.get_properties(block_id);
51
52        let persistent = PersistentBlockState {
53            name: block.key.clone(),
54            properties: properties
55                .into_iter()
56                .map(|(k, v)| (k.to_string(), v.to_string()))
57                .collect(),
58        };
59
60        // Check if already exists
61        if let Some(idx) = self.block_states.iter().position(|s| s == &persistent) {
62            return idx as u16;
63        }
64
65        // Add new entry
66        let idx = self.block_states.len();
67        self.block_states.push(persistent);
68        idx as u16
69    }
70
71    /// Ensures a biome exists in the chunk's palette, returning its index.
72    fn ensure_biome(&mut self, biome_id: u8) -> u16 {
73        // Get biome identifier from registry
74        let biome = self
75            .registry
76            .biomes
77            .by_id(biome_id as usize)
78            .expect("Invalid biome ID");
79        let identifier = biome.key.clone();
80
81        if let Some(idx) = self.biomes.iter().position(|b| b == &identifier) {
82            return idx as u16;
83        }
84
85        let idx = self.biomes.len();
86        self.biomes.push(identifier);
87        idx as u16
88    }
89}

91/// Chunk storage backend.
92///
93/// This enum provides persistence for chunks, either to disk (region files)
94/// or in-memory (for testing/minigames).

104    /// Loads a chunk from storage.
105    ///
106    /// Returns `Ok(None)` if the chunk doesn't exist in storage.
107    /// For `RamOnly` with `create_empty_on_miss=true`, this always
108    /// returns an empty chunk (never `None`).

179    /// Saves a chunk to the appropriate region.
180    ///
181    /// The chunk is serialized, compressed, and written to disk immediately.
182    /// If the region was already open (has loaded chunks), the header update is
183    /// deferred. If this call opened the region, it will be closed after saving.
184    ///
185    /// If the chunk is not dirty, this is a no-op and returns `Ok(false)`.
186    /// Returns `Ok(true)` if the chunk was saved.
187    /// Prepares chunk data for saving. Call this while holding the chunk lock,
188    /// then pass the result to `save_chunk_data` after releasing the lock.
189    #[must_use]
190    pub fn prepare_chunk_save(chunk: &ChunkAccess) -> Option<PreparedChunkSave> {
191        if !chunk.is_dirty() {
192            return None;
193        }
194
195        let pos = chunk.pos();
196
197        // Get block entities if this is a full chunk
198        let block_entities: Vec<SharedBlockEntity> = chunk
199            .as_full()
200            .map(LevelChunk::get_block_entities)
201            .unwrap_or_default();
202
203        // Get saveable entities if this is a full chunk
204        let entities: Vec<SharedEntity> = chunk
205            .as_full()
206            .map(|c| c.entities.get_saveable_entities())
207            .unwrap_or_default();
208
209        let persistent = Self::to_persistent(chunk.sections(), &block_entities, &entities, pos);
210
211        Some(PreparedChunkSave { pos, persistent })
212    }

214    /// Converts chunk data to persistent format.
215    fn to_persistent(
216        sections: &Sections,
217        block_entities: &[SharedBlockEntity],
218        entities: &[SharedEntity],
219        chunk_pos: ChunkPos,
220    ) -> PersistentChunk {
221        let mut builder = ChunkBuilder::new(&REGISTRY);
222
223        let persistent_sections = sections
224            .sections
225            .iter()
226            .map(|section| Self::section_to_persistent(section, &mut builder))
227            .collect();
228
229        // Serialize block entities
230        let persistent_block_entities: Vec<PersistentBlockEntity> = block_entities
231            .iter()
232            .map(|entity| {
233                let guard = entity.lock();
234                let pos = guard.get_block_pos();
235
236                // Serialize NBT data
237                let mut nbt = NbtCompound::new();
238                guard.save_additional(&mut nbt);
239                let mut nbt_bytes = Vec::new();
240                nbt.write(&mut nbt_bytes);
241
242                PersistentBlockEntity {
243                    x: (pos.0.x - chunk_pos.0.x * 16) as u8,
244                    y: pos.0.y as i16,
245                    z: (pos.0.z - chunk_pos.0.y * 16) as u8,
246                    entity_type: guard.get_type().key.clone(),
247                    nbt_data: nbt_bytes,
248                }
249            })
250            .collect();
251
252        // Serialize entities
253        let persistent_entities: Vec<PersistentEntity> = entities
254            .iter()
255            .filter_map(|entity| {
256                let pos = entity.position();
257                let vel = entity.velocity();
258                let (yaw, pitch) = entity.rotation();
259
260                // Validate position is finite (discard corrupted entities)
261                if !pos.x.is_finite() || !pos.y.is_finite() || !pos.z.is_finite() {
262                    tracing::warn!(
263                        uuid = ?entity.uuid(),
264                        "Entity has non-finite position {:?}, skipping save",
265                        pos
266                    );
267                    return None;
268                }
269
270                // Serialize type-specific NBT data
271                let mut nbt = NbtCompound::new();
272                entity.save_additional(&mut nbt);
273                let mut nbt_bytes = Vec::new();
274                nbt.write(&mut nbt_bytes);
275
276                Some(PersistentEntity {
277                    entity_type: entity.entity_type().key.clone(),
278                    uuid: *entity.uuid().as_bytes(),
279                    pos: [pos.x, pos.y, pos.z],
280                    motion: [vel.x, vel.y, vel.z],
281                    rotation: [yaw, pitch],
282                    on_ground: entity.on_ground(),
283                    nbt_data: nbt_bytes,
284                })
285            })
286            .collect();
287
288        PersistentChunk {
289            last_modified: SystemTime::now()
290                .duration_since(UNIX_EPOCH)
291                .map_or(0, |d| d.as_secs() as u32),
292            block_states: builder.block_states,
293            biomes: builder.biomes,
294            sections: persistent_sections,
295            block_entities: persistent_block_entities,
296            entities: persistent_entities,
297        }
298    }

300    /// Converts a runtime section to persistent format.
301    fn section_to_persistent(
302        section: &SectionHolder,
303        builder: &mut ChunkBuilder,
304    ) -> PersistentSection {
305        let section = section.read();
306        let biomes = Self::biomes_to_persistent(&section.biomes, builder);
307
308        match &section.states {
309            PalettedContainer::Homogeneous(block_id) => {
310                let block_idx = builder.ensure_block_state(*block_id);
311                PersistentSection::Homogeneous {
312                    block_state: block_idx,
313                    biomes,
314                }
315            }
316            PalettedContainer::Heterogeneous(data) => {
317                // Build section-local palette (indices into chunk's block_states)
318                let palette: Vec<u16> = data
319                    .palette
320                    .iter()
321                    .map(|(block_id, _)| builder.ensure_block_state(*block_id))
322                    .collect();
323
324                // Pack block indices (indices into section-local palette)
325                let bits = bits_for_palette_len(palette.len())
326                    .expect("Heterogeneous section should have palette length >= 2");
327                let indices: Vec<u32> = data
328                    .cube
329                    .iter()
330                    .flatten()
331                    .flatten()
332                    .map(|block_id| {
333                        data.palette
334                            .iter()
335                            .position(|(v, _)| v == block_id)
336                            .unwrap_or(0) as u32
337                    })
338                    .collect();
339
340                let block_data = pack_indices(&indices, bits);
341
342                PersistentSection::Heterogeneous {
343                    palette,
344                    bits_per_entry: bits,
345                    block_data,
346                    biomes,
347                }
348            }
349        }
350    }

396    /// Converts a persistent chunk to runtime format.
397    /// The returned chunk is not dirty (freshly loaded from disk).
398    ///
399    /// # Arguments
400    /// * `persistent` - The persistent chunk data
401    /// * `pos` - The chunk position
402    /// * `status` - The chunk status
403    /// * `min_y` - The minimum Y coordinate of the world
404    /// * `height` - The total height of the world
405    /// * `level` - Weak reference to the world for `LevelChunk`
406    pub(crate) fn persistent_to_chunk(
407        persistent: &PersistentChunk,
408        pos: ChunkPos,
409        status: ChunkStatus,
410        min_y: i32,
411        height: i32,
412        level: Weak<World>,
413    ) -> ChunkAccess {
414        let sections: Vec<ChunkSection> = persistent
415            .sections
416            .iter()
417            .map(|section| Self::persistent_to_section(section, persistent))
418            .collect();
419
420        match status {
421            ChunkStatus::Full => {
422                let chunk = LevelChunk::from_disk(
423                    Sections::from_owned(sections.into_boxed_slice()),
424                    pos,
425                    min_y,
426                    height,
427                    level.clone(),
428                );
429
430                // Load block entities
431                for persistent_be in &persistent.block_entities {
432                    if let Some(block_entity) =
433                        Self::persistent_to_block_entity(persistent_be, pos, &chunk)
434                    {
435                        chunk.add_and_register_block_entity(block_entity);
436                    }
437                }
438
439                // Load entities
440                for persistent_entity in &persistent.entities {
441                    if let Some(entity) = Self::persistent_to_entity(persistent_entity, pos, &chunk)
442                    {
443                        chunk.add_and_register_entity(entity);
444                    }
445                }
446
447                // Clear dirty flag since we just loaded (add_and_register marks dirty)
448                chunk.dirty.store(false, Ordering::Release);
449
450                ChunkAccess::Full(chunk)
451            }
452            _ => ChunkAccess::Proto(ProtoChunk::from_disk(
453                Sections::from_owned(sections.into_boxed_slice()),
454                pos,
455                status,
456                min_y,
457                height,
458            )),
459        }
460    }

462    /// Converts a persistent block entity to runtime format.
463    fn persistent_to_block_entity(
464        persistent: &PersistentBlockEntity,
465        chunk_pos: ChunkPos,
466        chunk: &LevelChunk,
467    ) -> Option<SharedBlockEntity> {
468        // Calculate absolute position
469        let abs_x = chunk_pos.0.x * 16 + i32::from(persistent.x);
470        let abs_z = chunk_pos.0.y * 16 + i32::from(persistent.z);
471        let pos = BlockPos::new(abs_x, i32::from(persistent.y), abs_z);
472
473        // Get the block state at this position
474        let state = chunk.get_block_state(pos);
475
476        // Look up the block entity type
477        let block_entity_type = REGISTRY
478            .block_entity_types
479            .by_key(&persistent.entity_type)?;
480
481        // Get the world reference from the chunk
482        let level = chunk.level_weak();
483
484        // Parse and load NBT data
485        if persistent.nbt_data.is_empty() {
486            // No NBT data, just create the entity without loading
487            BLOCK_ENTITIES.create(block_entity_type, level, pos, state)
488        } else {
489            // Parse NBT from bytes as borrowed
490            let Ok(nbt) = read_borrowed_compound(&mut Cursor::new(&persistent.nbt_data)) else {
491                return BLOCK_ENTITIES.create(block_entity_type, level, pos, state);
492            };
493
494            // Create the block entity and load NBT
495            BLOCK_ENTITIES.create_and_load(block_entity_type, level, pos, state, &nbt)
496        }
497    }

499    /// Converts a persistent entity to runtime format.
500    fn persistent_to_entity(
501        persistent: &PersistentEntity,
502        chunk_pos: ChunkPos,
503        chunk: &LevelChunk,
504    ) -> Option<SharedEntity> {
505        use steel_utils::math::Vector3;
506        use uuid::Uuid;
507
508        // Reconstruct base fields
509        let pos = Vector3::new(persistent.pos[0], persistent.pos[1], persistent.pos[2]);
510        let mut velocity = Vector3::new(
511            persistent.motion[0],
512            persistent.motion[1],
513            persistent.motion[2],
514        );
515        let rotation = (persistent.rotation[0], persistent.rotation[1]);
516        let uuid = Uuid::from_bytes(persistent.uuid);
517
518        // Validate position is finite
519        if !pos.x.is_finite() || !pos.y.is_finite() || !pos.z.is_finite() {
520            tracing::warn!(
521                ?uuid,
522                "Entity has non-finite position {:?}, skipping load",
523                pos
524            );
525            return None;
526        }
527
528        // Validate position is within expected chunk (sanity check)
529        let expected_chunk_x = (pos.x as i32) >> 4;
530        let expected_chunk_z = (pos.z as i32) >> 4;
531        if chunk_pos.0.x != expected_chunk_x || chunk_pos.0.y != expected_chunk_z {
532            tracing::warn!(
533                ?uuid,
534                "Entity position {:?} doesn't match chunk {:?}, loading anyway",
535                pos,
536                chunk_pos
537            );
538        }
539
540        // Clamp motion values > 10.0 to 0 (vanilla behavior to prevent corruption)
541        if velocity.x.abs() > 10.0 {
542            velocity.x = 0.0;
543        }
544        if velocity.y.abs() > 10.0 {
545            velocity.y = 0.0;
546        }
547        if velocity.z.abs() > 10.0 {
548            velocity.z = 0.0;
549        }
550
551        // Look up entity type
552        let entity_type = REGISTRY.entity_types.by_key(&persistent.entity_type)?;
553
554        // Check if we have a load factory for this entity type
555        if !ENTITIES.has_load_factory(entity_type) {
556            tracing::debug!(
557                entity_type = %persistent.entity_type,
558                "No load factory for entity type, skipping"
559            );
560            return None;
561        }
562
563        // Get world reference
564        let level = chunk.level_weak();
565
566        // Parse NBT from bytes (or use empty compound data)
567        let nbt_bytes = if persistent.nbt_data.is_empty() {
568            // Empty NBT compound: type byte (10 = compound), empty name (2 zero bytes), end tag (0)
569            &[0x0a, 0x00, 0x00, 0x00][..]
570        } else {
571            &persistent.nbt_data[..]
572        };
573
574        let Ok(nbt) = read_borrowed_compound(&mut Cursor::new(nbt_bytes)) else {
575            tracing::warn!(?uuid, "Failed to parse entity NBT, skipping");
576            return None;
577        };
578
579        ENTITIES.create_and_load(
580            entity_type,
581            pos,
582            uuid,
583            velocity,
584            rotation,
585            persistent.on_ground,
586            level,
587            &nbt,
588        )
589    }

591    /// Converts a persistent section to runtime format.
592    fn persistent_to_section(
593        persistent: &PersistentSection,
594        chunk: &PersistentChunk,
595    ) -> ChunkSection {
596        match persistent {
597            PersistentSection::Homogeneous {
598                block_state,
599                biomes,
600            } => {
601                let block_id = Self::resolve_block_state(chunk, *block_state);
602                let biome_data = Self::persistent_to_biomes(biomes, chunk);
603
604                ChunkSection::new_with_biomes(PalettedContainer::Homogeneous(block_id), biome_data)
605            }
606            PersistentSection::Heterogeneous {
607                palette,
608                bits_per_entry,
609                block_data,
610                biomes,
611            } => {
612                // Unpack indices (into section-local palette)
613                let indices = unpack_indices(block_data, *bits_per_entry, BLOCKS_PER_SECTION);
614
615                // Build runtime palette by resolving section-local -> chunk -> runtime
616                let runtime_palette: Vec<BlockStateId> = palette
617                    .iter()
618                    .map(|&idx| Self::resolve_block_state(chunk, idx))
619                    .collect();
620
621                // Build cube
622                let mut cube = Box::new([[[BlockStateId(0); 16]; 16]; 16]);
623                for (i, &idx) in indices.iter().enumerate() {
624                    let y = i / 256;
625                    let z = (i / 16) % 16;
626                    let x = i % 16;
627                    cube[y][z][x] = runtime_palette
628                        .get(idx as usize)
629                        .copied()
630                        .unwrap_or(BlockStateId(0));
631                }
632
633                let states = PalettedContainer::from_cube(cube);
634                let biome_data = Self::persistent_to_biomes(biomes, chunk);
635
636                ChunkSection::new_with_biomes(states, biome_data)
637            }
638        }
639    }

641    /// Converts persistent biome data to runtime format.
642    fn persistent_to_biomes(
643        persistent: &PersistentBiomeData,
644        chunk: &PersistentChunk,
645    ) -> PalettedContainer<u8, 4> {
646        match persistent {
647            PersistentBiomeData::Homogeneous { biome } => {
648                let biome_id = Self::resolve_biome(chunk, *biome);
649                PalettedContainer::Homogeneous(biome_id)
650            }
651            PersistentBiomeData::Heterogeneous {
652                palette,
653                bits_per_entry,
654                biome_data,
655            } => {
656                let indices = unpack_indices(biome_data, *bits_per_entry, BIOMES_PER_SECTION);
657
658                // Resolve section-local palette -> chunk palette -> runtime
659                let runtime_palette: Vec<u8> = palette
660                    .iter()
661                    .map(|&idx| Self::resolve_biome(chunk, idx))
662                    .collect();
663
664                let mut cube = Box::new([[[0u8; 4]; 4]; 4]);
665                for (i, &idx) in indices.iter().enumerate() {
666                    let y = i / 16;
667                    let z = (i / 4) % 4;
668                    let x = i % 4;
669                    cube[y][z][x] = runtime_palette.get(idx as usize).copied().unwrap_or(0);
670                }
671
672                PalettedContainer::from_cube(cube)
673            }
674        }
675    }

677    /// Resolves a chunk palette index to a runtime `BlockStateId`.
678    fn resolve_block_state(chunk: &PersistentChunk, index: u16) -> BlockStateId {
679        if let Some(state) = chunk.block_states.get(index as usize) {
680            // Convert properties to the format expected by the registry
681            let properties: Vec<(&str, &str)> = state
682                .properties
683                .iter()
684                .map(|(k, v)| (k.as_str(), v.as_str()))
685                .collect();
686
687            if let Some(state_id) = REGISTRY
688                .blocks
689                .state_id_from_properties(&state.name, &properties)
690            {
691                return state_id;
692            }
693        }
694        BlockStateId(0) // Air fallback
695    }

697    /// Resolves a chunk palette index to a runtime biome ID.
698    fn resolve_biome(chunk: &PersistentChunk, index: u16) -> u8 {
699        if let Some(biome_key) = chunk.biomes.get(index as usize)
700            && let Some(id) = REGISTRY.biomes.id_from_key(biome_key)
701        {
702            return id as u8;
703        }
704        0 // Plains fallback
705    }
