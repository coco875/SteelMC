32/// The main server struct.
33pub struct Server {
34    /// The cancellation token for graceful shutdown.
35    pub cancel_token: CancellationToken,
36    /// The key store for the server.
37    pub key_store: KeyStore,
38    /// The registry cache for the server.
39    pub registry_cache: RegistryCache,
40    /// A list of all the worlds on the server.
41    pub worlds: Vec<Arc<World>>,
42    /// The tick rate manager for the server.
43    pub tick_rate_manager: SyncRwLock<TickRateManager>,
44    /// Saves and dispatches commands to appropriate handlers.
45    pub command_dispatcher: SyncRwLock<CommandDispatcher>,
46    /// Counter for assigning unique entity IDs.
47    next_entity_id: AtomicI32,
48}

50impl Server {
51    /// Creates a new server.
52    ///
53    /// # Panics
54    ///
55    /// Panics if the global registry has already been initialized.
56    pub async fn new(chunk_runtime: Arc<Runtime>, cancel_token: CancellationToken) -> Self {
57        let start = Instant::now();
58        let mut registry = Registry::new_vanilla();
59        registry.freeze();
60        log::info!("Vanilla registry loaded in {:?}", start.elapsed());
61
62        REGISTRY
63            .init(registry)
64            .expect("We should be the ones who init the REGISTRY");
65
66        // Initialize behavior registries after the main registry is frozen
67        init_behaviors();
68        log::info!("Behavior registries initialized");
69
70        let registry_cache = RegistryCache::new().await;
71
72        Server {
73            cancel_token,
74            key_store: KeyStore::create(),
75            worlds: vec![World::new(chunk_runtime, OVERWORLD)],
76            registry_cache,
77            tick_rate_manager: SyncRwLock::new(TickRateManager::new()),
78            command_dispatcher: SyncRwLock::new(CommandDispatcher::new()),
79            next_entity_id: AtomicI32::new(1), // Start at 1, 0 is reserved
80        }
81    }
82
83    /// Allocates a new unique entity ID.
84    #[must_use]
85    pub fn next_entity_id(&self) -> i32 {
86        self.next_entity_id.fetch_add(1, Ordering::Relaxed)
87    }

89    /// Adds a player to the server.
90    pub fn add_player(&self, player: Arc<Player>) {
91        player.connection.send_packet(CLogin {
92            player_id: player.entity_id,
93            hardcore: false,
94            levels: vec![Identifier::vanilla_static("overworld")],
95            max_players: 5,
96            chunk_radius: STEEL_CONFIG.view_distance.into(),
97            simulation_distance: STEEL_CONFIG.simulation_distance.into(),
98            reduced_debug_info: false,
99            show_death_screen: false,
100            do_limited_crafting: false,
101            common_player_spawn_info: CommonPlayerSpawnInfo {
102                dimension_type: 0,
103                dimension: Identifier::vanilla_static("overworld"),
104                seed: 0,
105                game_type: GameType::Survival,
106                previous_game_type: None,
107                is_debug: false,
108                is_flat: true,
109                last_death_location: None,
110                portal_cooldown: 0,
111                sea_level: 64,
112            },
113            enforces_secure_chat: STEEL_CONFIG.enforce_secure_chat,
114        });
115
116        let commands = self.command_dispatcher.read().get_commands();
117        player.connection.send_packet(commands);
118
119        self.worlds[0].add_player(player);
120    }

122    /// Runs the server tick loop.
123    pub async fn run(self: Arc<Self>, cancel_token: CancellationToken) {
124        let mut next_tick_time = Instant::now();
125
126        loop {
127            if cancel_token.is_cancelled() {
128                break;
129            }
130
131            let (nanoseconds_per_tick, is_sprinting, should_sprint_this_tick) = {
132                let mut tick_manager = self.tick_rate_manager.write();
133                let nanoseconds_per_tick = tick_manager.nanoseconds_per_tick;
134
135                // Handle sprinting
136                let is_sprinting = tick_manager.is_sprinting();
137                let should_sprint_this_tick = if is_sprinting {
138                    tick_manager.check_should_sprint_this_tick()
139                } else {
140                    false
141                };
142                (nanoseconds_per_tick, is_sprinting, should_sprint_this_tick)
143            };
144
145            if is_sprinting && should_sprint_this_tick {
146                // If sprinting, we don't wait
147                next_tick_time = Instant::now();
148            } else {
149                // Normal wait logic
150                let now = Instant::now();
151                if now < next_tick_time {
152                    tokio::select! {
153                        () = cancel_token.cancelled() => break,
154                        () = sleep(next_tick_time - now) => {}
155                    }
156                }
157                next_tick_time += Duration::from_nanos(nanoseconds_per_tick);
158            }
159
160            if cancel_token.is_cancelled() {
161                break;
162            }
163
164            let tick_count = {
165                let mut tick_manager = self.tick_rate_manager.write();
166                tick_manager.tick();
167                tick_manager.tick_count
168            };
169
170            // Tick worlds
171            self.tick_worlds(tick_count).await;
172
173            if is_sprinting && should_sprint_this_tick {
174                let mut tick_manager = self.tick_rate_manager.write();
175                tick_manager.end_tick_work();
176            }
177        }
178    }
179
180    async fn tick_worlds(&self, tick_count: u64) {
181        let mut tasks = Vec::with_capacity(self.worlds.len());
182        for world in &self.worlds {
183            let world_clone = world.clone();
184            tasks.push(spawn_blocking(move || world_clone.tick_b(tick_count)));
185        }
186        let start = Instant::now();
187        for task in tasks {
188            let _ = task.await;
189        }
190        if start.elapsed().as_millis() > 1 {
191            log::warn!(
192                "Worlds ticked in {:?}, tick count: {tick_count}",
193                start.elapsed()
194            );
195        }
196    }
