22/// Vanilla `BubbleColumnBlock` column propagation and fluid state.
23#[block_behavior]
24pub struct BubbleColumnBlock {
25    block: BlockRef,
26}
27
28impl BubbleColumnBlock {
29    /// Creates a bubble column block behavior.
30    #[must_use]
31    pub const fn new(block: BlockRef) -> Self {
32        Self { block }
33    }
34
35    pub(super) fn update_column(
36        bubble_column: BlockRef,
37        level: &dyn LevelAccessor,
38        occupy_at: BlockPos,
39        below_state: BlockStateId,
40    ) {
41        let occupy_state = level.get_block_state(occupy_at);
42        if !Self::can_occupy(bubble_column, occupy_state) {
43            return;
44        }
45
46        let column_state = Self::column_state(bubble_column, below_state, occupy_state);
47        level.set_block_state(occupy_at, column_state, UpdateFlags::UPDATE_CLIENTS);
48
49        let mut pos = occupy_at.above();
50        while Self::can_occupy(bubble_column, level.get_block_state(pos)) {
51            if !level.set_block_state(pos, column_state, UpdateFlags::UPDATE_CLIENTS) {
52                return;
53            }
54            pos = pos.above();
55        }
56    }
57
58    pub(super) fn can_occupy(bubble_column: BlockRef, occupy_state: BlockStateId) -> bool {
59        if occupy_state.get_block() == bubble_column {
60            return true;
61        }
62
63        let fluid_state = occupy_state.get_fluid_state();
64        fluid_state
65            .fluid_id
66            .has_tag(&FluidTag::BUBBLE_COLUMN_CAN_OCCUPY)
67            && occupy_state.get_block() == &vanilla_blocks::WATER
68            && fluid_state.is_source()
69            && fluid_state.amount >= 8
70    }
71
72    fn column_state(
73        bubble_column: BlockRef,
74        below_state: BlockStateId,
75        occupy_state: BlockStateId,
76    ) -> BlockStateId {
77        if below_state.get_block() == bubble_column {
78            return below_state;
79        }
80        if below_state
81            .get_block()
82            .has_tag(&BlockTag::ENABLES_BUBBLE_COLUMN_PUSH_UP)
83        {
84            return bubble_column
85                .default_state()
86                .set_value(&BlockStateProperties::DRAG, false);
87        }
88        if below_state
89            .get_block()
90            .has_tag(&BlockTag::ENABLES_BUBBLE_COLUMN_DRAG_DOWN)
91        {
92            return bubble_column
93                .default_state()
94                .set_value(&BlockStateProperties::DRAG, true);
95        }
96
97        if occupy_state.get_block() == bubble_column {
98            vanilla_blocks::WATER.default_state()
99        } else {
100            occupy_state
101        }
102    }
103}

105impl BlockBehavior for BubbleColumnBlock {
106    fn get_state_for_placement(&self, _context: &BlockPlaceContext<'_>) -> Option<BlockStateId> {
107        Some(self.block.default_state())
108    }
109
110    fn can_survive(&self, _state: BlockStateId, world: &dyn LevelReader, pos: BlockPos) -> bool {
111        let below = world.get_block_state(pos.below());
112        below.get_block() == self.block
113            || below
114                .get_block()
115                .has_tag(&BlockTag::ENABLES_BUBBLE_COLUMN_PUSH_UP)
116            || below
117                .get_block()
118                .has_tag(&BlockTag::ENABLES_BUBBLE_COLUMN_DRAG_DOWN)
119    }
120
121    fn update_shape(
122        &self,
123        state: BlockStateId,
124        world: &dyn ScheduledTickAccess,
125        pos: BlockPos,
126        direction: Direction,
127        _neighbor_pos: BlockPos,
128        neighbor_state: BlockStateId,
129    ) -> BlockStateId {
130        let delay = world.fluid_tick_delay(&vanilla_fluids::WATER);
131        let _ = world.schedule_fluid_tick_default(pos, &vanilla_fluids::WATER, delay);
132
133        if !self.can_survive(state, world, pos)
134            || direction == Direction::Down
135            || (direction == Direction::Up
136                && neighbor_state.get_block() != self.block
137                && Self::can_occupy(self.block, neighbor_state))
138        {
139            let _ = world.schedule_block_tick_default(pos, self.block, 5);
140        }
141
142        state
143    }
144
145    fn tick(&self, _state: BlockStateId, world: &Arc<World>, pos: BlockPos) {
146        Self::update_column(self.block, world, pos, world.get_block_state(pos.below()));
147    }
148
149    fn get_fluid_state(&self, _state: BlockStateId) -> FluidState {
150        FluidState::source(&vanilla_fluids::WATER)
151    }
152
153    fn pickup_block(
154        &self,
155        world: &Arc<World>,
156        pos: BlockPos,
157        _state: BlockStateId,
158        _player: Option<&Player>,
159    ) -> Option<PickupResult> {
160        world.set_block(
161            pos,
162            vanilla_blocks::AIR.default_state(),
163            UpdateFlags::UPDATE_ALL_IMMEDIATE,
164        );
165        Some(PickupResult {
166            filled_bucket: &vanilla_items::ITEMS.water_bucket,
167            sound: Some(&sound_events::ITEM_BUCKET_FILL),
168        })
169    }
170}

172#[cfg(test)]
173mod tests {
174    use super::*;
175    use crate::behavior::init_behaviors;
176    use crate::test_support::TestLevel;
177    use steel_registry::test_support::init_test_registry;

179    #[test]
180    fn bubble_column_update_shape_schedules_water_and_column_tick() {
181        init_test_registry();
182        let behavior = BubbleColumnBlock::new(&vanilla_blocks::BUBBLE_COLUMN);
183        let level = TestLevel::default();
184        let state = vanilla_blocks::BUBBLE_COLUMN.default_state();
185
186        let updated = behavior.update_shape(
187            state,
188            &level,
189            BlockPos::ZERO,
190            Direction::Down,
191            BlockPos::ZERO.below(),
192            vanilla_blocks::SOUL_SAND.default_state(),
193        );
194
195        assert_eq!(updated, state);
196        assert!(level.scheduled_water_tick());
197        assert!(
198            level
199                .scheduled_block_ticks
200                .borrow()
201                .iter()
202                .any(|tick| tick.block == &vanilla_blocks::BUBBLE_COLUMN && tick.delay == 5)
203        );
204    }
205
206    #[test]
207    fn bubble_column_update_column_uses_push_up_and_drag_down_blocks() {
208        init_test_registry();
209        init_behaviors();
210        let level = TestLevel::default()
211            .with_block(BlockPos::ZERO, vanilla_blocks::WATER.default_state())
212            .with_block(
213                BlockPos::ZERO.above(),
214                vanilla_blocks::WATER.default_state(),
215            );
216
217        BubbleColumnBlock::update_column(
218            &vanilla_blocks::BUBBLE_COLUMN,
219            &level,
220            BlockPos::ZERO,
221            vanilla_blocks::SOUL_SAND.default_state(),
222        );
223
224        let placed = level.placed_blocks.borrow();
225        assert_eq!(placed.len(), 2);
226        assert!(placed.iter().all(|placed| {
227            placed.state.get_block() == &vanilla_blocks::BUBBLE_COLUMN
228                && !placed.state.get_value(&BlockStateProperties::DRAG)
229        }));
230    }
