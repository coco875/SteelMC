15/// Result of an interaction (item use, block use, etc.)
16#[derive(Debug, Clone, Copy, PartialEq, Eq)]
17pub enum InteractionResult {
18    /// The interaction succeeded and consumed the action.
19    Success,
20    /// The interaction failed and consumed the action.
21    Fail,
22    /// The interaction did not apply; try the next handler.
23    Pass,
24    /// Try the empty-hand interaction on the block.
25    TryEmptyHandInteraction,
26}
27
28impl InteractionResult {
29    /// Returns true if this result consumes the action (Success or Fail).
30    /// Pass and `TryEmptyHandInteraction` do not consume the action.
31    #[must_use]
32    pub fn consumes_action(&self) -> bool {
33        matches!(self, InteractionResult::Success | InteractionResult::Fail)
34    }
35}

37/// Context for placing a block.
38pub struct BlockPlaceContext<'a> {
39    /// The position that was clicked.
40    pub clicked_pos: BlockPos,
41    /// The face of the block that was clicked.
42    pub clicked_face: Direction,
43    /// The exact location where the click occurred.
44    pub click_location: Vector3<f64>,
45    /// Whether the click was inside the block.
46    pub inside: bool,
47    /// The position where the block will be placed.
48    pub relative_pos: BlockPos,
49    /// Whether the clicked block is being replaced.
50    pub replace_clicked: bool,
51    /// The horizontal direction the player is facing.
52    pub horizontal_direction: Direction,
53    /// The player's rotation (yaw).
54    pub rotation: f32,
55    /// The world where the block is being placed.
56    pub world: &'a World,
57}

59/// Context for using an item on a block.
60pub struct UseOnContext<'a> {

69    /// The item stack being used (mutable for consumption).
