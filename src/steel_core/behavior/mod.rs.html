1//! Block and item behavior system.
2//!
3//! This module contains the behavior traits and registries that define how
4//! blocks and items behave dynamically. This is separate from the static data
5//! in steel-registry to maintain a clean separation between constant data and
6//! functional/dynamic behavior.
7//!
8//! # Architecture
9//!
10//! After the main registry (`steel-registry`) is frozen, behavior registries
11//! are created:
12//! - `BlockBehaviorRegistry` - assigns default or custom behaviors to each block
13//! - `ItemBehaviorRegistry` - assigns default or custom behaviors to each item
14//!
15//! # Usage
16//!
17//! ```ignore
18//! use steel_core::behavior::{init_behaviors, BLOCK_BEHAVIORS, ITEM_BEHAVIORS};
19//!
20//! // After registry is frozen, call once at startup:
21//! init_behaviors();
22//!
23//! // Then access behaviors via the global registries:
24//! let behavior = BLOCK_BEHAVIORS.get().unwrap().get_behavior(block);
25//! ```

51/// Global block behavior registry.
52pub static BLOCK_BEHAVIORS: OnceLock<BlockBehaviorRegistry> = OnceLock::new();
53
54/// Global item behavior registry.
55pub static ITEM_BEHAVIORS: OnceLock<ItemBehaviorRegistry> = OnceLock::new();
56
57/// Initializes the global behavior registries.
58///
59/// This should be called once after the main registry is frozen.
60///
61/// # Panics
62///
63/// Panics if called more than once.
64pub fn init_behaviors() {
65    let mut block_behaviors = BlockBehaviorRegistry::new();
66    register_block_behaviors(&mut block_behaviors);
67
68    assert!(
69        BLOCK_BEHAVIORS.set(block_behaviors).is_ok(),
70        "Block behavior registry already initialized"
71    );
72
73    let mut item_behaviors = ItemBehaviorRegistry::new();
74    register_item_behaviors(&mut item_behaviors);
75
76    assert!(
77        ITEM_BEHAVIORS.set(item_behaviors).is_ok(),
78        "Item behavior registry already initialized"
79    );
80}
