21/// Factory function type for creating entities.
22///
23/// Takes the entity ID, spawn position, and world reference.
24/// Returns a new entity instance. The entity ID should be obtained from
25/// `next_entity_id()`.
26pub type EntityFactory = fn(i32, DVec3, Weak<World>) -> SharedEntity;
27
28/// Factory function type for loading entities from disk.
29///
30/// Takes all base entity fields needed for reconstruction.
31pub type EntityLoadFactory = fn(EntityBaseLoad) -> SharedEntity;
32
33/// Entity load request before the registry assigns a runtime ID.
34pub struct EntityLoadRequest {
35    /// Entity type to instantiate.
36    pub entity_type: EntityTypeRef,
37    /// Restored entity position.
38    pub position: DVec3,
39    /// Persisted entity UUID.
40    pub uuid: Uuid,
41    /// Restored velocity.
42    pub velocity: DVec3,
43    /// Restored yaw and pitch.
44    pub rotation: (f32, f32),
45    /// Restored accumulated fall distance.
46    pub fall_distance: f64,
47    /// Restored vanilla fire/freeze state.
48    pub fire_freeze: EntityFireFreezeState,
49    /// Restored ground-contact flag.
50    pub on_ground: bool,
51    /// Restored shared vanilla `NoGravity` flag.
52    pub no_gravity: bool,
53    /// World reference for the loaded entity.
54    pub world: Weak<World>,
55}
56
57impl EntityLoadRequest {
58    fn into_base_load(self) -> (EntityTypeRef, EntityBaseLoad) {
59        (
60            self.entity_type,
61            EntityBaseLoad {
62                id: next_entity_id(),
63                position: self.position,
64                uuid: self.uuid,
65                velocity: self.velocity,
66                rotation: self.rotation,
67                fall_distance: self.fall_distance,
68                fire_freeze: self.fire_freeze,
69                on_ground: self.on_ground,
70                no_gravity: self.no_gravity,
71                world: self.world,
72            },
73        )
74    }
75}

77/// Registry entry for an entity type.
78struct EntityEntry {
79    /// Factory function to create new instances.
80    factory: Option<EntityFactory>,
81    /// Factory function to load instances from disk.
82    load_factory: Option<EntityLoadFactory>,
83}
84
85/// Registry for entity factories.
86///
87/// Maps `EntityType` to factory functions that can create entity instances.
88/// This is used when loading entities from disk or when entities are spawned.
89pub struct EntityRegistry {
90    entries: Vec<EntityEntry>,
91}
92
93impl EntityRegistry {
94    /// Creates a new empty registry with entries for all entity types.
95    #[must_use]
96    pub fn new() -> Self {
97        let count = REGISTRY.entity_types.len();
98        let entries = (0..count)
99            .map(|_| EntityEntry {
100                factory: None,
101                load_factory: None,
102            })
103            .collect();
104
105        Self { entries }
106    }
107
108    /// Registers a factory function for an entity type.
109    pub fn register(&mut self, entity_type: EntityTypeRef, factory: EntityFactory) {
110        let id = entity_type.id();
111        self.entries[id].factory = Some(factory);
112    }
113
114    /// Registers a load factory function for an entity type.
115    ///
116    /// The load factory is used when loading entities from disk.
117    pub fn register_load(&mut self, entity_type: EntityTypeRef, factory: EntityLoadFactory) {
118        let id = entity_type.id();
119        self.entries[id].load_factory = Some(factory);
120    }
121
122    /// Creates a new entity instance.
123    ///
124    /// Returns `None` if no factory is registered for the given type.
125    #[must_use]
126    pub fn create(
127        &self,
128        entity_type: EntityTypeRef,
129        entity_id: i32,
130        pos: DVec3,
131        world: Weak<World>,
132    ) -> Option<SharedEntity> {
133        let id = entity_type.id();
134        self.entries
135            .get(id)?
136            .factory
137            .map(|f| f(entity_id, pos, world))
138    }
139
140    /// Creates an entity from persisted data and loads its type-specific NBT.
141    ///
142    /// Returns `None` if no load factory is registered for the entity type.
143    #[must_use]
144    pub fn create_and_load(
145        &self,
146        request: EntityLoadRequest,
147        nbt: &BorrowedNbtCompound<'_>,
148    ) -> Option<SharedEntity> {
149        let (entity_type, load) = request.into_base_load();
150        let id = entity_type.id();
151        let load_factory = self.entries.get(id)?.load_factory?;
152        let no_gravity = load.no_gravity;
153
154        let entity = load_factory(load);
155        entity.set_no_gravity(no_gravity);
156        entity.load_additional(nbt);
157        entity.sync_base_fire_freeze_entity_data();
158        Some(entity)
159    }
160
161    /// Creates an entity from persisted data, falling back to raw NBT preservation.
162    #[must_use]
163    pub fn create_and_load_or_raw(
164        &self,
165        request: EntityLoadRequest,
166        nbt: &BorrowedNbtCompound<'_>,
167    ) -> SharedEntity {
168        let (entity_type, load) = request.into_base_load();
169        let id = entity_type.id();
170        let no_gravity = load.no_gravity;
171        if let Some(load_factory) = self.entries.get(id).and_then(|entry| entry.load_factory) {
172            let entity = load_factory(load);
173            entity.set_no_gravity(no_gravity);
174            entity.load_additional(nbt);
175            entity.sync_base_fire_freeze_entity_data();
176            return entity;
177        }
178
179        let entity: SharedEntity = Arc::new(RawEntity::from_saved(load, entity_type));
180        entity.set_no_gravity(no_gravity);
181        entity.load_additional(nbt);
182        entity
183    }
184
185    /// Returns whether a factory is registered for the given type.
186    #[must_use]
187    pub fn has_factory(&self, entity_type: EntityTypeRef) -> bool {
188        let id = entity_type.id();
189        self.entries.get(id).is_some_and(|e| e.factory.is_some())
190    }
191
192    /// Returns whether a load factory is registered for the given type.
193    #[must_use]
194    pub fn has_load_factory(&self, entity_type: EntityTypeRef) -> bool {
195        let id = entity_type.id();
196        self.entries
197            .get(id)
198            .is_some_and(|e| e.load_factory.is_some())
199    }
200}
201
202impl Default for EntityRegistry {
203    fn default() -> Self {
204        Self::new()
205    }
206}

208/// Wrapper for the global entity registry that implements `Deref`.
209pub struct EntityRegistryLock(OnceLock<EntityRegistry>);
210
211impl Deref for EntityRegistryLock {
212    type Target = EntityRegistry;
213
214    fn deref(&self) -> &Self::Target {
215        self.0.get().expect("Entity registry not initialized")
216    }
217}
218
219impl EntityRegistryLock {
220    /// Sets the registry. Returns `Err` if already initialized.
221    pub fn set(&self, registry: EntityRegistry) -> Result<(), EntityRegistry> {
222        self.0.set(registry)
223    }
224}
225
226/// Global entity registry.
227///
228/// Access via deref: `ENTITIES.create(type, entity_id, pos)`
229pub static ENTITIES: EntityRegistryLock = EntityRegistryLock(OnceLock::new());
230
231/// Initializes the global entity registry.
232///
233/// This should be called once after the main registry is frozen.
234///
235/// # Panics
236///
237/// Panics if called more than once.
238pub fn init_entities() {
239    let mut registry = EntityRegistry::new();
240
241    // Register block display entity factory
242    registry.register(&vanilla_entities::BLOCK_DISPLAY, |id, pos, world| {
243        Arc::new(BlockDisplayEntity::new(id, pos, world))
244    });
245    registry.register_load(&vanilla_entities::BLOCK_DISPLAY, |load| {
246        Arc::new(BlockDisplayEntity::from_saved(load))
247    });
248
249    // Register item entity factory
250    registry.register(&vanilla_entities::ITEM, |id, pos, world| {
251        Arc::new(ItemEntity::new(id, pos, world))
252    });
253    registry.register_load(&vanilla_entities::ITEM, |load| {
254        Arc::new(ItemEntity::from_saved(load))
255    });
256
257    // Register end crystal entity factory
258    registry.register(&vanilla_entities::END_CRYSTAL, |id, pos, world| {
259        Arc::new(EndCrystalEntity::new(id, pos, world))
260    });
261    registry.register_load(&vanilla_entities::END_CRYSTAL, |load| {
262        Arc::new(EndCrystalEntity::from_saved(load))
263    });
264
265    // Register chest minecart entity factory
266    registry.register(&vanilla_entities::CHEST_MINECART, |id, pos, world| {
267        Arc::new(ChestMinecartEntity::new(id, pos, world))
268    });
269    registry.register_load(&vanilla_entities::CHEST_MINECART, |load| {
270        Arc::new(ChestMinecartEntity::from_saved(load))
271    });
272
273    registry.register(&vanilla_entities::ITEM_FRAME, |id, pos, world| {
274        Arc::new(ItemFrameEntity::new(
275            id,
276            BlockPos::new(
277                pos.x.floor() as i32,
278                pos.y.floor() as i32,
279                pos.z.floor() as i32,
280            ),
281            Direction::South,
282            world,
283        ))
284    });
285    registry.register_load(&vanilla_entities::ITEM_FRAME, |load| {
286        Arc::new(ItemFrameEntity::from_saved(load))
287    });
288
289    assert!(
290        ENTITIES.set(registry).is_ok(),
291        "Entity registry already initialized"
292    );
293}

295#[cfg(test)]
296mod tests {
297    use std::io::Cursor;
298
299    use simdnbt::borrow::read_compound as read_borrowed_compound;
300    use simdnbt::owned::NbtCompound;
301    use steel_registry::test_support::init_test_registry;
302
303    use super::*;
304
305    #[test]
306    fn create_and_load_or_raw_preserves_unregistered_entity_data() {
307        init_test_registry();
308        let registry = EntityRegistry::new();
309        let mut nbt = NbtCompound::new();
310        nbt.insert("CustomName", "raw");
311        let mut bytes = Vec::new();
312        nbt.write(&mut bytes);
313        let borrowed =
314            read_borrowed_compound(&mut Cursor::new(&bytes)).expect("test nbt should reborrow");
315
316        let entity = registry.create_and_load_or_raw(
317            EntityLoadRequest {
318                entity_type: &vanilla_entities::VILLAGER,
319                position: DVec3::new(1.0, 2.0, 3.0),
320                uuid: Uuid::from_u128(1),
321                velocity: DVec3::new(0.1, 0.0, 0.2),
322                rotation: (45.0, 10.0),
323                fall_distance: 2.25,
324                fire_freeze: EntityFireFreezeState::new(),
325                on_ground: true,
326                no_gravity: true,
327                world: Weak::new(),
328            },
329            &borrowed,
330        );
331
332        assert_eq!(&entity.entity_type().key, &vanilla_entities::VILLAGER.key);
333        assert_eq!(entity.position(), DVec3::new(1.0, 2.0, 3.0));
334        assert_eq!(entity.velocity(), DVec3::new(0.1, 0.0, 0.2));
335        assert_eq!(entity.rotation(), (45.0, 10.0));
336        assert!((entity.fall_distance() - 2.25).abs() <= f64::EPSILON);
337        assert!(entity.on_ground());
338        assert!(entity.is_no_gravity());
339
340        let mut saved = NbtCompound::new();
341        entity.save_additional(&mut saved);
342        assert_eq!(
343            saved.string("CustomName").map(ToString::to_string),
344            Some("raw".to_owned())
345        );
346    }
