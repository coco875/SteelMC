21/// Chest minecart entity state used by mineshaft generation.
22///
23/// Steel does not yet implement minecart movement or container interaction, so this
24/// entity currently preserves the vanilla placement and loot-table state that
25/// structure generation creates.
26pub struct ChestMinecartEntity {
27    base: EntityBase,
28    velocity: SyncMutex<DVec3>,
29    rotation: AtomicCell<(f32, f32)>,
30    on_ground: AtomicBool,
31    first_tick: AtomicBool,
32    loot_table: SyncMutex<Option<Identifier>>,
33    loot_table_seed: AtomicI64,
34}
35
36impl ChestMinecartEntity {
37    /// Creates a new chest minecart entity.
38    #[must_use]
39    pub fn new(id: i32, position: DVec3, world: Weak<World>) -> Self {
40        Self {
41            base: EntityBase::new(id, position, world),
42            velocity: SyncMutex::new(DVec3::ZERO),
43            rotation: AtomicCell::new((0.0, 0.0)),
44            on_ground: AtomicBool::new(false),
45            first_tick: AtomicBool::new(true),
46            loot_table: SyncMutex::new(None),
47            loot_table_seed: AtomicI64::new(0),
48        }
49    }
50
51    /// Creates a chest minecart entity from saved data.
52    #[must_use]
53    pub fn from_saved(
54        id: i32,
55        position: DVec3,
56        uuid: Uuid,
57        velocity: DVec3,
58        rotation: (f32, f32),
59        on_ground: bool,
60        world: Weak<World>,
61    ) -> Self {
62        Self {
63            base: EntityBase::with_uuid(id, uuid, position, world),
64            velocity: SyncMutex::new(velocity),
65            rotation: AtomicCell::new(rotation),
66            on_ground: AtomicBool::new(on_ground),
67            first_tick: AtomicBool::new(false),
68            loot_table: SyncMutex::new(None),
69            loot_table_seed: AtomicI64::new(0),
70        }
71    }
72
73    /// Sets the deferred loot table used when the container is first opened.
74    pub fn set_loot_table(&self, loot_table: Identifier, seed: i64) {
75        *self.loot_table.lock() = Some(loot_table);
76        self.loot_table_seed.store(seed, Ordering::Relaxed);
77    }
78
79    const fn nbt_bool(value: bool) -> i8 {
80        if value { 1 } else { 0 }
81    }
82}
83
84impl Entity for ChestMinecartEntity {
85    fn base(&self) -> Option<&EntityBase> {
86        Some(&self.base)
87    }
88
89    fn entity_type(&self) -> EntityTypeRef {
90        &vanilla_entities::CHEST_MINECART
91    }
92
93    fn bounding_box(&self) -> AABBd {
94        let pos = self.position();
95        let dims = self.entity_type().dimensions;
96        let half_width = f64::from(dims.width) / 2.0;
97        let height = f64::from(dims.height);
98        AABBd {
99            min_x: pos.x - half_width,
100            min_y: pos.y,
101            min_z: pos.z - half_width,
102            max_x: pos.x + half_width,
103            max_y: pos.y + height,
104            max_z: pos.z + half_width,
105        }
106    }
107
108    fn rotation(&self) -> (f32, f32) {
109        self.rotation.load()
110    }
111
112    fn velocity(&self) -> DVec3 {
113        *self.velocity.lock()
114    }
115
116    fn set_velocity(&self, velocity: DVec3) {
117        *self.velocity.lock() = velocity;
118    }
119
120    fn on_ground(&self) -> bool {
121        self.on_ground.load(Ordering::Relaxed)
122    }
123
124    fn set_on_ground(&self, on_ground: bool) {
125        self.on_ground.store(on_ground, Ordering::Relaxed);
126    }
127
128    fn save_additional(&self, nbt: &mut NbtCompound) {
129        nbt.insert("FlippedRotation", Self::nbt_bool(false));
130        nbt.insert(
131            "HasTicked",
132            Self::nbt_bool(self.first_tick.load(Ordering::Relaxed)),
133        );
134
135        if let Some(loot_table) = self.loot_table.lock().as_ref() {
136            nbt.insert("LootTable", loot_table.to_string());
137            let seed = self.loot_table_seed.load(Ordering::Relaxed);
138            if seed != 0 {
139                nbt.insert("LootTableSeed", NbtTag::Long(seed));
140            }
141        }
142    }
143
144    fn load_additional(&self, nbt: &BorrowedNbtCompound<'_>) {
145        let nbt: NbtCompoundView<'_, '_> = nbt.into();
146
147        if let Some(first_tick) = nbt.byte("HasTicked") {
148            self.first_tick.store(first_tick != 0, Ordering::Relaxed);
149        }
150
151        let loot_table = nbt
152            .string("LootTable")
153            .and_then(|value| Identifier::from_str(&value.to_string()).ok());
154        *self.loot_table.lock() = loot_table;
155        self.loot_table_seed
156            .store(nbt.long("LootTableSeed").unwrap_or(0), Ordering::Relaxed);
157    }
158}

160#[cfg(test)]
161mod tests {
162    use super::*;
163
164    #[test]
165    fn chest_minecart_saves_structure_loot_table_state() {
166        let minecart = ChestMinecartEntity::new(1, DVec3::new(1.5, 2.5, 3.5), Weak::new());
167        minecart.set_loot_table(
168            Identifier::new_static("minecraft", "chests/abandoned_mineshaft"),
169            42,
170        );
171
172        let mut nbt = NbtCompound::new();
173        minecart.save_additional(&mut nbt);
174
175        assert_eq!(
176            nbt.string("LootTable").map(ToString::to_string),
177            Some("minecraft:chests/abandoned_mineshaft".to_owned())
178        );
179        assert_eq!(nbt.long("LootTableSeed"), Some(42));
180        assert_eq!(nbt.byte("HasTicked"), Some(1));
181        assert_eq!(nbt.byte("FlippedRotation"), Some(0));
182    }
