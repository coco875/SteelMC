40/// Handles using an item on a block.
41///
42/// This implements the logic from Java's `ServerPlayerGameMode.useItemOn()`.
43///
44/// # Flow
45/// 1. Spectator mode: Only allow opening menus (currently returns Pass)
46/// 2. Check if block interaction should be suppressed (sneaking + holding items)
47/// 3. If not suppressed: Call block's `use_item_on` method
48/// 4. If block returns `TryEmptyHandInteraction` and main hand: Call block's `use_without_item`
49/// 5. If item not empty: Call item behavior's `use_on` for placement
50/// 6. Handle creative mode infinite materials
51pub fn use_item_on(
52    player: &Player,
53    world: &Arc<World>,
54    hand: InteractionHand,
55    hit_result: &BlockHitResult,
56) -> InteractionResult {
57    let pos = hit_result.block_pos;
58    let state = world.get_block_state(pos);
59
60    // Spectator mode: can only open menus
61    // TODO: Implement menu providers for blocks like chests
62    if player.game_mode.load() == GameType::Spectator {
63        return InteractionResult::Pass;
64    }
65
66    // Check if block interaction should be suppressed (sneaking + holding items in either hand)
67    let have_something = {
68        let inv = player.inventory.lock();
69        !inv.get_item_in_hand(InteractionHand::MainHand).is_empty()
70            || !inv.get_item_in_hand(InteractionHand::OffHand).is_empty()
71    };
72
73    let suppress_block_use = player.is_secondary_use_active() && have_something;
74
75    // Get behavior registries
76    let block_behaviors = &*BLOCK_BEHAVIORS;
77    let item_behaviors = &*ITEM_BEHAVIORS;
78
79    // Try block interaction first (if not suppressed).
80    // No inventory lock held — block behaviors may need inventory access (e.g. opening chests).
81    if !suppress_block_use {
82        let Some(block) = REGISTRY.blocks.by_state_id(state) else {
83            return InteractionResult::Pass;
84        };
85        let behavior = block_behaviors.get_behavior(block);
86
87        // Brief lock for an immutable snapshot used during block interaction check
88        let item_snapshot = player.inventory.lock().get_item_in_hand(hand).clone();
89
90        let block_result =
91            behavior.use_item_on(&item_snapshot, state, world, pos, player, hand, hit_result);
92
93        if block_result.consumes_action() {
94            return block_result;
95        }
96
97        if matches!(block_result, InteractionResult::TryEmptyHandInteraction)
98            && hand == InteractionHand::MainHand
99        {
100            let empty_result = behavior.use_without_item(state, world, pos, player, hit_result);
101
102            if empty_result.consumes_action() {
103                return empty_result;
104            }
105        }
106    }
107
108    // Item use (block placement, etc.) — acquire inventory lock via ContainerLockGuard
109    let inv_ref = ContainerRef::PlayerInventory(player.inventory.clone());
110    let mut guard = ContainerLockGuard::lock_all(&[&inv_ref]);
111
112    let inv_id = inv_ref.container_id();
113
114    let is_empty = {
115        let Some(inv) = guard.get_player_inventory_mut(inv_id) else {
116            return InteractionResult::Pass;
117        };
118        inv.get_item_in_hand(hand).is_empty()
119    };
120
121    if !is_empty {
122        // TODO: Check item cooldowns
123        // if player.getCooldowns().isOnCooldown(item_stack.item) { return Pass }
124
125        let mut context =
126            UseOnContext::new(player, hand, hit_result.clone(), world, &mut guard, inv_id);
127
128        let original_count = context.inv.item().count;
129        let item_ref = context.inv.item().item;
130        let item_behavior = item_behaviors.get_behavior(item_ref);
131        let result = item_behavior.use_on(&mut context);
132
133        // Restore count for creative mode (infinite materials)
134        if player.has_infinite_materials() && context.inv.item().count < original_count {
135            context.inv.item().count = original_count;
136        }
137
138        return result;
139    }
140
141    InteractionResult::Pass
142}

144/// Handles using an item (general usage like right-clicking air).
145///
146/// This implements logic similar to `ServerPlayerGameMode.useItem()`.
147pub fn use_item(player: &Player, world: &Arc<World>, hand: InteractionHand) -> InteractionResult {
148    // Spectator mode: can only open menus
149    if player.game_mode.load() == GameType::Spectator {
150        return InteractionResult::Pass;
151    }
152
153    // TODO: Check item cooldowns
154    // if player.getCooldowns().isOnCooldown(item_stack) { return InteractionResult::Pass }
155
156    let inv_ref = ContainerRef::PlayerInventory(player.inventory.clone());
157    let mut guard = ContainerLockGuard::lock_all(&[&inv_ref]);
158    let inv_id = inv_ref.container_id();
159
160    let is_empty = {
161        let Some(inv) = guard.get_player_inventory_mut(inv_id) else {
162            return InteractionResult::Pass;
163        };
164        inv.get_item_in_hand(hand).is_empty()
165    };
166
167    if !is_empty {
168        let mut context =
169            crate::behavior::UseItemContext::new(player, hand, world, &mut guard, inv_id);
170
171        let original_count = context.inv.item().count;
172
173        // Get behavior registries
174        let item_behaviors = &*ITEM_BEHAVIORS;
175        let item_ref = context.inv.item().item;
176        let item_behavior = item_behaviors.get_behavior(item_ref);
177
178        let result = item_behavior.use_item(&mut context);
179
180        // Restore count for creative mode (infinite materials)
181        if player.has_infinite_materials() && context.inv.item().count < original_count {
182            context.inv.item().count = original_count;
183        }
184
185        return result;
186    }
187
188    InteractionResult::Pass
189}

194    /// Returns `true` if the game mode was changed, `false` if the player was already in the requested game mode.
195    pub fn set_game_mode(&self, gamemode: GameType) -> bool {
196        let current_gamemode = self.game_mode.load();
197        if current_gamemode == gamemode {
198            return false;
199        }
200
201        self.prev_game_mode.store(self.game_mode.load());
202        self.game_mode.store(gamemode);
203
204        // Update abilities based on new game mode (mirrors vanilla GameType.updatePlayerAbilities)
205        self.abilities.lock().update_for_game_mode(gamemode);
206        self.send_abilities();
207
208        self.send_packet(CGameEvent {
209            event: GameEventType::ChangeGameMode,
210            data: gamemode.into(),
211        });
212
213        let update_packet =
214            CPlayerInfoUpdate::update_game_mode(self.gameprofile.id, gamemode as i32);
215        self.get_world().broadcast_to_all(update_packet);
216
217        self.send_message(
218            &translations::COMMANDS_GAMEMODE_SUCCESS_SELF
219                .message([get_gamemode_translation(gamemode)])
220                .into(),
221        );
222
223        true
224    }

237    pub fn handle_change_difficulty(&self, difficulty: Difficulty) {
238        // TODO: implement op-level permission check
239        let world = self.get_world();
240        {
241            let level_data = world.level_data.read();
242            if level_data.data().difficulty_locked {
243                let current = level_data.data().difficulty;
244                drop(level_data);
245                self.send_packet(CChangeDifficulty {
246                    difficulty: current,
247                    locked: true,
248                });
249                return;
250            }
251        }
252
253        let domain = self.get_world().domain().to_owned();
254        for w in self.server().worlds.worlds_in_domain(&domain) {
255            let mut level_data = w.level_data.write();
256            level_data.data_mut().difficulty = difficulty;
257            let locked = level_data.data().difficulty_locked;
258            drop(level_data);
259
260            w.broadcast_to_all(CChangeDifficulty { difficulty, locked });
261        }
262    }

265    ///

269        let is_creative = self.game_mode.load() == GameType::Creative;

279                },
280                false,
281            );
282            attrs.set_modifier(
283                vanilla_attributes::ENTITY_INTERACTION_RANGE,
284                AttributeModifier {
285                    id: Identifier::vanilla_static("creative_mode_entity_range"),
286                    amount: CREATIVE_ENTITY_RANGE_MODIFIER_AMOUNT,
287                    operation: AttributeModifierOperation::AddValue,
288                },
289                false,
290            );
291        } else {
292            attrs.remove_modifier(
293                vanilla_attributes::BLOCK_INTERACTION_RANGE,
294                &Identifier::vanilla_static("creative_mode_block_range"),
295            );
296            attrs.remove_modifier(

298                &Identifier::vanilla_static("creative_mode_entity_range"),

309    /// Acknowledges block changes up to the given sequence number.
310    ///
311    /// The ack is batched and sent once per tick (in `tick_ack_block_changes`),
312    /// matching vanilla behavior.
313    pub fn ack_block_changes_up_to(&self, sequence: i32) {
314        let current = self.ack_block_changes_up_to.load(Ordering::Relaxed);
315        if sequence > current {
316            self.ack_block_changes_up_to
317                .store(sequence, Ordering::Relaxed);
318        }
319    }

321    /// Sends pending block change ack if any. Called once per tick.
322    pub(super) fn tick_ack_block_changes(&self) {
323        let sequence = self.ack_block_changes_up_to.swap(-1, Ordering::Relaxed);
324        if sequence > -1 {
325            self.send_packet(CBlockChangedAck { sequence });
326        }
327    }

329    /// Returns true if player is within block interaction range.
330    ///
331    /// Uses eye position and AABB distance (nearest point on block surface),
332    /// matching vanilla's `Player.isWithinBlockInteractionRange(pos, 1.0)`.
333    #[must_use]
334    pub fn is_within_block_interaction_range(&self, pos: BlockPos) -> bool {
335        let player_pos = *self.position.lock();
336        let eye_y = player_pos.y + self.get_eye_height();
337
338        let min_x = f64::from(pos.x());
339        let min_y = f64::from(pos.y());
340        let min_z = f64::from(pos.z());
341        let max_x = min_x + 1.0;
342        let max_y = min_y + 1.0;
343        let max_z = min_z + 1.0;
344
345        let dx = f64::max(f64::max(min_x - player_pos.x, player_pos.x - max_x), 0.0);
346        let dy = f64::max(f64::max(min_y - eye_y, eye_y - max_y), 0.0);
347        let dz = f64::max(f64::max(min_z - player_pos.z, player_pos.z - max_z), 0.0);
348        let dist_sq = dx * dx + dy * dy + dz * dz;
349
350        let base_range = self
351            .attributes
352            .lock()
353            .get_value(vanilla_attributes::BLOCK_INTERACTION_RANGE)
354            .unwrap_or(4.5);
355        let max_range = base_range + 1.0;
356        dist_sq < max_range * max_range
357    }

397    /// Handles the use of an item on a block.
398    ///
399    /// Implements the logic from Java's `ServerGamePacketListenerImpl.handleUseItemOn()`.
400    pub fn handle_use_item_on(&self, packet: SUseItemOn) {
401        if !self.client_loaded.load(Ordering::Relaxed) {
402            return;
403        }
404
405        self.ack_block_changes_up_to(packet.sequence);
406
407        let pos = packet.block_hit.block_pos;
408        let direction = packet.block_hit.direction;
409
410        if !self.is_within_block_interaction_range(pos) {
411            self.send_block_updates(pos, direction);
412            return;
413        }
414
415        let center_x = f64::from(pos.x()) + 0.5;
416        let center_y = f64::from(pos.y()) + 0.5;
417        let center_z = f64::from(pos.z()) + 0.5;
418        let location = &packet.block_hit.location;
419        let limit = 1.000_000_1;
420
421        if (location.x - center_x).abs() >= limit
422            || (location.y - center_y).abs() >= limit
423            || (location.z - center_z).abs() >= limit
424        {
425            log::warn!(
426                "Rejecting UseItemOnPacket from {}: location {:?} too far from block {:?}",
427                self.gameprofile.name,
428                location,
429                pos
430            );
431            self.send_block_updates(pos, direction);
432            return;
433        }
434
435        let world = self.get_world();
436
437        if pos.y() >= world.max_build_height() {
438            // TODO: Send "build.tooHigh" message to player
439            self.send_block_updates(pos, direction);
440            return;
441        }
442
443        if self.is_awaiting_teleport() {
444            self.send_block_updates(pos, direction);
445            return;
446        }
447
448        if !world.may_interact(self, pos) {
449            self.send_block_updates(pos, direction);
450            return;
451        }
452
453        let result = use_item_on(self, &world, packet.hand, &packet.block_hit);
454
455        if let InteractionResult::Success = result {
456            // TODO: Trigger arm swing animation if needed
457            self.swing(packet.hand, true);
458        }
459
460        self.send_block_updates(pos, direction);
461        self.broadcast_inventory_changes();
462    }

464    /// Handles a player action packet (block breaking, item dropping, etc.).
465    pub fn handle_player_action(&self, packet: SPlayerAction) {
466        let world = self.get_world();
467        match packet.action {
468            PlayerAction::StartDestroyBlock => {
469                self.block_breaking.lock().handle_block_break_action(
470                    self,
471                    &world,
472                    packet.pos,
473                    BlockBreakAction::Start,
474                    packet.direction,
475                );
476                self.ack_block_changes_up_to(packet.sequence);
477            }
478            PlayerAction::StopDestroyBlock => {
479                self.block_breaking.lock().handle_block_break_action(
480                    self,
481                    &world,
482                    packet.pos,
483                    BlockBreakAction::Stop,
484                    packet.direction,
485                );
486                self.ack_block_changes_up_to(packet.sequence);
487            }
488            PlayerAction::AbortDestroyBlock => {
489                self.block_breaking.lock().handle_block_break_action(
490                    self,
491                    &world,
492                    packet.pos,
493                    BlockBreakAction::Abort,
494                    packet.direction,
495                );
496                self.ack_block_changes_up_to(packet.sequence);
497            }
498            PlayerAction::DropAllItems => {
499                self.drop_from_selected(true);
500            }
501            PlayerAction::DropItem => {
502                self.drop_from_selected(false);
503            }
504            PlayerAction::ReleaseUseItem => {
505                // TODO: Implement release use item (releasing bow, etc.)
506                log::debug!("Player {} released use item", self.gameprofile.name);
507            }
508            PlayerAction::SwapItemWithOffhand => {
509                // TODO: Implement swap item with offhand (F key)
510                log::debug!("Player {} wants to swap items", self.gameprofile.name);
511            }
512            PlayerAction::Stab => {
513                log::debug!("Player {} performed stab action", self.gameprofile.name);
514            }
515        }
516    }

541    /// Handles the pick block action (middle click on a block).
542    ///
543    /// # Panics
544    ///
545    /// Panics if the behavior registry has not been initialized.
546    pub fn handle_pick_item_from_block(&self, packet: SPickItemFromBlock) {
547        if !self.is_within_block_interaction_range(packet.pos) {
548            return;
549        }
550
551        let state = self.get_world().get_block_state(packet.pos);
552        if state.is_air() {
553            return;
554        }
555
556        let block = state.get_block();
557        let block_behaviors = &*BLOCK_BEHAVIORS;
558        let behavior = block_behaviors.get_behavior(block);
559
560        let include_data = self.has_infinite_materials() && packet.include_data;
561
562        let Some(item_stack) = behavior.get_clone_item_stack(block, state, include_data) else {
563            return;
564        };
565
566        if item_stack.is_empty() {
567            return;
568        }
569
570        // TODO: If include_data, add block entity NBT data to the item stack
571        // This requires block entity support which isn't implemented yet
572
573        let mut inventory = self.inventory.lock();
574
575        let slot_with_item = inventory.find_slot_matching_item(&item_stack);
576
577        if slot_with_item != -1 {
578            if PlayerInventory::is_hotbar_slot(slot_with_item as usize) {
579                inventory.set_selected_slot(slot_with_item as u8);
580            } else {
581                inventory.pick_slot(slot_with_item);
582            }
583        } else if self.has_infinite_materials() {
584            inventory.add_and_pick_item(item_stack);
585        } else {
586            return;
587        }
588
589        self.send_packet(CSetHeldSlot {
590            slot: i32::from(inventory.get_selected_slot()),
591        });
592
593        drop(inventory);
594        self.inventory_menu
595            .lock()
596            .behavior_mut()
597            .broadcast_changes(&self.connection);
598    }

600    /// Handles a sign update packet from the client.
601    pub fn handle_sign_update(&self, packet: SSignUpdate) {
602        if !self.is_within_block_interaction_range(packet.pos) {
603            return;
604        }
605
606        let world = self.get_world();
607
608        let Some(block_entity) = world.get_block_entity(packet.pos) else {
609            return;
610        };
611
612        let mut guard = block_entity.lock();
613        let Some(sign) = guard.as_any_mut().downcast_mut::<SignBlockEntity>() else {
614            return;
615        };
616
617        if sign.is_waxed {
618            return;
619        }
620
621        if sign.get_player_who_may_edit() != Some(self.gameprofile.id) {
622            log::warn!(
623                "Player {} tried to edit sign they're not allowed to edit",
624                self.gameprofile.name
625            );
626            return;
627        }
628
629        let text = sign.get_text_mut(packet.is_front_text);
630        for (i, line) in packet.lines.iter().enumerate() {
631            if i < 4 {
632                let stripped = strip_formatting_codes(line);
633                text.set_message(i, TextComponent::plain(stripped));
634            }
635        }
636
637        sign.set_player_who_may_edit(None);
638        sign.set_changed();
639
640        let update_tag = sign.get_update_tag();
641        let block_entity_type = sign.get_type();
642        let pos = packet.pos;
643
644        drop(guard);
645
646        if let Some(nbt) = update_tag {
647            world.broadcast_block_entity_update(pos, block_entity_type, nbt);
648        }
649    }

651    /// Opens the sign editor for the player.
652    ///
653    /// # Arguments
654    /// * `pos` - Position of the sign block
655    /// * `is_front_text` - Whether to edit front (true) or back (false) text
656    pub fn open_sign_editor(&self, pos: BlockPos, is_front_text: bool) {
657        let world = self.get_world();
658
659        if let Some(block_entity) = world.get_block_entity(pos) {
660            let mut guard = block_entity.lock();
661            if let Some(sign) = guard.as_any_mut().downcast_mut::<SignBlockEntity>() {
662                sign.set_player_who_may_edit(Some(self.gameprofile.id));
663            }
664        }
665
666        let state = world.get_block_state(pos);
667        self.send_packet(CBlockUpdate {
668            pos,
669            block_state: state,
670        });
671
672        self.send_packet(COpenSignEditor { pos, is_front_text });
673    }
674}

676/// Strips Minecraft formatting codes (§ followed by a character) from a string.
677///
678/// This is equivalent to vanilla's `ChatFormatting.stripFormatting()`.
679fn strip_formatting_codes(text: &str) -> String {
680    let mut result = String::with_capacity(text.len());
681    let mut chars = text.chars().peekable();
682
683    while let Some(c) = chars.next() {
684        if c == '§' {
685            chars.next();
686        } else {
687            result.push(c);
688        }
689    }
690
691    result
692}
