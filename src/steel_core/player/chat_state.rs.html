26/// All chat-related state for a player.
27///
28/// Stored behind a single `SyncMutex` on `Player`. The fields were previously
29/// individual atomics/mutexes but are always accessed within short critical
30/// sections per-player, so a single lock is simpler with no real contention cost.
31pub struct ChatState {
32    /// Counter for chat messages sent BY this player.
33    pub messages_sent: i32,
34    /// Counter for chat messages received BY this player.
35    pub messages_received: i32,
36    /// Message signature cache for tracking chat messages.
37    pub signature_cache: MessageCache,
38    /// Validator for client acknowledgements of messages we've sent.
39    pub message_validator: LastSeenMessagesValidator,
40    /// Remote chat session containing the player's public key (if signed chat is enabled).
41    pub chat_session: Option<RemoteChatSession>,
42    /// Message chain state for tracking signed message sequence.
43    pub message_chain: Option<SignedMessageChain>,
44}
45
46impl ChatState {
47    pub fn new() -> Self {
48        Self {
49            messages_sent: 0,
50            messages_received: 0,
51            signature_cache: MessageCache::new(),
52            message_validator: LastSeenMessagesValidator::new(),
53            chat_session: None,
54            message_chain: None,
55        }
56    }
57}

60    /// Gets the next `messages_received` counter and increments it
61    pub fn get_and_increment_messages_received(&self) -> i32 {
62        let mut chat = self.chat.lock();
63        let val = chat.messages_received;
64        chat.messages_received += 1;
65        val
66    }

68    fn verify_chat_signature(
69        &self,
70        packet: &SChat,
71    ) -> Result<(message_chain::SignedMessageLink, LastSeen), String> {
72        const MESSAGE_EXPIRES_AFTER: Duration = Duration::from_mins(5);
73
74        let mut chat = self.chat.lock();
75        let session = chat.chat_session.clone().ok_or("No chat session")?;
76        let signature = packet.signature.as_ref().ok_or("No signature present")?;
77
78        if session
79            .profile_public_key
80            .data()
81            .has_expired_with_grace(profile_key::EXPIRY_GRACE_PERIOD)
82        {
83            return Err("Profile key has expired".to_string());
84        }
85
86        let chain = chat.message_chain.as_mut().ok_or("No message chain")?;
87
88        if chain.is_broken() {
89            return Err("Message chain is broken".to_string());
90        }
91
92        let timestamp =
93            UNIX_EPOCH + Duration::from_millis(packet.timestamp.try_into().unwrap_or(0));
94
95        let now = SystemTime::now();
96        let message_age = now
97            .duration_since(timestamp)
98            .unwrap_or(Duration::from_secs(0));
99
100        if message_age > MESSAGE_EXPIRES_AFTER {
101            return Err(format!(
102                "Message expired (age: {}s, max: 300s)",
103                message_age.as_secs()
104            ));
105        }
106
107        let last_seen_signatures = chat
108            .message_validator
109            .apply_update(packet.acknowledged, packet.offset, packet.checksum)
110            .map_err(|e| {
111                log::error!("Message acknowledgment validation failed: {e}");
112                e
113            })?;
114
115        let last_seen = LastSeen::new(last_seen_signatures);
116
117        let body = message_chain::SignedMessageBody::new(
118            packet.message.clone(),
119            timestamp,
120            packet.salt,
121            last_seen,
122        );
123
124        let chain = chat.message_chain.as_mut().ok_or("No message chain")?;
125        let link = chain
126            .validate_and_advance(&body)
127            .map_err(|e| format!("Chain validation failed: {e}"))?;
128
129        let updater = message_chain::MessageSignatureUpdater::new(&link, &body);
130        let validator = session.profile_public_key.create_signature_validator();
131
132        let is_valid = SignatureValidator::validate(&validator, &updater, signature)
133            .map_err(|e| format!("Signature validation error: {e}"))?;
134
135        if is_valid {
136            Ok((link, body.last_seen.clone()))
137        } else {
138            Err("Invalid signature".to_string())
139        }
140    }

142    /// Handles a chat message from the player.
143    pub fn handle_chat(&self, packet: SChat, player: Arc<Player>) {
144        let chat_message = packet.message.clone();
145
146        let verification_result = if let Some(_signature) = &packet.signature {
147            match self.verify_chat_signature(&packet) {
148                Ok((link, last_seen)) => Some(Ok((link, last_seen))),
149                Err(err) => {
150                    log::warn!(
151                        "Player {} sent message with invalid signature: {err}",
152                        self.gameprofile.name
153                    );
154                    Some(Err(err))
155                }
156            }
157        } else {
158            None
159        };
160
161        if self.config.enforce_secure_chat {
162            match &verification_result {
163                Some(Ok(_)) => {}
164                Some(Err(err)) => {
165                    self.disconnect(format!("Chat message validation failed: {err}"));
166                    return;
167                }
168                None => {
169                    self.disconnect(
170                        "Secure chat is enforced on this server, but your message was not signed",
171                    );
172                    return;
173                }
174            }
175        }
176
177        let signature = if matches!(verification_result, Some(Ok(_))) {
178            packet.signature.map(|sig| Box::new(sig) as Box<[u8]>)
179        } else {
180            None
181        };
182
183        let sender_index = {
184            let mut chat = player.chat.lock();
185            let idx = chat.messages_sent;
186            chat.messages_sent += 1;
187            idx
188        };
189
190        let registry_id = vanilla_chat_types::CHAT.id() as i32;
191
192        let chat_packet = CPlayerChat::new(
193            0,
194            player.gameprofile.id,
195            sender_index,
196            signature.clone(),
197            chat_message.clone(),
198            packet.timestamp,
199            packet.salt,
200            Box::new([]),
201            Some(TextComponent::plain(chat_message.clone())),
202            FilterType::PassThrough,
203            ChatTypeBound {
204                registry_id,
205                sender_name: TextComponent::plain(player.gameprofile.name.clone())
206                    .insertion(player.gameprofile.name.clone())
207                    .click_event(ClickEvent::suggest_command(format!(
208                        "/tell {} ",
209                        player.gameprofile.name
210                    )))
211                    .hover_event(HoverEvent::show_entity(
212                        "minecraft:player",
213                        self.uuid(),
214                        Some(player.gameprofile.name.clone()),
215                    )),
216                target_name: None,
217            },
218        );
219
220        steel_utils::chat!(player.gameprofile.name.clone(), "{}", chat_message);
221        if let Some(sig_box) = &signature
222            && sig_box.len() == 256
223        {
224            let mut sig_array = [0u8; 256];
225            sig_array.copy_from_slice(&sig_box[..]);
226
227            let last_seen = if let Some(Ok((_, ref last_seen))) = verification_result {
228                last_seen.clone()
229            } else {
230                LastSeen::default()
231            };
232
233            for world in self.server().worlds.values() {
234                world.broadcast_chat(
235                    chat_packet.clone(),
236                    Arc::clone(&player),
237                    last_seen.clone(),
238                    Some(&sig_array),
239                );
240            }
241        } else {
242            for world in self.server().worlds.values() {
243                world.broadcast_unsigned_chat(chat_packet.clone());
244            }
245        }
246    }

253    /// Updates the player's chat session and initializes the message chain.
254    ///
255    /// This should be called when receiving a `ChatSessionUpdate` packet from the client.
256    pub fn set_chat_session(&self, session: RemoteChatSession) {
257        let chain = SignedMessageChain::new(self.gameprofile.id, session.session_id);
258
259        let session_data = session.as_data();
260        let protocol_data = match session_data.to_protocol_data() {
261            Ok(data) => data,
262            Err(err) => {
263                log::error!(
264                    "Failed to convert chat session to protocol data for {}: {:?}",
265                    self.gameprofile.name,
266                    err
267                );
268                let mut chat = self.chat.lock();
269                chat.chat_session = Some(session);
270                chat.message_chain = Some(chain);
271                return;
272            }
273        };
274
275        {
276            let mut chat = self.chat.lock();
277            chat.chat_session = Some(session);
278            chat.message_chain = Some(chain);
279        }
280
281        log::info!(
282            "Player {} initialized signed chat session",
283            self.gameprofile.name
284        );
285
286        let update_packet =
287            CPlayerInfoUpdate::update_chat_session(self.gameprofile.id, protocol_data);
288        self.get_world().broadcast_to_all(update_packet);
289    }

301    /// Handles a chat session update packet from the client.
302    ///
303    /// This validates the player's profile key and initializes signed chat if valid.
304    pub fn handle_chat_session_update(&self, packet: SChatSessionUpdate) {
305        log::info!("Player {} sent chat session update", self.gameprofile.name);
306
307        let expires_at = UNIX_EPOCH + Duration::from_millis(packet.expires_at as u64);
308
309        let public_key = match public_key_from_bytes(&packet.public_key) {
310            Ok(key) => key,
311            Err(err) => {
312                log::warn!(
313                    "Player {} sent invalid public key: {err}",
314                    self.gameprofile.name
315                );
316                if self.config.enforce_secure_chat {
317                    log::error!(
318                        "Player {} kicked for invalid public key",
319                        self.gameprofile.name
320                    );
321                    self.disconnect("Invalid profile public key");
322                }
323                return;
324            }
325        };
326
327        let profile_key_data =
328            profile_key::ProfilePublicKeyData::new(expires_at, public_key, packet.key_signature);
329
330        let validator = Box::new(NoValidation) as Box<dyn SignatureValidator>;
331
332        let session_data = profile_key::RemoteChatSessionData {
333            session_id: packet.session_id,
334            profile_public_key: profile_key_data,
335        };
336
337        match session_data.validate(self.gameprofile.id, &*validator) {
338            Ok(session) => {
339                self.set_chat_session(session);
340            }
341            Err(err) => {
342                log::warn!(
343                    "Player {} sent invalid chat session: {err}",
344                    self.gameprofile.name
345                );
346                if self.config.enforce_secure_chat {
347                    self.disconnect(format!("Chat session validation failed: {err}"));
348                }
349            }
350        }
351    }

353    /// Handles a chat acknowledgment packet from the client.
354    pub fn handle_chat_ack(&self, packet: SChatAck) {
355        if let Err(err) = self
356            .chat
357            .lock()
358            .message_validator
359            .apply_offset(packet.offset.0)
360        {
361            log::warn!(
362                "Player {} sent invalid chat acknowledgment: {err}",
363                self.gameprofile.name
364            );
365        }
366    }
