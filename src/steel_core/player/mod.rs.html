277        self.known_move_packet_count.store(

279            Ordering::Relaxed,
280        );
281
282        // Send pending block change acks (batched, once per tick like vanilla)
283        self.tick_ack_block_changes();
284
285        if !self.client_loaded.load(Ordering::Relaxed) {
286            //return;
287        }
288
289        let current_pos = *self.position.lock();
290        let chunk_x = (current_pos.x as i32) >> 4;
291        let chunk_z = (current_pos.z as i32) >> 4;

336    fn verify_chat_signature(
337        &self,
338        packet: &SChat,
339    ) -> Result<(message_chain::SignedMessageLink, LastSeen), String> {
340        const MESSAGE_EXPIRES_AFTER: Duration = Duration::from_secs(5 * 60);
341
342        let session = self.chat_session.lock().clone().ok_or("No chat session")?;
343        let signature = packet.signature.as_ref().ok_or("No signature present")?;
344
345        if session
346            .profile_public_key
347            .data()
348            .has_expired_with_grace(profile_key::EXPIRY_GRACE_PERIOD)
349        {
350            return Err("Profile key has expired".to_string());
351        }
352
353        let mut chain_guard = self.message_chain.lock();
354        let chain = chain_guard.as_mut().ok_or("No message chain")?;
355
356        if chain.is_broken() {
357            return Err("Message chain is broken".to_string());
358        }
359
360        let timestamp =
361            UNIX_EPOCH + Duration::from_millis(packet.timestamp.try_into().unwrap_or(0));
362
363        let now = SystemTime::now();
364        let message_age = now
365            .duration_since(timestamp)
366            .unwrap_or(Duration::from_secs(0));
367
368        if message_age > MESSAGE_EXPIRES_AFTER {
369            return Err(format!(
370                "Message expired (age: {}s, max: 300s)",
371                message_age.as_secs()
372            ));
373        }
374
375        let last_seen_signatures = self
376            .message_validator
377            .lock()
378            .apply_update(packet.acknowledged, packet.offset, packet.checksum)
379            .map_err(|e| {
380                log::error!("Message acknowledgment validation failed: {e}");
381                e
382            })?;
383
384        let last_seen = LastSeen::new(last_seen_signatures);
385
386        let body = message_chain::SignedMessageBody::new(
387            packet.message.clone(),
388            timestamp,
389            packet.salt,
390            last_seen,
391        );
392
393        let link = chain
394            .validate_and_advance(&body)
395            .map_err(|e| format!("Chain validation failed: {e}"))?;
396
397        let updater = message_chain::MessageSignatureUpdater::new(&link, &body);
398        let validator = session.profile_public_key.create_signature_validator();
399
400        let is_valid = SignatureValidator::validate(&validator, &updater, signature)
401            .map_err(|e| format!("Signature validation error: {e}"))?;
402
403        if is_valid {
404            Ok((link, body.last_seen.clone()))
405        } else {
406            Err("Invalid signature".to_string())
407        }
408    }

410    /// Handles a chat message from the player.
411    #[allow(clippy::too_many_lines)]
412    pub fn handle_chat(&self, packet: SChat, player: Arc<Player>) {
413        let chat_message = packet.message.clone();
414
415        let verification_result = if let Some(_signature) = &packet.signature {
416            match self.verify_chat_signature(&packet) {
417                Ok((link, last_seen)) => {
418                    // Don't add to cache here - will be added during broadcast
419                    // to avoid cache state mismatch with client
420                    Some(Ok((link, last_seen)))
421                }
422                Err(err) => {
423                    log::warn!(
424                        "Player {} sent message with invalid signature: {err}",
425                        self.gameprofile.name
426                    );
427                    Some(Err(err))
428                }
429            }
430        } else {
431            None
432        };
433
434        if STEEL_CONFIG.enforce_secure_chat {
435            match &verification_result {
436                Some(Ok(_)) => {}
437                Some(Err(err)) => {
438                    self.connection.disconnect(
439                        TextComponent::new().text(format!("Chat message validation failed: {err}")),
440                    );
441                    return;
442                }
443                None => {
444                    self.connection.disconnect(TextComponent::new().text(
445                        "Secure chat is enforced on this server, but your message was not signed",
446                    ));
447                    return;
448                }
449            }
450        }
451
452        let signature = if matches!(verification_result, Some(Ok(_))) {
453            packet.signature.map(|sig| Box::new(sig) as Box<[u8]>)
454        } else {
455            None
456        };
457
458        let sender_index = player.messages_sent.fetch_add(1, Ordering::SeqCst);
459
460        let chat_packet = CPlayerChat::new(
461            0,
462            player.gameprofile.id,
463            sender_index,
464            signature.clone(),
465            chat_message.clone(),
466            packet.timestamp,
467            packet.salt,
468            Box::new([]),
469            Some(TextComponent::new().text(chat_message.clone())),
470            FilterType::PassThrough,
471            ChatTypeBound {
472                //TODO: Use the registry to derive this instead of hardcoding it
473                registry_id: 0,
474                sender_name: TextComponent::new().text(player.gameprofile.name.clone()),
475                target_name: None,
476            },
477        );
478
479        if let Some(ref sig_box) = signature {
480            if sig_box.len() == 256 {
481                let mut sig_array = [0u8; 256];
482                sig_array.copy_from_slice(&sig_box[..]);
483
484                let last_seen = if let Some(Ok((_, ref last_seen))) = verification_result {
485                    last_seen.clone()
486                } else {
487                    LastSeen::default()
488                };
489
490                log::info!("<{}> {}", player.gameprofile.name, chat_message);
491                self.world.broadcast_chat(
492                    chat_packet,
493                    Arc::clone(&player),
494                    last_seen,
495                    Some(sig_array),
496                );
497            } else {
498                self.world.broadcast_unsigned_chat(
499                    chat_packet,
500                    &player.gameprofile.name,
501                    &chat_message,
502                );
503            }
504        } else {
505            self.world.broadcast_unsigned_chat(
506                chat_packet,
507                &player.gameprofile.name,
508                &chat_message,
509            );
510        }
511    }

544    /// Checks if the player would collide with any NEW blocks when moving to the new position.
545    ///
546    /// This allows movement when already stuck in blocks (e.g., sand fell on player).
547    /// Only rejects movement if it would cause collision with blocks the player
548    /// wasn't already colliding with at the old position.
549    ///
550    /// Matches vanilla `ServerGamePacketListenerImpl.isEntityCollidingWithAnythingNew()`.
551    #[allow(clippy::cast_possible_truncation)]
552    fn is_colliding_with_new_blocks(&self, old_pos: Vector3<f64>, new_pos: Vector3<f64>) -> bool {
553        // Old and new player AABBs (slightly deflated like vanilla does)
554        let old_aabb = Self::make_player_aabb(old_pos);
555        let new_aabb = Self::make_player_aabb(new_pos);
556
557        // Calculate the block positions that the new AABB could intersect
558        let min_x = new_aabb.min_x.floor() as i32;
559        let max_x = new_aabb.max_x.ceil() as i32;
560        let min_y = new_aabb.min_y.floor() as i32;
561        let max_y = new_aabb.max_y.ceil() as i32;
562        let min_z = new_aabb.min_z.floor() as i32;
563        let max_z = new_aabb.max_z.ceil() as i32;
564
565        // Check each block position
566        for bx in min_x..max_x {
567            for by in min_y..max_y {
568                for bz in min_z..max_z {
569                    let block_pos = BlockPos::new(bx, by, bz);
570                    let block_state = self.world.get_block_state(&block_pos);
571                    let collision_shape = block_state.get_collision_shape();
572
573                    // Check each AABB in the collision shape
574                    for aabb in collision_shape {
575                        // Convert block-local AABB to world coordinates
576                        let world_aabb = aabb.at_block(bx, by, bz);
577
578                        // Check if new position collides with this shape
579                        if !new_aabb.intersects_block_aabb(&world_aabb) {
580                            continue;
581                        }
582
583                        // Check if old position also collided with this shape
584                        // If new position collides but old didn't, this is a NEW collision
585                        if !old_aabb.intersects_block_aabb(&world_aabb) {
586                            return true;
587                        }
588                    }
589                }
590            }
591        }
592
593        false
594    }

596    /// Checks if we're awaiting a teleport confirmation and handles timeout/resend.
597    ///
598    /// Returns `true` if awaiting teleport (movement should be rejected),
599    /// `false` if normal movement processing should continue.
600    fn update_awaiting_teleport(&self) -> bool {
601        let awaiting = self.awaiting_position_from_client.lock();
602        if let Some(pos) = *awaiting {
603            let current_tick = self.tick_count.load(Ordering::Relaxed);
604            let last_time = self.awaiting_teleport_time.load(Ordering::Relaxed);
605
606            // Resend teleport after 20 ticks (~1 second) timeout
607            if current_tick.wrapping_sub(last_time) > 20 {
608                self.awaiting_teleport_time
609                    .store(current_tick, Ordering::Relaxed);
610                drop(awaiting);
611
612                // Resend the teleport packet
613                let (yaw, pitch) = self.rotation.load();
614                let teleport_id = self.awaiting_teleport_id.load(Ordering::Relaxed);
615                self.connection.send_packet(CPlayerPosition::absolute(
616                    teleport_id,
617                    pos.x,
618                    pos.y,
619                    pos.z,
620                    yaw,
621                    pitch,
622                ));
623            }
624            return true; // Still awaiting, reject movement
625        }
626
627        self.awaiting_teleport_time
628            .store(self.tick_count.load(Ordering::Relaxed), Ordering::Relaxed);
629        false
630    }

675    /// Checks if player is currently in old collision (already stuck in blocks).
676    /// Used by vanilla to allow movement when already stuck.
677    fn is_in_collision_at(&self, pos: Vector3<f64>) -> bool {
678        let aabb = Self::make_player_aabb(pos);
679
680        let min_x = aabb.min_x.floor() as i32;
681        let max_x = aabb.max_x.ceil() as i32;
682        let min_y = aabb.min_y.floor() as i32;
683        let max_y = aabb.max_y.ceil() as i32;
684        let min_z = aabb.min_z.floor() as i32;
685        let max_z = aabb.max_z.ceil() as i32;
686
687        for bx in min_x..max_x {
688            for by in min_y..max_y {
689                for bz in min_z..max_z {
690                    let block_pos = BlockPos::new(bx, by, bz);
691                    let block_state = self.world.get_block_state(&block_pos);
692                    let collision_shape = block_state.get_collision_shape();
693
694                    for block_aabb in collision_shape {
695                        let world_aabb = block_aabb.at_block(bx, by, bz);
696                        if aabb.intersects_block_aabb(&world_aabb) {
697                            return true;
698                        }
699                    }
700                }
701            }
702        }
703
704        false
705    }

713    /// Handles a move player packet.
714    ///
715    /// Matches vanilla `ServerGamePacketListenerImpl.handleMovePlayer()`.
716    #[allow(clippy::cast_lossless, clippy::too_many_lines, clippy::similar_names)]
717    pub fn handle_move_player(&self, packet: SMovePlayer) {
718        if Self::is_invalid_position(
719            packet.get_x(0.0),
720            packet.get_y(0.0),
721            packet.get_z(0.0),
722            packet.get_x_rot(0.0),
723            packet.get_y_rot(0.0),
724        ) {
725            self.connection
726                .disconnect(translations::MULTIPLAYER_DISCONNECT_INVALID_PLAYER_MOVEMENT.msg());
727            return;
728        }
729
730        // Check awaiting teleport - if so, only update rotation (vanilla: absSnapRotationTo)
731        if self.update_awaiting_teleport() {
732            // While awaiting teleport, still allow rotation updates
733            if packet.has_rot {
734                self.rotation.store((packet.y_rot, packet.x_rot));
735            }
736            return;
737        }
738
739        if !self.client_loaded.load(Ordering::Relaxed) {
740            return;
741        }
742
743        let prev_pos = *self.prev_position.lock();
744        let prev_rot = self.prev_rotation.load();
745        let start_pos = *self.position.lock();
746        let game_mode = self.game_mode.load();
747        let is_spectator = game_mode == GameType::Spectator;
748        let is_creative = game_mode == GameType::Creative;
749        let is_sleeping = self.sleeping.load(Ordering::Relaxed);
750        let is_fall_flying = self.fall_flying.load(Ordering::Relaxed);
751        let was_on_ground = self.on_ground.load(Ordering::Relaxed);
752        // Skip movement checks when tick rate is frozen (vanilla: tickRateManager().runsNormally())
753        let tick_frozen = !self.world.tick_runs_normally();
754
755        // Handle position updates
756        if packet.has_pos {
757            // Clamp position to vanilla limits
758            let target_pos = Vector3::new(
759                Self::clamp_horizontal(packet.position.x),
760                Self::clamp_vertical(packet.position.y),
761                Self::clamp_horizontal(packet.position.z),
762            );
763            let first_good = *self.first_good_position.lock();
764            let last_good = *self.last_good_position.lock();
765
766            // Sleeping check - only allow small movements when sleeping
767            if is_sleeping {
768                let dx = target_pos.x - first_good.x;
769                let dy = target_pos.y - first_good.y;
770                let dz = target_pos.z - first_good.z;
771                let moved_dist_sq = dx * dx + dy * dy + dz * dz;
772
773                if moved_dist_sq > 1.0 {
774                    let (yaw, pitch) = self.rotation.load();
775                    self.teleport(start_pos.x, start_pos.y, start_pos.z, yaw, pitch);
776                    return;
777                }
778            } else {
779                // Increment received packet count
780                self.received_move_packet_count
781                    .fetch_add(1, Ordering::Relaxed);
782
783                // Calculate delta packets since last tick (for rate limiting)
784                let received = self.received_move_packet_count.load(Ordering::Relaxed);
785                let known = self.known_move_packet_count.load(Ordering::Relaxed);
786                let mut delta_packets = received - known;
787
788                // Cap delta packets to prevent abuse (vanilla caps at 5)
789                if delta_packets > 5 {
790                    delta_packets = 1;
791                }
792
793                // Speed check: distance from first_good position
794                let dx = target_pos.x - first_good.x;
795                let dy = target_pos.y - first_good.y;
796                let dz = target_pos.z - first_good.z;
797                let moved_dist_sq = dx * dx + dy * dy + dz * dz;
798
799                // Skip checks for spectators, creative mode, or when tick is frozen
800                let skip_checks = is_spectator || is_creative || tick_frozen;
801
802                // Speed check (configurable)
803                if !skip_checks && STEEL_CONFIG.checks.speed {
804                    let expected_dist_sq = self.get_delta_movement_length_sq();
805                    let threshold = if is_fall_flying {
806                        Self::SPEED_THRESHOLD_FLYING
807                    } else {
808                        Self::SPEED_THRESHOLD_NORMAL
809                    } * (delta_packets as f64);
810
811                    if moved_dist_sq - expected_dist_sq > threshold {
812                        // Player moved too fast - teleport back to current position
813                        let (yaw, pitch) = self.rotation.load();
814                        self.teleport(start_pos.x, start_pos.y, start_pos.z, yaw, pitch);
815                        return;
816                    }
817                }
818
819                // Calculate movement delta from last_good position
820                let move_dx = target_pos.x - last_good.x;
821                let mut move_dy = target_pos.y - last_good.y;
822                let move_dz = target_pos.z - last_good.z;
823
824                // Y-axis tolerance: ignore small Y discrepancies (vanilla behavior)
825                if move_dy > -Self::Y_TOLERANCE && move_dy < Self::Y_TOLERANCE {
826                    move_dy = 0.0;
827                }
828
829                // Movement error check (configurable)
830                // Vanilla checks if (moved_dist_sq > 0.0625) after physics simulation
831                // Since we don't have full physics, we check the squared distance directly
832                let movement_dist_sq = move_dx * move_dx + move_dy * move_dy + move_dz * move_dz;
833                let error_check_failed = STEEL_CONFIG.checks.movement_error
834                    && !self.is_in_post_impulse_grace_time()
835                    && movement_dist_sq > Self::MOVEMENT_ERROR_THRESHOLD;
836
837                // Collision check (configurable)
838                // Vanilla only runs collision check if error was detected AND player was
839                // already in collision at old position (to allow movement when stuck)
840                let collision_check_failed = STEEL_CONFIG.checks.collision
841                    && error_check_failed
842                    && self.is_in_collision_at(last_good)
843                    && self.is_colliding_with_new_blocks(last_good, target_pos);
844
845                // Also check collision without error if movement > 0 and not in old collision
846                let new_collision_without_error = STEEL_CONFIG.checks.collision
847                    && !error_check_failed
848                    && self.is_colliding_with_new_blocks(last_good, target_pos);
849
850                // Check for movement errors and collisions
851                let movement_failed = !skip_checks
852                    && ((error_check_failed && !self.is_in_collision_at(last_good))
853                        || collision_check_failed
854                        || new_collision_without_error);
855
856                if movement_failed {
857                    // Teleport back to start position
858                    let (yaw, pitch) = prev_rot;
859                    self.teleport(start_pos.x, start_pos.y, start_pos.z, yaw, pitch);
860                    return;
861                }
862
863                // Movement accepted - update last good position
864                *self.last_good_position.lock() = target_pos;
865
866                // Update velocity based on actual movement
867                *self.delta_movement.lock() = Vector3::new(move_dx, move_dy, move_dz);
868
869                // Jump detection (vanilla: jumpFromGround)
870                let moved_upwards = move_dy > 0.0;
871                if was_on_ground && !packet.on_ground && moved_upwards {
872                    // Player jumped - could trigger jump-related mechanics here
873                    // For now, this is a placeholder for future jump handling
874                }
875            }
876        }
877
878        // Update on_ground state from packet
879        self.on_ground.store(packet.on_ground, Ordering::Relaxed);
880
881        // Update current state
882        if packet.has_pos {
883            *self.position.lock() = packet.position;
884        }
885        if packet.has_rot {
886            self.rotation.store((packet.y_rot, packet.x_rot));
887        }
888
889        // Broadcast movement to other players
890        let pos = if packet.has_pos {
891            packet.position
892        } else {
893            prev_pos
894        };
895        let (yaw, pitch) = if packet.has_rot {
896            (packet.y_rot, packet.x_rot)
897        } else {
898            prev_rot
899        };
900
901        if packet.has_pos || packet.has_rot {
902            let new_chunk = ChunkPos::new((pos.x as i32) >> 4, (pos.z as i32) >> 4);
903
904            // Note: player_area_map is updated in chunk_map.update_player_status
905            // which is called every tick and computes view diffs efficiently
906
907            if packet.has_pos {
908                let move_packet = CMoveEntityPosRot {
909                    entity_id: self.entity_id,
910                    dx: calc_delta(pos.x, prev_pos.x),
911                    dy: calc_delta(pos.y, prev_pos.y),
912                    dz: calc_delta(pos.z, prev_pos.z),
913                    y_rot: to_angle_byte(yaw),
914                    x_rot: to_angle_byte(pitch),
915                    on_ground: packet.on_ground,
916                };
917                self.world
918                    .broadcast_to_nearby(new_chunk, move_packet, Some(self.entity_id));
919            } else {
920                let rot_packet = CMoveEntityRot {
921                    entity_id: self.entity_id,
922                    y_rot: to_angle_byte(yaw),
923                    x_rot: to_angle_byte(pitch),
924                    on_ground: packet.on_ground,
925                };
926                self.world
927                    .broadcast_to_nearby(new_chunk, rot_packet, Some(self.entity_id));
928            }
929
930            if packet.has_rot {
931                let head_packet = CRotateHead {
932                    entity_id: self.entity_id,
933                    head_y_rot: to_angle_byte(yaw),
934                };
935                self.world
936                    .broadcast_to_nearby(new_chunk, head_packet, Some(self.entity_id));
937            }
938
939            *self.prev_position.lock() = pos;
940            self.prev_rotation.store((yaw, pitch));
941        }
942    }

1180        }

1363    ///
1364    /// Sends a `CPlayerPosition` packet and waits for client acknowledgment.
1365    /// Until acknowledged, movement packets from the client will be rejected.
1366    ///

1368    pub fn teleport(&self, x: f64, y: f64, z: f64, yaw: f32, pitch: f32) {
1369        let current_tick = self.tick_count.load(Ordering::Relaxed);
1370        self.awaiting_teleport_time
1371            .store(current_tick, Ordering::Relaxed);
1372
1373        // Pre-increment teleport ID, wrapping at i32::MAX (matches vanilla: ++awaitingTeleport)
1374        let new_id = self
1375            .awaiting_teleport_id
1376            .fetch_update(Ordering::Relaxed, Ordering::Relaxed, |id| {
1377                Some(if id == i32::MAX { 0 } else { id + 1 })
1378            })
1379            .map(|old| if old == i32::MAX { 0 } else { old + 1 })
1380            .unwrap_or(1);
1381
1382        // Update player position (vanilla: player.teleportSetPosition)
1383        *self.position.lock() = Vector3::new(x, y, z);
1384        self.rotation.store((yaw, pitch));
1385
1386        // Store the position we're waiting for confirmation of
1387        // (vanilla stores player.position() after teleportSetPosition)
1388        *self.awaiting_position_from_client.lock() = Some(Vector3::new(x, y, z));
1389
1390        // Send the teleport packet with the new ID
1391        self.connection
1392            .send_packet(CPlayerPosition::absolute(new_id, x, y, z, yaw, pitch));

1461    /// Handles the use of an item on a block.
1462    ///
1463    /// Implements the logic from Java's `ServerGamePacketListenerImpl.handleUseItemOn()`.
1464    pub fn handle_use_item_on(&self, packet: SUseItemOn) {
1465        // 1. Check client loaded
1466        if !self.client_loaded.load(Ordering::Relaxed) {
1467            return;
1468        }
1469
1470        // 2. Ack block changes
1471        self.ack_block_changes_up_to(packet.sequence);
1472
1473        let pos = &packet.block_hit.block_pos;
1474        let direction = packet.block_hit.direction;
1475
1476        // 3. Validate interaction range
1477        if !self.is_within_block_interaction_range(pos) {
1478            self.send_block_updates(pos, direction);
1479            return;
1480        }
1481
1482        // 4. Validate hit location precision (must be within 1.0000001 of block center)
1483        let center_x = f64::from(pos.x()) + 0.5;
1484        let center_y = f64::from(pos.y()) + 0.5;
1485        let center_z = f64::from(pos.z()) + 0.5;
1486        let location = &packet.block_hit.location;
1487        let limit = 1.000_000_1;
1488
1489        if (location.x - center_x).abs() >= limit
1490            || (location.y - center_y).abs() >= limit
1491            || (location.z - center_z).abs() >= limit
1492        {
1493            log::warn!(
1494                "Rejecting UseItemOnPacket from {}: location {:?} too far from block {:?}",
1495                self.gameprofile.name,
1496                location,
1497                pos
1498            );
1499            self.send_block_updates(pos, direction);
1500            return;
1501        }
1502
1503        // 5. Validate Y height
1504        if pos.y() >= self.world.max_build_height() {
1505            // TODO: Send "build.tooHigh" message to player
1506            self.send_block_updates(pos, direction);
1507            return;
1508        }
1509
1510        // 6. Check awaiting teleport
1511        if self.is_awaiting_teleport() {
1512            self.send_block_updates(pos, direction);
1513            return;
1514        }
1515
1516        // 7. Check may_interact permission
1517        if !self.world.may_interact(self, pos) {
1518            self.send_block_updates(pos, direction);
1519            return;
1520        }
1521
1522        // 8. Call use_item_on
1523        let result = game_mode::use_item_on(self, &self.world, packet.hand, &packet.block_hit);
1524
1525        // 9. Handle result
1526        if let InteractionResult::Success = result {
1527            // TODO: Trigger arm swing animation if needed
1528            self.swing(packet.hand, true);
1529        }
1530
1531        // 10. Always send block updates to resync client
1532        self.send_block_updates(pos, direction);
1533
1534        // 11. Broadcast inventory changes
1535        self.broadcast_inventory_changes();
1536    }

1613    ///
1614    /// # Panics
1615    ///
1616    /// Panics if the behavior registry has not been initialized.
1617    pub fn handle_pick_item_from_block(&self, packet: SPickItemFromBlock) {
1618        // Check if player is within interaction range (with 1.0 buffer like vanilla)
1619        if !self.is_within_block_interaction_range(&packet.pos) {
1620            return;
1621        }
1622
1623        // Get block state at position
1624        let state = self.world.get_block_state(&packet.pos);
1625        if state.is_air() {
1626            return;
1627        }
1628
1629        // Get the block and its behavior
1630        let block = state.get_block();
1631        let block_behaviors = BLOCK_BEHAVIORS.get().expect("Behaviors not initialized");
1632        let behavior = block_behaviors.get_behavior(block);
1633
1634        // Only include data if player has infinite materials (creative mode)
1635        let include_data = self.has_infinite_materials() && packet.include_data;
1636
1637        // Get clone item stack from behavior (handles blocks with different item keys)
1638        let Some(item_stack) = behavior.get_clone_item_stack(block, state, include_data) else {
1639            // No corresponding item for this block (e.g., fire, portal)
1640            return;
1641        };
1642
1643        if item_stack.is_empty() {
1644            return;
1645        }
1646
1647        // TODO: If include_data, add block entity NBT data to the item stack
1648        // This requires block entity support which isn't implemented yet
1649
1650        let mut inventory = self.inventory.lock();
1651
1652        // Find existing slot with this item
1653        let slot_with_item = inventory.find_slot_matching_item(&item_stack);
1654
1655        if slot_with_item != -1 {
1656            // Item found in inventory
1657            if PlayerInventory::is_hotbar_slot(slot_with_item as usize) {
1658                // Already in hotbar, just switch to that slot
1659                inventory.set_selected_slot(slot_with_item as u8);

1662                inventory.pick_slot(slot_with_item);

1674            slot: i32::from(inventory.get_selected_slot()),

1680            .lock()
