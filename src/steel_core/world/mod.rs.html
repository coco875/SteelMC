62/// Generates a random value using triangle distribution.
63///
64/// Mirrors vanilla's `RandomSource.triangle(mode, deviation)`.
65/// Produces values centered around `mode` with a spread of `deviation`.
66fn triangle_random(mode: f64, deviation: f64) -> f64 {
67    mode + deviation * (rand::random::<f64>() - rand::random::<f64>())
68}

115    /// Creates a new world with custom configuration.
116    ///
117    /// This allows specifying storage backend (disk or RAM-only) and other options.
118    /// Uses `Arc::new_cyclic` to create a cyclic reference between
119    /// the World and its `ChunkMap`'s `WorldGenContext`.
120    ///
121    /// # Arguments
122    /// * `chunk_runtime` - The Tokio runtime for chunk operations
123    /// * `dimension` - The dimension type (overworld, nether, end)
124    /// * `seed` - The world seed
125    /// * `config` - World configuration including storage options
126    pub async fn new_with_config(
127        chunk_runtime: Arc<Runtime>,
128        dimension: DimensionTypeRef,
129        seed: i64,
130        config: WorldConfig,
131    ) -> io::Result<Arc<Self>> {
132        // Create storage backend based on config
133        let storage: Arc<ChunkStorage> = match &config.storage {
134            WorldStorageConfig::Disk { path } => {
135                Arc::new(ChunkStorage::Disk(RegionManager::new(path.clone())))
136            }
137            WorldStorageConfig::RamOnly => {
138                Arc::new(ChunkStorage::RamOnly(RamOnlyStorage::empty_world()))
139            }
140        };
141
142        // Create or skip level data based on config
143
144        let path = match &config.storage {
145            WorldStorageConfig::Disk { path } => Some(Path::new(path)),
146            WorldStorageConfig::RamOnly => None,
147        };
148        let level_data = LevelDataManager::new(path, seed).await?;
149        // let generator = Arc::new(ChunkGeneratorType::Flat(FlatChunkGenerator::new(
150        //     REGISTRY
151        //         .blocks
152        //         .get_default_state_id(vanilla_blocks::BEDROCK), // Bedrock
153        //     REGISTRY.blocks.get_default_state_id(vanilla_blocks::DIRT), // Dirt
154        //     REGISTRY
155        //         .blocks
156        //         .get_default_state_id(vanilla_blocks::GRASS_BLOCK), // Grass Block
157        // )));
158
159        Ok(Arc::new_cyclic(|weak_self: &Weak<World>| Self {
160            chunk_map: Arc::new(ChunkMap::new_with_storage(
161                chunk_runtime,
162                weak_self.clone(),
163                &dimension,
164                storage,
165                config.generator,
166            )),
167            players: PlayerMap::new(),
168            player_area_map: PlayerAreaMap::new(),
169            dimension,
170            level_data: SyncRwLock::new(level_data),
171            tick_runs_normally: AtomicBool::new(true),
172            entity_cache: EntityCache::new(),
173            entity_tracker: EntityTracker::new(),
174        }))
175    }

246    /// Checks if a block's collision shape at the given position is unobstructed by entities.
247    ///
248    /// This is the Rust equivalent of vanilla's `Level.isUnobstructed(BlockState, BlockPos, CollisionContext)`.
249    /// In vanilla, this checks all entities with `blocksBuilding=true` (players, mobs, boats, etc.).
250    /// Currently only checks players since other entities aren't fully implemented.
251    ///
252    /// Returns `true` if the position is clear, `false` if an entity would obstruct placement.
253    #[must_use]
254    pub fn is_unobstructed(&self, collision_shape: VoxelShape, pos: &BlockPos) -> bool {
255        if collision_shape.is_empty() {
256            return true;
257        }
258
259        // TODO: Check other entities with blocksBuilding=true (mobs, boats, minecarts, etc.)
260        let mut obstructed = false;
261        self.players.iter_players(|_uuid, player| {
262            let player_pos = player.position.lock();
263            let half_width = Self::PLAYER_WIDTH / 2.0;
264            let player_aabb = AABBd::new(
265                player_pos.x - half_width,
266                player_pos.y,
267                player_pos.z - half_width,
268                player_pos.x + half_width,
269                player_pos.y + Self::PLAYER_HEIGHT,
270                player_pos.z + half_width,
271            );
272
273            // Check if any block AABB intersects with the player
274            for block_aabb in collision_shape {
275                let world_aabb = block_aabb.at_block(pos.x(), pos.y(), pos.z());
276                if player_aabb.intersects_block_aabb(&world_aabb) {
277                    obstructed = true;
278                    return false; // stop iteration
279                }
280            }
281
282            true // continue iteration
283        });
284
285        !obstructed
286    }

330    /// Gets the obfuscated seed for sending to clients.
331    ///
332    /// This uses SHA-256 hashing to prevent clients from easily extracting
333    /// the actual world seed, matching vanilla's `BiomeManager.obfuscateSeed()`.
334    #[must_use]
335    #[allow(clippy::missing_panics_doc)] // SHA-256 always produces 32 bytes
336    pub fn obfuscated_seed(&self) -> i64 {
337        let seed = self.seed();
338        let mut hasher = Sha256::new();
339        hasher.update(seed.to_be_bytes());
340        let result = hasher.finalize();
341        // SHA-256 always produces 32 bytes, so taking 8 bytes always succeeds
342        let bytes: [u8; 8] = result[0..8].try_into().expect("SHA-256 produces 32 bytes");
343        i64::from_be_bytes(bytes)
344    }

369    /// Sets a block at the given position with a custom update limit.
370    ///
371    /// The update limit prevents infinite recursion when shape updates trigger
372    /// further block changes. Each recursive call decrements the limit.
373    ///
374    /// Returns `true` if the block was successfully set, `false` otherwise.
375    pub fn set_block_with_limit(
376        &self,
377        pos: BlockPos,
378        block_state: BlockStateId,
379        flags: UpdateFlags,
380        update_limit: i32,
381    ) -> bool {
382        if update_limit <= 0 {
383            return false;
384        }
385
386        if !self.is_in_valid_bounds(&pos) {
387            return false;
388        }
389
390        let chunk_pos = Self::chunk_pos_for_block(&pos);
391        let Some(old_state) = self
392            .chunk_map
393            .with_full_chunk(&chunk_pos, |chunk| {
394                chunk.set_block_state(pos, block_state, flags)
395            })
396            .flatten()
397        else {
398            return false;
399        };
400
401        // Record the block change for broadcasting to clients
402        log::debug!("Block changed at {pos:?}: {old_state:?} -> {block_state:?}");
403        self.chunk_map.block_changed(&pos);
404
405        // Neighbor updates (when UPDATE_NEIGHBORS is set)
406        if flags.contains(UpdateFlags::UPDATE_NEIGHBORS) {
407            self.update_neighbors_at(&pos, old_state.get_block());
408            // TODO: if block has analog output signal, update comparator neighbors
409            // via updateNeighbourForOutputSignal
410        }
411
412        // Shape updates (unless UPDATE_KNOWN_SHAPE is set)
413        if !flags.contains(UpdateFlags::UPDATE_KNOWN_SHAPE) && update_limit > 0 {
414            // Clear UPDATE_NEIGHBORS and UPDATE_SUPPRESS_DROPS for propagation
415            let neighbor_flags =
416                flags & !(UpdateFlags::UPDATE_NEIGHBORS | UpdateFlags::UPDATE_SUPPRESS_DROPS);
417
418            // Notify all 6 neighbors about our shape change
419            for direction in Direction::UPDATE_SHAPE_ORDER {
420                let neighbor_pos = pos.relative(direction);
421
422                // Tell the neighbor that we (at pos) changed
423                self.neighbor_shape_changed(
424                    direction.opposite(), // Direction from us to neighbor
425                    neighbor_pos,         // Neighbor's position
426                    pos,                  // Our position (the one that changed)
427                    block_state,          // Our new state
428                    neighbor_flags,
429                    update_limit - 1,
430                );
431            }
432        }
433
434        true
435    }

437    /// Order in which neighbors are updated (matches vanilla's `NeighborUpdater.UPDATE_ORDER`).
438    const NEIGHBOR_UPDATE_ORDER: [Direction; 6] = [
439        Direction::West,
440        Direction::East,
441        Direction::Down,
442        Direction::Up,
443        Direction::North,
444        Direction::South,
445    ];

447    /// Updates all neighbors of the given position about a block change.
448    ///
449    /// This is the Rust equivalent of vanilla's `Level.updateNeighborsAt()`.
450    fn update_neighbors_at(&self, pos: &BlockPos, source_block: BlockRef) {
451        for direction in Self::NEIGHBOR_UPDATE_ORDER {
452            let neighbor_pos = pos.relative(direction);
453            self.neighbor_changed(neighbor_pos, source_block, false);
454        }
455    }

457    /// Called when a neighbor's shape changes, to update this block's state.
458    ///
459    /// This is the Rust equivalent of vanilla's `NeighborUpdater.executeShapeUpdate()`.
460    fn neighbor_shape_changed(
461        &self,
462        direction: Direction,
463        pos: BlockPos,
464        neighbor_pos: BlockPos,
465        neighbor_state: BlockStateId,
466        flags: UpdateFlags,
467        update_limit: i32,
468    ) {
469        if !self.is_in_valid_bounds(&pos) {
470            return;
471        }
472
473        let current_state = self.get_block_state(&pos);
474
475        // TODO: Skip redstone wire if UPDATE_SKIP_SHAPE_UPDATE_ON_WIRE is set
476        // if flags.contains(UpdateFlags::UPDATE_SKIP_SHAPE_UPDATE_ON_WIRE)
477        //     && current_state.is_redstone_wire() { return; }
478
479        let block_behaviors = &*BLOCK_BEHAVIORS;
480        let behavior = block_behaviors.get_behavior(current_state.get_block());
481        let new_state = behavior.update_shape(
482            current_state,
483            self,
484            pos,
485            direction,
486            neighbor_pos,
487            neighbor_state,
488        );
489
490        if new_state != current_state {
491            log::debug!(
492                "Shape update at {pos:?}: {current_state:?} -> {new_state:?} (neighbor {neighbor_pos:?} changed)"
493            );
494            // Use set_block_with_limit to prevent infinite recursion
495            self.set_block_with_limit(pos, new_state, flags, update_limit);
496        }
497    }

499    /// Notifies a block that one of its neighbors changed.
500    ///
501    /// This is the Rust equivalent of vanilla's `Level.neighborChanged()`.
502    fn neighbor_changed(&self, pos: BlockPos, source_block: BlockRef, moved_by_piston: bool) {
503        if !self.is_in_valid_bounds(&pos) {
504            return;
505        }
506
507        let state = self.get_block_state(&pos);
508        let block_behaviors = &*BLOCK_BEHAVIORS;
509        let behavior = block_behaviors.get_behavior(state.get_block());
510        behavior.handle_neighbor_changed(state, self, pos, source_block, moved_by_piston);
511    }

552    /// Ticks the world.
553    ///
554    /// * `tick_count` - The current tick number
555    /// * `runs_normally` - Whether game elements (random ticks, entities) should run.
556    ///   When false (frozen), only essential operations like chunk loading run.
557    ///
558    /// Returns timing information for the world tick.
559    #[tracing::instrument(level = "trace", skip(self), name = "world_tick")]
560    pub fn tick_b(&self, tick_count: u64, runs_normally: bool) -> WorldTickTimings {
561        if runs_normally {
562            self.tick_time();
563        }
564
565        let random_tick_speed = self.get_game_rule(RANDOM_TICK_SPEED).as_int().unwrap_or(3) as u32;
566
567        let chunk_map_timings = self
568            .chunk_map
569            .tick_b(tick_count, random_tick_speed, runs_normally);
570
571        // Tick players (always tick players - they can move when frozen)
572        let player_tick = {
573            let _span = tracing::trace_span!("player_tick").entered();
574            let start = Instant::now();
575            self.players.iter_players(|_uuid, player| {
576                player.tick();
577                true
578            });
579            start.elapsed()
580        };
581
582        // Broadcast player latency updates periodically
583        if tick_count.is_multiple_of(SEND_PLAYER_INFO_INTERVAL) {
584            let _span = tracing::trace_span!("broadcast_latency").entered();
585            self.broadcast_player_latency_updates();
586        }
587
588        WorldTickTimings {
589            chunk_map: chunk_map_timings,
590            player_tick,
591        }
592    }

594    /// Advances the gametime and the daytime (if `ADVANCE_TIME` gamerule is true) by one tick, and
595    /// then sends an update to all clients in this world every 20th tick.
596    fn tick_time(&self) {
597        let advance_time = self
598            .get_game_rule(ADVANCE_TIME)
599            .as_bool()
600            .expect("gamerule advance_time should always be a bool.");
601
602        let (game_time, day_time) = {
603            let mut lock = self.level_data.write();
604            let updated_game_time = lock.game_time() + 1;
605            lock.set_game_time(updated_game_time);
606            let current_day_time = lock.day_time();
607
608            if advance_time {
609                let updated_day_time = (current_day_time + 1) % 24000;
610                lock.set_day_time(updated_day_time);
611                (updated_game_time, updated_day_time)
612            } else {
613                (updated_game_time, current_day_time)
614            }
615        };
616
617        if game_time % 20 == 0 {
618            self.broadcast_to_all(CSetTime {
619                game_time,
620                day_time,
621                time_of_day_increasing: advance_time,
622            });
623        }
624    }

626    /// Broadcasts latency updates for all players to all players.
627    /// This is called every `SEND_PLAYER_INFO_INTERVAL` ticks to update the ping display.
628    fn broadcast_player_latency_updates(&self) {
629        // Collect all player latencies
630        let mut latency_entries = Vec::new();
631        self.players.iter_players(|uuid, player| {
632            latency_entries.push((*uuid, player.connection.latency()));
633            true
634        });
635
636        // Only broadcast if there are players
637        if !latency_entries.is_empty() {
638            let packet = CPlayerInfoUpdate::update_latency(latency_entries);
639            self.broadcast_to_all(packet);
640        }

643    /// Broadcasts a signed chat message to all players in the world.
644    ///
645    /// # Panics
646    /// Panics if `message_signature` is `None` after checking `is_some()` (should never happen).
647    pub fn broadcast_chat(
648        &self,
649        mut packet: CPlayerChat,
650        _sender: Arc<Player>,
651        sender_last_seen: LastSeen,
652        message_signature: Option<[u8; 256]>,
653    ) {
654        log::debug!(
655            "broadcast_chat: sender_last_seen has {} signatures, message_signature present: {}",
656            sender_last_seen.len(),
657            message_signature.is_some()
658        );
659
660        self.players.iter_players(|_, recipient| {
661            let messages_received = recipient.get_and_increment_messages_received();
662            packet.global_index = messages_received;
663
664            log::debug!(
665                "Broadcasting to player {} (UUID: {}), global_index={}",
666                recipient.gameprofile.name,
667                recipient.gameprofile.id,
668                messages_received
669            );
670
671            // IMPORTANT: Index previous messages BEFORE updating the cache
672            // This matches vanilla's order: pack() then push()
673            let previous_messages = {
674                let recipient_cache = recipient.signature_cache.lock();
675                recipient_cache.index_previous_messages(&sender_last_seen)
676            };
677
678            log::debug!(
679                "  Indexed {} previous messages for recipient",
680                previous_messages.len()
681            );
682
683            packet.previous_messages.clone_from(&previous_messages);
684
685            // Send the packet
686            recipient.send_packet(packet.clone());
687
688            // AFTER sending, update the recipient's cache using vanilla's push algorithm
689            // This adds all lastSeen signatures + current signature to the cache
690            if let Some(signature) = message_signature {
691                recipient
692                    .signature_cache
693                    .lock()
694                    .push(&sender_last_seen, Some(&signature));
695
696                log::debug!("  Added signature to recipient's cache and pending list");
697
698                // Add to pending messages for acknowledgment tracking
699                recipient
700                    .message_validator
701                    .lock()
702                    .add_pending(Some(Box::new(signature) as Box<[u8]>));
703            } else {
704                // Even unsigned messages update the pending tracker
705                recipient.message_validator.lock().add_pending(None);
706                log::debug!("  Added unsigned message to pending list");
707            }
708
709            true
710        });
711    }

879    /// Drops an item stack at the given position with scatter behavior.
880    ///
881    /// Mirrors vanilla's `Containers.dropItemStack`. Splits large stacks into
882    /// multiple item entities (10-30 items each) and scatters them with random
883    /// positions and velocities.
884    ///
885    /// # Arguments
886    /// * `pos` - The block position to drop the item at
887    /// * `item` - The item stack to drop
888    pub fn drop_item_stack(self: &Arc<Self>, pos: BlockPos, mut item: ItemStack) {
889        use crate::entity::next_entity_id;
890        use steel_registry::vanilla_entities;
891
892        // Random velocity using triangle distribution (vanilla uses random.triangle)
893        // Vanilla constant: 0.05F * Mth.SQRT_OF_TWO (sqrt(2) * 0.05  0.1148...)
894        const VELOCITY_SPREAD: f64 = 0.114_850_001_711_398_36;
895
896        if item.is_empty() {
897            return;
898        }
899
900        // Vanilla uses EntityType.ITEM dimensions for position calculation
901        let item_width = f64::from(vanilla_entities::ITEM.dimensions.width);
902        let center_range = 1.0 - item_width;
903        let half_size = item_width / 2.0;
904
905        // Keep spawning item entities until the stack is empty
906        // Vanilla splits stacks into 10-30 items each
907        while !item.is_empty() {
908            // Split off 10-30 items (or remaining if less)
909            let split_count = (rand::random::<u32>() % 21 + 10) as i32;
910            let split_stack = item.split(split_count);
911
912            if split_stack.is_empty() {
913                break;
914            }
915
916            // Random position within the block (vanilla logic)
917            let x = f64::from(pos.x()).floor() + rand::random::<f64>() * center_range + half_size;
918            let y = f64::from(pos.y()).floor() + rand::random::<f64>() * center_range;
919            let z = f64::from(pos.z()).floor() + rand::random::<f64>() * center_range + half_size;
920
921            // triangle(mode, deviation) produces values centered around mode with spread of deviation
922            let vx = triangle_random(0.0, VELOCITY_SPREAD);
923            let vy = triangle_random(0.2, VELOCITY_SPREAD);
924            let vz = triangle_random(0.0, VELOCITY_SPREAD);
925
926            let entity_id = next_entity_id();
927            let entity = Arc::new(ItemEntity::with_item_and_velocity(
928                entity_id,
929                Vector3::new(x, y, z),
930                split_stack,
931                Vector3::new(vx, vy, vz),
932                Arc::downgrade(self),
933            ));
934            entity.set_default_pickup_delay();
935            self.add_entity(entity);
936        }
937    }

939    /// Broadcasts a level event to nearby players within 64 blocks.
940    ///
941    /// Level events trigger sounds, particles, and animations on the client.
942    /// See `steel_registry::level_events` for available event type constants.
943    ///
944    /// # Arguments
945    /// * `event_type` - The event type ID from `steel_registry::level_events`
946    /// * `pos` - The position where the event occurs
947    /// * `data` - Event-specific data (e.g., block state ID for block destruction)
948    /// * `exclude` - Optional entity ID to exclude from receiving the event
949    pub fn level_event(&self, event_type: i32, pos: BlockPos, data: i32, exclude: Option<i32>) {
950        const MAX_DISTANCE_SQ: f64 = 64.0 * 64.0;
951
952        let chunk = ChunkPos::new(
953            SectionPos::block_to_section_coord(pos.x()),
954            SectionPos::block_to_section_coord(pos.z()),
955        );
956        let packet = CLevelEvent::new(event_type, pos, data, false);
957        let Ok(encoded) =
958            EncodedPacket::from_bare(packet, STEEL_CONFIG.compression, ConnectionProtocol::Play)
959        else {
960            log::warn!("Failed to encode level event packet");
961            return;
962        };
963
964        // Get players tracking this chunk, then filter by 64-block distance
965        let event_pos = (
966            f64::from(pos.x()) + 0.5,
967            f64::from(pos.y()) + 0.5,
968            f64::from(pos.z()) + 0.5,
969        );
970
971        for entity_id in self.player_area_map.get_tracking_players(chunk) {
972            // Skip excluded player (they hear the effect client-side)
973            if exclude == Some(entity_id) {
974                continue;
975            }
976            if let Some(player) = self.players.get_by_entity_id(entity_id) {
977                let player_pos = *player.position.lock();
978                let dx = player_pos.x - event_pos.0;
979                let dy = player_pos.y - event_pos.1;
980                let dz = player_pos.z - event_pos.2;
981                let dist_sq = dx * dx + dy * dy + dz * dz;
982
983                if dist_sq <= MAX_DISTANCE_SQ {
984                    player.connection.send_encoded(encoded.clone());
985                }
986            }
987        }
988    }

1261
1262        // Vanilla uses EntityType.ITEM dimensions for offset calculation
1263        let half_height = f64::from(vanilla_entities::ITEM.dimensions.height) / 2.0;
1264
1265        // Random offset within block (vanilla: nextDouble(-0.25, 0.25))
1266        let x = f64::from(pos.x()) + 0.5 + (rand::random::<f64>() - 0.5) * 0.5;
1267        let y = f64::from(pos.y()) + 0.5 + (rand::random::<f64>() - 0.5) * 0.5 - half_height;
1268        let z = f64::from(pos.z()) + 0.5 + (rand::random::<f64>() - 0.5) * 0.5;
1269
1270        self.spawn_item(Vector3::new(x, y, z), item)
1271    }

1273    /// Drops an item from a block face with directional velocity.
1274    ///
1275    /// Mirrors vanilla's `Block.popResourceFromFace()`. Used for items ejected
1276    /// from a specific side of a block.
1277    pub fn pop_resource_from_face(
1278        self: &Arc<Self>,
1279        pos: &BlockPos,
1280        face: Direction,
1281        item: ItemStack,
1282    ) -> Option<Arc<ItemEntity>> {
1283        use steel_registry::vanilla_entities;
1284
1285        if item.is_empty() {
1286            return None;
1287        }
1288
1289        let half_width = f64::from(vanilla_entities::ITEM.dimensions.width) / 2.0;
1290        let half_height = f64::from(vanilla_entities::ITEM.dimensions.height) / 2.0;
1291
1292        let (step_x, step_y, step_z) = face.offset();
1293
1294        // Position calculation (vanilla logic)
1295        let x = f64::from(pos.x())
1296            + 0.5
1297            + if step_x == 0 {
1298                (rand::random::<f64>() - 0.5) * 0.5
1299            } else {
1300                f64::from(step_x) * (0.5 + half_width)
1301            };
1302        let y = f64::from(pos.y())
1303            + 0.5
1304            + if step_y == 0 {
1305                (rand::random::<f64>() - 0.5) * 0.5
1306            } else {
1307                f64::from(step_y) * (0.5 + half_height)
1308            }
1309            - half_height;
1310        let z = f64::from(pos.z())
1311            + 0.5
1312            + if step_z == 0 {
1313                (rand::random::<f64>() - 0.5) * 0.5
1314            } else {
1315                f64::from(step_z) * (0.5 + half_width)
1316            };
1317
1318        // Velocity in direction of face
1319        let delta_x = if step_x == 0 {
1320            (rand::random::<f64>() - 0.5) * 0.2
1321        } else {
1322            f64::from(step_x) * 0.1
1323        };
1324        let delta_y = if step_y == 0 {
1325            rand::random::<f64>() * 0.1
1326        } else {
1327            f64::from(step_y) * 0.1 + 0.1
1328        };
1329        let delta_z = if step_z == 0 {
1330            (rand::random::<f64>() - 0.5) * 0.2
1331        } else {
1332            f64::from(step_z) * 0.1
1333        };
1334
1335        self.spawn_item_with_velocity(
1336            Vector3::new(x, y, z),
1337            item,
1338            Vector3::new(delta_x, delta_y, delta_z),
1339        )
1340    }

1366    /// Moves an entity's Arc between chunks when it crosses a chunk boundary.
1367    ///
1368    /// Called by `EntityChunkCallback` when an entity moves between chunks.
1369    pub fn move_entity_between_chunks(&self, entity_id: i32, from: ChunkPos, to: ChunkPos) {
1370        // Remove Arc from old chunk
1371        let entity = self
1372            .chunk_map
1373            .with_full_chunk(&from, |chunk| {
1374                chunk.as_full().and_then(|c| c.entities.remove(entity_id))
1375            })
1376            .flatten();
1377
1378        // Add Arc to new chunk
1379        if let Some(entity) = entity {
1380            self.chunk_map.with_full_chunk(&to, |chunk| {
1381                if let Some(c) = chunk.as_full() {
1382                    c.entities.add(entity);
1383                }
1384            });
1385        }
1386    }

1388    /// Internal method to remove an entity from the world.
1389    ///
1390    /// Called by `EntityChunkCallback::on_remove`.
1391    pub fn remove_entity_internal(
1392        &self,
1393        entity_id: i32,
1394        chunk_pos: ChunkPos,
1395        reason: RemovalReason,
1396    ) {
1397        // Remove from chunk storage
1398        let entity: Option<SharedEntity> = self
1399            .chunk_map
1400            .with_full_chunk(&chunk_pos, |chunk| {
1401                chunk.as_full().and_then(|c| c.entities.remove(entity_id))
1402            })
1403            .flatten();
1404
1405        // Unregister from cache
1406        if let Some(entity) = entity {
1407            let pos = entity.position();
1408            let section = SectionPos::new(
1409                (pos.x as i32) >> 4,
1410                (pos.y as i32) >> 4,
1411                (pos.z as i32) >> 4,
1412            );
1413            self.entity_cache
1414                .unregister(entity_id, entity.uuid(), section);
1415
1416            // Broadcast remove packet if entity was destroyed
1417            if reason.should_destroy() {
1418                let packet = CRemoveEntities::single(entity_id);
1419                self.broadcast_to_nearby(chunk_pos, packet, None);
1420            }
1421        }
1422    }
