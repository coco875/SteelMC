19    /// Removes a player from the world.
20    pub async fn remove_player(self: &Arc<Self>, player: Arc<Player>) {
21        let uuid = player.gameprofile.id;
22        let entity_id = player.id;
23
24        if self.players.remove(&uuid).await.is_none() {
25            return;
26        }
27
28        let start = Instant::now();
29
30        // Save player data before removal
31        let server = player.server();
32        if let Err(e) = server.player_data_storage.save(&player).await {
33            log::error!("Failed to save player data for {uuid}: {e}");
34        }
35
36        // Unregister from entity cache
37        let pos = player.position();
38        let section = SectionPos::from_entity_pos(pos);
39        self.entity_cache.unregister(entity_id, uuid, section);
40
41        // Remove player from entity tracking (stop tracking all entities for this player)
42        self.entity_tracker().on_player_leave(entity_id);
43
44        self.player_area_map.on_player_leave(&player);
45        self.broadcast_to_all(CRemoveEntities::single(entity_id));
46        self.broadcast_to_all(CRemovePlayerInfo::single(uuid));
47
48        self.chunk_map.remove_player(&player);
49        player.cleanup();
50        log::info!("Player {uuid} removed in {:?}", start.elapsed());
51    }

53    /// Removes a player from the world during a world change.
54    ///
55    /// Unlike `remove_player`, this is synchronous and skips player data saving and tab list
56    /// removal — the player stays in the global tab list since they are only switching worlds.
57    pub fn remove_player_for_world_change(self: &Arc<Self>, player: &Arc<Player>) {
58        let uuid = player.gameprofile.id;
59        let entity_id = player.id;
60
61        if self.players.remove_sync(&uuid).is_none() {
62            return;
63        }
64
65        let pos = player.position();
66        let section = SectionPos::from_entity_pos(pos);
67        self.entity_cache.unregister(entity_id, uuid, section);
68        self.entity_tracker().on_player_leave(entity_id);
69        self.player_area_map.on_player_leave(player);
70        self.broadcast_to_all(CRemoveEntities::single(entity_id));
71        // Note: no CRemovePlayerInfo — player stays in the global tab list
72        self.chunk_map.remove_player(player);
73    }

75    /// Adds a player to the world.
76    ///
77    /// On `InitialJoin`, sends full tab list + entity spawn synchronization to/from all
78    /// players. On `WorldChange`, this is skipped — the player already exists in all
79    /// clients' tab lists and the entity tracker handles spawning as chunks load.
80    pub fn add_player(self: &Arc<Self>, player: Arc<Player>, reason: ResetReason) {
81        if !self.players.insert(player.clone()) {
82            player.connection.close();
83            return;
84        }
85
86        // Set up level callback for section tracking
87        let pos = player.position();
88        let callback = Arc::new(PlayerEntityCallback::new(
89            player.id,
90            pos,
91            Arc::downgrade(self),
92        ));
93        player.set_level_callback(callback);
94
95        // Register player in entity cache for unified entity lookups
96        self.entity_cache
97            .register(&(player.clone() as SharedEntity));
98
99        // Note: player_area_map.on_player_join is called in chunk_map.update_player_status
100        // when the player's view is first computed
101
102        // Tab list + entity spawn sync only needed on first join
103        if reason == ResetReason::InitialJoin {
104            self.sync_tab_list_and_entities(&player);
105        }
106
107        player.send_packet(CGameEvent {
108            event: GameEventType::LevelChunksLoadStart,
109            data: 0.0,
110        });
111
112        player.send_packet(CGameEvent {
113            event: GameEventType::ChangeGameMode,
114            data: player.game_mode.load().into(),
115        });
116    }

118    /// Sends full tab list + entity spawn synchronization for a newly joined player.
119    ///
120    /// Sends all existing players' info to the new player, and broadcasts the new
121    /// player's info + entity spawn to all existing players.
122    fn sync_tab_list_and_entities(self: &Arc<Self>, player: &Arc<Player>) {
123        let pos = *player.position.lock();
124        let (yaw, pitch) = player.rotation.load();
125
126        // Send existing players to the new player (tab list + entity spawn)
127        self.players.iter_players(|_, existing_player| {
128            if existing_player.gameprofile.id == player.gameprofile.id {
129                return true;
130            }
131
132            // Add to tab list with full player info
133            let add_existing = CPlayerInfoUpdate::create_player_initializing(
134                existing_player.gameprofile.id,
135                existing_player.gameprofile.name.clone(),
136                existing_player.gameprofile.properties.clone(),
137                existing_player.game_mode.load().into(),
138                existing_player.connection.latency(),
139                None, // display_name
140                true, // show_hat
141            );
142            player.send_packet(add_existing);
143
144            // Send chat session if available
145            if let Some(session) = existing_player.chat_session()
146                && let Ok(protocol_data) = session.as_data().to_protocol_data()
147            {
148                let session_packet = CPlayerInfoUpdate::update_chat_session(
149                    existing_player.gameprofile.id,
150                    protocol_data,
151                );
152                player.send_packet(session_packet);
153            }
154
155            // Spawn existing player entity for new player (bundled for atomic processing)
156            let existing_pos = *existing_player.position.lock();
157            let (existing_yaw, existing_pitch) = existing_player.rotation.load();
158            let player_type_id = vanilla_entities::PLAYER.id() as i32;
159            player.send_bundle(|bundle| {
160                bundle.add(CAddEntity::player(
161                    existing_player.id,
162                    existing_player.gameprofile.id,
163                    player_type_id,
164                    existing_pos.x,
165                    existing_pos.y,
166                    existing_pos.z,
167                    existing_yaw,
168                    existing_pitch,
169                ));
170                // TODO: Add entity metadata and equipment packets here when implemented
171            });
172
173            true
174        });
175
176        // Broadcast new player's tab list entry to all players
177        let player_info_packet = CPlayerInfoUpdate::create_player_initializing(
178            player.gameprofile.id,
179            player.gameprofile.name.clone(),
180            player.gameprofile.properties.clone(),
181            player.game_mode.load().into(),
182            player.connection.latency(),
183            None, // display_name
184            true, // show_hat
185        );
186        self.broadcast_to_all(player_info_packet);
187
188        // Spawn new player entity for all other players
189        // TODO: bundle with entity metadata + equipment packets when implemented
190        let player_type_id = vanilla_entities::PLAYER.id() as i32;
191        let spawn_packet = CAddEntity::player(
192            player.id,
193            player.gameprofile.id,
194            player_type_id,
195            pos.x,
196            pos.y,
197            pos.z,
198            yaw,
199            pitch,
200        );
201        self.broadcast_to_all_except(spawn_packet, player.id);
202    }
