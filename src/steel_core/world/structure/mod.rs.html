64/// Vanilla's `StructurePiece.makeBoundingBox`: north/south keep width/depth,
65/// east/west swap them.
66pub(crate) const fn make_oriented_piece_bounding_box(
67    chunk_min_x: i32,
68    y: i32,
69    chunk_min_z: i32,
70    orientation: Direction,
71    width: i32,
72    height: i32,
73    depth: i32,
74) -> BoundingBox {
75    let z_axis = matches!(orientation, Direction::North | Direction::South);
76    let (box_width, box_depth) = if z_axis {
77        (width, depth)
78    } else {
79        (depth, width)
80    };
81    BoundingBox::new(
82        chunk_min_x,
83        y,
84        chunk_min_z,
85        chunk_min_x + box_width - 1,
86        y + height - 1,
87        chunk_min_z + box_depth - 1,
88    )
89}

91/// A structure start placed in a chunk. Vanilla's `StructureStart` — invalid (empty)
92/// starts are not stored.
93#[derive(Debug, Clone)]
94pub struct StructureStart {
95    /// Structure id (e.g., `minecraft:village`).
96    pub structure: Identifier,
97    /// Origin chunk.
98    pub chunk_pos: ChunkPos,
99    /// Vanilla's map/locate reference counter. This is distinct from
100    /// [`StructureReferenceMap`]; generating per-chunk structure references does
101    /// not increment this counter.
102    pub references: i32,
103    /// Pieces composing this structure.
104    pub pieces: Vec<StructurePiece>,
105    /// Bounding-box inflation applied at construction. Vanilla inflates by 12
106    /// when `terrain_adaptation != NONE`. Stored for serialization parity; the
107    /// inflation is already baked into [`bounding_box`](Self::bounding_box).
108    pub bb_inflate: i32,
109    /// Terrain adaptation mode from the structure registry. Used by Beardifier.
110    pub terrain_adjustment: TerrainAdjustment,
111    /// Cached bounding box matching vanilla's `StructureStart.getBoundingBox()`:
112    /// the union of piece bounding boxes, then `inflatedBy(bb_inflate)`.
113    /// `None` iff `pieces` is empty.
114    pub bounding_box: Option<BoundingBox>,
115}
116
117impl StructureStart {
118    /// Creates a start, computing the inflated piece-union bounding box up-front.
119    #[must_use]
120    pub fn new(
121        structure: Identifier,
122        chunk_pos: ChunkPos,
123        pieces: Vec<StructurePiece>,
124        terrain_adjustment: TerrainAdjustment,
125    ) -> Self {
126        let bb_inflate = terrain_adjustment.bb_inflate();
127        let bounding_box = Self::compute_bounding_box(&pieces, bb_inflate);
128        Self {
129            structure,
130            chunk_pos,
131            references: 0,
132            pieces,
133            bb_inflate,
134            terrain_adjustment,
135            bounding_box,
136        }
137    }
138
139    /// Union of all pieces' bounding boxes, inflated by `bb_inflate` on every
140    /// axis. Returns `None` if `pieces` is empty. Mirrors vanilla's
141    /// `StructureStart.getBoundingBox()` (= `adjustBoundingBox(union)`).
142    #[must_use]
143    pub fn compute_bounding_box(pieces: &[StructurePiece], bb_inflate: i32) -> Option<BoundingBox> {
144        let (first, rest) = pieces.split_first()?;
145        let mut bb = first.bounding_box;
146        for piece in rest {
147            bb = BoundingBox::new(
148                bb.min_x.min(piece.bounding_box.min_x),
149                bb.min_y.min(piece.bounding_box.min_y),
150                bb.min_z.min(piece.bounding_box.min_z),
151                bb.max_x.max(piece.bounding_box.max_x),
152                bb.max_y.max(piece.bounding_box.max_y),
153                bb.max_z.max(piece.bounding_box.max_z),
154            );
155        }
156        Some(bb.inflated_by(bb_inflate, bb_inflate, bb_inflate))
157    }
158
159    /// Vanilla `StructureStart.placeInChunk` reference position: the first
160    /// piece center X/Z and first piece minimum Y.
161    #[must_use]
162    pub fn placement_reference_pos(&self) -> Option<BlockPos> {
163        let first_piece = self.pieces.first()?;
164        let center = first_piece.bounding_box.get_center();
165        Some(BlockPos::new(
166            center.x(),
167            first_piece.bounding_box.min_y,
168            center.z(),
169        ))
170    }
171}

177///
178/// Vanilla stores these as a fastutil `LongOpenHashSet`, so duplicates are
179/// ignored and feature-stage iteration follows that table order.
180pub type StructureReferenceMap = FxHashMap<Identifier, StructureReferenceSet>;

182/// Set of structure-start chunk positions with vanilla iteration order.
183///
184/// Reference generation discovers sources in a stable scan order, but vanilla
185/// stores the packed chunk longs in fastutil's `LongOpenHashSet`. Feature-stage
186/// placement consumes the set through that table iteration order, so Steel keeps
187/// the insertion order for persistence and exposes the vanilla iteration order
188/// for worldgen.
189#[derive(Debug, Clone, Default, PartialEq, Eq)]
190pub struct StructureReferenceSet {
191    insertion_order: Vec<ChunkPos>,
192    iteration_order: Vec<ChunkPos>,
193}
194
195impl StructureReferenceSet {
196    /// Inserts a chunk position if it was not already present.
197    pub fn insert(&mut self, pos: ChunkPos) -> bool {
198        if self.insertion_order.contains(&pos) {
199            return false;
200        }
201        self.insertion_order.push(pos);
202        self.rebuild_iteration_order();
203        true
204    }
205
206    /// Extends this set with insertion-order duplicate removal.
207    pub fn extend(&mut self, positions: impl IntoIterator<Item = ChunkPos>) {
208        for pos in positions {
209            self.insert(pos);
210        }
211    }
212
213    /// Returns an iterator over positions in vanilla `LongOpenHashSet` order.
214    pub fn iter(&self) -> slice::Iter<'_, ChunkPos> {
215        self.iteration_order.iter()
216    }
217
218    /// Returns an iterator over positions in discovery order.
219    pub fn insertion_order_iter(&self) -> slice::Iter<'_, ChunkPos> {
220        self.insertion_order.iter()
221    }
222
223    /// Returns `true` when no positions are stored.
224    #[must_use]
225    pub const fn is_empty(&self) -> bool {
226        self.insertion_order.is_empty()
227    }
228
229    fn rebuild_iteration_order(&mut self) {
230        self.iteration_order = Self::vanilla_long_open_hash_set_order(&self.insertion_order);
231    }
232
233    fn vanilla_long_open_hash_set_order(insertion_order: &[ChunkPos]) -> Vec<ChunkPos> {
234        let Some(table_size) = Self::vanilla_long_open_hash_set_table_size(insertion_order.len())
235        else {
236            return Vec::new();
237        };
238        let mask = (table_size - 1) as u64;
239        let mut table = vec![None; table_size];
240        let mut zero_key = None;
241
242        for &pos in insertion_order {
243            let packed = Self::pack_chunk_pos(pos);
244            if packed == 0 {
245                zero_key = Some(pos);
246                continue;
247            }
248
249            let mut slot = (Self::fastutil_mix(packed) & mask) as usize;
250            loop {
251                if table[slot].is_none() {
252                    table[slot] = Some(pos);
253                    break;
254                }
255                slot = (slot + 1) & (table_size - 1);
256            }
257        }
258
259        let mut ordered = Vec::with_capacity(insertion_order.len());
260        if let Some(pos) = zero_key {
261            ordered.push(pos);
262        }
263        for slot in (0..table_size).rev() {
264            if let Some(pos) = table[slot] {
265                ordered.push(pos);
266            }
267        }
268        ordered
269    }
270
271    fn vanilla_long_open_hash_set_table_size(len: usize) -> Option<usize> {
272        if len == 0 {
273            return None;
274        }
275
276        let mut table_size = Self::fastutil_array_size(16);
277        let mut max_fill = Self::fastutil_max_fill(table_size);
278        let mut size = 0;
279        for _ in 0..len {
280            let old_size = size;
281            size += 1;
282            if old_size >= max_fill {
283                table_size = Self::fastutil_array_size(size + 1);
284                max_fill = Self::fastutil_max_fill(table_size);
285            }
286        }
287        Some(table_size)
288    }
289
290    fn fastutil_array_size(expected: usize) -> usize {
291        let needed = ((expected as f64) / 0.75).ceil() as usize;
292        needed.max(2).next_power_of_two()
293    }
294
295    const fn fastutil_max_fill(table_size: usize) -> usize {
296        let fill = table_size - table_size / 4;
297        if fill < table_size {
298            fill
299        } else {
300            table_size - 1
301        }
302    }
303
304    const fn pack_chunk_pos(pos: ChunkPos) -> u64 {
305        (pos.0.x as u32 as u64) | ((pos.0.y as u32 as u64) << 32)
306    }
307
308    const fn fastutil_mix(value: u64) -> u64 {
309        let mixed = value.wrapping_mul(0x9E37_79B9_7F4A_7C15);
310        let mixed = mixed ^ (mixed >> 32);
311        mixed ^ (mixed >> 16)
312    }
313}

352/// Per-chunk context shared by every structure's `findGenerationPoint`.
353///
354/// Holds mutable per-chunk state (biome sampler, height cache, aquifer) so structures
355/// don't each allocate their own. Wraps `VanillaGenerator`'s helpers.
356pub struct GenerationContext<'ctx, 'src, N: DimensionNoises>
357where
358    'src: 'ctx,
359{
360    /// World seed.
361    pub seed: i64,
362    /// Chunk being populated.
363    pub chunk_x: i32,
364    /// Chunk being populated.
365    pub chunk_z: i32,
366    /// `chunk_x * 16`.
367    pub chunk_min_x: i32,
368    /// `chunk_z * 16`.
369    pub chunk_min_z: i32,
370    /// `chunk_min_x + 8`.
371    pub center_block_x: i32,
372    /// `chunk_min_z + 8`.
373    pub center_block_z: i32,
374    /// Sea level for this dimension.
375    pub sea_level: i32,
376    /// Shared memoisation slot for the chunk-center surface Y.
377    pub(crate) surface_y_cache: &'ctx mut Option<i32>,
378    /// Whether `height_cache`'s 5×5 quart grid has been populated. Shared across
379    /// per-structure contexts in the same chunk.
380    pub(crate) height_cache_grid_ready: &'ctx mut bool,
381
382    /// Dimension noise router.
383    pub noises: &'src N,
384    /// Positional splitter for per-chunk RNG.
385    pub splitter: &'src RandomSplitter,
386    /// Template pool registry for jigsaw assembly.
387    pub template_pools: &'src FxHashMap<Identifier, TemplatePoolData>,
388    /// Structure templates (piece definitions + sizes).
389    pub templates: &'src FxHashMap<Identifier, TemplateData>,
390
391    /// Biome sampler scoped to this chunk.
392    pub biome_sampler: &'ctx mut ChunkBiomeSampler<'src>,
393    /// Column cache for height/density queries (grid-initialized on demand).
394    pub height_cache: &'ctx mut N::ColumnCache,
395    /// Aquifer built on first query; skipped on chunks where no structure needs it.
396    pub aquifer: &'ctx mut LazyAquifer<'src, N>,
397    pub(crate) terrain_height_cache: RefCell<FxHashMap<(i32, i32, bool), i32>>,
398    pub(crate) terrain_opaque_cache: RefCell<FxHashMap<(i32, i32, i32, bool), bool>>,
399    pub(crate) terrain_probes: RefCell<FxHashMap<(i32, i32), TerrainProbe<N>>>,
400}

432/// Terrain, biome, and template queries exposed to structure algorithms.
433///
434/// Vanilla calls these through `ChunkGenerator`/`WorldGenLevel`; keeping the
435/// interface here lets structure algorithms stay independent of a concrete
436/// chunk generator while preserving their vanilla query order.
437pub trait StructureGenerationContext {
438    /// World seed.
439    fn seed(&self) -> i64;
440    /// Chunk X being populated.
441    fn chunk_x(&self) -> i32;
442    /// Chunk Z being populated.
443    fn chunk_z(&self) -> i32;
444    /// Minimum block X of the chunk.
445    fn chunk_min_x(&self) -> i32;
446    /// Minimum block Z of the chunk.
447    fn chunk_min_z(&self) -> i32;
448    /// Center block X of the chunk.
449    fn center_block_x(&self) -> i32;
450    /// Center block Z of the chunk.
451    fn center_block_z(&self) -> i32;
452    /// Sea level for this generator/dimension.
453    fn sea_level(&self) -> i32;
454    /// Minimum build Y.
455    fn min_y(&self) -> i32;
456    /// Total build height.
457    fn height(&self) -> i32;
458    /// One-past-maximum build Y.
459    fn max_y(&self) -> i32 {
460        self.min_y() + self.height()
461    }
462    /// Template pool registry for jigsaw assembly.
463    fn template_pools(&self) -> &FxHashMap<Identifier, TemplatePoolData>;
464    /// Structure templates (piece definitions + sizes).
465    fn templates(&self) -> &FxHashMap<Identifier, TemplateData>;
466    /// Base height at a column.
467    fn base_height(&mut self, x: i32, z: i32, ocean_floor: bool) -> i32;
468    /// Full-column base height scan.
469    fn base_height_full(&mut self, x: i32, z: i32, ocean_floor: bool) -> i32;
470    /// Biome at a block position.
471    fn biome_at(&mut self, block_x: i32, block_y: i32, block_z: i32) -> BiomeRef;
472    /// Classify a block in the generator's base terrain.
473    fn column_state(&mut self, x: i32, y: i32, z: i32) -> ColumnBlock;
474    /// Highest solid base-terrain block directly below air in `[min_solid_y, start_y)`.
475    fn solid_block_below_air(
476        &mut self,
477        x: i32,
478        z: i32,
479        start_y: i32,
480        min_solid_y: i32,
481    ) -> Option<i32> {
482        if start_y <= min_solid_y {
483            return None;
484        }
485
486        let mut above = self.column_state(x, start_y, z);
487        for y in (min_solid_y..start_y).rev() {
488            let current = self.column_state(x, y, z);
489            if above == ColumnBlock::Air && current == ColumnBlock::Solid {
490                return Some(y);
491            }
492            above = current;
493        }
494        None
495    }
496    /// Chunk-center surface Y, memoised by the concrete context.
497    fn surface_y(&mut self) -> i32;
498    /// Surface height for off-chunk terrain queries used by piece placement.
499    fn terrain_surface_height(&self, x: i32, z: i32, ocean_floor: bool) -> i32;
500    /// Opaque terrain test for off-chunk terrain queries used by piece placement.
501    fn terrain_is_opaque(&self, x: i32, y: i32, z: i32, ocean_floor: bool) -> bool;
502}

505/// collision checks, and biome check.
506pub trait Structure: Send + Sync {
507    /// `structure` carries registry data; per-set metadata stays in placement.
508    /// `rng` is a fresh `LegacyRandom` seeded with `setLargeFeatureSeed`.
509    fn find_generation_point(

514    ) -> Option<GenerationStub>;

517impl<'ctx, 'src, N: DimensionNoises> GenerationContext<'ctx, 'src, N>
518where
519    'src: 'ctx,
520{
521    /// `getBaseHeight(WORLD_SURFACE_WG)` — aquifer-aware, scans from
522    /// `preliminary_surface_level + 16`.
523    ///
524    /// `ocean_floor=false` → opaque is Solid+Fluid; `true` → opaque is Solid only.
525    ///
526    /// In dimensions with a constant `preliminary_surface_level` (End), use
527    /// [`base_height_full`](Self::base_height_full) instead.
528    pub fn base_height(&mut self, x: i32, z: i32, ocean_floor: bool) -> i32 {
529        self.ensure_height_cache_grid();
530        let aq = self.aquifer.ensure(self.height_cache);
531        column_base_height::<N>(self.height_cache, self.noises, aq, x, z, ocean_floor)
532    }
533
534    /// Full-column scan from chunk top. Matches vanilla's `iterateNoiseColumn`.
535    pub fn base_height_full(&mut self, x: i32, z: i32, ocean_floor: bool) -> i32 {
536        self.ensure_height_cache_grid();
537        let aq = self.aquifer.ensure(self.height_cache);
538        iterate_noise_column_with_aquifer::<N>(
539            self.height_cache,
540            self.noises,
541            aq,
542            x,
543            z,
544            ocean_floor,
545        )
546    }
547
548    /// Biome at a block position (quantized to quart).
549    pub fn biome_at(&mut self, block_x: i32, block_y: i32, block_z: i32) -> BiomeRef {
550        self.biome_sampler
551            .sample(block_x >> 2, block_y >> 2, block_z >> 2)
552    }
553
554    /// Classify a single block in the base-noise column.
555    pub fn column_state(&mut self, x: i32, y: i32, z: i32) -> ColumnBlock {
556        self.ensure_height_cache_grid();
557        let cw = N::Settings::CELL_WIDTH;
558        let ch = N::Settings::CELL_HEIGHT;
559        let density =
560            column_interpolated_density::<N>(self.height_cache, self.noises, x, y, z, cw, ch);
561        let aq = self.aquifer.ensure(self.height_cache);
562        match aq.compute_substance(self.noises, x, y, z, density) {
563            AquiferResult::Solid => ColumnBlock::Solid,
564            AquiferResult::Fluid(_) => ColumnBlock::Fluid,
565            AquiferResult::Air => ColumnBlock::Air,
566        }
567    }
568
569    /// Highest solid base-terrain block directly below air in `[min_solid_y, start_y)`.
570    pub fn solid_block_below_air(
571        &mut self,
572        x: i32,
573        z: i32,
574        start_y: i32,
575        min_solid_y: i32,
576    ) -> Option<i32> {
577        self.ensure_height_cache_grid();
578        let aq = self.aquifer.ensure(self.height_cache);
579        find_solid_block_below_air::<N>(
580            self.height_cache,
581            self.noises,
582            aq,
583            x,
584            z,
585            start_y,
586            min_solid_y,
587        )
588    }
589
590    /// Surface Y at chunk center, memoised across per-structure contexts.
591    pub fn surface_y(&mut self) -> i32 {
592        if let Some(y) = *self.surface_y_cache {
593            return y;
594        }
595        let y = self.base_height(self.center_block_x, self.center_block_z, false) - 1;
596        *self.surface_y_cache = Some(y);
597        y
598    }
599
600    fn ensure_height_cache_grid(&mut self) {
601        if *self.height_cache_grid_ready {
602            return;
603        }
604        self.height_cache
605            .init_grid(self.chunk_min_x, self.chunk_min_z, self.noises);
606        *self.height_cache_grid_ready = true;
607    }
608}

689    fn terrain_surface_height(&self, x: i32, z: i32, ocean_floor: bool) -> i32 {
690        if let Some(height) = self
691            .terrain_height_cache
692            .borrow()
693            .get(&(x, z, ocean_floor))
694            .copied()
695        {
696            return height;
697        }
698
699        let cell_w = N::Settings::CELL_WIDTH;
700        let cell_x = x.div_euclid(cell_w) * cell_w;
701        let cell_z = z.div_euclid(cell_w) * cell_w;
702        let aq_chunk_x = (cell_x >> 4) * 16;
703        let aq_chunk_z = (cell_z >> 4) * 16;
704        let height = {
705            let mut probes = self.terrain_probes.borrow_mut();
706            let probe = probes.entry((aq_chunk_x, aq_chunk_z)).or_insert_with(|| {
707                TerrainProbe::<N>::new(aq_chunk_x, aq_chunk_z, self.splitter, self.noises)
708            });
709            iterate_noise_column_with_aquifer::<N>(
710                &mut probe.cache,
711                self.noises,
712                &mut probe.aquifer,
713                x,
714                z,
715                ocean_floor,
716            )
717        };
718        self.terrain_height_cache
719            .borrow_mut()
720            .insert((x, z, ocean_floor), height);
721        height
722    }

724    fn terrain_is_opaque(&self, x: i32, y: i32, z: i32, ocean_floor: bool) -> bool {
725        if let Some(opaque) = self
726            .terrain_opaque_cache
727            .borrow()
728            .get(&(x, y, z, ocean_floor))
729            .copied()
730        {
731            return opaque;
732        }
733
734        let cell_w = N::Settings::CELL_WIDTH;
735        let cell_h = N::Settings::CELL_HEIGHT;
736        let cell_x = x.div_euclid(cell_w) * cell_w;
737        let cell_z = z.div_euclid(cell_w) * cell_w;
738        let aq_chunk_x = (cell_x >> 4) * 16;
739        let aq_chunk_z = (cell_z >> 4) * 16;
740        let opaque = {
741            let mut probes = self.terrain_probes.borrow_mut();
742            let probe = probes.entry((aq_chunk_x, aq_chunk_z)).or_insert_with(|| {
743                TerrainProbe::<N>::new(aq_chunk_x, aq_chunk_z, self.splitter, self.noises)
744            });
745            let density = column_interpolated_density::<N>(
746                &mut probe.cache,
747                self.noises,
748                x,
749                y,
750                z,
751                cell_w,
752                cell_h,
753            );
754            match probe
755                .aquifer
756                .compute_substance(self.noises, x, y, z, density)
757            {
758                AquiferResult::Solid => true,
759                AquiferResult::Fluid(_) => !ocean_floor,
760                AquiferResult::Air => false,
761            }
762        };
763        self.terrain_opaque_cache
764            .borrow_mut()
765            .insert((x, y, z, ocean_floor), opaque);
766        opaque
767    }
