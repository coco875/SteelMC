76/// Resolves pool aliases for a specific structure instance.
77pub fn resolve_aliases(
78    aliases: &[PoolAlias],
79    rng: &mut impl Random,
80) -> FxHashMap<Identifier, Identifier> {
81    let mut map = FxHashMap::default();
82    for alias in aliases {
83        match alias {
84            PoolAlias::Direct { alias, target } => {
85                map.insert(alias.clone(), target.clone());
86            }
87            PoolAlias::Random { alias, targets } => {
88                let total: i32 = targets.iter().map(|(_, w)| *w).sum();
89                if total > 0 {
90                    let mut pick = rng.next_i32_bounded(total);
91                    for (target, weight) in targets {
92                        pick -= weight;
93                        if pick < 0 {
94                            map.insert(alias.clone(), target.clone());
95                            break;
96                        }
97                    }
98                }
99            }
100            PoolAlias::RandomGroup { groups } => {
101                let total: i32 = groups.iter().map(|(_, w)| *w).sum();
102                if total > 0 {
103                    let mut pick = rng.next_i32_bounded(total);
104                    for (bindings, weight) in groups {
105                        pick -= weight;
106                        if pick < 0 {
107                            for (alias, target) in bindings {
108                                map.insert(alias.clone(), target.clone());
109                            }
110                            break;
111                        }
112                    }
113                }
114            }
115        }
116    }
117    map
118}

147/// Vanilla-matching shuffle (reverse Fisher-Yates).
148fn vanilla_shuffle<T>(list: &mut [T], rng: &mut LegacyRandom) {
149    for i in (1..list.len()).rev() {
150        let j = rng.next_i32_bounded((i + 1) as i32) as usize;
151        list.swap(i, j);
152    }
153}

155fn shuffle_jigsaws(jigsaws: &mut [TransformedJigsaw<'_>], rng: &mut LegacyRandom) {
156    vanilla_shuffle(jigsaws, rng);
157    let priorities = descending_priorities(jigsaws.iter().map(|jigsaw| jigsaw.selection_priority));
158    order_jigsaws_by_priorities(jigsaws, &priorities);
159}

161fn shuffled_jigsaw_indices(
162    cached: &CachedTemplateJigsaws<'_>,
163    rng: &mut LegacyRandom,
164) -> Vec<usize> {
165    let mut indices: Vec<usize> = (0..cached.jigsaws.len()).collect();
166    vanilla_shuffle(&mut indices, rng);
167    order_jigsaw_indices_by_priorities(cached, &mut indices);
168    indices
169}

171fn descending_priorities(priorities: impl Iterator<Item = i32>) -> Vec<i32> {
172    let mut unique = Vec::new();
173    for priority in priorities {
174        if !unique.contains(&priority) {
175            unique.push(priority);
176        }
177    }
178    if unique.len() > 1 {
179        unique.sort_unstable_by_key(|priority| Reverse(*priority));
180    }
181    unique
182}

184fn order_jigsaws_by_priorities(jigsaws: &mut [TransformedJigsaw<'_>], priorities: &[i32]) {
185    if priorities.len() <= 1 {
186        return;
187    }
188
189    let original = jigsaws.to_vec();
190    let mut out = Vec::with_capacity(original.len());
191    for &priority in priorities {
192        out.extend(
193            original
194                .iter()
195                .copied()
196                .filter(|jigsaw| jigsaw.selection_priority == priority),
197        );
198    }
199    jigsaws.copy_from_slice(&out);
200}
201
202fn order_jigsaw_indices_by_priorities(
203    cached: &CachedTemplateJigsaws<'_>,
204    indices: &mut Vec<usize>,
205) {
206    if cached.priorities_desc.len() <= 1 {
207        return;
208    }
209
210    let original = mem::take(indices);
211    let mut ordered = Vec::with_capacity(original.len());
212    for &priority in &cached.priorities_desc {
213        ordered.extend(
214            original
215                .iter()
216                .copied()
217                .filter(|&idx| cached.jigsaws[idx].selection_priority == priority),
218        );
219    }
220    *indices = ordered;
221}

225/// For `List` elements, delegates to the first sub-element matching vanilla's
226/// `ListPoolElement` which uses `elements.get(0)` for jigsaws and BB.

228    match element {
229        PoolElement::Single { location, .. } | PoolElement::LegacySingle { location, .. } => {

232        PoolElement::List { elements, .. } => elements.first().and_then(element_location),

237/// Gets shuffled jigsaws for a pool element at a given position and rotation.
238///
239/// Returns the jigsaws with their positions transformed by rotation, sorted
240/// by `selection_priority` (descending), then shuffled within equal priorities.
241fn get_shuffled_jigsaws<'a>(
242    element: &PoolElement,
243    templates: &'a FxHashMap<Identifier, TemplateData>,
244    rotation: Rotation,
245    rng: &mut LegacyRandom,
246) -> Vec<TransformedJigsaw<'a>> {
247    let Some(location) = element_location(element) else {
248        // Feature/Empty elements: synthetic jigsaw at origin facing down
249        let mut jigsaws = vec![TransformedJigsaw {
250            pos: (0, 0, 0),
251            orientation: JigsawOrientation::DownSouth,
252            name: &SYNTHETIC_BOTTOM_JIGSAW,
253            target: &SYNTHETIC_EMPTY_POOL,
254            pool: &SYNTHETIC_EMPTY_POOL,
255            joint: JointType::Rollable,
256            selection_priority: 0,
257            placement_priority: 0,
258        }];
259        shuffle_jigsaws(&mut jigsaws, rng);
260        return jigsaws;
261    };
262
263    let Some(template) = templates.get(location) else {
264        return vec![];
265    };
266
267    let mut jigsaws = transform_template_jigsaws(template, rotation);
268    shuffle_jigsaws(&mut jigsaws, rng);
269    jigsaws
270}

297fn get_cached_shuffled_template_jigsaws<'a>(
298    location: &Identifier,
299    templates: &'a FxHashMap<Identifier, TemplateData>,
300    cache: &mut JigsawTransformCache<'a>,
301    rotation: Rotation,
302    rng: &mut LegacyRandom,
303) -> Vec<TransformedJigsaw<'a>> {
304    let cached = get_cached_template_jigsaws(location, templates, cache, rotation);
305    let mut jigsaws = cached.jigsaws.clone();
306    vanilla_shuffle(&mut jigsaws, rng);
307    order_jigsaws_by_priorities(&mut jigsaws, &cached.priorities_desc);
308    jigsaws
309}

311fn get_cached_template_jigsaws<'cache, 'a>(
312    location: &Identifier,
313    templates: &'a FxHashMap<Identifier, TemplateData>,
314    cache: &'cache mut JigsawTransformCache<'a>,
315    rotation: Rotation,
316) -> &'cache CachedTemplateJigsaws<'a> {
317    let idx = rotation_index(rotation);
318    let by_rotation = cache
319        .entry(location.clone())
320        .or_insert_with(|| [(); 4].map(|()| None));
321    by_rotation[idx].get_or_insert_with(|| {
322        templates.get(location).map_or_else(
323            || CachedTemplateJigsaws {
324                jigsaws: Vec::new(),
325                priorities_desc: Vec::new(),
326            },
327            |template| {
328                let jigsaws = transform_template_jigsaws(template, rotation);
329                let priorities_desc =
330                    descending_priorities(jigsaws.iter().map(|jigsaw| jigsaw.selection_priority));
331                CachedTemplateJigsaws {
332                    jigsaws,
333                    priorities_desc,
334                }
335            },
336        )
337    })
338}

340fn transform_template_jigsaws(
341    template: &TemplateData,
342    rotation: Rotation,
343) -> Vec<TransformedJigsaw<'_>> {
344    template
345        .jigsaws
346        .iter()
347        .map(|j| {
348            // Transform position by rotation (pivot = 0,0,0)
349            let (tx, ty, tz) = rotation.transform_pos(j.pos[0], j.pos[1], j.pos[2], 0, 0);
350            // Rotate orientation
351            let orientation = j.orientation.rotate(rotation);
352            TransformedJigsaw {
353                pos: (tx, ty, tz),
354                orientation,
355                name: &j.name,
356                target: &j.target,
357                pool: &j.pool,
358                joint: j.joint,
359                selection_priority: j.selection_priority,
360                placement_priority: j.placement_priority,
361            }
362        })
363        .collect()
364}

384/// Vanilla's `StructureTemplatePool.getMaxSize` — max Y span across all templates.
385fn pool_max_y_size(
386    pool: &TemplatePoolData,
387    templates: &FxHashMap<Identifier, TemplateData>,
388) -> i32 {
389    pool.elements
390        .iter()
391        .filter_map(|(element, _)| {
392            let (PoolElement::Single { location: loc, .. }
393            | PoolElement::LegacySingle { location: loc, .. }) = element
394            else {
395                return None;
396            };
397            templates.get(loc).map(|t| t.size[1])
398        })
399        .max()
400        .unwrap_or(0)
401}

403/// Checks if two jigsaws can connect.
404///
405/// Vanilla's `JigsawBlock.canAttach`: opposite facing, name match, joint compatibility.
406fn can_attach(source: &TransformedJigsaw, target: &TransformedJigsaw) -> bool {
407    let source_front = source.orientation.front_direction();
408    let target_front = target.orientation.front_direction();
409
410    // Fronts must be opposite
411    if source_front != target_front.opposite() {
412        return false;
413    }
414
415    // Names must match: source.target == target.name
416    if source.target != target.name {
417        return false;
418    }
419
420    // Joint compatibility: if aligned, tops must match
421    if source.joint == JointType::Aligned {
422        let source_top = source.orientation.top_direction();
423        let target_top = target.orientation.top_direction();
424        if source_top != target_top {
425            return false;
426        }
427    }
428
429    true
430}

432/// Gets the bounding box for a pool element at a position with rotation.
433///
434/// Feature elements return a 1×1×1 BB at the given position, matching
435/// vanilla's `FeaturePoolElement.getBoundingBox`.
436/// List elements return the encapsulating BB of all sub-elements, matching
437/// vanilla's `ListPoolElement.getBoundingBox`.
438fn element_bounding_box(
439    element: &PoolElement,
440    templates: &FxHashMap<Identifier, TemplateData>,
441    pos_x: i32,
442    pos_y: i32,
443    pos_z: i32,
444    rotation: Rotation,
445) -> Option<BoundingBox> {
446    match element {
447        PoolElement::Feature { .. } => {
448            Some(BoundingBox::new(pos_x, pos_y, pos_z, pos_x, pos_y, pos_z))
449        }
450        PoolElement::List { elements, .. } => {
451            // Vanilla: encapsulating BB of all non-empty sub-elements
452            let mut result: Option<BoundingBox> = None;
453            for sub in elements {
454                if let Some(sub_bb) =
455                    element_bounding_box(sub, templates, pos_x, pos_y, pos_z, rotation)
456                {
457                    result = Some(match result {
458                        Some(prev) => BoundingBox::new(
459                            prev.min_x.min(sub_bb.min_x),
460                            prev.min_y.min(sub_bb.min_y),
461                            prev.min_z.min(sub_bb.min_z),
462                            prev.max_x.max(sub_bb.max_x),
463                            prev.max_y.max(sub_bb.max_y),
464                            prev.max_z.max(sub_bb.max_z),
465                        ),
466                        None => sub_bb,
467                    });
468                }
469            }
470            result
471        }
472        _ => {
473            let location = element_location(element)?;
474            let template = templates.get(location)?;
475            Some(rotation.get_bounding_box(
476                pos_x,
477                pos_y,
478                pos_z,
479                template.size[0],
480                template.size[1],
481                template.size[2],
482            ))
483        }
484    }
485}

487fn expand_pool_weights(pool: &TemplatePoolData) -> Vec<&PoolElement> {
488    let mut expanded = Vec::with_capacity(pool.elements.iter().map(|(_, w)| *w as usize).sum());
489    for (element, weight) in &pool.elements {
490        for _ in 0..*weight {
491            expanded.push(element);
492        }
493    }
494    expanded
495}

497/// Appends vanilla's `StructureTemplatePool.getShuffledTemplates` to `out`.
498fn append_shuffled_templates_cached<'a>(
499    pool: &'a TemplatePoolData,
500    cache: &mut PoolTemplateCache<'a>,
501    rng: &mut LegacyRandom,
502    out: &mut Vec<&'a PoolElement>,
503) {
504    let expanded = cache
505        .entry(pool.key.clone())
506        .or_insert_with(|| expand_pool_weights(pool));
507    let start = out.len();
508    out.extend(expanded.iter().copied());
509    vanilla_shuffle(&mut out[start..], rng);
510}

512/// Vanilla's `StructureTemplatePool.getRandomTemplate`.
513fn get_random_template<'a>(pool: &'a TemplatePoolData, rng: &mut LegacyRandom) -> &'a PoolElement {
514    let expanded = expand_pool_weights(pool);
515    if expanded.is_empty() {
516        // `PoolElement::Empty` is a unit variant with no interior mutability,
517        // so `&'static` sharing is sound.
518        static EMPTY: PoolElement = PoolElement::Empty;
519        return &EMPTY;
520    }
521    let idx = rng.next_i32_bounded(expanded.len() as i32) as usize;
522    expanded[idx]
523}

525/// Hierarchical free-space tracker. Vanilla uses `MutableObject<VoxelShape>`
526/// with subtraction; for integer-aligned BBs, `constraint + occupied` is
527/// equivalent. Internal children share the source's internal free space;
528/// external children share the parent's context.
529struct FreeSpace {
530    constraint: BoundingBox,
531    occupied: Vec<BoundingBox>,
532}
533
534impl FreeSpace {
535    fn collides(&self, candidate: &BoundingBox) -> bool {
536        if candidate.min_x < self.constraint.min_x
537            || candidate.max_x > self.constraint.max_x
538            || candidate.min_y < self.constraint.min_y
539            || candidate.max_y > self.constraint.max_y
540            || candidate.min_z < self.constraint.min_z
541            || candidate.max_z > self.constraint.max_z
542        {
543            return true;
544        }
545        self.occupied.iter().any(|p| candidate.intersects(p))
546    }
547}

563/// Vanilla's `JigsawPlacement.addPieces` before the lazy `GenerationStub` child builder.
564/// This must stop after the center piece so invalid start biomes do not assemble
565/// full village/trial/ancient-city trees.
566#[expect(
567    clippy::too_many_arguments,
568    reason = "matches vanilla's addPieces call surface"
569)]
570fn start_assembly(
571    config: &JigsawConfig,
572    rng: &mut LegacyRandom,
573    chunk_x: i32,
574    chunk_z: i32,
575    pools: &FxHashMap<Identifier, TemplatePoolData>,
576    templates: &FxHashMap<Identifier, TemplateData>,
577    alias_map: &FxHashMap<Identifier, Identifier>,
578    get_height: &mut dyn FnMut(i32, i32) -> i32,
579    min_y: i32,
580    max_y: i32,
581) -> Option<StartedAssembly> {
582    let start_y = sample_start_height(config, rng);
583    let start_x = chunk_x * 16;
584    let start_z = chunk_z * 16;
585    let center_rotation = Rotation::get_random(rng);
586
587    let start_pool_key = alias_map
588        .get(&config.start_pool)
589        .unwrap_or(&config.start_pool);
590    let start_pool = pools.get(start_pool_key)?;
591    let center_element = get_random_template(start_pool, rng);
592    if center_element.is_empty() {
593        return None;
594    }
595
596    let (anchor_offset_x, anchor_offset_y, anchor_offset_z) =
597        if let Some(ref jigsaw_name) = config.start_jigsaw_name {
598            let jigsaws = get_shuffled_jigsaws(center_element, templates, center_rotation, rng);
599            let j = jigsaws.iter().find(|j| j.name == jigsaw_name)?;
600            (j.pos.0, j.pos.1, j.pos.2)
601        } else {
602            (0, 0, 0)
603        };
604
605    // Move piece so anchor aligns with start position.
606    let adjusted_x = start_x - anchor_offset_x;
607    let adjusted_y = start_y - anchor_offset_y;
608    let adjusted_z = start_z - anchor_offset_z;
609
610    let center_bb = element_bounding_box(
611        center_element,
612        templates,
613        adjusted_x,
614        adjusted_y,
615        adjusted_z,
616        center_rotation,
617    )?;
618
619    let bottom_y = if config.project_start_to_heightmap.is_some() {
620        let mid_x = java_center(center_bb.min_x, center_bb.max_x);
621        let mid_z = java_center(center_bb.min_z, center_bb.max_z);
622        start_y + get_height(mid_x, mid_z)
623    } else {
624        adjusted_y
625    };
626
627    // Move center piece to projected height.
628    let ground_level_delta = center_element.projection().ground_level_delta();
629    let dy = bottom_y - (center_bb.min_y + ground_level_delta);
630    let center_bb = BoundingBox::new(
631        center_bb.min_x,
632        center_bb.min_y + dy,
633        center_bb.min_z,
634        center_bb.max_x,
635        center_bb.max_y + dy,
636        center_bb.max_z,
637    );
638    let adjusted_y = adjusted_y + dy;
639
640    // Dimension padding. Vanilla's `getMaxY()` is inclusive (= minY + height - 1).
641    let padding = &config.dimension_padding;
642    if center_bb.min_y < min_y + padding.bottom || center_bb.max_y > max_y - 1 - padding.top {
643        return None;
644    }
645
646    let pieces = vec![PlacedPiece {
647        element: center_element.clone(),
648        template_location: element_location(center_element).cloned(),
649        position: (adjusted_x, adjusted_y, adjusted_z),
650        rotation: center_rotation,
651        bounding_box: center_bb,
652        assembly_bb: center_bb,
653        ground_level_delta,
654        projection: center_element.projection(),
655        depth: 0,
656        junctions: Vec::new(),
657    }];
658
659    // GenerationStub center.
660    let center_stub_x = java_center(center_bb.min_x, center_bb.max_x);
661    let center_stub_z = java_center(center_bb.min_z, center_bb.max_z);
662    let center_stub_y = bottom_y + anchor_offset_y;
663    let biome_check_pos = (center_stub_x, center_stub_y, center_stub_z);
664
665    Some(StartedAssembly {
666        pieces,
667        biome_check_pos,
668    })
669}

671#[expect(
672    clippy::too_many_arguments,
673    reason = "matches vanilla's addPieces child-builder call surface"
674)]
675fn finish_assembly<'a>(
676    mut started: StartedAssembly,
677    config: &JigsawConfig,
678    rng: &mut LegacyRandom,
679    pools: &'a FxHashMap<Identifier, TemplatePoolData>,
680    templates: &'a FxHashMap<Identifier, TemplateData>,
681    alias_map: &FxHashMap<Identifier, Identifier>,
682    get_height: &mut dyn FnMut(i32, i32) -> i32,
683    min_y: i32,
684    max_y: i32,
685) -> AssemblyResult {
686    let biome_check_pos = started.biome_check_pos;
687
688    if config.max_depth <= 0 {
689        return AssemblyResult {
690            pieces: started.pieces,
691            biome_check_pos,
692        };
693    }
694
695    let Some(center_piece) = started.pieces.first() else {
696        return AssemblyResult {
697            pieces: started.pieces,
698            biome_check_pos,
699        };
700    };
701    let center_bb = center_piece.assembly_bb;
702    let (center_stub_x, center_stub_y, center_stub_z) = biome_check_pos;
703
704    // Vanilla centers the constraint on `(centerX, centerY, centerZ)`, NOT on BB
705    // corners. Uses `+1` on the max side for AABB, but integer-BB collision with
706    // `[center - maxDist, center + maxDist]` is equivalent.
707    let max_dist = config.max_distance_from_center;
708    let constraint_bb = BoundingBox::new(
709        center_stub_x - max_dist,
710        (center_stub_y - max_dist).max(min_y + config.dimension_padding.bottom),
711        center_stub_z - max_dist,
712        center_stub_x + max_dist,
713        (center_stub_y + max_dist).min(max_y - 1 - config.dimension_padding.top),
714        center_stub_z + max_dist,
715    );
716
717    // Index 0 = global collision context.
718    let mut free_spaces: Vec<FreeSpace> = vec![FreeSpace {
719        constraint: constraint_bb,
720        occupied: vec![center_bb],
721    }];
722    let mut jigsaw_cache = JigsawTransformCache::default();
723    let mut pool_template_cache = PoolTemplateCache::default();
724    // (piece_index, depth, placement_priority, context_idx)
725    let mut queue: Vec<(usize, i32, i32, usize)> = Vec::new();
726
727    try_placing_children(
728        0,
729        0,
730        0,
731        config,
732        pools,
733        templates,
734        alias_map,
735        &mut jigsaw_cache,
736        &mut pool_template_cache,
737        &mut started.pieces,
738        &mut free_spaces,
739        &mut queue,
740        rng,
741        get_height,
742    );
743
744    while !queue.is_empty() {
745        // Stable sort keeps insertion order within equal priorities.
746        queue.sort_by_key(|entry| Reverse(entry.2));
747        let (piece_idx, depth, _priority, context_idx) = queue.remove(0);
748        try_placing_children(
749            piece_idx,
750            depth,
751            context_idx,
752            config,
753            pools,
754            templates,
755            alias_map,
756            &mut jigsaw_cache,
757            &mut pool_template_cache,
758            &mut started.pieces,
759            &mut free_spaces,
760            &mut queue,
761            rng,
762            get_height,
763        );
764    }
765
766    AssemblyResult {
767        pieces: started.pieces,
768        biome_check_pos,
769    }
770}

772/// Vanilla's `JigsawPlacement.addPieces`. Returns `None` on failure (empty start
773/// pool, dimension padding violation, etc.).
774#[expect(
775    clippy::too_many_arguments,
776    reason = "matches vanilla's addPieces call surface"
777)]
778#[expect(
779    clippy::implicit_hasher,
780    reason = "FxHashMap avoids SipHash overhead on Identifier lookups"
781)]
782pub fn assemble(
783    config: &JigsawConfig,
784    rng: &mut LegacyRandom,
785    chunk_x: i32,
786    chunk_z: i32,
787    pools: &FxHashMap<Identifier, TemplatePoolData>,
788    templates: &FxHashMap<Identifier, TemplateData>,
789    alias_map: &FxHashMap<Identifier, Identifier>,
790    get_height: &mut dyn FnMut(i32, i32) -> i32,
791    min_y: i32,
792    max_y: i32,
793) -> Option<AssemblyResult> {
794    let started = start_assembly(
795        config, rng, chunk_x, chunk_z, pools, templates, alias_map, get_height, min_y, max_y,
796    )?;
797    Some(finish_assembly(
798        started, config, rng, pools, templates, alias_map, get_height, min_y, max_y,
799    ))
800}

806impl Structure for JigsawStructure {
807    fn find_generation_point(
808        &self,
809        ctx: &mut dyn StructureGenerationContext,
810        structure: &StructureData,
811        _rng: &mut LegacyRandom,
812    ) -> Option<GenerationStub> {
813        let config = structure.config.as_jigsaw()?;
814
815        let mut alias_position_rng = LegacyRandom::from_seed(0);
816        alias_position_rng.set_large_feature_seed(ctx.seed(), ctx.chunk_x(), ctx.chunk_z());
817        let start_y = sample_start_height(config, &mut alias_position_rng);
818        let mut alias_source = LegacyRandom::from_seed(ctx.seed() as u64);
819        let mut alias_rng =
820            alias_source
821                .next_positional()
822                .at(ctx.chunk_min_x(), start_y, ctx.chunk_min_z());
823        let alias_map = resolve_aliases(&config.pool_aliases, &mut alias_rng);
824
825        let mut assembly_rng = LegacyRandom::from_seed(0);
826        assembly_rng.set_large_feature_seed(ctx.seed(), ctx.chunk_x(), ctx.chunk_z());
827
828        let started = {
829            let mut get_height = |x: i32, z: i32| ctx.terrain_surface_height(x, z, false);
830            start_assembly(
831                config,
832                &mut assembly_rng,
833                ctx.chunk_x(),
834                ctx.chunk_z(),
835                ctx.template_pools(),
836                ctx.templates(),
837                &alias_map,
838                &mut get_height,
839                ctx.min_y(),
840                ctx.max_y(),
841            )?
842        };
843
844        if started.pieces.is_empty() {
845            return None;
846        }
847
848        let (bx, by, bz) = started.biome_check_pos;
849        let biome = ctx.biome_at(bx, by, bz);
850        if !structure.allowed_biomes.contains(&biome.key) {
851            return None;
852        }
853
854        let assembly = {
855            let mut get_height = |x: i32, z: i32| ctx.terrain_surface_height(x, z, false);
856            finish_assembly(
857                started,
858                config,
859                &mut assembly_rng,
860                ctx.template_pools(),
861                ctx.templates(),
862                &alias_map,
863                &mut get_height,
864                ctx.min_y(),
865                ctx.max_y(),
866            )
867        };
868
869        let pieces = assembly
870            .pieces
871            .into_iter()
872            .map(|piece| StructurePiece {
873                piece_type: Identifier::new_static("minecraft", "jigsaw"),
874                // Vanilla's `JigsawPlacement.tryPlacingChildren` mutates
875                // `targetBB` in place via the expansion hack, then stores it on
876                // `PoolElementStructurePiece`.
877                bounding_box: piece.assembly_bb,
878                // Vanilla's `PoolElementStructurePiece` constructor hardcodes
879                // `genDepth = 0` for every jigsaw piece.
880                gen_depth: 0,
881                orientation: None,
882                payload: StructurePiecePayload::Jigsaw(JigsawPieceData {
883                    pool_element: piece.element,
884                    position: piece.position,
885                    rotation: piece.rotation,
886                    liquid_settings: config.liquid_settings,
887                }),
888                ground_level_delta: piece.ground_level_delta,
889                junctions: piece.junctions,
890                projection: Some(piece.projection),
891            })
892            .collect();
893
894        Some(GenerationStub {
895            position: assembly.biome_check_pos,
896            pieces,
897        })
898    }
899}

901/// Vanilla's `tryPlacingChildren`. `context_idx` is this piece's collision context
902/// in `free_spaces` — external children get the parent's context, internal
903/// children get the parent's internal free space.
904#[expect(
905    clippy::too_many_arguments,
906    reason = "matches vanilla's tryPlacingChildren signature"
907)]
908#[expect(
909    clippy::too_many_lines,
910    reason = "inlined to mirror vanilla's source-jigsaw/child-pool loop"
911)]
912fn try_placing_children<'a>(
913    source_idx: usize,
914    depth: i32,
915    context_idx: usize,
916    config: &JigsawConfig,
917    pools: &'a FxHashMap<Identifier, TemplatePoolData>,
918    templates: &'a FxHashMap<Identifier, TemplateData>,
919    alias_map: &FxHashMap<Identifier, Identifier>,
920    jigsaw_cache: &mut JigsawTransformCache<'a>,
921    pool_template_cache: &mut PoolTemplateCache<'a>,
922    pieces: &mut Vec<PlacedPiece>,
923    free_spaces: &mut Vec<FreeSpace>,
924    queue: &mut Vec<(usize, i32, i32, usize)>,
925    rng: &mut LegacyRandom,
926    get_height: &mut dyn FnMut(i32, i32) -> i32,
927) {
928    let (source_jigsaws, source_bb, source_projection, source_ground_level_delta) = {
929        let source_piece = &pieces[source_idx];
930        let mut jigsaws = get_cached_shuffled_jigsaws(
931            &source_piece.element,
932            templates,
933            jigsaw_cache,
934            source_piece.rotation,
935            rng,
936        );
937        if jigsaws.is_empty() {
938            return;
939        }
940        for jigsaw in &mut jigsaws {
941            jigsaw.pos.0 += source_piece.position.0;
942            jigsaw.pos.1 += source_piece.position.1;
943            jigsaw.pos.2 += source_piece.position.2;
944        }
945        (
946            jigsaws,
947            source_piece.assembly_bb,
948            source_piece.projection,
949            source_piece.ground_level_delta,
950        )
951    };
952    let source_box_y = source_bb.min_y;
953    let source_rigid = source_projection == Projection::Rigid;
954
955    // Vanilla's sourceFree: tracks free space inside source piece for internal
956    // placements. Lazily initialized on first internal placement (matching
957    // vanilla's `MutableObject<VoxelShape>` that starts null).
958    let mut internal_ctx_idx: Option<usize> = None;
959    let mut candidates: Vec<&PoolElement> = Vec::new();
960
961    // For each source jigsaw, try to place one child
962    'source_jigsaw: for source_jigsaw in &source_jigsaws {
963        candidates.clear();
964        let front = source_jigsaw.orientation.front_direction();
965        let (fdx, fdy, fdz) = front.offset();
966        let source_jigsaw_pos = source_jigsaw.pos;
967        let target_jigsaw_world = (
968            source_jigsaw_pos.0 + fdx,
969            source_jigsaw_pos.1 + fdy,
970            source_jigsaw_pos.2 + fdz,
971        );
972
973        let source_jigsaw_local_y = source_jigsaw_pos.1 - source_box_y;
974
975        // Resolve target pool — vanilla always resolves fallback from the pool,
976        // even if the main pool is empty
977        let pool_key = alias_map
978            .get(source_jigsaw.pool)
979            .unwrap_or(source_jigsaw.pool);
980        let raw_pool = pools.get(pool_key);
981        let target_pool = raw_pool.filter(|p| !p.elements.is_empty());
982        let fallback_pool = raw_pool
983            .and_then(|p| pools.get(&p.fallback))
984            .filter(|p| !p.elements.is_empty());
985
986        // Determine whether target attaches inside source
987        let attach_inside = source_bb.contains_xyz(
988            target_jigsaw_world.0,
989            target_jigsaw_world.1,
990            target_jigsaw_world.2,
991        );
992
993        if depth != config.max_depth
994            && let Some(pool) = target_pool
995        {
996            append_shuffled_templates_cached(pool, pool_template_cache, rng, &mut candidates);
997        }
998        if let Some(fallback) = fallback_pool {
999            append_shuffled_templates_cached(fallback, pool_template_cache, rng, &mut candidates);
1000        }
1001
1002        let placement_priority = source_jigsaw.placement_priority;
1003
1004        // Track source jigsaw base height (lazy, for terrain matching)
1005        let mut source_jigsaw_base_height: Option<i32> = None;
1006
1007        // Try each candidate
1008        for &candidate_element in &candidates {
1009            if candidate_element.is_empty() {
1010                break;
1011            }
1012
1013            let candidate_projection = candidate_element.projection();
1014            let candidate_rigid = candidate_projection == Projection::Rigid;
1015
1016            // Try each rotation
1017            let rotations = Rotation::get_shuffled(rng);
1018            for candidate_rotation in rotations {
1019                let fallback_jigsaws;
1020                let (candidate_jigsaws, candidate_jigsaw_order) =
1021                    if let Some(location) = element_location(candidate_element) {
1022                        let cached = get_cached_template_jigsaws(
1023                            location,
1024                            templates,
1025                            jigsaw_cache,
1026                            candidate_rotation,
1027                        );
1028                        (
1029                            cached.jigsaws.as_slice(),
1030                            shuffled_jigsaw_indices(cached, rng),
1031                        )
1032                    } else {
1033                        fallback_jigsaws = get_cached_shuffled_jigsaws(
1034                            candidate_element,
1035                            templates,
1036                            jigsaw_cache,
1037                            candidate_rotation,
1038                            rng,
1039                        );
1040                        (
1041                            fallback_jigsaws.as_slice(),
1042                            (0..fallback_jigsaws.len()).collect(),
1043                        )
1044                    };
1045
1046                // Expansion hack: compute max child pool size for Y expansion.
1047                // Vanilla: getBoundingBox(manager, ZERO, rotation) uses default
1048                // StructurePlaceSettings with pivot=ZERO and mirror=NONE.
1049                // Jigsaw positions are also transformed with pivot=ZERO.
1050                let expand_to = if config.use_expansion_hack {
1051                    let hack_data =
1052                        element_location(candidate_element).and_then(|loc| templates.get(loc));
1053                    if let Some(template_data) = hack_data {
1054                        let hack_box = candidate_rotation.get_bounding_box(
1055                            0,
1056                            0,
1057                            0,
1058                            template_data.size[0],
1059                            template_data.size[1],
1060                            template_data.size[2],
1061                        );
1062                        if hack_box.max_y - hack_box.min_y < 16 {
1063                            template_data
1064                                .jigsaws
1065                                .iter()
1066                                .map(|j| {
1067                                    let (rx, ry, rz) = candidate_rotation
1068                                        .transform_pos(j.pos[0], j.pos[1], j.pos[2], 0, 0);
1069                                    let front =
1070                                        j.orientation.rotate(candidate_rotation).front_direction();
1071                                    let front_off = front.offset();
1072                                    let front_pos =
1073                                        (rx + front_off.0, ry + front_off.1, rz + front_off.2);
1074                                    if !hack_box.contains_xyz(front_pos.0, front_pos.1, front_pos.2)
1075                                    {
1076                                        return 0;
1077                                    }
1078                                    let child_pool_key = alias_map.get(&j.pool).unwrap_or(&j.pool);
1079                                    let child_pool_size = pools
1080                                        .get(child_pool_key)
1081                                        .map_or(0, |p| pool_max_y_size(p, templates));
1082                                    let child_fallback_size = pools
1083                                        .get(child_pool_key)
1084                                        .and_then(|p| pools.get(&p.fallback))
1085                                        .map_or(0, |p| pool_max_y_size(p, templates));
1086                                    child_pool_size.max(child_fallback_size)
1087                                })
1088                                .max()
1089                                .unwrap_or(0)
1090                        } else {
1091                            0
1092                        }
1093                    } else {
1094                        0
1095                    }
1096                } else {
1097                    0
1098                };
1099
1100                let mut candidate_bb_at_origin: Option<BoundingBox> = None;
1101
1102                // Try each target jigsaw
1103                for target_jigsaw_idx in candidate_jigsaw_order {
1104                    let target_jigsaw = candidate_jigsaws[target_jigsaw_idx];
1105                    if !can_attach(source_jigsaw, &target_jigsaw) {
1106                        continue;
1107                    }
1108
1109                    let target_jigsaw_local = target_jigsaw.pos;
1110
1111                    // Compute raw target position
1112                    let raw_target_x = target_jigsaw_world.0 - target_jigsaw_local.0;
1113                    let raw_target_z = target_jigsaw_world.2 - target_jigsaw_local.2;
1114
1115                    // Compute raw bounding box at that position
1116                    let raw_bb = if let Some(bb) = candidate_bb_at_origin {
1117                        bb.moved(raw_target_x, 0, raw_target_z)
1118                    } else {
1119                        let Some(bb) = element_bounding_box(
1120                            candidate_element,
1121                            templates,
1122                            0,
1123                            0,
1124                            0,
1125                            candidate_rotation,
1126                        ) else {
1127                            continue;
1128                        };
1129                        candidate_bb_at_origin = Some(bb);
1130                        bb.moved(raw_target_x, 0, raw_target_z)
1131                    };
1132
1133                    let target_jigsaw_local_y = target_jigsaw_local.1;
1134
1135                    // Compute Y position
1136                    let delta_y = source_jigsaw_local_y - target_jigsaw_local_y + fdy;
1137
1138                    let target_box_y = if source_rigid && candidate_rigid {
1139                        source_box_y + delta_y
1140                    } else {
1141                        let base_height = *source_jigsaw_base_height.get_or_insert_with(|| {
1142                            get_height(source_jigsaw_pos.0, source_jigsaw_pos.2)
1143                        });
1144                        base_height - target_jigsaw_local_y
1145                    };
1146
1147                    let y_offset = target_box_y - raw_bb.min_y;
1148                    let candidate_bb = BoundingBox::new(
1149                        raw_bb.min_x,
1150                        raw_bb.min_y + y_offset,
1151                        raw_bb.min_z,
1152                        raw_bb.max_x,
1153                        raw_bb.max_y + y_offset,
1154                        raw_bb.max_z,
1155                    );
1156                    let target_position = (raw_target_x, raw_bb.min_y + y_offset, raw_target_z);
1157
1158                    // Apply expansion hack: expand BB vertically to reserve space
1159                    // for potential children during assembly. The expanded BB is used
1160                    // for collision and is_inside checks; the original BB is stored
1161                    // in the piece for beardifier and world save.
1162                    let expanded_bb = if expand_to > 0 {
1163                        let new_size = (expand_to + 1).max(candidate_bb.max_y - candidate_bb.min_y);
1164                        BoundingBox::new(
1165                            candidate_bb.min_x,
1166                            candidate_bb.min_y,
1167                            candidate_bb.min_z,
1168                            candidate_bb.max_x,
1169                            candidate_bb.min_y + new_size,
1170                            candidate_bb.max_z,
1171                        )
1172                    } else {
1173                        candidate_bb
1174                    };
1175
1176                    // Collision check — vanilla tracks free space hierarchically:
1177                    // internal children use sourceFree (this piece's internal space),
1178                    // external children use contextFree (parent's context).
1179                    let effective_ctx = if attach_inside {
1180                        *internal_ctx_idx.get_or_insert_with(|| {
1181                            free_spaces.push(FreeSpace {
1182                                constraint: source_bb,
1183                                occupied: Vec::new(),
1184                            });
1185                            free_spaces.len() - 1
1186                        })
1187                    } else {
1188                        context_idx
1189                    };
1190
1191                    if free_spaces[effective_ctx].collides(&expanded_bb) {
1192                        continue;
1193                    }
1194
1195                    // Success! Place this piece — subtract from the collision domain.
1196                    free_spaces[effective_ctx].occupied.push(expanded_bb);
1197
1198                    // Compute ground level delta
1199                    let target_ground_level_delta = if candidate_rigid {
1200                        source_ground_level_delta - delta_y
1201                    } else {
1202                        candidate_projection.ground_level_delta()
1203                    };
1204
1205                    // Compute junction Y
1206                    let junction_y = if source_rigid {
1207                        source_box_y + source_jigsaw_local_y
1208                    } else if candidate_rigid {
1209                        target_box_y + target_jigsaw_local_y
1210                    } else {
1211                        let base_height = *source_jigsaw_base_height.get_or_insert_with(|| {
1212                            get_height(source_jigsaw_pos.0, source_jigsaw_pos.2)
1213                        });
1214                        base_height + delta_y / 2
1215                    };
1216
1217                    // Add junction to source piece
1218                    pieces[source_idx].junctions.push(JigsawJunction {
1219                        source_x: target_jigsaw_world.0,
1220                        source_ground_y: junction_y - source_jigsaw_local_y
1221                            + source_ground_level_delta,
1222                        source_z: target_jigsaw_world.2,
1223                        delta_y,
1224                        dest_projection: candidate_projection,
1225                    });
1226
1227                    let new_piece_idx = pieces.len();
1228
1229                    // Create target piece
1230                    let mut target_piece = PlacedPiece {
1231                        element: candidate_element.clone(),
1232                        template_location: element_location(candidate_element).cloned(),
1233                        position: target_position,
1234                        rotation: candidate_rotation,
1235                        bounding_box: candidate_bb,
1236                        assembly_bb: expanded_bb,
1237                        ground_level_delta: target_ground_level_delta,
1238                        projection: candidate_projection,
1239                        depth: depth + 1,
1240                        junctions: Vec::new(),
1241                    };
1242
1243                    // Add junction to target piece
1244                    target_piece.junctions.push(JigsawJunction {
1245                        source_x: source_jigsaw_pos.0,
1246                        source_ground_y: junction_y - target_jigsaw_local_y
1247                            + target_ground_level_delta,
1248                        source_z: source_jigsaw_pos.2,
1249                        delta_y: -delta_y,
1250                        dest_projection: source_projection,
1251                    });
1252
1253                    pieces.push(target_piece);
1254
1255                    // Queue for further expansion if within depth limit.
1256                    // The child inherits the effective collision context:
1257                    // internal children get sourceFree, external get contextFree.
1258                    if depth < config.max_depth {
1259                        queue.push((new_piece_idx, depth + 1, placement_priority, effective_ctx));
1260                    }
1261
1262                    // Break to next source jigsaw (one target per jigsaw)
1263                    continue 'source_jigsaw;
1264                }
1265            }
1266        }
1267    }
1268}
