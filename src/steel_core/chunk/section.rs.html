1//! This module contains the `Sections` and `ChunkSection` structs.
2use std::{fmt::Debug, io::Cursor, sync::Arc};
3
4use steel_utils::{BlockStateId, locks::SyncRwLock, serial::WriteTo};
5
6use crate::chunk::paletted_container::{BiomePalette, BlockPalette};

8/// A collection of chunk sections.
9#[derive(Debug, Clone)]
10pub struct Sections {
11    /// The sections in the collection.
12    pub sections: Arc<[SyncRwLock<ChunkSection>]>,
13}
14
15impl Sections {
16    /// Creates a new `Sections` from a box of owned `ChunkSection`s.
17    #[must_use]
18    pub fn from_owned(sections: Box<[ChunkSection]>) -> Self {
19        Self {
20            sections: sections.into_iter().map(SyncRwLock::new).collect(),
21        }
22    }
23
24    /// Gets a block at a relative position in the chunk.
25    #[must_use]
26    pub fn get_relative_block(
27        &self,
28        relative_x: usize,
29        relative_y: usize,
30        relative_z: usize,
31    ) -> Option<BlockStateId> {
32        debug_assert!(relative_x < BlockPalette::SIZE);
33        debug_assert!(relative_z < BlockPalette::SIZE);
34
35        let section_index = relative_y / BlockPalette::SIZE;
36        let relative_y = relative_y % BlockPalette::SIZE;
37        self.sections.get(section_index).map(|section| {
38            section
39                .read()
40                .states
41                .get(relative_x, relative_y, relative_z)
42        })
43    }
44
45    /// Sets a block at a relative position in the chunk.
46    pub fn set_relative_block(
47        &self,
48        relative_x: usize,
49        relative_y: usize,
50        relative_z: usize,
51        value: BlockStateId,
52    ) {
53        debug_assert!(relative_x < BlockPalette::SIZE);
54        debug_assert!(relative_z < BlockPalette::SIZE);
55
56        let idx = relative_y / BlockPalette::SIZE;
57        let relative_y = relative_y % BlockPalette::SIZE;
58        //println!(
59        //    "setting block at {}, {}, {} to {}",
60        //    relative_x, relative_y, relative_z, value.0
61        //);
62        self.sections[idx]
63            .write()
64            .states
65            .set(relative_x, relative_y, relative_z, value);
66    }
67}

69/// A chunk section.
70#[derive(Debug, Clone)]
71pub struct ChunkSection {
72    /// The block states in the section.
73    pub states: BlockPalette,
74    /// The biomes in the section.
75    pub biomes: BiomePalette,
76}
77
78impl ChunkSection {
79    /// Creates a new chunk section.
80    #[must_use]
81    pub fn new(states: BlockPalette) -> Self {
82        Self {
83            states,
84            biomes: BiomePalette::Homogeneous(0),
85        }
86    }
87
88    /// Creates a new empty chunk section.
89    #[must_use]
90    pub fn new_empty() -> Self {
91        Self {
92            states: BlockPalette::Homogeneous(BlockStateId(0)),
93            biomes: BiomePalette::Homogeneous(0),
94        }
95    }
96
97    /// Writes the chunk section to a writer.
98    ///
99    /// # Panics
100    /// - If the writer fails to write.
101    pub fn write(&self, writer: &mut Cursor<Vec<u8>>) {
102        self.states
103            .non_empty_block_count()
104            .write(writer)
105            .expect("Failed to write block count");
106
107        self.states
108            .write(writer)
109            .expect("Failed to write block states");
110        self.biomes.write(writer).expect("Failed to write biomes");
111    }
112}
