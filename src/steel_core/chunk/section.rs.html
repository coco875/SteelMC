12/// A wrapper around a chunk section.
13#[derive(Debug)]
14pub struct SectionHolder {
15    /// The chunk section data (requires lock to access).
16    pub section: SyncRwLock<ChunkSection>,
17}
18
19impl SectionHolder {
20    /// Creates a new section holder.
21    #[must_use]
22    pub const fn new(section: ChunkSection) -> Self {
23        Self {
24            section: SyncRwLock::new(section),
25        }
26    }
27
28    /// Returns true if this section contains any randomly-ticking blocks.
29    ///
30    /// Performs an unsynchronized read of the ticking block count to avoid
31    /// lock overhead on every section during random ticks. A stale read is
32    /// acceptable: worst case we acquire an unnecessary lock.
33    #[inline]
34    #[must_use]
35    pub fn is_randomly_ticking(&self) -> bool {
36        // SAFETY: `ticking_block_count` is a `u16` — reads are atomic on all
37        // supported platforms. A torn/stale value only causes a harmless
38        // false-positive (we take the lock when we didn't need to).
39        unsafe { (*self.section.data_ptr()).ticking_block_count > 0 }
40    }
41
42    /// Acquires a read lock on the section.
43    #[inline]
44    pub fn read(&self) -> parking_lot::RwLockReadGuard<'_, ChunkSection> {
45        self.section.read()
46    }
47
48    /// Acquires a write lock on the section.
49    #[inline]
50    pub fn write(&self) -> parking_lot::RwLockWriteGuard<'_, ChunkSection> {
51        self.section.write()
52    }
53}

62/// Cached section counter traits for one block state.
63#[derive(Clone, Copy, Debug, PartialEq, Eq)]
64pub(crate) struct BlockStateSectionCounts {
65    is_air: bool,
66    has_fluid: bool,

70static BLOCK_STATE_SECTION_COUNTS: LazyLock<Box<[BlockStateSectionCounts]>> = LazyLock::new(|| {
71    let mut counts = Vec::with_capacity(REGISTRY.blocks.state_to_block_lookup.len());
72    for state_index in 0..REGISTRY.blocks.state_to_block_lookup.len() {
73        let Ok(raw_state_id) = u16::try_from(state_index) else {
74            panic!("block state registry exceeded BlockStateId range");
75        };
76        counts.push(ChunkSection::block_state_section_counts_with(
77            BlockStateId(raw_state_id),
78            &BLOCK_BEHAVIORS,
79        ));
80    }
81    counts.into_boxed_slice()
82});

84impl Sections {
85    /// Creates a new `Sections` from a box of owned `ChunkSection`s.
86    #[must_use]
87    pub fn from_owned(sections: Box<[ChunkSection]>) -> Self {
88        let holders: Box<[SectionHolder]> = sections
89            .into_vec()
90            .into_iter()
91            .map(SectionHolder::new)
92            .collect();
93        Self { sections: holders }
94    }

96    /// Gets a block at a relative position in the chunk.
97    #[must_use]
98    pub fn get_relative_block(
99        &self,
100        relative_x: usize,
101        relative_y: usize,
102        relative_z: usize,
103    ) -> Option<BlockStateId> {
104        debug_assert!(relative_x < BlockPalette::SIZE);
105        debug_assert!(relative_z < BlockPalette::SIZE);
106
107        let section_index = relative_y / BlockPalette::SIZE;
108        let relative_y = relative_y % BlockPalette::SIZE;
109        self.sections.get(section_index).map(|section| {
110            section
111                .read()
112                .states
113                .get(relative_x, relative_y, relative_z)
114        })
115    }

117    /// Reads an entire column at `(x, z)` across all sections into a caller-owned buffer.
118    ///
119    /// Holds each section's read lock once for 16 Y reads instead of acquiring
120    /// a lock per block. Indexed by `relative_y` (0 = chunk min-y).
121    /// The buffer is resized if needed and reused across calls to avoid allocation.
122    pub fn read_column_into(&self, x: usize, z: usize, buf: &mut Vec<BlockStateId>) {
123        debug_assert!(x < BlockPalette::SIZE);
124        debug_assert!(z < BlockPalette::SIZE);
125
126        let total = self.sections.len() * 16;
127        buf.clear();
128        buf.resize(total, BlockStateId(0));
129        for (i, holder) in self.sections.iter().enumerate() {
130            let guard = holder.read();
131            let base = i * 16;
132            for ly in 0..16 {
133                buf[base + ly] = guard.states.get(x, ly, z);
134            }
135        }
136    }
137
138    /// Reads all biome palette values into a flat array.
139    ///
140    /// Indexed as `[section_idx * 64 + qy * 16 + qz * 4 + qx]`.
141    /// Holds each section's read lock once for all 64 biome reads.
142    #[must_use]
143    pub fn read_all_biomes(&self) -> Box<[u16]> {
144        let total = self.sections.len() * 64;
145        let mut biomes = vec![0u16; total];
146        for (i, holder) in self.sections.iter().enumerate() {
147            let guard = holder.read();
148            let base = i * 64;
149            for qy in 0..4 {
150                for qz in 0..4 {
151                    for qx in 0..4 {
152                        biomes[base + qy * 16 + qz * 4 + qx] = guard.biomes.get(qx, qy, qz);
153                    }
154                }
155            }
156        }
157        biomes.into_boxed_slice()
158    }
159
160    /// Visits every biome palette value in section order while holding each
161    /// section's read lock once.
162    pub fn for_each_biome_id(&self, mut visitor: impl FnMut(u16)) {
163        for holder in &self.sections {
164            let guard = holder.read();
165            for qy in 0..4 {
166                for qz in 0..4 {
167                    for qx in 0..4 {
168                        visitor(guard.biomes.get(qx, qy, qz));
169                    }
170                }
171            }
172        }
173    }
174
175    /// Writes multiple blocks in one column, holding each section's write guard
176    /// across all writes to that section. Most efficient when blocks are grouped
177    /// by section (e.g. descending `relative_y` from a top-to-bottom scan).
178    pub fn write_column_blocks(&self, x: usize, z: usize, blocks: &[(usize, BlockStateId)]) {
179        debug_assert!(x < BlockPalette::SIZE);
180        debug_assert!(z < BlockPalette::SIZE);
181
182        let mut i = 0;
183        while i < blocks.len() {
184            let section_idx = blocks[i].0 / BlockPalette::SIZE;
185            let mut guard = self.sections[section_idx].write();
186            while i < blocks.len() && blocks[i].0 / BlockPalette::SIZE == section_idx {
187                let (rel_y, value) = blocks[i];
188                guard.states.set(x, rel_y % BlockPalette::SIZE, z, value);
189                i += 1;
190            }
191        }
192    }
193
194    /// Writes a batch of blocks at arbitrary positions, holding each section's
195    /// write guard across consecutive entries in the same section. Blocks should
196    /// be roughly grouped by section index for best performance.
197    pub fn write_block_batch(&self, blocks: &[(usize, usize, usize, BlockStateId)]) {
198        let mut i = 0;
199        while i < blocks.len() {
200            let section_idx = blocks[i].1 / BlockPalette::SIZE;
201            let mut guard = self.sections[section_idx].write();
202            while i < blocks.len() && blocks[i].1 / BlockPalette::SIZE == section_idx {
203                let (x, rel_y, z, value) = blocks[i];
204                guard.states.set(x, rel_y % BlockPalette::SIZE, z, value);
205                i += 1;
206            }
207        }
208    }

210    /// Sets a block at a relative position in the chunk.
211    pub fn set_relative_block(
212        &self,
213        relative_x: usize,
214        relative_y: usize,
215        relative_z: usize,
216        value: BlockStateId,
217    ) {
218        debug_assert!(relative_x < BlockPalette::SIZE);
219        debug_assert!(relative_z < BlockPalette::SIZE);
220
221        let idx = relative_y / BlockPalette::SIZE;
222        let relative_y = relative_y % BlockPalette::SIZE;
223        self.sections[idx]
224            .write()
225            .states
226            .set(relative_x, relative_y, relative_z, value);
227    }

230/// A chunk section.
231///
232/// Contains a 16x16x16 cube of block states and biomes, along with cached
233/// counts for optimization (similar to vanilla's `LevelChunkSection`).
234#[derive(Debug)]
235pub struct ChunkSection {
236    /// The block states in the section.
237    pub states: BlockPalette,
238    /// The biomes in the section.
239    pub biomes: BiomePalette,
240    /// Number of non-air blocks in this section (0-4096).
241    /// Used to quickly check if a section is empty.
242    non_empty_block_count: u16,
243    /// Number of fluid-containing blocks in this section (0-4096).
244    /// Includes water, lava, and waterlogged blocks.
245    fluid_count: u16,
246    /// Number of randomly-ticking blocks in this section (0-4096).
247    pub ticking_block_count: u16,
248}

315    /// Recalculates both cached counters by iterating all blocks.
316    ///
317    /// This should be called after chunk loading or generation to initialize
318    /// the counters. It requires the block behavior registry to be initialized.
319    ///
320    /// # Panics
321    /// Panics if the block behavior registry has not been initialized.
322    pub fn recalculate_counts(&mut self) {
323        self.recalculate_counts_with(&BLOCK_BEHAVIORS);
324    }
325
326    /// Recalculates all cached counters using the provided behavior registry.
327    pub fn recalculate_counts_with(&mut self, block_behaviors: &BlockBehaviorRegistry) {
328        let mut non_empty: u16 = 0;
329        let mut fluid: u16 = 0;
330        let mut ticking: u16 = 0;
331
332        for y in 0..16 {
333            for z in 0..16 {
334                for x in 0..16 {
335                    let state = self.states.get(x, y, z);
336                    if !state.is_air() {
337                        non_empty += 1;
338                        let block = state.get_block();
339                        let behavior = block_behaviors.get_behavior(block);
340                        if behavior.is_randomly_ticking(state) {
341                            ticking += 1;
342                        }
343                    }
344                    let fluid_state = block_behaviors
345                        .get_behavior(state.get_block())
346                        .get_fluid_state(state);
347                    if !fluid_state.is_empty() {
348                        fluid += 1;
349                    }
350                }
351            }
352        }
353
354        self.non_empty_block_count = non_empty;
355        self.fluid_count = fluid;
356        self.ticking_block_count = ticking;
357    }

359    /// Sets a block state and updates the cached counters.
360    ///
361    /// Returns the old block state.
362    ///
363    /// # Panics
364    /// Panics if the block behavior registry has not been initialized.
365    pub fn set_block_state(
366        &mut self,
367        x: usize,
368        y: usize,
369        z: usize,
370        new_state: BlockStateId,
371    ) -> BlockStateId {
372        self.set_block_state_with(x, y, z, new_state, &BLOCK_BEHAVIORS)
373    }
374
375    /// Sets a block state and updates the cached counters using the provided behavior registry.
376    ///
377    /// Returns the old block state.
378    pub fn set_block_state_with(
379        &mut self,
380        x: usize,
381        y: usize,
382        z: usize,
383        new_state: BlockStateId,
384        block_behaviors: &BlockBehaviorRegistry,
385    ) -> BlockStateId {
386        let old_state = self.states.set(x, y, z, new_state);
387
388        if old_state != new_state {
389            let old_counts = Self::block_state_section_counts_with(old_state, block_behaviors);
390            let new_counts = Self::block_state_section_counts_with(new_state, block_behaviors);
391            self.apply_count_change(old_counts, new_counts);
392        }
393
394        old_state
395    }

397    /// Sets a block state and updates counters when the caller already knows
398    /// the replacement state's counter traits.
399    pub(crate) fn set_block_state_with_known_new_counts(
400        &mut self,
401        x: usize,
402        y: usize,
403        z: usize,
404        new_state: BlockStateId,
405        new_counts: BlockStateSectionCounts,
406    ) -> BlockStateId {
407        let old_state = self.states.set(x, y, z, new_state);
408        if old_state != new_state {
409            let old_counts = Self::block_state_section_counts(old_state);
410            self.apply_count_change(old_counts, new_counts);
411        }
412
413        old_state
414    }

416    /// Returns the cached-counter traits for a block state using the global
417    /// behavior registry.
418    pub(crate) fn block_state_section_counts(state: BlockStateId) -> BlockStateSectionCounts {
419        let Some(&counts) = BLOCK_STATE_SECTION_COUNTS.get(state.0 as usize) else {
420            panic!("invalid block state id {}", state.0);
421        };
422        counts
423    }
424
425    fn block_state_section_counts_with(
426        state: BlockStateId,
427        block_behaviors: &BlockBehaviorRegistry,
428    ) -> BlockStateSectionCounts {
429        let behavior = block_behaviors.get_behavior(state.get_block());
430        BlockStateSectionCounts {
431            is_air: state.is_air(),
432            has_fluid: !behavior.get_fluid_state(state).is_empty(),
433            randomly_ticking: behavior.is_randomly_ticking(state),
434        }
435    }
436
437    const fn apply_count_change(
438        &mut self,
439        old_counts: BlockStateSectionCounts,
440        new_counts: BlockStateSectionCounts,
441    ) {
442        if !old_counts.is_air && new_counts.is_air {
443            self.non_empty_block_count -= 1;
444        } else if old_counts.is_air && !new_counts.is_air {
445            self.non_empty_block_count += 1;
446        }
447
448        if old_counts.has_fluid && !new_counts.has_fluid {
449            self.fluid_count -= 1;
450        } else if !old_counts.has_fluid && new_counts.has_fluid {
451            self.fluid_count += 1;
452        }
453
454        if old_counts.randomly_ticking && !new_counts.randomly_ticking {
455            self.ticking_block_count -= 1;
456        } else if !old_counts.randomly_ticking && new_counts.randomly_ticking {
457            self.ticking_block_count += 1;
458        }
459    }

461    /// Writes the chunk section to a writer.
462    ///
463    /// # Panics
464    /// - If the writer fails to write.
465    pub fn write(&self, writer: &mut Cursor<Vec<u8>>) {
466        self.non_empty_block_count
467            .write(writer)
468            .expect("Failed to write block count");
469        self.fluid_count
470            .write(writer)
471            .expect("Failed to write fluid count");
472
473        self.states
474            .write(writer)
475            .expect("Failed to write block states");
476        self.biomes.write(writer).expect("Failed to write biomes");
477    }
