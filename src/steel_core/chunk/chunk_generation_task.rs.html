1//! This module contains the `ChunkGenerationTask` struct, which is used to generate chunks.
2use std::{
3    collections::HashMap,
4    future::Future,
5    sync::{Arc, atomic::AtomicBool},
6};
7
8use steel_utils::ChunkPos;
9
10use crate::chunk::{chunk_access::ChunkStatus, chunk_holder::ChunkHolder};
11
12/// A task that generates a chunk.
13pub struct ChunkGenerationTask {
14    /// The position of the chunk.
15    pub pos: ChunkPos,
16    /// The target status of the chunk.
17    pub target_status: ChunkStatus,
18    /// The status that the chunk is scheduled to be generated to.
19    pub scheduled_status: Option<ChunkStatus>,
20    /// Whether the task is marked for cancellation.
21    pub marked_for_cancel: AtomicBool,
22
23    /// A list of futures that will be ready when the neighbors are ready.
24    pub neighbor_ready: Vec<Box<dyn Future<Output = ()> + Send>>,
25    //TODO: We should make a custom struct in the future that can treat this as a fixed size array.
26    /// A cache of chunks that are needed for generation.
27    pub cache: HashMap<ChunkPos, Arc<ChunkHolder>>,
28    /// Whether the chunk needs to be generated.
29    pub needs_generation: bool,
30}
31
32impl ChunkGenerationTask {
33    /// Creates a new chunk generation task.
34    #[must_use]
35    pub fn new(pos: ChunkPos, target_status: ChunkStatus) -> Self {
36        Self {
37            pos,
38            target_status,
39            scheduled_status: None,
40            marked_for_cancel: AtomicBool::new(false),
41            neighbor_ready: Vec::new(),
42            cache: HashMap::new(),
43            needs_generation: true,
44        }
45    }
46}
