100    pub fn get_full_chunk(&self, pos: &ChunkPos) -> Option<Arc<ChunkAccess>> {
101        let chunk_holder = self.chunks.get_sync(pos)?;

104            .map(|guard| guard.as_ref().expect("Not empty by this stage").clone())

107    /// Records a block change at the given position.
108    /// This marks the chunk as having pending changes to broadcast.
109    pub fn block_changed(&self, pos: &BlockPos) {
110        let chunk_pos = ChunkPos::new(
111            SectionPos::block_to_section_coord(pos.0.x),
112            SectionPos::block_to_section_coord(pos.0.z),
113        );
114
115        if let Some(holder) = self.chunks.read_sync(&chunk_pos, |_, h| h.clone())
116            && holder.block_changed(pos)
117        {
118            // First change for this chunk - add to broadcast list
119            self.chunks_to_broadcast.lock().push(holder);
120        }
121    }

123    /// Broadcasts all pending block changes to nearby players.
124    ///
125    /// # Panics
126    /// Panics if a section has exactly one change (should never happen).
127    pub fn broadcast_changed_chunks(&self) {
128        let holders = {
129            let mut guard = self.chunks_to_broadcast.lock();
130            if guard.is_empty() {
131                return;
132            }
133            mem::take(&mut *guard)
134        };
135
136        let world = self.world_gen_context.world();
137
138        for holder in holders {
139            let chunk_pos = holder.get_pos();
140            let min_y = holder.min_y();
141
142            // Take all pending changes from this chunk holder
143            let changes_by_section = holder.take_changed_blocks();
144
145            if changes_by_section.is_empty() {
146                continue;
147            }
148
149            // Get players tracking this chunk
150            let tracking_players = world.player_area_map.get_tracking_players(chunk_pos);
151            if tracking_players.is_empty() {
152                continue;
153            }
154
155            // For each section with changes, send appropriate packet
156            for (section_index, changed_positions) in changes_by_section {
157                let section_y = min_y / 16 + section_index as i32;
158                let section_pos = SectionPos::new(chunk_pos.0.x, section_y, chunk_pos.0.y);
159
160                if changed_positions.len() == 1 {
161                    // Single block change - use CBlockUpdate
162                    let packed = *changed_positions.iter().next().expect("len == 1");
163                    let block_pos = section_pos.relative_to_block_pos(packed);
164                    let block_state = world.get_block_state(&block_pos);
165
166                    log::debug!(
167                        "Broadcasting single block update at {:?} state={:?} to {} players",
168                        block_pos,
169                        block_state,
170                        tracking_players.len()
171                    );
172
173                    let update_packet = CBlockUpdate {
174                        pos: block_pos,
175                        block_state,
176                    };
177
178                    for entity_id in &tracking_players {
179                        if let Some(player) = world.players.get_by_entity_id(*entity_id) {
180                            player.connection.send_packet(update_packet.clone());
181                        }
182                    }
183                } else {
184                    // Multiple block changes - use CSectionBlocksUpdate
185                    let changes: Vec<BlockChange> = changed_positions
186                        .iter()
187                        .map(|&packed| {
188                            let block_pos = section_pos.relative_to_block_pos(packed);
189                            let block_state = world.get_block_state(&block_pos);
190                            BlockChange {
191                                pos: block_pos,
192                                block_state,
193                            }
194                        })
195                        .collect();
196
197                    log::debug!(
198                        "Broadcasting {} block updates in section {:?} to {} players",
199                        changes.len(),
200                        section_pos,
201                        tracking_players.len()
202                    );
203
204                    let packet = CSectionBlocksUpdate {
205                        section_pos,
206                        changes,
207                    };
208
209                    for entity_id in &tracking_players {
210                        if let Some(player) = world.players.get_by_entity_id(*entity_id) {
211                            player.connection.send_packet(packet.clone());
212                        }
213                    }
214                }
215            }
216        }
217    }

219    /// Schedules a new generation task.
220    #[inline]
221    pub(crate) fn schedule_generation_task_b(
222        self: &Arc<Self>,
223        target_status: ChunkStatus,
224        pos: ChunkPos,
225    ) -> Arc<ChunkGenerationTask> {
226        let start = Instant::now();
227        let task = Arc::new(ChunkGenerationTask::new(
228            pos,
229            target_status,
230            self.clone(),
231            self.thread_pool.clone(),
232        ));
233        if start.elapsed() >= Duration::from_millis(1) {
234            log::warn!("schedule_generation_task_b took: {:?}", start.elapsed());
235        }
236        self.pending_generation_tasks.lock().push(task.clone());
237        task
238    }

240    /// Runs queued generation tasks.
241    pub fn run_generation_tasks_b(&self) {
242        let mut pending = self.pending_generation_tasks.lock();
243        if pending.is_empty() {
244            return;
245        }
246        //log::info!("Running {} generation tasks", pending.len());
247        let tasks = pending.drain(..).collect::<Vec<_>>();
248        tasks.into_par_iter().for_each(|task| {
249            self.task_tracker
250                .spawn_on(async move { task.run().await }, self.chunk_runtime.handle());
251        });
252    }

254    /// Updates scheduling for a chunk based on its new level.
255    /// Returns the chunk holder if it is active.
256    #[inline]
257    #[allow(clippy::missing_panics_doc, clippy::unwrap_used)]
258    pub fn update_chunk_level(
259        self: &Arc<Self>,
260        pos: &ChunkPos,
261        new_level: Option<u8>,
262    ) -> Option<Arc<ChunkHolder>> {
263        // Recover from unloading if possible, else create new holder.
264        let chunk_holder =
265            if let Some(holder) = self.chunks.read_sync(pos, |_, holder| holder.clone()) {
266                holder
267            } else {
268                new_level?;
269
270                if let Some(entry) = self.unloading_chunks.remove_sync(pos) {
271                    let _ = self.chunks.insert_sync(*pos, entry.1.clone());
272                    entry.1
273                } else {
274                    let holder = Arc::new(ChunkHolder::new(
275                        *pos,
276                        new_level.unwrap(),
277                        self.world_gen_context.min_y(),
278                        self.world_gen_context.height(),
279                    ));
280                    let _ = self.chunks.insert_sync(*pos, holder.clone());
281                    holder
282                }
283            };
284
285        if let Some(level) = new_level {
286            let old = chunk_holder.ticket_level.swap(level, Ordering::Relaxed);
287            if old != level {
288                chunk_holder.update_highest_allowed_status(level);
289            }
290            Some(chunk_holder)
291        } else {
292            //log::info!("Unloading chunk at {pos:?}");
293            chunk_holder.cancel_generation_task();
294
295            // Check for two cause we are also holding a reference to the chunk
296            if let Some((_, holder)) = self
297                .chunks
298                .remove_if_sync(pos, |chunk| Arc::strong_count(chunk) == 2)
299            {
300                let _ = self.unloading_chunks.insert_sync(*pos, holder.clone());
301            } else {
302                chunk_holder.ticket_level.store(u8::MAX, Ordering::Relaxed);
303                chunk_holder.update_highest_allowed_status(u8::MAX);
304            }
305            None
306        }
307    }

309    /// Processes chunk updates.
310    pub fn tick_b(self: &Arc<Self>, tick_count: u64) {
311        let start = Instant::now();
312
313        {
314            let mut ct = self.chunk_tickets.lock();
315
316            let updates_start = Instant::now();
317            // Only process chunks that actually changed
318            let changes: Vec<LevelChange> = ct.run_all_updates().to_vec();
319            let updates_elapsed = updates_start.elapsed();
320
321            let holder_creation_start = Instant::now();
322            let holders_to_schedule: Vec<_> = changes
323                .iter()
324                .filter_map(|change| {
325                    self.update_chunk_level(&change.pos, change.new_level)
326                        .map(|holder| (holder, change.new_level))
327                })
328                .collect();
329            let holder_creation_elapsed = holder_creation_start.elapsed();
330
331            let schedule_start = Instant::now();
332            holders_to_schedule.par_iter().for_each(|(holder, level)| {
333                if let Some(level) = level
334                    && is_full(*level)
335                {
336                    holder.schedule_chunk_generation_task_b(ChunkStatus::Full, self);
337                }
338            });
339            let schedule_elapsed = schedule_start.elapsed();
340
341            if updates_elapsed >= SLOW_TASK_WARN_THRESHOLD {
342                log::warn!("chunk_tickets run_updates slow: {updates_elapsed:?}");
343            }
344            if holder_creation_elapsed >= SLOW_TASK_WARN_THRESHOLD {
345                log::warn!("holder_creation slow: {holder_creation_elapsed:?}");
346            }
347            if schedule_elapsed >= SLOW_TASK_WARN_THRESHOLD {
348                log::warn!("schedule slow: {schedule_elapsed:?}");
349            }
350        };
351
352        let start_gen = Instant::now();
353        self.run_generation_tasks_b();
354        let gen_elapsed = start_gen.elapsed();
355        if gen_elapsed >= SLOW_TASK_WARN_THRESHOLD {
356            log::warn!("run_generation_tasks_b slow: {gen_elapsed:?}");
357        }
358
359        let start_broadcast = Instant::now();
360        self.broadcast_changed_chunks();
361        let broadcast_elapsed = start_broadcast.elapsed();
362        if broadcast_elapsed >= SLOW_TASK_WARN_THRESHOLD {
363            log::warn!("broadcast_changed_chunks slow: {broadcast_elapsed:?}");
364        }
365
366        let start_unload = Instant::now();
367        self.process_unloads();
368        let unload_elapsed = start_unload.elapsed();
369        if unload_elapsed >= SLOW_TASK_WARN_THRESHOLD {
370            log::warn!("process_unloads slow: {unload_elapsed:?}");
371        }
372
373        let tick_elapsed = start.elapsed();
374        if tick_elapsed >= SLOW_TASK_WARN_THRESHOLD {
375            log::warn!("Tick_b slow: total {tick_elapsed:?}");
376        }
377
378        if tick_count.is_multiple_of(100) {
379            log::debug!(
380                "Chunk map entries: {}, unloading chunks: {}",
381                self.chunks.len(),
382                self.unloading_chunks.len()
383            );
384        }
385    }

387    /// Saves a chunk to disk.
388    ///
389    /// This function is currently a placeholder for the actual saving logic.
390    #[allow(clippy::missing_panics_doc, clippy::unwrap_used)]
391    pub async fn save_chunk(&self, chunk_holder: &Arc<ChunkHolder>, chunk_map: &Arc<ChunkMap>) {
392        let chunk = chunk_holder.try_chunk(ChunkStatus::StructureStarts);
393
394        if let Some(chunk) = chunk {
395            let status = chunk_holder
396                .persisted_status()
397                .expect("The check above confirmed it exists");
398
399            let result = chunk_map.region_manager.save_chunk(&chunk, status).await;
400
401            match result {
402                Ok(_) => {
403                    let res = chunk_map
404                        .unloading_chunks
405                        .remove_async(&chunk_holder.get_pos())
406                        .await;
407                    if res.is_some() {
408                        if let Err(e) = chunk_map
409                            .region_manager
410                            .release_chunk(chunk_holder.get_pos())
411                            .await
412                        {
413                            log::error!("Error releasing chunk: {e}");
414                        }
415                    } else {
416                        // Chunk was recovered
417                    }
418                }
419                Err(e) => {
420                    log::error!("Error saving chunk: {e}");
421                }
422            }
423        } else {
424            // Chunk was at Empty stage so no need to save it
425            let _ = chunk_map
426                .unloading_chunks
427                .remove_async(&chunk_holder.get_pos())
428                .await;
429        }
430    }

432    /// Processes chunks that are pending unload.
433    ///
434    /// This method iterates over the chunks in the `unloading_chunks` map.
435    /// If a chunk is only held by the map (strong count is 1), it is removed
436    /// and a background task is spawned to save it.
437    pub fn process_unloads(self: &Arc<Self>) {
438        self.unloading_chunks.iter_sync(|_pos, holder| {
439            // If the strong count is 1, it means only this map holds a reference to the chunk.
440            // We can safely unload it.
441            if Arc::strong_count(holder) == 1 {
442                let holder_clone = holder.clone();
443                let map_clone = self.clone();
444
445                self.task_tracker.spawn(async move {
446                    map_clone.save_chunk(&holder_clone, &map_clone).await;
447                });
448            }
449            true
450        });
451    }

453    /// Updates the player's status in the chunk map.
454    pub fn update_player_status(&self, player: &Player) {
455        let current_chunk_pos = *player.last_chunk_pos.lock();
456        let view_distance = STEEL_CONFIG.view_distance;
457
458        let new_view = PlayerChunkView::new(current_chunk_pos, view_distance);
459        let mut last_view_guard = player.last_tracking_view.lock();
460
461        if last_view_guard.as_ref() != Some(&new_view) {
462            let mut chunk_tickets = self.chunk_tickets.lock();
463
464            let connection = &player.connection;
465            let world = self.world_gen_context.world();
466
467            if let Some(last_view) = last_view_guard.as_ref() {
468                if last_view.center != new_view.center
469                    || last_view.view_distance != new_view.view_distance
470                {
471                    chunk_tickets.remove_ticket(
472                        last_view.center,
473                        MAX_VIEW_DISTANCE.saturating_sub(last_view.view_distance),
474                    );
475                    chunk_tickets.add_ticket(
476                        new_view.center,
477                        MAX_VIEW_DISTANCE.saturating_sub(new_view.view_distance),
478                    );
479
480                    connection.send_packet(CSetChunkCenter {
481                        x: new_view.center.0.x,
482                        y: new_view.center.0.y,
483                    });
484                }
485
486                // Track chunks for PlayerAreaMap update
487                let mut added_chunks = Vec::new();
488                let mut removed_chunks = Vec::new();
489
490                // We lock here to ensure we have unique access for the duration of the diff
491                let mut chunk_sender = player.chunk_sender.lock();
492                PlayerChunkView::difference(
493                    last_view,
494                    &new_view,
495                    |pos, ctx: &mut (&mut _, &mut Vec<_>, &mut Vec<_>)| {
496                        ctx.0.mark_chunk_pending_to_send(pos);
497                        ctx.1.push(pos);
498                    },
499                    |pos, ctx: &mut (&mut _, &mut Vec<_>, &mut Vec<_>)| {
500                        ctx.0.drop_chunk(connection, pos);
501                        ctx.2.push(pos);
502                    },
503                    &mut (&mut chunk_sender, &mut added_chunks, &mut removed_chunks),
504                );
505                drop(chunk_sender);
506
507                // Update the player area map with the diff
508                world.player_area_map.on_player_view_change(
509                    player.entity_id,
510                    &added_chunks,
511                    &removed_chunks,
512                );
513            } else {
514                chunk_tickets.add_ticket(
515                    new_view.center,
516                    MAX_VIEW_DISTANCE.saturating_sub(new_view.view_distance),
517                );
518
519                let mut chunk_sender = player.chunk_sender.lock();
520                new_view.for_each(|pos| {
521                    chunk_sender.mark_chunk_pending_to_send(pos);
522                });
523                drop(chunk_sender);
524
525                // First time - add all chunks in view to player area map
526                world.player_area_map.on_player_join(player, &new_view);
527            }
528
529            *last_view_guard = Some(new_view);
530        }
531    }

533    /// Removes a player from the chunk map.
534    pub fn remove_player(&self, player: &Player) {
535        // Okay to lock sync lock here cause it has low contention
536        let mut last_view_guard = player.last_tracking_view.lock();
537        if let Some(last_view) = last_view_guard.take() {
538            drop(last_view_guard);
539            let mut chunk_tickets = self.chunk_tickets.lock();
540            chunk_tickets.remove_ticket(
541                last_view.center,
542                MAX_VIEW_DISTANCE.saturating_sub(last_view.view_distance),
543            );
544        }
545    }

547    /// Saves all dirty chunks to disk.
548    ///
549    /// This method should be called during graceful shutdown to ensure all
550    /// modified chunks are persisted. It saves:
551    /// 1. All dirty chunks in the active `chunks` map
552    /// 2. All chunks pending unload in the `unloading_chunks` map
553    /// 3. Closes all region file handles (flushing headers)
554    ///
555    /// Returns the number of chunks saved.
556    pub async fn save_all_chunks(self: &Arc<Self>) -> io::Result<usize> {
557        let mut saved_count = 0;
558
559        // Collect all chunks from both maps
560        let all_chunks: Vec<Arc<ChunkHolder>> = {
561            let mut chunks = Vec::new();
562            self.chunks.iter_sync(|_, holder| {
563                chunks.push(holder.clone());
564                true
565            });
566            self.unloading_chunks.iter_sync(|_, holder| {
567                chunks.push(holder.clone());
568                true
569            });
570            chunks
571        };
572
573        log::info!("Saving {} chunks...", all_chunks.len());
574
575        // Save all chunks that have data
576        for holder in &all_chunks {
577            if let Some(chunk) = holder.try_chunk(ChunkStatus::StructureStarts)
578                && let Some(status) = holder.persisted_status()
579            {
580                match self.region_manager.save_chunk(&chunk, status).await {
581                    Ok(true) => saved_count += 1,
582                    Ok(false) => {} // Not dirty
583                    Err(e) => {
584                        log::error!("Failed to save chunk at {:?}: {e}", holder.get_pos());
585                    }
586                }
587            }
588        }
589
590        // Close all region files (flushes headers and releases file handles)
591        if let Err(e) = self.region_manager.close_all().await {
592            log::error!("Failed to close region files: {e}");
593        }
594
595        log::info!(
596            "Saved {saved_count} dirty chunks (checked {} total)",
597            all_chunks.len()
598        );
599
600        Ok(saved_count)
601    }
