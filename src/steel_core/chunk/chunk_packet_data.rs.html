1//! This module contains the `ChunkPacketData` struct, which is used to create the data for a chunk packet.
2use std::io::Cursor;
3
4use crate::chunk::level_chunk::LevelChunk;
5
6/// A struct that contains the data for a chunk packet.
7/// This is equivalent to `ClientboundLevelChunkPacketData` in vanilla.
8///
9/// We have it in world because protocol has no access to world.
10pub struct ChunkPacketData<'a> {
11    /// The chunk to create the packet data for.
12    pub chunk: &'a LevelChunk,
13}
14
15impl<'a> ChunkPacketData<'a> {
16    /// Creates a new `ChunkPacketData`.
17    #[must_use]
18    pub fn new(chunk: &'a LevelChunk) -> Self {
19        Self { chunk }
20    }
21
22    /// Extracts the chunk data from the chunk.
23    #[must_use]
24    pub fn extract_chunk_data(&self) -> Vec<u8> {
25        let mut buf = Cursor::new(Vec::new());
26
27        for section in &self.chunk.sections.sections {
28            section.write(&mut buf);
29        }
30
31        buf.into_inner()
32    }
33}
