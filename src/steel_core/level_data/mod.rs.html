1//! Level data persistence module.
2//!
3//! This module handles saving and loading world-level data like game rules,
4//! time, weather, spawn point, and seed. This data is stored in `level.toml`
5//! in each world's directory.

20/// Persistent level data that gets saved to disk.
21#[derive(Debug, Clone, Serialize, Deserialize)]
22pub struct LevelData {
23    /// World seed for terrain generation.
24    pub seed: i64,
25    /// Total game time in ticks.
26    pub game_time: i64,
27    /// Time of day in ticks (0-24000).
28    pub day_time: i64,
29    /// World spawn point.
30    pub spawn: SpawnPoint,
31    /// Weather state.
32    pub weather: WeatherState,
33    /// World difficulty.
34    #[serde(default)]
35    pub difficulty: Difficulty,
36    /// Whether the difficulty is locked.
37    #[serde(default)]
38    pub difficulty_locked: bool,
39    /// Game rules (stored as name -> value pairs for serialization).
40    pub game_rules: FxHashMap<String, GameRuleValue>,
41    /// Runtime game rule values (not serialized, loaded from `game_rules`).
42    #[serde(skip)]
43    pub game_rules_values: GameRuleValues,
44    /// Whether the world has been initialized.
45    pub initialized: bool,
46    /// Generator settings this persisted world was created with.
47    #[serde(default)]
48    pub generation: Option<WorldGenerationSettings>,
49}

91fn describe_generation_settings(settings: &WorldGenerationSettings) -> String {
92    format!(
93        "generator {}, dimension_type {}, min_y {}, height {}, config {}",
94        settings.generator,
95        settings.dimension_type,
96        settings.min_y,
97        settings.height,
98        generation_config_string(&settings.config),
99    )
100}
101
102fn generation_config_string(config: &toml::Value) -> String {
103    match toml::to_string(config) {
104        Ok(value) => value.trim().to_owned(),
105        Err(_) => "<invalid generator config>".to_owned(),
106    }
107}

154impl LevelData {
155    /// Creates new level data with the given seed.
156    #[must_use]
157    pub fn new_with_seed(seed: i64) -> Self {
158        Self::new_with_seed_and_difficulty(seed, Difficulty::default())
159    }
160
161    /// Creates new level data with the given seed and difficulty.
162    #[must_use]
163    pub fn new_with_seed_and_difficulty(seed: i64, difficulty: Difficulty) -> Self {
164        Self {
165            seed,
166            game_time: 0,
167            day_time: 0,
168            spawn: SpawnPoint::default(),
169            weather: WeatherState::default(),
170            difficulty,
171            difficulty_locked: false,
172            game_rules: FxHashMap::default(),
173            game_rules_values: GameRuleValues::new(&REGISTRY.game_rules),
174            initialized: false,
175            generation: None,
176        }
177    }
178
179    /// Verifies saved generator metadata against the current config.
180    ///
181    /// Returns whether missing metadata was adopted and should be saved.
182    pub fn validate_generation_settings(
183        &mut self,
184        expected: WorldGenerationSettings,
185    ) -> io::Result<bool> {
186        match self.generation.as_ref() {
187            Some(saved) if saved == &expected => Ok(false),
188            Some(saved) => Err(io::Error::new(
189                io::ErrorKind::InvalidData,
190                format!(
191                    "world generation settings do not match saved level data: saved {}; configured {}. Delete or regenerate this world's saved chunks, or restore the previous generator config.",
192                    describe_generation_settings(saved),
193                    describe_generation_settings(&expected),
194                ),
195            )),
196            None => {
197                self.generation = Some(expected);
198                Ok(true)
199            }
200        }
201    }
202
203    /// Loads game rules from the serialized map into the runtime values.
204    pub fn load_game_rules(&mut self) {
205        self.game_rules_values = GameRuleValues::new(&REGISTRY.game_rules);
206        for (name, value) in &self.game_rules {
207            self.game_rules_values
208                .set_by_name(name, *value, &REGISTRY.game_rules);
209        }
210    }
211
212    /// Saves game rules from the runtime values to the serialized map.
213    pub fn save_game_rules(&mut self) {
214        self.game_rules.clear();
215        for (_, rule) in REGISTRY.game_rules.iter() {
216            let name = rule.key.path.to_string();
217            let value = self.game_rules_values.get(rule, &REGISTRY.game_rules);
218            self.game_rules.insert(name, value);
219        }
220    }
221
222    /// Gets the spawn position as a `BlockPos`.
223    #[must_use]
224    pub const fn spawn_pos(&self) -> BlockPos {
225        BlockPos::new(self.spawn.x, self.spawn.y, self.spawn.z)
226    }
227
228    /// Sets the spawn position from a `BlockPos`.
229    pub const fn set_spawn_pos(&mut self, pos: BlockPos) {
230        self.spawn.x = pos.x();
231        self.spawn.y = pos.y();
232        self.spawn.z = pos.z();
233    }
234}

236/// Manages level data persistence for a world.
237pub struct LevelDataManager {
238    /// Path to the level.toml file.
239    path: Option<PathBuf>,
240    /// Cached level data.
241    data: LevelData,
242    /// Whether data has been modified since last save.
243    dirty: bool,
244}
245
246impl LevelDataManager {
247    /// Creates a new level data manager for the given world directory.
248    ///
249    /// If `level.toml` exists, it will be loaded (the provided seed is ignored).
250    /// Otherwise, new data will be created with the provided seed.
251    pub async fn new(
252        world_dir: Option<impl AsRef<Path>>,
253        seed: i64,
254        difficulty: Difficulty,
255        generation: WorldGenerationSettings,
256    ) -> io::Result<Self> {
257        let (data, path, dirty) = if let Some(dir) = &world_dir {
258            let path = dir.as_ref().join("level.toml");
259
260            let (data, dirty) = if path.exists() {
261                // Load existing level data (seed from file takes precedence)
262                let content = fs::read_to_string(&path).await?;
263                let mut loaded: LevelData = toml::from_str(&content).map_err(|e| {
264                    io::Error::new(
265                        io::ErrorKind::InvalidData,
266                        format!("Invalid level.toml: {e}"),
267                    )
268                })?;
269                // Initialize runtime game rules from serialized values
270                loaded.load_game_rules();
271                let adopted_generation = loaded.validate_generation_settings(generation)?;
272                (loaded, adopted_generation)
273            } else {
274                // Create new level data with the provided defaults.
275                let mut data = LevelData::new_with_seed_and_difficulty(seed, difficulty);
276                data.generation = Some(generation);
277                (data, true)
278            };
279            (data, Some(path), dirty)
280        } else {
281            let mut data = LevelData::new_with_seed_and_difficulty(seed, difficulty);
282            data.generation = Some(generation);
283            (data, None, false)
284        };
285
286        Ok(Self { path, data, dirty })
287    }
288
289    /// Loads the saved world seed from `level.toml`, or returns the provided default.
290    pub async fn load_seed_or_default(
291        world_dir: Option<impl AsRef<Path>>,
292        default_seed: i64,
293    ) -> io::Result<i64> {
294        let Some(dir) = world_dir else {
295            return Ok(default_seed);
296        };
297
298        let path = dir.as_ref().join("level.toml");
299        if !path.exists() {
300            return Ok(default_seed);
301        }
302
303        let content = fs::read_to_string(path).await?;
304        let saved: SavedLevelSeed = toml::from_str(&content).map_err(|e| {
305            io::Error::new(
306                io::ErrorKind::InvalidData,
307                format!("Invalid level.toml: {e}"),
308            )
309        })?;
310        Ok(saved.seed)
311    }

334    }
335
336    /// Saves the level data to disk if it has been modified.
337    pub async fn save(&mut self) -> io::Result<()> {
338        if !self.dirty {
339            return Ok(());
340        }
341
342        let Some(world_path) = &self.path else {
343            self.dirty = false;
344            return Ok(());
345        };
346        if let Some(parent) = world_path.parent() {
347            fs::create_dir_all(parent).await?;
348        }
349
350        // Export runtime game rules to serializable format before saving
351        self.data.save_game_rules();
352
353        let content = toml::to_string_pretty(&self.data)
354            .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e))?;
355        fs::write(world_path, content).await?;
356        self.dirty = false;
357
358        log::debug!("Saved level data to {}", world_path.display());
359        Ok(())
360    }

477    fn settings(dimension_type: &str, height: i32) -> WorldGenerationSettings {
478        let mut config = Map::new();
479        config.insert(
480            "dimension_type".to_owned(),
481            toml::Value::String(dimension_type.to_owned()),
482        );
483        WorldGenerationSettings {
484            generator: Identifier::vanilla_static("flat"),
485            config: toml::Value::Table(config),
486            dimension_type: dimension_type
487                .parse()
488                .expect("valid dimension type identifier"),
489            min_y: 0,
490            height,
491        }
492    }

494    fn temp_level_data_dir(test_name: &str) -> PathBuf {
495        let unique = SystemTime::now()
496            .duration_since(UNIX_EPOCH)
497            .expect("system clock should be after unix epoch")
498            .as_nanos();
499        let path = env::temp_dir().join(format!(
500            "steel-level-data-{test_name}-{}-{unique}",
501            process::id()
502        ));
503        std_fs::create_dir_all(&path).expect("temp level data dir should be created");
504        path
505    }

507    #[tokio::test]
508    async fn load_seed_prefers_saved_level_toml() {
509        let dir = temp_level_data_dir("saved-seed");
510        std_fs::write(dir.join("level.toml"), "seed = 42\n").expect("level.toml should be written");
511
512        let seed = LevelDataManager::load_seed_or_default(Some(dir.as_path()), 7)
513            .await
514            .expect("saved level seed should load");
515        let _ = std_fs::remove_dir_all(&dir);

517        assert_eq!(seed, 42);
518    }
