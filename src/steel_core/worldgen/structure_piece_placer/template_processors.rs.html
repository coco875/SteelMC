17impl StructurePiecePlacer {
18    pub(super) fn template_processors<'a>(
19        registry: &'a Registry,
20        processors: &'a TemplateProcessorList,
21        hardcoded_processors: &'a mut Vec<StructureProcessorKind>,
22    ) -> &'a [StructureProcessorKind] {
23        match processors {
24            TemplateProcessorList::Empty => &[],
25            TemplateProcessorList::Registry(key) => {
26                let Some(processor_list) = registry.structure_processors.by_key(key) else {
27                    panic!("template piece references unknown processor list {key}");
28                };
29                &processor_list.data.processors
30            }
31            TemplateProcessorList::OceanRuin {
32                biome_temp,
33                integrity,
34            } => {
35                hardcoded_processors.extend(Self::ocean_ruin_processors(*biome_temp, *integrity));
36                hardcoded_processors.as_slice()
37            }
38            TemplateProcessorList::RuinedPortal {
39                vertical_placement,
40                properties,
41            } => {
42                hardcoded_processors.extend(Self::ruined_portal_processors(
43                    *vertical_placement,
44                    *properties,
45                ));
46                hardcoded_processors.as_slice()
47            }
48        }
49    }
50
51    fn ocean_ruin_processors(
52        biome_temp: OceanRuinBiomeTempData,
53        integrity: f32,
54    ) -> Vec<StructureProcessorKind> {
55        let (source, target, loot_table) = match biome_temp {
56            OceanRuinBiomeTempData::Warm => (
57                "sand",
58                "suspicious_sand",
59                Identifier::vanilla_static("archaeology/ocean_ruin_warm"),
60            ),
61            OceanRuinBiomeTempData::Cold => (
62                "gravel",
63                "suspicious_gravel",
64                Identifier::vanilla_static("archaeology/ocean_ruin_cold"),
65            ),
66        };
67
68        vec![
69            StructureProcessorKind::BlockRot {
70                rottable_blocks: None,
71                integrity,
72            },
73            StructureProcessorKind::Capped {
74                delegate: Box::new(StructureProcessorKind::Rule {
75                    rules: vec![Self::append_loot_replace_rule(source, target, loot_table)],
76                }),
77                limit: IntProvider::Constant(5),
78            },
79        ]
80    }
81
82    fn ruined_portal_processors(
83        vertical_placement: RuinedPortalPlacementData,
84        properties: RuinedPortalProperties,
85    ) -> Vec<StructureProcessorKind> {
86        let mut rules = vec![
87            Self::random_block_replace_rule("gold_block", 0.3, "air"),
88            Self::ruined_portal_lava_rule(vertical_placement, properties),
89        ];
90        if !properties.cold {
91            rules.push(Self::random_block_replace_rule(
92                "netherrack",
93                0.07,
94                "magma_block",
95            ));
96        }
97
98        let mut processors = vec![
99            StructureProcessorKind::Rule { rules },
100            StructureProcessorKind::BlockAge {
101                mossiness: properties.mossiness,
102            },
103            StructureProcessorKind::ProtectedBlocks {
104                cannot_replace: vanilla_block_tags::FEATURES_CANNOT_REPLACE_TAG,
105            },
106            StructureProcessorKind::LavaSubmergedBlock,
107        ];
108        if properties.replace_with_blackstone {
109            processors.push(StructureProcessorKind::BlackstoneReplace);
110        }
111        processors
112    }
113
114    fn ruined_portal_lava_rule(
115        vertical_placement: RuinedPortalPlacementData,
116        properties: RuinedPortalProperties,
117    ) -> ProcessorRuleData {
118        if vertical_placement == RuinedPortalPlacementData::OnOceanFloor {
119            Self::block_replace_rule("lava", "magma_block")
120        } else if properties.cold {
121            Self::block_replace_rule("lava", "netherrack")
122        } else {
123            Self::random_block_replace_rule("lava", 0.2, "magma_block")
124        }
125    }
126
127    const fn block_replace_rule(source: &'static str, target: &'static str) -> ProcessorRuleData {
128        ProcessorRuleData {
129            input_predicate: StructureRuleTestData::BlockMatch {
130                block: Identifier::vanilla_static(source),
131            },
132            location_predicate: StructureRuleTestData::AlwaysTrue,
133            position_predicate: PosRuleTestData::AlwaysTrue,
134            output_state: Self::block_state_data(target),
135            block_entity_modifier: RuleBlockEntityModifierData::Passthrough,
136        }
137    }
138
139    const fn random_block_replace_rule(
140        source: &'static str,
141        probability: f32,
142        target: &'static str,
143    ) -> ProcessorRuleData {
144        ProcessorRuleData {
145            input_predicate: StructureRuleTestData::RandomBlockMatch {
146                block: Identifier::vanilla_static(source),
147                probability,
148            },
149            location_predicate: StructureRuleTestData::AlwaysTrue,
150            position_predicate: PosRuleTestData::AlwaysTrue,
151            output_state: Self::block_state_data(target),
152            block_entity_modifier: RuleBlockEntityModifierData::Passthrough,
153        }
154    }
155
156    const fn append_loot_replace_rule(
157        source: &'static str,
158        target: &'static str,
159        loot_table: Identifier,
160    ) -> ProcessorRuleData {
161        ProcessorRuleData {
162            input_predicate: StructureRuleTestData::BlockMatch {
163                block: Identifier::vanilla_static(source),
164            },
165            location_predicate: StructureRuleTestData::AlwaysTrue,
166            position_predicate: PosRuleTestData::AlwaysTrue,
167            output_state: Self::block_state_data(target),
168            block_entity_modifier: RuleBlockEntityModifierData::AppendLoot { loot_table },
169        }
170    }
171
172    const fn block_state_data(block: &'static str) -> BlockStateData {
173        BlockStateData {
174            name: Identifier::vanilla_static(block),
175            properties: BTreeMap::new(),
176        }
177    }
