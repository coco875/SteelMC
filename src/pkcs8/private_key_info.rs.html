35/// PKCS#8 `PrivateKeyInfo`.
36///
37/// ASN.1 structure containing an `AlgorithmIdentifier`, private key
38/// data in an algorithm specific format, and optional attributes
39/// (ignored by this implementation).
40///
41/// Supports PKCS#8 v1 as described in [RFC 5208] and PKCS#8 v2 as described
42/// in [RFC 5958]. PKCS#8 v2 keys include an additional public key field.
43///
44/// # PKCS#8 v1 `PrivateKeyInfo`
45///
46/// Described in [RFC 5208 Section 5]:
47///
48/// ```text
49/// PrivateKeyInfo ::= SEQUENCE {
50///         version                   Version,
51///         privateKeyAlgorithm       PrivateKeyAlgorithmIdentifier,
52///         privateKey                PrivateKey,
53///         attributes           [0]  IMPLICIT Attributes OPTIONAL }
54///
55/// Version ::= INTEGER
56///
57/// PrivateKeyAlgorithmIdentifier ::= AlgorithmIdentifier
58///
59/// PrivateKey ::= OCTET STRING
60///
61/// Attributes ::= SET OF Attribute
62/// ```
63///
64/// # PKCS#8 v2 `OneAsymmetricKey`
65///
66/// PKCS#8 `OneAsymmetricKey` as described in [RFC 5958 Section 2]:
67///
68/// ```text
69/// PrivateKeyInfo ::= OneAsymmetricKey
70///
71/// OneAsymmetricKey ::= SEQUENCE {
72///     version                   Version,
73///     privateKeyAlgorithm       PrivateKeyAlgorithmIdentifier,
74///     privateKey                PrivateKey,
75///     attributes            [0] Attributes OPTIONAL,
76///     ...,
77///     [[2: publicKey        [1] PublicKey OPTIONAL ]],
78///     ...
79///   }
80///
81/// Version ::= INTEGER { v1(0), v2(1) } (v1, ..., v2)
82///
83/// PrivateKeyAlgorithmIdentifier ::= AlgorithmIdentifier
84///
85/// PrivateKey ::= OCTET STRING
86///
87/// Attributes ::= SET OF Attribute
88///
89/// PublicKey ::= BIT STRING
90/// ```
91///
92/// [RFC 5208]: https://tools.ietf.org/html/rfc5208
93/// [RFC 5958]: https://datatracker.ietf.org/doc/html/rfc5958
94/// [RFC 5208 Section 5]: https://tools.ietf.org/html/rfc5208#section-5
95/// [RFC 5958 Section 2]: https://datatracker.ietf.org/doc/html/rfc5958#section-2
96#[derive(Clone)]
97pub struct PrivateKeyInfo<Params, Key, PubKey> {
98    /// X.509 `AlgorithmIdentifier` for the private key type.
99    pub algorithm: AlgorithmIdentifier<Params>,
100
101    /// Private key data. Exact content format is different between algorithms.
102    pub private_key: Key,
103
104    /// Public key data, optionally available if version is V2.
105    pub public_key: Option<PubKey>,
106}
107
108impl<Params, Key, PubKey> PrivateKeyInfo<Params, Key, PubKey> {
109    /// Create a new PKCS#8 [`PrivateKeyInfo`] message.
110    ///
111    /// This is a helper method which initializes `attributes` and `public_key`
112    /// to `None`, helpful if you aren't using those.
113    pub fn new(algorithm: AlgorithmIdentifier<Params>, private_key: Key) -> Self {
114        Self {
115            algorithm,
116            private_key,
117            public_key: None,
118        }
119    }
120
121    /// Get the PKCS#8 [`Version`] for this structure.
122    ///
123    /// [`Version::V1`] if `public_key` is `None`, [`Version::V2`] if `Some`.
124    pub fn version(&self) -> Version {
125        if self.public_key.is_some() {
126            Version::V2
127        } else {
128            Version::V1
129        }
130    }
131}

133impl<'a, Params, Key, PubKey> PrivateKeyInfo<Params, Key, PubKey>
134where
135    Params: der::Choice<'a, Error = der::Error> + Encode,
136    Key: DecodeValue<'a, Error = der::Error> + FixedTag + 'a,
137    Key: EncodeValue,
138    PubKey: DecodeValue<'a, Error = der::Error> + FixedTag + 'a,
139    PubKey: BitStringLike,
140{
141    /// Encrypt this private key using a symmetric encryption key derived
142    /// from the provided password.
143    ///
144    /// Uses the following algorithms for encryption:
145    /// - PBKDF: scrypt with default parameters:
146    ///   - logâ‚‚(N): 15
147    ///   - r: 8
148    ///   - p: 1
149    /// - Cipher: AES-256-CBC (best available option for PKCS#5 encryption)
150    #[cfg(feature = "encryption")]
151    pub fn encrypt<R: CryptoRng>(
152        &self,
153        rng: &mut R,
154        password: impl AsRef<[u8]>,
155    ) -> Result<SecretDocument> {
156        let der = Zeroizing::new(self.to_der()?);
157        EncryptedPrivateKeyInfoRef::encrypt(rng, password, der.as_ref())
158    }
159
160    /// Encrypt this private key using a symmetric encryption key derived
161    /// from the provided password and [`pbes2::Parameters`].
162    #[cfg(feature = "encryption")]
163    pub fn encrypt_with_params(
164        &self,
165        pbes2_params: pbes2::Parameters,
166        password: impl AsRef<[u8]>,
167    ) -> Result<SecretDocument> {
168        let der = Zeroizing::new(self.to_der()?);
169        EncryptedPrivateKeyInfoRef::encrypt_with(pbes2_params, password, der.as_ref())
170    }
171}

173impl<'a, Params, Key, PubKey> PrivateKeyInfo<Params, Key, PubKey>
174where
175    Params: der::Choice<'a> + Encode,
176    PubKey: BitStringLike,
177{
178    /// Get a `BIT STRING` representation of the public key, if present.
179    fn public_key_bit_string(&self) -> Option<ContextSpecific<BitStringRef<'_>>> {
180        self.public_key.as_ref().map(|pk| {
181            let value = pk.as_bit_string();
182            ContextSpecific {
183                tag_number: PUBLIC_KEY_TAG,
184                tag_mode: TagMode::Implicit,
185                value,
186            }
187        })
188    }
189}

191impl<'a, Params, Key, PubKey> DecodeValue<'a> for PrivateKeyInfo<Params, Key, PubKey>
192where
193    Params: der::Choice<'a, Error = der::Error> + Encode,
194    Key: DecodeValue<'a, Error = der::Error> + FixedTag + 'a,
195    PubKey: DecodeValue<'a, Error = der::Error> + FixedTag + 'a,
196{
197    type Error = der::Error;
198
199    fn decode_value<R: Reader<'a>>(reader: &mut R, _header: Header) -> der::Result<Self> {
200        // Parse and validate `version` INTEGER.
201        let version = Version::decode(reader)?;
202        let algorithm = reader.decode()?;
203        let private_key = Key::decode(reader)?;
204
205        let _attributes =
206            reader.context_specific::<SequenceRef<'_>>(ATTRIBUTES_TAG, TagMode::Implicit)?;
207
208        let public_key = reader.context_specific::<PubKey>(PUBLIC_KEY_TAG, TagMode::Implicit)?;
209
210        if version.has_public_key() != public_key.is_some() {
211            return Err(reader.error(
212                der::Tag::ContextSpecific {
213                    constructed: true,
214                    number: PUBLIC_KEY_TAG,
215                }
216                .value_error(),
217            ));
218        }
219
220        // Ignore any remaining extension fields
221        while !reader.is_finished() {
222            reader.decode::<ContextSpecific<AnyRef<'_>>>()?;
223        }
224
225        Ok(Self {
226            algorithm,
227            private_key,
228            public_key,
229        })
230    }
231}

233impl<'a, Params, Key, PubKey> EncodeValue for PrivateKeyInfo<Params, Key, PubKey>
234where
235    Params: der::Choice<'a, Error = der::Error> + Encode,
236    Key: EncodeValue + FixedTag,
237    PubKey: BitStringLike,
238{
239    fn value_len(&self) -> der::Result<Length> {
240        self.version().encoded_len()?
241            + self.algorithm.encoded_len()?
242            + self.private_key.encoded_len()?
243            + self.public_key_bit_string().encoded_len()?
244    }
245
246    fn encode_value(&self, writer: &mut impl Writer) -> der::Result<()> {
247        self.version().encode(writer)?;
248        self.algorithm.encode(writer)?;
249        self.private_key.encode(writer)?;
250        self.public_key_bit_string().encode(writer)?;
251        Ok(())
252    }
253}

331#[cfg(feature = "subtle")]
332impl<Params, Key, PubKey> ConstantTimeEq for PrivateKeyInfo<Params, Key, PubKey>
333where
334    Params: Eq,
335    Key: PartialEq + AsRef<[u8]>,
336    PubKey: PartialEq,
337{
338    fn ct_eq(&self, other: &Self) -> Choice {
339        // NOTE: public fields are not compared in constant time
340        let public_fields_eq =
341            self.algorithm == other.algorithm && self.public_key == other.public_key;
342
343        self.private_key.as_ref().ct_eq(other.private_key.as_ref())
344            & Choice::from(public_fields_eq as u8)
345    }
346}
