61/// Configuration for the Loggers
62///
63/// All loggers print the message in the following form:
64/// `00:00:00 [LEVEL] crate::module: [lib.rs::100] your_message`
65/// Every space delimited part except the actual message is optional.
66///
67/// Pass this struct to your logger to change when these information shall
68/// be logged.
69///
70/// Construct using [`Default`](Config::default) or using [`ConfigBuilder`]
71#[derive(Debug, Clone)]
72pub struct Config {
73    pub(crate) time: LevelFilter,
74    pub(crate) level: LevelFilter,
75    pub(crate) level_padding: LevelPadding,
76    pub(crate) thread: LevelFilter,
77    pub(crate) thread_log_mode: ThreadLogMode,
78    pub(crate) thread_padding: ThreadPadding,
79    pub(crate) target: LevelFilter,
80    pub(crate) target_padding: TargetPadding,
81    pub(crate) location: LevelFilter,
82    pub(crate) time_format: TimeFormat,
83    pub(crate) time_offset: UtcOffset,
84    pub(crate) filter_allow: Cow<'static, [Cow<'static, str>]>,
85    pub(crate) filter_ignore: Cow<'static, [Cow<'static, str>]>,
86    #[cfg(feature = "termcolor")]
87    pub(crate) level_color: [Option<Color>; 6],
88    pub(crate) write_log_enable_colors: bool,
89}

91/// Builder for the Logger Configurations (`Config`)
92///
93/// All loggers print the message in the following form:
94/// `00:00:00 [LEVEL] crate::module: [lib.rs::100] your_message`
95/// Every space delimited part except the actual message is optional.
96///
97/// Use this struct to create a custom `Config` changing when these information shall
98/// be logged. Every part can be enabled for a specific Level and is then
99/// automatically enable for all lower levels as well.
100///
101/// The Result is that the logging gets more detailed the more verbose it gets.
102/// E.g. to have one part shown always use `Level::Error`. But if you
103/// want to show the source line only on `Trace` use that.
104#[derive(Debug, Clone)]
105#[non_exhaustive]
106pub struct ConfigBuilder(Config);

176    /// Sets the time format to a custom representation.
177    ///
178    /// The easiest way to satisfy the static lifetime of the argument is to directly use the
179    /// re-exported [`time::macros::format_description`] macro.
180    ///
181    /// *Note*: The default time format is "[hour]:[minute]:[second]".
182    ///
183    /// The syntax for the format_description macro can be found in the
184    /// [`time` crate book](https://time-rs.github.io/book/api/format-description.html).
185    ///
186    /// # Usage
187    ///
188    /// ```
189    /// # use simplelog::{ConfigBuilder, format_description};
190    /// let config = ConfigBuilder::new()
191    ///     .set_time_format_custom(format_description!("[hour]:[minute]:[second].[subsecond]"))
192    ///     .build();
193    /// ```
194    pub fn set_time_format_custom(
195        &mut self,
196        time_format: &'static [FormatItem<'static>],
197    ) -> &mut ConfigBuilder {
198        self.0.time_format = TimeFormat::Custom(time_format);
199        self
200    }

220    /// Sets the offset used to the current local time offset
221    /// (overriding values previously set by [`ConfigBuilder::set_time_offset`]).
222    ///
223    /// This function may fail if the offset cannot be determined soundly.
224    /// This may be the case, when the program is multi-threaded by the time of calling this function.
225    /// One can opt-out of this behavior by setting `RUSTFLAGS="--cfg unsound_local_offset"`.
226    /// Doing so is not recommended, completely untested and may cause unexpected segfaults.
227    #[cfg(feature = "local-offset")]
228    pub fn set_time_offset_to_local(&mut self) -> Result<&mut ConfigBuilder, &mut ConfigBuilder> {
229        match UtcOffset::current_local_offset() {
230            Ok(offset) => {
231                self.0.time_offset = offset;
232                Ok(self)
233            }
234            Err(_) => Err(self),
235        }
236    }

245    /// Add allowed module filters.
246    /// If any are specified, only records from modules starting with one of these entries will be printed
247    ///

249    pub fn add_filter_allow_str(&mut self, filter_allow: &'static str) -> &mut ConfigBuilder {
